42    public void shouldCreateAProxyForAPartiallyLoadedBean() throws Exception {
43      ResultLoaderMap loader = new ResultLoaderMap();
44      loader.addLoader("id", null, null);
45      Object proxy = proxyFactory.createProxy(author, loader, new Configuration(), new DefaultObjectFactory(), new ArrayList<Class<?>>(), new ArrayList<Object>());
46      Author author2 = (Author) deserialize(serialize((Serializable) proxy));
47  //    assertTrue(author2.getClass().getName().contains("CGLIB"));
48      assertTrue(author2 instanceof Proxy);
49    }

51    @Test(expected = ExecutorException.class)
52    public void shouldFailCallingAnUnloadedProperty() throws Exception {
53      // yes, it must go in uppercase
54      HashSet<String> unloadedProperties = new HashSet<String>();
55      unloadedProperties.add("ID");
56      Author author2 = (Author) ((JavassistProxyFactory)proxyFactory).createDeserializationProxy(author, unloadedProperties, new DefaultObjectFactory(), new ArrayList<Class<?>>(), new ArrayList<Object>());
57      author2.getId();
58    }
59  
60    @Test
61    public void shouldLetCallALoadedProperty() throws Exception {
62      Author author2 = (Author) ((JavassistProxyFactory)proxyFactory).createDeserializationProxy(author, new HashSet<String>(), new DefaultObjectFactory(), new ArrayList<Class<?>>(), new ArrayList<Object>());
63      assertEquals(999, author2.getId());
64    }
65  
66    @Test
67    public void shouldSerizalizeADeserlizaliedProxy() throws Exception {
68      Object proxy = ((JavassistProxyFactory)proxyFactory).createDeserializationProxy(author, new HashSet<String>(), new DefaultObjectFactory(), new ArrayList<Class<?>>(), new ArrayList<Object>());
69      Author author2 = (Author) deserialize(serialize((Serializable) proxy));
70      assertEquals(author, author2);
71      assertFalse(author.getClass().equals(author2.getClass()));

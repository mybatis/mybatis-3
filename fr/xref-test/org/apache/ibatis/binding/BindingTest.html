662   @Test
663   public void shouldCacheMapperMethod() throws Exception {
664     final SqlSession session = sqlSessionFactory.openSession();
665     try {
666 
667       // Create another mapper instance with a method cache we can test against:
668       final MapperProxyFactory<BoundBlogMapper> mapperProxyFactory = new MapperProxyFactory<BoundBlogMapper>(BoundBlogMapper.class);
669       assertEquals(BoundBlogMapper.class, mapperProxyFactory.getMapperInterface());
670       final BoundBlogMapper mapper = mapperProxyFactory.newInstance(session);
671       assertNotSame(mapper, mapperProxyFactory.newInstance(session));
672       assertTrue(mapperProxyFactory.getMethodCache().isEmpty());
673 
674       // Mapper methods we will call later:
675       final Method selectBlog = BoundBlogMapper.class.getMethod("selectBlog", Integer.TYPE);
676       final Method selectBlogByIdUsingConstructor = BoundBlogMapper.class.getMethod("selectBlogByIdUsingConstructor", Integer.TYPE);
677 
678       // Call mapper method and verify it is cached:
679       mapper.selectBlog(1);
680       assertEquals(1, mapperProxyFactory.getMethodCache().size());
681       assertTrue(mapperProxyFactory.getMethodCache().containsKey(selectBlog));
682       final MapperMethod cachedSelectBlog = mapperProxyFactory.getMethodCache().get(selectBlog);
683 
684       // Call mapper method again and verify the cache is unchanged:
685       session.clearCache();
686       mapper.selectBlog(1);
687       assertEquals(1, mapperProxyFactory.getMethodCache().size());
688       assertSame(cachedSelectBlog, mapperProxyFactory.getMethodCache().get(selectBlog));
689 
690       // Call another mapper method and verify that it shows up in the cache as well:
691       session.clearCache();
692       mapper.selectBlogByIdUsingConstructor(1);

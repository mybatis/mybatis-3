303   @Test
304   public void shouldIterateOnceForEachItemInCollection() throws Exception {
305     final HashMap<String, String[]> parameterObject = new HashMap<String, String[]>() {{
306       put("array", new String[]{"one", "two", "three"});
307     }};
308     final String expected = "SELECT * FROM BLOG WHERE ID in (  one = ? AND two = ? AND three = ? )";
309     DynamicSqlSource source = createDynamicSqlSource(
310         new TextSqlNode("SELECT * FROM BLOG WHERE ID in"),
311         new ForEachSqlNode(new Configuration(),mixedContents(new TextSqlNode("${item} = #{item}")), "array", "index", "item", "(", ")", "AND"));
312     BoundSql boundSql = source.getBoundSql(parameterObject);
313     assertEquals(expected, boundSql.getSql());
314     assertEquals(3, boundSql.getParameterMappings().size());
315     assertEquals("__frch_item_0", boundSql.getParameterMappings().get(0).getProperty());
316     assertEquals("__frch_item_1", boundSql.getParameterMappings().get(1).getProperty());
317     assertEquals("__frch_item_2", boundSql.getParameterMappings().get(2).getProperty());
318   }

345   @Test
346   public void shouldPerformStrictMatchOnForEachVariableSubstitution() throws Exception {
347     final Map<String, Object> param = new HashMap<String, Object>();
348     final Map<String, String> uuu = new HashMap<String, String>();
349     uuu.put("u", "xyz");
350     List<Bean> uuuu = new ArrayList<Bean>();
351     uuuu.add(new Bean("bean id"));
352     param.put("uuu", uuu);
353     param.put("uuuu", uuuu);
354     DynamicSqlSource source = createDynamicSqlSource(
355         new TextSqlNode("INSERT INTO BLOG (ID, NAME, NOTE, COMMENT) VALUES"),
356         new ForEachSqlNode(new Configuration(),mixedContents(
357             new TextSqlNode("#{uuu.u}, #{u.id}, #{ u,typeHandler=org.apache.ibatis.type.StringTypeHandler},"
358                 + " #{u:VARCHAR,typeHandler=org.apache.ibatis.type.StringTypeHandler}")), "uuuu", "uu", "u", "(", ")", ","));
359     BoundSql boundSql = source.getBoundSql(param);
360     assertEquals(4, boundSql.getParameterMappings().size());
361     assertEquals("uuu.u", boundSql.getParameterMappings().get(0).getProperty());
362     assertEquals("__frch_u_0.id", boundSql.getParameterMappings().get(1).getProperty());
363     assertEquals("__frch_u_0", boundSql.getParameterMappings().get(2).getProperty());
364     assertEquals("__frch_u_0", boundSql.getParameterMappings().get(3).getProperty());
365   }

367   private DynamicSqlSource createDynamicSqlSource(SqlNode... contents) throws IOException, SQLException {
368     createBlogDataSource();
369     final String resource = "org/apache/ibatis/builder/MapperConfig.xml";
370     final Reader reader = Resources.getResourceAsReader(resource);
371     SqlSessionFactory sqlMapper = new SqlSessionFactoryBuilder().build(reader);
372     Configuration configuration = sqlMapper.getConfiguration();
373     MixedSqlNode sqlNode = mixedContents(contents);
374     return new DynamicSqlSource(configuration, sqlNode);
375   }
376 
377   private MixedSqlNode mixedContents(SqlNode... contents) {
378     return new MixedSqlNode(Arrays.asList(contents));
379   }

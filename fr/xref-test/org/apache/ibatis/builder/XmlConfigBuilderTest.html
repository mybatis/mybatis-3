105   public static class EnumOrderTypeHandler<E extends Enum<E>> extends BaseTypeHandler<E> {
106 
107     private E[] constants;
108 
109     public EnumOrderTypeHandler(Class<E> javaType) {
110       constants = javaType.getEnumConstants();
111     }
112 
113     @Override
114     public void setNonNullParameter(PreparedStatement ps, int i, E parameter, JdbcType jdbcType) throws SQLException {
115       ps.setInt(i, parameter.ordinal() + 1); // 0 means NULL so add +1
116     }
117 
118     @Override
119     public E getNullableResult(ResultSet rs, String columnName) throws SQLException {
120       int index = rs.getInt(columnName) - 1;
121       return index < 0 ? null : constants[index];
122     }
123 
124     @Override
125     public E getNullableResult(ResultSet rs, int columnIndex) throws SQLException {
126       int index = rs.getInt(rs.getInt(columnIndex)) - 1;
127       return index < 0 ? null : constants[index];
128     }
129 
130     @Override
131     public E getNullableResult(CallableStatement cs, int columnIndex) throws SQLException {
132       int index = cs.getInt(columnIndex) - 1;
133       return index < 0 ? null : constants[index];
134     }
135   }

138   public void registerJavaTypeInitializingTypeHandler() {
139     final String MAPPER_CONFIG = "<?xml version=\"1.0\" encoding=\"UTF-8\" ?>\n"
140         + "<!DOCTYPE configuration PUBLIC \"-//mybatis.org//DTD Config 3.0//EN\" \"http://mybatis.org/dtd/mybatis-3-config.dtd\">\n" 
141         + "<configuration>\n" 
142         + "  <typeHandlers>\n"
143         + "    <typeHandler javaType=\"org.apache.ibatis.builder.XmlConfigBuilderTest$MyEnum\"\n"
144         + "      handler=\"org.apache.ibatis.builder.XmlConfigBuilderTest$EnumOrderTypeHandler\"/>\n" 
145         + "  </typeHandlers>\n" 
146         + "</configuration>\n";
147 
148     XMLConfigBuilder builder = new XMLConfigBuilder(new StringReader(MAPPER_CONFIG));
149     builder.parse();
150 
151     TypeHandlerRegistry typeHandlerRegistry = builder.getConfiguration().getTypeHandlerRegistry();
152     TypeHandler<MyEnum> typeHandler = typeHandlerRegistry.getTypeHandler(MyEnum.class);
153 
154     assertTrue(typeHandler instanceof EnumOrderTypeHandler);
155     assertArrayEquals(MyEnum.values(), ((EnumOrderTypeHandler) typeHandler).constants);
156   }

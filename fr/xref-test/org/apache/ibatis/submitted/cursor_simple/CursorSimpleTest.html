59      @Test
60      public void shouldGetAllUser() {
61          SqlSession sqlSession = sqlSessionFactory.openSession();
62          Mapper mapper = sqlSession.getMapper(Mapper.class);
63          Cursor<User> usersCursor = mapper.getAllUsers();
64          try {
65              Assert.assertFalse(usersCursor.isOpen());
66  
67              // Cursor is just created, current index is -1
68              Assert.assertEquals(-1, usersCursor.getCurrentIndex());
69  
70              Iterator<User> iterator = usersCursor.iterator();
71  
72              // Check if hasNext, fetching is started
73              Assert.assertTrue(iterator.hasNext());
74              Assert.assertTrue(usersCursor.isOpen());
75              Assert.assertFalse(usersCursor.isConsumed());
76  
77              // next() has not been called, index is still -1
78              Assert.assertEquals(-1, usersCursor.getCurrentIndex());
79  
80              User user = iterator.next();
81              Assert.assertEquals("User1", user.getName());
82              Assert.assertEquals(0, usersCursor.getCurrentIndex());
83  
84              user = iterator.next();
85              Assert.assertEquals("User2", user.getName());
86              Assert.assertEquals(1, usersCursor.getCurrentIndex());
87  
88              user = iterator.next();
89              Assert.assertEquals("User3", user.getName());
90              Assert.assertEquals(2, usersCursor.getCurrentIndex());
91  
92              user = iterator.next();
93              Assert.assertEquals("User4", user.getName());
94              Assert.assertEquals(3, usersCursor.getCurrentIndex());
95  
96              user = iterator.next();
97              Assert.assertEquals("User5", user.getName());
98              Assert.assertEquals(4, usersCursor.getCurrentIndex());
99  
100             // Check no more elements
101             Assert.assertFalse(iterator.hasNext());
102             Assert.assertFalse(usersCursor.isOpen());
103             Assert.assertTrue(usersCursor.isConsumed());
104         } finally {
105             sqlSession.close();
106         }
107     }

112         Mapper mapper = sqlSession.getMapper(Mapper.class);
113         Cursor<User> usersCursor = mapper.getAllUsers();
114         try {

120             Assert.assertTrue(iterator.hasNext());
121             Assert.assertTrue(usersCursor.isOpen());
122             Assert.assertFalse(usersCursor.isConsumed());
123 
124             // Consume only the first result
125             User user = iterator.next();
126             Assert.assertEquals("User1", user.getName());
127 
128             // Check there is still remaining elements
129             Assert.assertTrue(iterator.hasNext());
130             Assert.assertTrue(usersCursor.isOpen());
131             Assert.assertFalse(usersCursor.isConsumed());
132         } finally {
133             sqlSession.close();
134         }
135 
136         // The cursor was not fully consumed, but it should be close since we closed the session
137         Assert.assertFalse(usersCursor.isOpen());
138         Assert.assertFalse(usersCursor.isConsumed());

141     @Test
142     public void testCursorWithRowBound() {
143         SqlSession sqlSession = sqlSessionFactory.openSession();
144 
145         try {
146             // RowBound starting at offset 1 and limiting to 2 items
147             Cursor<User> usersCursor = sqlSession.selectCursor("getAllUsers", null, new RowBounds(1, 3));
148 
149             Iterator<User> iterator = usersCursor.iterator();
150 
151             User user = iterator.next();
152             Assert.assertEquals("User2", user.getName());
153             Assert.assertEquals(1, usersCursor.getCurrentIndex());
154 
155             // Calling hasNext() before next()
156             Assert.assertTrue(iterator.hasNext());
157             user = iterator.next();
158             Assert.assertEquals("User3", user.getName());
159             Assert.assertEquals(2, usersCursor.getCurrentIndex());
160 
161             // Calling next() without a previous hasNext() call
162             user = iterator.next();
163             Assert.assertEquals("User4", user.getName());
164             Assert.assertEquals(3, usersCursor.getCurrentIndex());
165 
166             Assert.assertFalse(iterator.hasNext());
167             Assert.assertFalse(usersCursor.isOpen());
168             Assert.assertTrue(usersCursor.isConsumed());
169         } finally {
170             sqlSession.close();
171         }
172     }

320     @Test
321     public void testCursorUsageAfterClose() throws IOException {
322         SqlSession sqlSession = sqlSessionFactory.openSession();
323         Mapper mapper = sqlSession.getMapper(Mapper.class);
324         Cursor<User> usersCursor = mapper.getAllUsers();
325 
326         try {
327             Iterator<User> iterator = usersCursor.iterator();
328             User user = iterator.next();
329             Assert.assertEquals("User1", user.getName());
330             Assert.assertEquals(0, usersCursor.getCurrentIndex());
331 
332             user = iterator.next();
333             Assert.assertEquals("User2", user.getName());
334             Assert.assertEquals(1, usersCursor.getCurrentIndex());
335 
336             usersCursor.close();
337 
338             // hasNext now return false, since the cursor is closed
339             Assert.assertFalse(iterator.hasNext());
340             Assert.assertFalse(usersCursor.isOpen());
341             Assert.assertFalse(usersCursor.isConsumed());
342 
343             // trying next() will fail
344             iterator.next();
345 
346             Assert.fail("We should have failed with NoSuchElementException since Cursor is closed");
347         } catch (NoSuchElementException e) {
348             // We had an exception and current index has not changed
349             Assert.assertEquals(1, usersCursor.getCurrentIndex());
350             return;
351         } finally {
352             sqlSession.close();
353         }
354 
355         Assert.fail("Should have returned earlier");
356     }

358     @Test
359     public void shouldGetAllUserUsingAnnotationBasedMapper() {
360         SqlSession sqlSession = sqlSessionFactory.openSession();
361         sqlSession.getConfiguration().getMapperRegistry().addMapper(AnnotationMapper.class);
362         AnnotationMapper mapper = sqlSession.getMapper(AnnotationMapper.class);
363         Cursor<User> usersCursor = mapper.getAllUsers();
364         try {
365             Assert.assertFalse(usersCursor.isOpen());
366             Assert.assertFalse(usersCursor.isConsumed());
367             Assert.assertEquals(-1, usersCursor.getCurrentIndex());
368 
369             List<User> userList = new ArrayList<User>();
370             for (User user : usersCursor){
371                 userList.add(user);
372                 Assert.assertEquals(userList.size() -1, usersCursor.getCurrentIndex());
373             }
374 
375             Assert.assertFalse(usersCursor.isOpen());
376             Assert.assertTrue(usersCursor.isConsumed());
377             Assert.assertEquals(4, usersCursor.getCurrentIndex());
378 
379             Assert.assertEquals(5, userList.size());
380             User user = userList.get(0);
381             Assert.assertEquals("User1", user.getName());
382             user = userList.get(1);
383             Assert.assertEquals("User2", user.getName());
384             user = userList.get(2);
385             Assert.assertEquals("User3", user.getName());
386             user = userList.get(3);
387             Assert.assertEquals("User4", user.getName());
388             user = userList.get(4);
389             Assert.assertEquals("User5", user.getName());
390 
391         } finally {
392             sqlSession.close();
393         }

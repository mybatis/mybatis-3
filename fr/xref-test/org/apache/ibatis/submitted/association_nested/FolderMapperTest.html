34  public class FolderMapperTest {
35  
36    @Test
37    public void testFindWithChildren() throws Exception {
38      Connection conn = DriverManager.getConnection("jdbc:hsqldb:mem:association_nested", "SA", "");
39      Statement stmt = conn.createStatement();
40      stmt.execute("create table folder (id int, name varchar(100), parent_id int)");
41  
42  
43      stmt.execute("insert into folder (id, name) values(1, 'Root')");
44      stmt.execute("insert into folder values(2, 'Folder 1', 1)");
45      stmt.execute("insert into folder values(3, 'Folder 2', 1)");
46      stmt.execute("insert into folder values(4, 'Folder 2_1', 3)");
47      stmt.execute("insert into folder values(5, 'Folder 2_2', 3)");
48  
49      /**
50       * Root/
51       *    Folder 1/
52       *    Folder 2/
53       *      Folder 2_1
54       *      Folder 2_2
55       */
56  
57      String resource = "org/apache/ibatis/submitted/association_nested/mybatis-config.xml";
58      InputStream inputStream = Resources.getResourceAsStream(resource);
59      SqlSessionFactory sqlSessionFactory = new SqlSessionFactoryBuilder().build(inputStream);
60  
61      SqlSession session = sqlSessionFactory.openSession();
62      FolderMapper postMapper = session.getMapper(FolderMapper.class);
63  
64      List<FolderFlatTree> folders = postMapper.findWithSubFolders("Root");
65  
66      Assert.assertEquals(3, folders.size());
67  
68      session.close();
69    }
70  
71  }

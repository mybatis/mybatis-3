16  package org.apache.ibatis.submitted.overwritingproperties;
17  
18  import org.apache.ibatis.io.Resources;
19  import org.apache.ibatis.jdbc.ScriptRunner;
20  import org.apache.ibatis.session.*;
21  import org.junit.*;
22  
23  import java.io.Reader;
24  import java.sql.Connection;
25  
26  /*
27   * @author jjensen
28   */
29  public class FooMapperTest {
30  
31    private final static String SQL_MAP_CONFIG = "org/apache/ibatis/submitted/overwritingproperties/sqlmap.xml";
32    private static SqlSession session;
33  
34    @BeforeClass
35    public static void setUpBeforeClass() {
36      try {
37        final SqlSessionFactory factory = new SqlSessionFactoryBuilder().build(Resources.getResourceAsReader(SQL_MAP_CONFIG));
38        session = factory.openSession();
39        Connection conn = session.getConnection();
40        ScriptRunner runner = new ScriptRunner(conn);
41        runner.setLogWriter(null);
42        runner.setErrorLogWriter(null);
43        Reader reader = Resources.getResourceAsReader("org/apache/ibatis/submitted/overwritingproperties/create-schema-mysql.sql");
44        runner.runScript(reader);
45      } catch (Exception ex) {
46        ex.printStackTrace();
47      }
48    }
49  
50    @Before
51    public void setUp() {
52      final FooMapper mapper = session.getMapper(FooMapper.class);
53      mapper.deleteAllFoo();
54      session.commit();
55    }
56  
57    @Test
58    public void testOverwriteWithDefault() {
59      final FooMapper mapper = session.getMapper(FooMapper.class);
60      final Bar bar = new Bar(2L);
61      final Foo inserted = new Foo(1L, bar, 3, 4);
62      mapper.insertFoo(inserted);
63  
64      final Foo selected = mapper.selectFoo();
65      
66      // field1 is explicitly mapped properly
67      // <result property="field1" column="field1" jdbcType="INTEGER"/>
68      Assert.assertEquals(inserted.getField1(), selected.getField1());
69  
70      // field4 is not mapped in the result map
71      // <result property="field4" column="field3" jdbcType="INTEGER"/>
72      Assert.assertEquals(inserted.getField3(), selected.getField4() );
73  
74      // field4 is explicitly remapped to field3 in the resultmap
75      // <result property="field3" column="field4" jdbcType="INTEGER"/>
76      Assert.assertEquals(inserted.getField4(), selected.getField3());
77  
78      // is automapped from the only column that matches... which is Field1
79      // probably not the intention, but it's working correctly given the code
80      // <association property="field2" javaType="Bar">
81      //  <result property="field1" column="bar_field1" jdbcType="INTEGER"/>
82      // </association>
83      Assert.assertEquals(inserted.getField2().getField1(), selected.getField2().getField1());
84    }
85  
86    @AfterClass
87    public static void tearDownAfterClass() {
88      session.close();
89    }
90  
91  }

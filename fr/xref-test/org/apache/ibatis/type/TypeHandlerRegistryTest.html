39    @Test
40    public void shouldRegisterAndRetrieveTypeHandler() {
41      TypeHandler<String> stringTypeHandler = typeHandlerRegistry.getTypeHandler(String.class);
42      typeHandlerRegistry.register(String.class, JdbcType.LONGVARCHAR, stringTypeHandler);
43      assertEquals(stringTypeHandler, typeHandlerRegistry.getTypeHandler(String.class, JdbcType.LONGVARCHAR));
44  
45      assertTrue(typeHandlerRegistry.hasTypeHandler(String.class));
46      assertFalse(typeHandlerRegistry.hasTypeHandler(RichType.class));
47      assertTrue(typeHandlerRegistry.hasTypeHandler(String.class, JdbcType.LONGVARCHAR));
48      assertTrue(typeHandlerRegistry.hasTypeHandler(String.class, JdbcType.INTEGER));
49      assertTrue(typeHandlerRegistry.getUnknownTypeHandler() instanceof UnknownTypeHandler);
50    }

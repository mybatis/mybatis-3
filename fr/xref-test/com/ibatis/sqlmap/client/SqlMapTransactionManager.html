16  package com.ibatis.sqlmap.client;
17  
18  import javax.sql.DataSource;
19  import java.sql.Connection;
20  import java.sql.SQLException;
21  
22  /*
23   * This interface declares methods for demarcating SQL Map transactions.
24   *
25   * @see com.ibatis.sqlmap.client.SqlMapSession
26   * @see SqlMapClient
27   */
28  public interface SqlMapTransactionManager {
29  
30    /*
31     * Demarcates the beginning of a transaction scope.  Transactions must be properly
32     * committed or rolled back to be effective.  Use the following pattern when working
33     * with transactions:
34     * <pre>
35     * try {
36     *   sqlMap.startTransaction();
37     *   // do work
38     *   sqlMap.commitTransaction();
39     * } finally {
40     *   sqlMap.endTransaction();
41     * }
42     * </pre>
43     * <p/>
44     * Always call endTransaction() once startTransaction() has been called.
45     *
46     * @throws java.sql.SQLException If an error occurs while starting the transaction, or
47     *                               the transaction could not be started.
48     */
49    public void startTransaction() throws SQLException;
50  
51  
52    /*
53     * Demarcates the beginning of a transaction scope using the specified transaction
54     * isolation.  Transactions must be properly committed or rolled back to be effective.
55     * Use the following pattern when working with transactions:
56     * <pre>
57     * try {
58     *   sqlMap.startTransaction(Connection.TRANSACTION_REPEATABLE_READ);
59     *   // do work
60     *   sqlMap.commitTransaction();
61     * } finally {
62     *   sqlMap.endTransaction();
63     * }
64     * </pre>
65     * <p/>
66     * Always call endTransaction() once startTransaction() has been called.
67     *
68     * @throws java.sql.SQLException If an error occurs while starting the transaction, or
69     *                               the transaction could not be started.
70     */
71    public void startTransaction(int transactionIsolation) throws SQLException;
72  
73    /*
74     * Commits the currently started transaction.
75     *
76     * @throws java.sql.SQLException If an error occurs while committing the transaction, or
77     *                               the transaction could not be committed.
78     */
79    public void commitTransaction() throws SQLException;
80  
81    /*
82     * Ends a transaction and rolls back if necessary.  If the transaction has
83     * been started, but not committed, it will be rolled back upon calling
84     * endTransaction().
85     *
86     * @throws java.sql.SQLException If an error occurs during rollback or the transaction could
87     *                               not be ended.
88     */
89    public void endTransaction() throws SQLException;
90  
91    /*
92     * Allows the developer to easily use an externally supplied connection
93     * when executing statements.
94     * <p/>
95     * <b>Important:</b> Using a user supplied connection basically sidesteps the transaction manager,
96     * so you are responsible for appropriately.  Here's a (very) simple example (throws SQLException):
97     * <pre>
98     * try {
99     *   Connection connection = dataSource.getConnection();
100    *   sqlMap.setUserConnection(connection);
101    *   // do work
102    *   connection.commit();
103    * } catch (SQLException e) {
104    *     try {
105    *       if (connection != null) commit.rollback();
106    *     } catch (SQLException ignored) {
107    *       // generally ignored
108    *     }
109    *     throw e;  // rethrow the exception
110    * } finally {
111    *   try {
112    *     if (connection != null) connection.close();
113    *   } catch (SQLException ignored) {
114    *     // generally ignored
115    *   }
116    * }
117    * </pre>
118    *
119    * @param connnection
120    * @throws java.sql.SQLException
121    */
122   public void setUserConnection(Connection connnection) throws SQLException;
123 
124   /*
125    * Returns the current connection in use.  If no connection exists null will
126    * be returned. There may be no connection if no transaction has been started,
127    * and if no user provided connection has been set.
128    *
129    * @return The current connection or null.
130    * @throws java.sql.SQLException
131    */
132   public Connection getCurrentConnection() throws SQLException;
133 
134   /*
135    * Returns the DataSource instance currently being used by the SqlMapSession.
136    *
137    * @return The DataSource instance currently being used by the SqlMapSession.
138    */
139   public DataSource getDataSource();
140

142 }

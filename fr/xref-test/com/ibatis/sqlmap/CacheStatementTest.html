265   private static class TestCacheThread extends Thread {
266     private SqlMapClient sqlMap;
267     private Map results;
268     private String statementName;
269 
270     public TestCacheThread(SqlMapClient sqlMap, Map results, String statementName) {
271       this.sqlMap = sqlMap;
272       this.results = results;
273       this.statementName = statementName;
274     }
275 
276     public void run() {
277       try {
278         SqlMapSession session = sqlMap.openSession();
279         List list = session.queryForList(statementName, null);
280         int firstId = System.identityHashCode(list);
281         list = session.queryForList(statementName, null);
282         int secondId = System.identityHashCode(list);
283         //assertEquals(firstId, secondId);
284         results.put("id", new Integer(System.identityHashCode(list)));
285         results.put("list", list);
286         session.close();
287       } catch (SQLException e) {
288         throw new RuntimeException("Error.  Cause: " + e);
289       }
290     }
291 
292     public static void startThread(SqlMapClient sqlMap, Map results, String statementName) {
293       Thread t = new TestCacheThread(sqlMap, results, statementName);
294       t.start();
295       try {
296         t.join();
297       } catch (InterruptedException e) {
298         throw new RuntimeException("Error.  Cause: " + e);
299       }
300     }
301   }

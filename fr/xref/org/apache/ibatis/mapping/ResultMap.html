66      public ResultMap build() {
67        if (resultMap.id == null) {
68          throw new IllegalArgumentException("ResultMaps must have an id");
69        }
70        resultMap.mappedColumns = new HashSet<String>();
71        resultMap.idResultMappings = new ArrayList<ResultMapping>();
72        resultMap.constructorResultMappings = new ArrayList<ResultMapping>();
73        resultMap.propertyResultMappings = new ArrayList<ResultMapping>();
74        for (ResultMapping resultMapping : resultMap.resultMappings) {
75          resultMap.hasNestedQueries = resultMap.hasNestedQueries || resultMapping.getNestedQueryId() != null;
76          resultMap.hasNestedResultMaps = resultMap.hasNestedResultMaps || (resultMapping.getNestedResultMapId() != null && resultMapping.getResultSet() == null);
77          final String column = resultMapping.getColumn();
78          if (column != null) {
79            resultMap.mappedColumns.add(column.toUpperCase(Locale.ENGLISH));
80          } else if (resultMapping.isCompositeResult()) {
81            for (ResultMapping compositeResultMapping : resultMapping.getComposites()) {
82              final String compositeColumn = compositeResultMapping.getColumn();
83              if (compositeColumn != null) {
84                resultMap.mappedColumns.add(compositeColumn.toUpperCase(Locale.ENGLISH));
85              }
86            }
87          }
88          if (resultMapping.getFlags().contains(ResultFlag.CONSTRUCTOR)) {
89            resultMap.constructorResultMappings.add(resultMapping);
90          } else {
91            resultMap.propertyResultMappings.add(resultMapping);
92          }
93          if (resultMapping.getFlags().contains(ResultFlag.ID)) {
94            resultMap.idResultMappings.add(resultMapping);
95          }
96        }
97        if (resultMap.idResultMappings.isEmpty()) {
98          resultMap.idResultMappings.addAll(resultMap.resultMappings);
99        }
100       // lock down collections
101       resultMap.resultMappings = Collections.unmodifiableList(resultMap.resultMappings);
102       resultMap.idResultMappings = Collections.unmodifiableList(resultMap.idResultMappings);
103       resultMap.constructorResultMappings = Collections.unmodifiableList(resultMap.constructorResultMappings);
104       resultMap.propertyResultMappings = Collections.unmodifiableList(resultMap.propertyResultMappings);
105       resultMap.mappedColumns = Collections.unmodifiableSet(resultMap.mappedColumns);
106       return resultMap;
107     }
108   }

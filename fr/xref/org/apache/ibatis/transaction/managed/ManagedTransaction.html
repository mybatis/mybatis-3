16  package org.apache.ibatis.transaction.managed;
17  
18  import java.sql.Connection;
19  import java.sql.SQLException;
20  
21  import javax.sql.DataSource;
22  
23  import org.apache.ibatis.logging.Log;
24  import org.apache.ibatis.logging.LogFactory;
25  import org.apache.ibatis.session.TransactionIsolationLevel;
26  import org.apache.ibatis.transaction.Transaction;
27  
28  /**
29   * {@link Transaction} that lets the container manage the full lifecycle of the transaction.
30   * Delays connection retrieval until getConnection() is called.
31   * Ignores all commit or rollback requests.
32   * By default, it closes the connection but can be configured not to do it.
33   *
34   * @see ManagedTransactionFactory
35   */
36  public class ManagedTransaction implements Transaction {
37  
38    private static final Log log = LogFactory.getLog(ManagedTransaction.class);
39  
40    private DataSource dataSource;
41    private TransactionIsolationLevel level;
42    private Connection connection;
43    private boolean closeConnection;
44  
45    public ManagedTransaction(Connection connection, boolean closeConnection) {
46      this.connection = connection;
47      this.closeConnection = closeConnection;
48    }
49  
50    public ManagedTransaction(DataSource ds, TransactionIsolationLevel level, boolean closeConnection) {
51      this.dataSource = ds;
52      this.level = level;
53      this.closeConnection = closeConnection;
54    }
55  
56    public Connection getConnection() throws SQLException {
57      if (this.connection == null) {
58        openConnection();
59      }
60      return this.connection;
61    }
62  
63    public void commit() throws SQLException {
64      // Does nothing
65    }
66  
67    public void rollback() throws SQLException {
68      // Does nothing
69    }
70  
71    public void close() throws SQLException {
72      if (this.closeConnection && this.connection != null) {
73        if (log.isDebugEnabled()) {
74          log.debug("Closing JDBC Connection [" + this.connection + "]");
75        }
76        this.connection.close();
77      }
78    }
79  
80    protected void openConnection() throws SQLException {
81      if (log.isDebugEnabled()) {
82        log.debug("Opening JDBC Connection");
83      }
84      this.connection = this.dataSource.getConnection();
85      if (this.level != null) {
86        this.connection.setTransactionIsolation(this.level.getLevel());
87      }
88    }
89  
90  }

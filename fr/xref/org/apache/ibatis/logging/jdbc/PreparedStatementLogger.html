16  package org.apache.ibatis.logging.jdbc;
17  
18  import java.lang.reflect.InvocationHandler;
19  import java.lang.reflect.Method;
20  import java.lang.reflect.Proxy;
21  import java.sql.CallableStatement;
22  import java.sql.PreparedStatement;
23  import java.sql.ResultSet;
24  
25  import org.apache.ibatis.logging.Log;
26  import org.apache.ibatis.reflection.ExceptionUtil;
27  
28  /*
29   * PreparedStatement proxy to add logging
30   */
31  public final class PreparedStatementLogger extends BaseJdbcLogger implements InvocationHandler {
32  
33    private PreparedStatement statement;
34  
35    private PreparedStatementLogger(PreparedStatement stmt, Log statementLog) {
36      super(statementLog);
37      this.statement = stmt;
38    }
39  
40    public Object invoke(Object proxy, Method method, Object[] params) throws Throwable {
41      try {
42        if (EXECUTE_METHODS.contains(method.getName())) {
43          if (isDebugEnabled()) {
44            debug("==> Parameters: " + getParameterValueString());
45          }
46          clearColumnInfo();
47          if ("executeQuery".equals(method.getName())) {
48            ResultSet rs = (ResultSet) method.invoke(statement, params);
49            if (rs != null) {
50              return ResultSetLogger.newInstance(rs, getStatementLog());
51            } else {
52              return null;
53            }
54          } else {
55            return method.invoke(statement, params);
56          }
57        } else if (SET_METHODS.contains(method.getName())) {
58          if ("setNull".equals(method.getName())) {
59            setColumn(params[0], null);
60          } else {
61            setColumn(params[0], params[1]);
62          }
63          return method.invoke(statement, params);
64        } else if ("getResultSet".equals(method.getName())) {
65          ResultSet rs = (ResultSet) method.invoke(statement, params);
66          if (rs != null) {
67            return ResultSetLogger.newInstance(rs, getStatementLog());
68          } else {
69            return null;
70          }
71        } else if ("getUpdateCount".equals(method.getName())) {
72          int updateCount = (Integer) method.invoke(statement, params);
73          if (updateCount != -1) {
74            debug("<==    Updates: " + updateCount);
75          }
76          return updateCount;
77        } else if ("equals".equals(method.getName())) {
78          Object ps = params[0];
79          return ps instanceof Proxy && proxy == ps;
80        } else if ("hashCode".equals(method.getName())) {
81          return proxy.hashCode();
82        } else {
83          return method.invoke(statement, params);
84        }
85      } catch (Throwable t) {
86        throw ExceptionUtil.unwrapThrowable(t);
87      }
88    }
89  
90    /*
91     * Creates a logging version of a PreparedStatement
92     *
93     * @param stmt - the statement
94     * @param sql  - the sql statement
95     * @return - the proxy
96     */
97    public static PreparedStatement newInstance(PreparedStatement stmt, Log statementLog) {
98      InvocationHandler handler = new PreparedStatementLogger(stmt, statementLog);
99      ClassLoader cl = PreparedStatement.class.getClassLoader();
100     return (PreparedStatement) Proxy.newProxyInstance(cl, new Class[]{PreparedStatement.class, CallableStatement.class}, handler);
101   }
102 
103   /*
104    * Return the wrapped prepared statement
105    *
106    * @return the PreparedStatement
107    */
108   public PreparedStatement getPreparedStatement() {
109     return statement;
110   }

47    public MapperMethod(Class<?> mapperInterface, Method method, Configuration config) {
48      this.command = new SqlCommand(config, mapperInterface, method);
49      this.method = new MethodSignature(config, mapperInterface, method);
50    }

52    public Object execute(SqlSession sqlSession, Object[] args) {
53      Object result;
54      switch (command.getType()) {
55        case INSERT: {
56      	Object param = method.convertArgsToSqlCommandParam(args);
57          result = rowCountResult(sqlSession.insert(command.getName(), param));
58          break;
59        }
60        case UPDATE: {
61          Object param = method.convertArgsToSqlCommandParam(args);
62          result = rowCountResult(sqlSession.update(command.getName(), param));
63          break;
64        }
65        case DELETE: {
66          Object param = method.convertArgsToSqlCommandParam(args);
67          result = rowCountResult(sqlSession.delete(command.getName(), param));
68          break;
69        }
70        case SELECT:
71          if (method.returnsVoid() && method.hasResultHandler()) {
72            executeWithResultHandler(sqlSession, args);
73            result = null;
74          } else if (method.returnsMany()) {
75            result = executeForMany(sqlSession, args);
76          } else if (method.returnsMap()) {
77            result = executeForMap(sqlSession, args);
78          } else if (method.returnsCursor()) {
79            result = executeForCursor(sqlSession, args);
80          } else {
81            Object param = method.convertArgsToSqlCommandParam(args);
82            result = sqlSession.selectOne(command.getName(), param);
83          }
84          break;
85        case FLUSH:
86          result = sqlSession.flushStatements();
87          break;
88        default:
89          throw new BindingException("Unknown execution method for: " + command.getName());
90      }
91      if (result == null && method.getReturnType().isPrimitive() && !method.returnsVoid()) {
92        throw new BindingException("Mapper method '" + command.getName() 
93            + " attempted to return null from a method with a primitive return type (" + method.getReturnType() + ").");
94      }
95      return result;
96    }

98    private Object rowCountResult(int rowCount) {
99      final Object result;
100     if (method.returnsVoid()) {
101       result = null;
102     } else if (Integer.class.equals(method.getReturnType()) || Integer.TYPE.equals(method.getReturnType())) {
103       result = rowCount;
104     } else if (Long.class.equals(method.getReturnType()) || Long.TYPE.equals(method.getReturnType())) {
105       result = (long)rowCount;
106     } else if (Boolean.class.equals(method.getReturnType()) || Boolean.TYPE.equals(method.getReturnType())) {
107       result = rowCount > 0;
108     } else {
109       throw new BindingException("Mapper method '" + command.getName() + "' has an unsupported return type: " + method.getReturnType());
110     }
111     return result;
112   }

114   private void executeWithResultHandler(SqlSession sqlSession, Object[] args) {
115     MappedStatement ms = sqlSession.getConfiguration().getMappedStatement(command.getName());
116     if (void.class.equals(ms.getResultMaps().get(0).getType())) {
117       throw new BindingException("method " + command.getName() 
118           + " needs either a @ResultMap annotation, a @ResultType annotation," 
119           + " or a resultType attribute in XML so a ResultHandler can be used as a parameter.");
120     }
121     Object param = method.convertArgsToSqlCommandParam(args);
122     if (method.hasRowBounds()) {
123       RowBounds rowBounds = method.extractRowBounds(args);
124       sqlSession.select(command.getName(), param, rowBounds, method.extractResultHandler(args));
125     } else {
126       sqlSession.select(command.getName(), param, method.extractResultHandler(args));
127     }
128   }

130   private <E> Object executeForMany(SqlSession sqlSession, Object[] args) {
131     List<E> result;
132     Object param = method.convertArgsToSqlCommandParam(args);
133     if (method.hasRowBounds()) {
134       RowBounds rowBounds = method.extractRowBounds(args);
135       result = sqlSession.<E>selectList(command.getName(), param, rowBounds);
136     } else {
137       result = sqlSession.<E>selectList(command.getName(), param);
138     }
139     // issue #510 Collections & arrays support
140     if (!method.getReturnType().isAssignableFrom(result.getClass())) {
141       if (method.getReturnType().isArray()) {
142         return convertToArray(result);
143       } else {
144         return convertToDeclaredCollection(sqlSession.getConfiguration(), result);
145       }
146     }
147     return result;
148   }
149 
150   private <T> Cursor<T> executeForCursor(SqlSession sqlSession, Object[] args) {
151     Cursor<T> result;
152     Object param = method.convertArgsToSqlCommandParam(args);
153     if (method.hasRowBounds()) {
154       RowBounds rowBounds = method.extractRowBounds(args);
155       result = sqlSession.<T>selectCursor(command.getName(), param, rowBounds);
156     } else {
157       result = sqlSession.<T>selectCursor(command.getName(), param);
158     }
159     return result;
160   }
161 
162   private <E> Object convertToDeclaredCollection(Configuration config, List<E> list) {
163     Object collection = config.getObjectFactory().create(method.getReturnType());
164     MetaObject metaObject = config.newMetaObject(collection);
165     metaObject.addAll(list);
166     return collection;
167   }
168 
169   @SuppressWarnings("unchecked")
170   private <E> Object convertToArray(List<E> list) {
171     Class<?> arrayComponentType = method.getReturnType().getComponentType();
172     Object array = Array.newInstance(arrayComponentType, list.size());
173     if (arrayComponentType.isPrimitive()) {
174       for (int i = 0; i < list.size(); i++) {
175         Array.set(array, i, list.get(i));
176       }
177       return array;
178     } else {
179       return list.toArray((E[])array);
180     }
181   }
182 
183   private <K, V> Map<K, V> executeForMap(SqlSession sqlSession, Object[] args) {
184     Map<K, V> result;
185     Object param = method.convertArgsToSqlCommandParam(args);
186     if (method.hasRowBounds()) {
187       RowBounds rowBounds = method.extractRowBounds(args);
188       result = sqlSession.<K, V>selectMap(command.getName(), param, method.getMapKey(), rowBounds);
189     } else {
190       result = sqlSession.<K, V>selectMap(command.getName(), param, method.getMapKey());
191     }
192     return result;
193   }

195   public static class ParamMap<V> extends HashMap<String, V> {

197     private static final long serialVersionUID = -2212268410512043556L;
198 
199     @Override
200     public V get(Object key) {
201       if (!super.containsKey(key)) {
202         throw new BindingException("Parameter '" + key + "' not found. Available parameters are " + keySet());
203       }
204       return super.get(key);
205     }

209   public static class SqlCommand {
210 
211     private final String name;
212     private final SqlCommandType type;
213 
214     public SqlCommand(Configuration configuration, Class<?> mapperInterface, Method method) {
215       String statementName = mapperInterface.getName() + "." + method.getName();
216       MappedStatement ms = null;
217       if (configuration.hasStatement(statementName)) {
218         ms = configuration.getMappedStatement(statementName);
219       } else if (!mapperInterface.equals(method.getDeclaringClass())) { // issue #35
220         String parentStatementName = method.getDeclaringClass().getName() + "." + method.getName();
221         if (configuration.hasStatement(parentStatementName)) {
222           ms = configuration.getMappedStatement(parentStatementName);
223         }
224       }
225       if (ms == null) {
226         if(method.getAnnotation(Flush.class) != null){
227           name = null;
228           type = SqlCommandType.FLUSH;
229         } else {
230           throw new BindingException("Invalid bound statement (not found): " + statementName);
231         }
232       } else {
233         name = ms.getId();
234         type = ms.getSqlCommandType();
235         if (type == SqlCommandType.UNKNOWN) {
236           throw new BindingException("Unknown execution method for: " + name);
237         }
238       }
239     }
240 
241     public String getName() {
242       return name;
243     }
244 
245     public SqlCommandType getType() {
246       return type;
247     }
248   }

250   public static class MethodSignature {
251 
252     private final boolean returnsMany;
253     private final boolean returnsMap;
254     private final boolean returnsVoid;
255     private final boolean returnsCursor;
256     private final Class<?> returnType;
257     private final String mapKey;
258     private final Integer resultHandlerIndex;
259     private final Integer rowBoundsIndex;
260     private final ParamNameResolver paramNameResolver;
261 
262     public MethodSignature(Configuration configuration, Class<?> mapperInterface, Method method) {
263       Type resolvedReturnType = TypeParameterResolver.resolveReturnType(method, mapperInterface);
264       if (resolvedReturnType instanceof Class<?>) {
265         this.returnType = (Class<?>) resolvedReturnType;
266       } else if (resolvedReturnType instanceof ParameterizedType) {
267         this.returnType = (Class<?>) ((ParameterizedType) resolvedReturnType).getRawType();
268       } else {
269         this.returnType = method.getReturnType();
270       }
271       this.returnsVoid = void.class.equals(this.returnType);
272       this.returnsMany = (configuration.getObjectFactory().isCollection(this.returnType) || this.returnType.isArray());
273       this.returnsCursor = Cursor.class.equals(this.returnType);
274       this.mapKey = getMapKey(method);
275       this.returnsMap = (this.mapKey != null);
276       this.rowBoundsIndex = getUniqueParamIndex(method, RowBounds.class);
277       this.resultHandlerIndex = getUniqueParamIndex(method, ResultHandler.class);
278       this.paramNameResolver = new ParamNameResolver(configuration, method);
279     }
280 
281     public Object convertArgsToSqlCommandParam(Object[] args) {
282       return paramNameResolver.getNamedParams(args);
283     }
284 
285     public boolean hasRowBounds() {
286       return rowBoundsIndex != null;
287     }
288 
289     public RowBounds extractRowBounds(Object[] args) {
290       return hasRowBounds() ? (RowBounds) args[rowBoundsIndex] : null;
291     }
292 
293     public boolean hasResultHandler() {
294       return resultHandlerIndex != null;
295     }
296 
297     public ResultHandler extractResultHandler(Object[] args) {
298       return hasResultHandler() ? (ResultHandler) args[resultHandlerIndex] : null;
299     }
300 
301     public String getMapKey() {
302       return mapKey;
303     }
304 
305     public Class<?> getReturnType() {
306       return returnType;
307     }
308 
309     public boolean returnsMany() {
310       return returnsMany;
311     }
312 
313     public boolean returnsMap() {
314       return returnsMap;
315     }
316 
317     public boolean returnsVoid() {
318       return returnsVoid;
319     }
320 
321     public boolean returnsCursor() {
322       return returnsCursor;
323     }
324 
325     private Integer getUniqueParamIndex(Method method, Class<?> paramType) {
326       Integer index = null;
327       final Class<?>[] argTypes = method.getParameterTypes();
328       for (int i = 0; i < argTypes.length; i++) {
329         if (paramType.isAssignableFrom(argTypes[i])) {
330           if (index == null) {
331             index = i;
332           } else {
333             throw new BindingException(method.getName() + " cannot have multiple " + paramType.getSimpleName() + " parameters");
334           }
335         }
336       }
337       return index;
338     }
339 
340     private String getMapKey(Method method) {
341       String mapKey = null;
342       if (Map.class.isAssignableFrom(method.getReturnType())) {
343         final MapKey mapKeyAnnotation = method.getAnnotation(MapKey.class);
344         if (mapKeyAnnotation != null) {
345           mapKey = mapKeyAnnotation.value();
346         }
347       }
348       return mapKey;
349     }
350   }

84    private XMLConfigBuilder(XPathParser parser, String environment, Properties props) {
85      super(new Configuration());
86      ErrorContext.instance().resource("SQL Mapper Configuration");
87      this.configuration.setVariables(props);
88      this.parsed = false;
89      this.environment = environment;
90      this.parser = parser;
91    }
92  
93    public Configuration parse() {
94      if (parsed) {
95        throw new BuilderException("Each XMLConfigBuilder can only be used once.");
96      }
97      parsed = true;
98      parseConfiguration(parser.evalNode("/configuration"));
99      return configuration;
100   }
101 
102   private void parseConfiguration(XNode root) {
103     try {
104       //issue #117 read properties first
105       propertiesElement(root.evalNode("properties"));
106       Properties settings = settingsAsProperties(root.evalNode("settings"));
107       loadCustomVfs(settings);
108       typeAliasesElement(root.evalNode("typeAliases"));
109       pluginElement(root.evalNode("plugins"));
110       objectFactoryElement(root.evalNode("objectFactory"));
111       objectWrapperFactoryElement(root.evalNode("objectWrapperFactory"));
112       reflectorFactoryElement(root.evalNode("reflectorFactory"));
113       settingsElement(settings);
114       // read it after objectFactory and objectWrapperFactory issue #631
115       environmentsElement(root.evalNode("environments"));
116       databaseIdProviderElement(root.evalNode("databaseIdProvider"));
117       typeHandlerElement(root.evalNode("typeHandlers"));
118       mapperElement(root.evalNode("mappers"));
119     } catch (Exception e) {
120       throw new BuilderException("Error parsing SQL Mapper Configuration. Cause: " + e, e);
121     }
122   }

124   private Properties settingsAsProperties(XNode context) {
125     if (context == null) {
126       return new Properties();
127     }
128     Properties props = context.getChildrenAsProperties();
129     // Check that all settings are known to the configuration class
130     MetaClass metaConfig = MetaClass.forClass(Configuration.class, localReflectorFactory);
131     for (Object key : props.keySet()) {
132       if (!metaConfig.hasSetter(String.valueOf(key))) {
133         throw new BuilderException("The setting " + key + " is not known.  Make sure you spelled it correctly (case sensitive).");
134       }
135     }
136     return props;
137   }

139   private void loadCustomVfs(Properties props) throws ClassNotFoundException {
140     String value = props.getProperty("vfsImpl");
141     if (value != null) {
142       String[] clazzes = value.split(",");
143       for (String clazz : clazzes) {
144         if (!clazz.isEmpty()) {
145           @SuppressWarnings("unchecked")
146           Class<? extends VFS> vfsImpl = (Class<? extends VFS>)Resources.classForName(clazz);
147           configuration.setVfsImpl(vfsImpl);
148         }
149       }
150     }
151   }

153   private void typeAliasesElement(XNode parent) {
154     if (parent != null) {
155       for (XNode child : parent.getChildren()) {
156         if ("package".equals(child.getName())) {
157           String typeAliasPackage = child.getStringAttribute("name");
158           configuration.getTypeAliasRegistry().registerAliases(typeAliasPackage);
159         } else {
160           String alias = child.getStringAttribute("alias");
161           String type = child.getStringAttribute("type");
162           try {
163             Class<?> clazz = Resources.classForName(type);
164             if (alias == null) {
165               typeAliasRegistry.registerAlias(clazz);
166             } else {
167               typeAliasRegistry.registerAlias(alias, clazz);
168             }
169           } catch (ClassNotFoundException e) {
170             throw new BuilderException("Error registering typeAlias for '" + alias + "'. Cause: " + e, e);
171           }
172         }
173       }
174     }
175   }

177   private void pluginElement(XNode parent) throws Exception {
178     if (parent != null) {
179       for (XNode child : parent.getChildren()) {
180         String interceptor = child.getStringAttribute("interceptor");
181         Properties properties = child.getChildrenAsProperties();
182         Interceptor interceptorInstance = (Interceptor) resolveClass(interceptor).newInstance();
183         interceptorInstance.setProperties(properties);
184         configuration.addInterceptor(interceptorInstance);
185       }
186     }
187   }

215   private void propertiesElement(XNode context) throws Exception {
216     if (context != null) {
217       Properties defaults = context.getChildrenAsProperties();
218       String resource = context.getStringAttribute("resource");
219       String url = context.getStringAttribute("url");
220       if (resource != null && url != null) {
221         throw new BuilderException("The properties element cannot specify both a URL and a resource based property file reference.  Please specify one or the other.");
222       }
223       if (resource != null) {
224         defaults.putAll(Resources.getResourceAsProperties(resource));
225       } else if (url != null) {
226         defaults.putAll(Resources.getUrlAsProperties(url));
227       }
228       Properties vars = configuration.getVariables();
229       if (vars != null) {
230         defaults.putAll(vars);
231       }
232       parser.setVariables(defaults);
233       configuration.setVariables(defaults);
234     }
235   }

237   private void settingsElement(Properties props) throws Exception {
238     configuration.setAutoMappingBehavior(AutoMappingBehavior.valueOf(props.getProperty("autoMappingBehavior", "PARTIAL")));
239     configuration.setAutoMappingUnknownColumnBehavior(AutoMappingUnknownColumnBehavior.valueOf(props.getProperty("autoMappingUnknownColumnBehavior", "NONE")));
240     configuration.setCacheEnabled(booleanValueOf(props.getProperty("cacheEnabled"), true));
241     configuration.setProxyFactory((ProxyFactory) createInstance(props.getProperty("proxyFactory")));
242     configuration.setLazyLoadingEnabled(booleanValueOf(props.getProperty("lazyLoadingEnabled"), false));
243     configuration.setAggressiveLazyLoading(booleanValueOf(props.getProperty("aggressiveLazyLoading"), false));
244     configuration.setMultipleResultSetsEnabled(booleanValueOf(props.getProperty("multipleResultSetsEnabled"), true));
245     configuration.setUseColumnLabel(booleanValueOf(props.getProperty("useColumnLabel"), true));
246     configuration.setUseGeneratedKeys(booleanValueOf(props.getProperty("useGeneratedKeys"), false));
247     configuration.setDefaultExecutorType(ExecutorType.valueOf(props.getProperty("defaultExecutorType", "SIMPLE")));
248     configuration.setDefaultStatementTimeout(integerValueOf(props.getProperty("defaultStatementTimeout"), null));
249     configuration.setDefaultFetchSize(integerValueOf(props.getProperty("defaultFetchSize"), null));
250     configuration.setMapUnderscoreToCamelCase(booleanValueOf(props.getProperty("mapUnderscoreToCamelCase"), false));
251     configuration.setSafeRowBoundsEnabled(booleanValueOf(props.getProperty("safeRowBoundsEnabled"), false));
252     configuration.setLocalCacheScope(LocalCacheScope.valueOf(props.getProperty("localCacheScope", "SESSION")));
253     configuration.setJdbcTypeForNull(JdbcType.valueOf(props.getProperty("jdbcTypeForNull", "OTHER")));
254     configuration.setLazyLoadTriggerMethods(stringSetValueOf(props.getProperty("lazyLoadTriggerMethods"), "equals,clone,hashCode,toString"));
255     configuration.setSafeResultHandlerEnabled(booleanValueOf(props.getProperty("safeResultHandlerEnabled"), true));
256     configuration.setDefaultScriptingLanguage(resolveClass(props.getProperty("defaultScriptingLanguage")));
257     configuration.setCallSettersOnNulls(booleanValueOf(props.getProperty("callSettersOnNulls"), false));
258     configuration.setUseActualParamName(booleanValueOf(props.getProperty("useActualParamName"), true));
259     configuration.setReturnInstanceForEmptyRow(booleanValueOf(props.getProperty("returnInstanceForEmptyRow"), false));
260     configuration.setLogPrefix(props.getProperty("logPrefix"));
261     @SuppressWarnings("unchecked")
262     Class<? extends Log> logImpl = (Class<? extends Log>)resolveClass(props.getProperty("logImpl"));
263     configuration.setLogImpl(logImpl);
264     configuration.setConfigurationFactory(resolveClass(props.getProperty("configurationFactory")));
265   }

267   private void environmentsElement(XNode context) throws Exception {
268     if (context != null) {
269       if (environment == null) {
270         environment = context.getStringAttribute("default");
271       }
272       for (XNode child : context.getChildren()) {
273         String id = child.getStringAttribute("id");
274         if (isSpecifiedEnvironment(id)) {
275           TransactionFactory txFactory = transactionManagerElement(child.evalNode("transactionManager"));
276           DataSourceFactory dsFactory = dataSourceElement(child.evalNode("dataSource"));
277           DataSource dataSource = dsFactory.getDataSource();
278           Environment.Builder environmentBuilder = new Environment.Builder(id)
279               .transactionFactory(txFactory)
280               .dataSource(dataSource);
281           configuration.setEnvironment(environmentBuilder.build());
282         }
283       }
284     }
285   }

287   private void databaseIdProviderElement(XNode context) throws Exception {
288     DatabaseIdProvider databaseIdProvider = null;
289     if (context != null) {
290       String type = context.getStringAttribute("type");
291       // awful patch to keep backward compatibility
292       if ("VENDOR".equals(type)) {
293           type = "DB_VENDOR";
294       }
295       Properties properties = context.getChildrenAsProperties();
296       databaseIdProvider = (DatabaseIdProvider) resolveClass(type).newInstance();
297       databaseIdProvider.setProperties(properties);
298     }
299     Environment environment = configuration.getEnvironment();
300     if (environment != null && databaseIdProvider != null) {
301       String databaseId = databaseIdProvider.getDatabaseId(environment.getDataSource());
302       configuration.setDatabaseId(databaseId);
303     }
304   }

328   private void typeHandlerElement(XNode parent) throws Exception {
329     if (parent != null) {
330       for (XNode child : parent.getChildren()) {
331         if ("package".equals(child.getName())) {
332           String typeHandlerPackage = child.getStringAttribute("name");
333           typeHandlerRegistry.register(typeHandlerPackage);
334         } else {
335           String javaTypeName = child.getStringAttribute("javaType");
336           String jdbcTypeName = child.getStringAttribute("jdbcType");
337           String handlerTypeName = child.getStringAttribute("handler");
338           Class<?> javaTypeClass = resolveClass(javaTypeName);
339           JdbcType jdbcType = resolveJdbcType(jdbcTypeName);
340           Class<?> typeHandlerClass = resolveClass(handlerTypeName);
341           if (javaTypeClass != null) {
342             if (jdbcType == null) {
343               typeHandlerRegistry.register(javaTypeClass, typeHandlerClass);
344             } else {
345               typeHandlerRegistry.register(javaTypeClass, jdbcType, typeHandlerClass);
346             }
347           } else {
348             typeHandlerRegistry.register(typeHandlerClass);
349           }
350         }
351       }
352     }
353   }

355   private void mapperElement(XNode parent) throws Exception {
356     if (parent != null) {
357       for (XNode child : parent.getChildren()) {
358         if ("package".equals(child.getName())) {
359           String mapperPackage = child.getStringAttribute("name");
360           configuration.addMappers(mapperPackage);
361         } else {
362           String resource = child.getStringAttribute("resource");
363           String url = child.getStringAttribute("url");
364           String mapperClass = child.getStringAttribute("class");
365           if (resource != null && url == null && mapperClass == null) {
366             ErrorContext.instance().resource(resource);
367             InputStream inputStream = Resources.getResourceAsStream(resource);
368             XMLMapperBuilder mapperParser = new XMLMapperBuilder(inputStream, configuration, resource, configuration.getSqlFragments());
369             mapperParser.parse();
370           } else if (resource == null && url != null && mapperClass == null) {
371             ErrorContext.instance().resource(url);
372             InputStream inputStream = Resources.getUrlAsStream(url);
373             XMLMapperBuilder mapperParser = new XMLMapperBuilder(inputStream, configuration, url, configuration.getSqlFragments());
374             mapperParser.parse();
375           } else if (resource == null && url == null && mapperClass != null) {
376             Class<?> mapperInterface = Resources.classForName(mapperClass);
377             configuration.addMapper(mapperInterface);
378           } else {
379             throw new BuilderException("A mapper element may only specify a url, resource or class, but not more than one.");
380           }
381         }
382       }
383     }
384   }

386   private boolean isSpecifiedEnvironment(String id) {
387     if (environment == null) {
388       throw new BuilderException("No environment specified.");
389     } else if (id == null) {
390       throw new BuilderException("Environment requires an id attribute.");
391     } else if (environment.equals(id)) {
392       return true;
393     }
394     return false;
395   }

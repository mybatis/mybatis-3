16  package org.apache.ibatis.annotations;
17  
18  import java.lang.annotation.Documented;
19  import java.lang.annotation.ElementType;
20  import java.lang.annotation.Retention;
21  import java.lang.annotation.RetentionPolicy;
22  import java.lang.annotation.Target;
23  
24  /**
25   * The annotation that specify the property name(or column name) for a key value of {@link java.util.Map}.
26   *
27   * <p>
28   * <b>How to use:</b>
29   *
30   * <pre>
31   * public interface UserMapper {
32   *   &#064;MapKey("id")
33   *   &#064;Select("SELECT id, name FROM users WHERE name LIKE #{name} || '%")
34   *   Map&lt;Integer, User&gt; selectByStartingWithName(String name);
35   * }
36   * </pre>
37   *
38   * @author Clinton Begin
39   */
40  @Documented
41  @Retention(RetentionPolicy.RUNTIME)
42  @Target(ElementType.METHOD)
43  public @interface MapKey {
44    /**
45     * Returns the property name(or column name) for a key value of {@link java.util.Map}.
46     *
47     * @return the property name(or column name)
48     */
49    String value();
50  }

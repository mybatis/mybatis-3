16  package org.apache.ibatis.executor;
17  
18  import org.apache.ibatis.reflection.MetaObject;
19  import org.apache.ibatis.reflection.factory.ObjectFactory;
20  import org.apache.ibatis.session.Configuration;
21  
22  import java.lang.reflect.Array;
23  import java.util.List;
24  
25  public class ResultExtractor {
26    private final Configuration configuration;
27    private final ObjectFactory objectFactory;
28  
29    public ResultExtractor(Configuration configuration, ObjectFactory objectFactory) {
30      this.configuration = configuration;
31      this.objectFactory = objectFactory;
32    }
33  
34    public Object extractObjectFromList(List<Object> list, Class<?> targetType) {
35      Object value = null;
36      if (targetType != null && targetType.isAssignableFrom(list.getClass())) {
37        value = list;
38      } else if (targetType != null && objectFactory.isCollection(targetType)) {
39        value = objectFactory.create(targetType);
40        MetaObject metaObject = configuration.newMetaObject(value);
41        metaObject.addAll(list);
42      } else if (targetType != null && targetType.isArray()) {
43        Object[] array = (Object[]) Array.newInstance(targetType.getComponentType(), list.size());
44        value = list.toArray(array);
45      } else {
46        if (list != null && list.size() > 1) {
47          throw new ExecutorException("Statement returned more than one row, where no more than one was expected.");
48        } else if (list != null && list.size() == 1) {
49          value = list.get(0);
50        }
51      }
52      return value;
53    }
54  }

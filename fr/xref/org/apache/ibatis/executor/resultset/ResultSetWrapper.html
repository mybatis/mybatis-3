40   */
41  public class ResultSetWrapper {
42  
43    private final ResultSet resultSet;
44    private final TypeHandlerRegistry typeHandlerRegistry;
45    private final List<String> columnNames = new ArrayList<String>();
46    private final List<String> classNames = new ArrayList<String>();
47    private final List<JdbcType> jdbcTypes = new ArrayList<JdbcType>();
48    private final Map<String, Map<Class<?>, TypeHandler<?>>> typeHandlerMap = new HashMap<String, Map<Class<?>, TypeHandler<?>>>();
49    private Map<String, List<String>> mappedColumnNamesMap = new HashMap<String, List<String>>();
50    private Map<String, List<String>> unMappedColumnNamesMap = new HashMap<String, List<String>>();
51  
52    public ResultSetWrapper(ResultSet rs, Configuration configuration) throws SQLException {
53      super();
54      this.typeHandlerRegistry = configuration.getTypeHandlerRegistry();
55      this.resultSet = rs;
56      final ResultSetMetaData metaData = rs.getMetaData();
57      final int columnCount = metaData.getColumnCount();
58      for (int i = 1; i <= columnCount; i++) {
59        columnNames.add(configuration.isUseColumnLabel() ? metaData.getColumnLabel(i) : metaData.getColumnName(i));
60        jdbcTypes.add(JdbcType.forCode(metaData.getColumnType(i)));
61        classNames.add(metaData.getColumnClassName(i));
62      }
63    }
64  
65    public ResultSet getResultSet() {
66      return resultSet;
67    }
68  
69    public List<String> getColumnNames() {
70      return this.columnNames;
71    }
72  
73    public List<String> getClassNames() {
74      return Collections.unmodifiableList(classNames);
75    }
76  
77    public JdbcType getJdbcType(String columnName) {
78      for (int i = 0 ; i < columnNames.size(); i++) {
79        if (columnNames.get(i).equalsIgnoreCase(columnName)) {
80          return jdbcTypes.get(i);
81        }
82      }
83      return null;
84    }
85  
86    /**
87     * Gets the type handler to use when reading the result set.
88     * Tries to get from the TypeHandlerRegistry by searching for the property type.
89     * If not found it gets the column JDBC type and tries to get a handler for it.
90     * 
91     * @param propertyType
92     * @param columnName
93     * @return
94     */
95    public TypeHandler<?> getTypeHandler(Class<?> propertyType, String columnName) {
96      TypeHandler<?> handler = null;
97      Map<Class<?>, TypeHandler<?>> columnHandlers = typeHandlerMap.get(columnName);
98      if (columnHandlers == null) {
99        columnHandlers = new HashMap<Class<?>, TypeHandler<?>>();
100       typeHandlerMap.put(columnName, columnHandlers);
101     } else {
102       handler = columnHandlers.get(propertyType);
103     }
104     if (handler == null) {
105       JdbcType jdbcType = getJdbcType(columnName);
106       handler = typeHandlerRegistry.getTypeHandler(propertyType, jdbcType);
107       // Replicate logic of UnknownTypeHandler#resolveTypeHandler
108       // See issue #59 comment 10
109       if (handler == null || handler instanceof UnknownTypeHandler) {
110         final int index = columnNames.indexOf(columnName);
111         final Class<?> javaType = resolveClass(classNames.get(index));
112         if (javaType != null && jdbcType != null) {
113           handler = typeHandlerRegistry.getTypeHandler(javaType, jdbcType);
114         } else if (javaType != null) {
115           handler = typeHandlerRegistry.getTypeHandler(javaType);
116         } else if (jdbcType != null) {
117           handler = typeHandlerRegistry.getTypeHandler(jdbcType);
118         }
119       }
120       if (handler == null || handler instanceof UnknownTypeHandler) {
121         handler = new ObjectTypeHandler();
122       }
123       columnHandlers.put(propertyType, handler);
124     }
125     return handler;
126   }
127 
128   private Class<?> resolveClass(String className) {
129     try {
130       // #699 className could be null
131       if (className != null) {
132         return Resources.classForName(className);
133       }
134     } catch (ClassNotFoundException e) {
135       // ignore
136     }
137     return null;
138   }
139 
140   private void loadMappedAndUnmappedColumnNames(ResultMap resultMap, String columnPrefix) throws SQLException {
141     List<String> mappedColumnNames = new ArrayList<String>();
142     List<String> unmappedColumnNames = new ArrayList<String>();
143     final String upperColumnPrefix = columnPrefix == null ? null : columnPrefix.toUpperCase(Locale.ENGLISH);
144     final Set<String> mappedColumns = prependPrefixes(resultMap.getMappedColumns(), upperColumnPrefix);
145     for (String columnName : columnNames) {
146       final String upperColumnName = columnName.toUpperCase(Locale.ENGLISH);
147       if (mappedColumns.contains(upperColumnName)) {
148         mappedColumnNames.add(upperColumnName);
149       } else {
150         unmappedColumnNames.add(columnName);
151       }
152     }
153     mappedColumnNamesMap.put(getMapKey(resultMap, columnPrefix), mappedColumnNames);
154     unMappedColumnNamesMap.put(getMapKey(resultMap, columnPrefix), unmappedColumnNames);
155   }
156 
157   public List<String> getMappedColumnNames(ResultMap resultMap, String columnPrefix) throws SQLException {
158     List<String> mappedColumnNames = mappedColumnNamesMap.get(getMapKey(resultMap, columnPrefix));
159     if (mappedColumnNames == null) {
160       loadMappedAndUnmappedColumnNames(resultMap, columnPrefix);
161       mappedColumnNames = mappedColumnNamesMap.get(getMapKey(resultMap, columnPrefix));
162     }
163     return mappedColumnNames;
164   }
165 
166   public List<String> getUnmappedColumnNames(ResultMap resultMap, String columnPrefix) throws SQLException {
167     List<String> unMappedColumnNames = unMappedColumnNamesMap.get(getMapKey(resultMap, columnPrefix));
168     if (unMappedColumnNames == null) {
169       loadMappedAndUnmappedColumnNames(resultMap, columnPrefix);
170       unMappedColumnNames = unMappedColumnNamesMap.get(getMapKey(resultMap, columnPrefix));
171     }
172     return unMappedColumnNames;
173   }
174 
175   private String getMapKey(ResultMap resultMap, String columnPrefix) {
176     return resultMap.getId() + ":" + columnPrefix;
177   }
178 
179   private Set<String> prependPrefixes(Set<String> columnNames, String prefix) {
180     if (columnNames == null || columnNames.isEmpty() || prefix == null || prefix.length() == 0) {
181       return columnNames;
182     }
183     final Set<String> prefixed = new HashSet<String>();
184     for (String columnName : columnNames) {
185       prefixed.add(prefix + columnName);
186     }
187     return prefixed;
188   }

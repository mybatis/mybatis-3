41  public class BatchExecutor extends BaseExecutor {
42  
43    public static final int BATCH_UPDATE_RETURN_VALUE = Integer.MIN_VALUE + 1002;
44  
45    private final List<Statement> statementList = new ArrayList<Statement>();
46    private final List<BatchResult> batchResultList = new ArrayList<BatchResult>();
47    private String currentSql;
48    private MappedStatement currentStatement;
49  
50    public BatchExecutor(Configuration configuration, Transaction transaction) {
51      super(configuration, transaction);
52    }
53  
54    @Override
55    public int doUpdate(MappedStatement ms, Object parameterObject) throws SQLException {
56      final Configuration configuration = ms.getConfiguration();
57      final StatementHandler handler = configuration.newStatementHandler(this, ms, parameterObject, RowBounds.DEFAULT, null, null);
58      final BoundSql boundSql = handler.getBoundSql();
59      final String sql = boundSql.getSql();
60      final Statement stmt;
61      if (sql.equals(currentSql) && ms.equals(currentStatement)) {
62        int last = statementList.size() - 1;
63        stmt = statementList.get(last);
64        applyTransactionTimeout(stmt);
65       handler.parameterize(stmt);//fix Issues 322
66        BatchResult batchResult = batchResultList.get(last);
67        batchResult.addParameterObject(parameterObject);
68      } else {
69        Connection connection = getConnection(ms.getStatementLog());
70        stmt = handler.prepare(connection, transaction.getTimeout());
71        handler.parameterize(stmt);    //fix Issues 322
72        currentSql = sql;
73        currentStatement = ms;
74        statementList.add(stmt);
75        batchResultList.add(new BatchResult(ms, sql, parameterObject));
76      }
77    // handler.parameterize(stmt);
78      handler.batch(stmt);
79      return BATCH_UPDATE_RETURN_VALUE;
80    }
81  
82    @Override
83    public <E> List<E> doQuery(MappedStatement ms, Object parameterObject, RowBounds rowBounds, ResultHandler resultHandler, BoundSql boundSql)
84        throws SQLException {
85      Statement stmt = null;
86      try {
87        flushStatements();
88        Configuration configuration = ms.getConfiguration();
89        StatementHandler handler = configuration.newStatementHandler(wrapper, ms, parameterObject, rowBounds, resultHandler, boundSql);
90        Connection connection = getConnection(ms.getStatementLog());
91        stmt = handler.prepare(connection, transaction.getTimeout());
92        handler.parameterize(stmt);
93        return handler.<E>query(stmt, resultHandler);
94      } finally {
95        closeStatement(stmt);
96      }
97    }
98  
99    @Override
100   protected <E> Cursor<E> doQueryCursor(MappedStatement ms, Object parameter, RowBounds rowBounds, BoundSql boundSql) throws SQLException {
101     flushStatements();
102     Configuration configuration = ms.getConfiguration();
103     StatementHandler handler = configuration.newStatementHandler(wrapper, ms, parameter, rowBounds, null, boundSql);
104     Connection connection = getConnection(ms.getStatementLog());
105     Statement stmt = handler.prepare(connection, transaction.getTimeout());
106     handler.parameterize(stmt);
107     return handler.<E>queryCursor(stmt);
108   }
109 
110   @Override
111   public List<BatchResult> doFlushStatements(boolean isRollback) throws SQLException {
112     try {
113       List<BatchResult> results = new ArrayList<BatchResult>();
114       if (isRollback) {
115         return Collections.emptyList();
116       }
117       for (int i = 0, n = statementList.size(); i < n; i++) {
118         Statement stmt = statementList.get(i);
119         applyTransactionTimeout(stmt);
120         BatchResult batchResult = batchResultList.get(i);
121         try {
122           batchResult.setUpdateCounts(stmt.executeBatch());
123           MappedStatement ms = batchResult.getMappedStatement();
124           List<Object> parameterObjects = batchResult.getParameterObjects();
125           KeyGenerator keyGenerator = ms.getKeyGenerator();
126           if (Jdbc3KeyGenerator.class.equals(keyGenerator.getClass())) {
127             Jdbc3KeyGenerator jdbc3KeyGenerator = (Jdbc3KeyGenerator) keyGenerator;
128             jdbc3KeyGenerator.processBatch(ms, stmt, parameterObjects);
129           } else if (!NoKeyGenerator.class.equals(keyGenerator.getClass())) { //issue #141
130             for (Object parameter : parameterObjects) {
131               keyGenerator.processAfter(this, ms, stmt, parameter);
132             }
133           }
134         } catch (BatchUpdateException e) {
135           StringBuilder message = new StringBuilder();
136           message.append(batchResult.getMappedStatement().getId())
137               .append(" (batch index #")
138               .append(i + 1)
139               .append(")")
140               .append(" failed.");
141           if (i > 0) {
142             message.append(" ")
143                 .append(i)
144                 .append(" prior sub executor(s) completed successfully, but will be rolled back.");
145           }
146           throw new BatchExecutorException(message.toString(), e, results, batchResult);
147         }
148         results.add(batchResult);
149       }
150       return results;
151     } finally {
152       for (Statement stmt : statementList) {
153         closeStatement(stmt);
154       }
155       currentSql = null;
156       statementList.clear();
157       batchResultList.clear();
158     }
159   }

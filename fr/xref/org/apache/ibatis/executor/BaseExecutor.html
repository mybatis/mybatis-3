73    public Transaction getTransaction() {
74      if (closed) throw new ExecutorException("Executor was closed.");

78    public void close(boolean forceRollback) {
79      try {
80        try {
81          rollback(forceRollback);
82        } finally {
83          if (transaction != null) transaction.close();
84        }
85      } catch (SQLException e) {
86        // Ignore.  There's nothing that can be done at this point.
87        log.warn("Unexpected exception on closing transaction.  Cause: " + e);
88      } finally {
89        transaction = null;
90        deferredLoads = null;
91        localCache = null;
92        localOutputParameterCache = null;
93        closed = true;
94      }
95    }

101   public int update(MappedStatement ms, Object parameter) throws SQLException {
102     ErrorContext.instance().resource(ms.getResource()).activity("executing an update").object(ms.getId());
103     if (closed) throw new ExecutorException("Executor was closed.");

123   @SuppressWarnings("unchecked")
124   public <E> List<E> query(MappedStatement ms, Object parameter, RowBounds rowBounds, ResultHandler resultHandler, CacheKey key, BoundSql boundSql) throws SQLException {
125     ErrorContext.instance().resource(ms.getResource()).activity("executing a query").object(ms.getId());
126     if (closed) throw new ExecutorException("Executor was closed.");
127     if (queryStack == 0 && ms.isFlushCacheRequired()) {
128       clearLocalCache();
129     }
130     List<E> list;
131     try {
132       queryStack++;
133       list = resultHandler == null ? (List<E>) localCache.getObject(key) : null;
134       if (list != null) {
135         handleLocallyCachedOutputParameters(ms, key, parameter, boundSql);
136       } else {
137         list = queryFromDatabase(ms, parameter, rowBounds, resultHandler, key, boundSql);
138       }
139     } finally {
140       queryStack--;
141     }
142     if (queryStack == 0) {
143       for (DeferredLoad deferredLoad : deferredLoads) {
144         deferredLoad.load();
145       }
146       deferredLoads.clear(); // issue #601
147       if (configuration.getLocalCacheScope() == LocalCacheScope.STATEMENT) {
148         clearLocalCache(); // issue #482
149       }
150     }
151     return list;
152   }

154   public void deferLoad(MappedStatement ms, MetaObject resultObject, String property, CacheKey key, Class<?> targetType) {
155     if (closed) throw new ExecutorException("Executor was closed.");
156     DeferredLoad deferredLoad = new DeferredLoad(resultObject, property, key, localCache, configuration, targetType);
157     if (deferredLoad.canLoad()) {
158     	deferredLoad.load();
159     } else {
160     	deferredLoads.add(new DeferredLoad(resultObject, property, key, localCache, configuration, targetType));
161     }
162   }

164   public CacheKey createCacheKey(MappedStatement ms, Object parameterObject, RowBounds rowBounds, BoundSql boundSql) {
165     if (closed) throw new ExecutorException("Executor was closed.");
166     CacheKey cacheKey = new CacheKey();
167     cacheKey.update(ms.getId());
168     cacheKey.update(rowBounds.getOffset());
169     cacheKey.update(rowBounds.getLimit());
170     cacheKey.update(boundSql.getSql());
171     List<ParameterMapping> parameterMappings = boundSql.getParameterMappings();
172     TypeHandlerRegistry typeHandlerRegistry = ms.getConfiguration().getTypeHandlerRegistry();
173     for (int i = 0; i < parameterMappings.size(); i++) { // mimic DefaultParameterHandler logic
174       ParameterMapping parameterMapping = parameterMappings.get(i);
175       if (parameterMapping.getMode() != ParameterMode.OUT) {
176         Object value;
177         String propertyName = parameterMapping.getProperty();
178         if (boundSql.hasAdditionalParameter(propertyName)) {
179           value = boundSql.getAdditionalParameter(propertyName);
180         } else if (parameterObject == null) {
181           value = null;
182         } else if (typeHandlerRegistry.hasTypeHandler(parameterObject.getClass())) {
183           value = parameterObject;
184         } else {
185           MetaObject metaObject = configuration.newMetaObject(parameterObject);
186           value = metaObject.getValue(propertyName);
187         }
188         cacheKey.update(value);
189       }
190     }
191     return cacheKey;
192   }

198   public void commit(boolean required) throws SQLException {
199     if (closed) throw new ExecutorException("Cannot commit, transaction is already closed");
200     clearLocalCache();
201     flushStatements();
202     if (required) {
203       transaction.commit();
204     }
205   }

207   public void rollback(boolean required) throws SQLException {
208     if (!closed) {
209       try {
210         clearLocalCache();
211         flushStatements(true);
212       } finally {
213         if (required) {
214           transaction.rollback();
215         }
216       }
217     }
218   }

220   public void clearLocalCache() {
221     if (!closed) {
222       localCache.clear();
223       localOutputParameterCache.clear();
224     }
225   }

246   private void handleLocallyCachedOutputParameters(MappedStatement ms, CacheKey key, Object parameter, BoundSql boundSql) {
247     if (ms.getStatementType() == StatementType.CALLABLE) {
248       final Object cachedParameter = localOutputParameterCache.getObject(key);
249       if (cachedParameter != null && parameter != null) {
250         final MetaObject metaCachedParameter = configuration.newMetaObject(cachedParameter);
251         final MetaObject metaParameter = configuration.newMetaObject(parameter);
252         for (ParameterMapping parameterMapping : boundSql.getParameterMappings()) {
253           if (parameterMapping.getMode() != ParameterMode.IN) {
254             final String parameterName = parameterMapping.getProperty();
255             final Object cachedValue = metaCachedParameter.getValue(parameterName);
256             metaParameter.setValue(parameterName, cachedValue);
257           }
258         }
259       }
260     }
261   }

263   private <E> List<E> queryFromDatabase(MappedStatement ms, Object parameter, RowBounds rowBounds, ResultHandler resultHandler, CacheKey key, BoundSql boundSql) throws SQLException {
264     List<E> list;
265     localCache.putObject(key, EXECUTION_PLACEHOLDER);
266     try {
267       list = doQuery(ms, parameter, rowBounds, resultHandler, boundSql);
268     } finally {
269       localCache.removeObject(key);
270     }
271     localCache.putObject(key, list);
272     if (ms.getStatementType() == StatementType.CALLABLE) {
273       localOutputParameterCache.putObject(key, parameter);
274     }
275     return list;
276   }

291   private static class DeferredLoad {
292 
293     private final MetaObject resultObject;
294     private final String property;
295     private final Class<?> targetType;
296     private final CacheKey key;
297     private final PerpetualCache localCache;
298     private final ObjectFactory objectFactory;
299     private final ResultExtractor resultExtractor;
300 
301     public DeferredLoad(MetaObject resultObject,
302                         String property,
303                         CacheKey key,
304                         PerpetualCache localCache,
305                         Configuration configuration,
306                         Class<?> targetType) { // issue #781
307       this.resultObject = resultObject;
308       this.property = property;
309       this.key = key;
310       this.localCache = localCache;
311       this.objectFactory = configuration.getObjectFactory();
312       this.resultExtractor = new ResultExtractor(configuration, objectFactory);
313       this.targetType = targetType;
314     }
315 
316     public boolean canLoad() {
317       return localCache.getObject(key) != null && localCache.getObject(key) != EXECUTION_PLACEHOLDER;
318     }
319 
320     public void load() {
321       @SuppressWarnings( "unchecked" ) // we suppose we get back a List
322       List<Object> list = (List<Object>) localCache.getObject(key);
323       Object value = resultExtractor.extractObjectFromList(list, targetType);
324       resultObject.setValue(property, value);
325     }

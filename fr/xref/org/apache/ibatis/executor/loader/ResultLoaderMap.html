50    private final Map<String, LoadPair> loaderMap = new HashMap<String, LoadPair>();
51  
52    public void addLoader(String property, MetaObject metaResultObject, ResultLoader resultLoader) {
53      String upperFirst = getUppercaseFirstProperty(property);
54      if (!upperFirst.equalsIgnoreCase(property) && loaderMap.containsKey(upperFirst)) {
55        throw new ExecutorException("Nested lazy loaded result property '" + property +
56                "' for query id '" + resultLoader.mappedStatement.getId() +
57                " already exists in the result map. The leftmost property of all lazy loaded properties must be unique within a result map.");
58      }
59      loaderMap.put(upperFirst, new LoadPair(property, metaResultObject, resultLoader));
60    }
61  
62    public final Map<String, LoadPair> getProperties() {
63      return new HashMap<String, LoadPair>(this.loaderMap);
64    }
65  
66    public Set<String> getPropertyNames() {
67      return loaderMap.keySet();
68    }
69  
70    public int size() {
71      return loaderMap.size();
72    }
73  
74    public boolean hasLoader(String property) {
75      return loaderMap.containsKey(property.toUpperCase(Locale.ENGLISH));
76    }
77  
78    public boolean load(String property) throws SQLException {
79      LoadPair pair = loaderMap.remove(property.toUpperCase(Locale.ENGLISH));
80      if (pair != null) {
81        pair.load();
82        return true;
83      }
84      return false;
85    }
86  
87    public void loadAll() throws SQLException {
88      final Set<String> methodNameSet = loaderMap.keySet();
89      String[] methodNames = methodNameSet.toArray(new String[methodNameSet.size()]);
90      for (String methodName : methodNames) {
91        load(methodName);
92      }
93    }
94  
95    private static String getUppercaseFirstProperty(String property) {
96      String[] parts = property.split("\\.");
97      return parts[0].toUpperCase(Locale.ENGLISH);
98    }

100   /**
101    * Property which was not loaded yet.
102    */
103   public static class LoadPair implements Serializable {
104 
105     private static final long serialVersionUID = 20130412;
106     /**
107      * Name of factory method which returns database connection.
108      */
109     private static final String FACTORY_METHOD = "getConfiguration";
110     /**
111      * Object to check whether we went through serialization..
112      */
113     private final transient Object serializationCheck = new Object();
114     /**
115      * Meta object which sets loaded properties.
116      */
117     private transient MetaObject metaResultObject;
118     /**
119      * Result loader which loads unread properties.
120      */
121     private transient ResultLoader resultLoader;
122     /**
123      * Wow, logger.
124      */
125     private transient Log log;
126     /**
127      * Factory class through which we get database connection.
128      */
129     private Class<?> configurationFactory;
130     /**
131      * Name of the unread property.
132      */
133     private String property;
134     /**
135      * ID of SQL statement which loads the property.
136      */
137     private String mappedStatement;
138     /**
139      * Parameter of the sql statement.
140      */
141     private Serializable mappedParameter;
142 
143     private LoadPair(final String property, MetaObject metaResultObject, ResultLoader resultLoader) {
144       this.property = property;
145       this.metaResultObject = metaResultObject;
146       this.resultLoader = resultLoader;
147 
148       /* Save required information only if original object can be serialized. */
149       if (metaResultObject != null && metaResultObject.getOriginalObject() instanceof Serializable) {
150         final Object mappedStatementParameter = resultLoader.parameterObject;
151 
152         /* @todo May the parameter be null? */
153         if (mappedStatementParameter instanceof Serializable) {
154           this.mappedStatement = resultLoader.mappedStatement.getId();
155           this.mappedParameter = (Serializable) mappedStatementParameter;
156 
157           this.configurationFactory = resultLoader.configuration.getConfigurationFactory();
158         } else {
159           Log log = this.getLogger();
160           if (log.isDebugEnabled()) {
161             log.debug("Property [" + this.property + "] of ["
162                     + metaResultObject.getOriginalObject().getClass() + "] cannot be loaded "
163                     + "after deserialization. Make sure it's loaded before serializing "
164                     + "forenamed object.");
165           }
166         }
167       }
168     }
169 
170     public void load() throws SQLException {
171       /* These field should not be null unless the loadpair was serialized.
172        * Yet in that case this method should not be called. */
173       if (this.metaResultObject == null) {
174         throw new IllegalArgumentException("metaResultObject is null");
175       }
176       if (this.resultLoader == null) {
177         throw new IllegalArgumentException("resultLoader is null");
178       }
179 
180       this.load(null);
181     }
182 
183     public void load(final Object userObject) throws SQLException {
184       if (this.metaResultObject == null || this.resultLoader == null) {
185         if (this.mappedParameter == null) {
186           throw new ExecutorException("Property [" + this.property + "] cannot be loaded because "
187                   + "required parameter of mapped statement ["
188                   + this.mappedStatement + "] is not serializable.");
189         }
190 
191         final Configuration config = this.getConfiguration();
192         final MappedStatement ms = config.getMappedStatement(this.mappedStatement);
193         if (ms == null) {
194           throw new ExecutorException("Cannot lazy load property [" + this.property
195                   + "] of deserialized object [" + userObject.getClass()
196                   + "] because configuration does not contain statement ["
197                   + this.mappedStatement + "]");
198         }
199 
200         this.metaResultObject = config.newMetaObject(userObject);
201         this.resultLoader = new ResultLoader(config, new ClosedExecutor(), ms, this.mappedParameter,
202                 metaResultObject.getSetterType(this.property), null, null);
203       }
204 
205       /* We are using a new executor because we may be (and likely are) on a new thread
206        * and executors aren't thread safe. (Is this sufficient?)
207        *
208        * A better approach would be making executors thread safe. */
209       if (this.serializationCheck == null) {
210         final ResultLoader old = this.resultLoader;
211         this.resultLoader = new ResultLoader(old.configuration, new ClosedExecutor(), old.mappedStatement,
212                 old.parameterObject, old.targetType, old.cacheKey, old.boundSql);
213       }
214 
215       this.metaResultObject.setValue(property, this.resultLoader.loadResult());
216     }
217 
218     private Configuration getConfiguration() {
219       if (this.configurationFactory == null) {
220         throw new ExecutorException("Cannot get Configuration as configuration factory was not set.");
221       }
222 
223       Object configurationObject = null;
224       try {
225         final Method factoryMethod = this.configurationFactory.getDeclaredMethod(FACTORY_METHOD);
226         if (!Modifier.isStatic(factoryMethod.getModifiers())) {
227           throw new ExecutorException("Cannot get Configuration as factory method ["
228                   + this.configurationFactory + "]#["
229                   + FACTORY_METHOD + "] is not static.");
230         }
231 
232         if (!factoryMethod.isAccessible()) {
233           configurationObject = AccessController.doPrivileged(new PrivilegedExceptionAction<Object>() {
234             @Override
235             public Object run() throws Exception {
236               try {
237                 factoryMethod.setAccessible(true);
238                 return factoryMethod.invoke(null);
239               } finally {
240                 factoryMethod.setAccessible(false);
241               }
242             }
243           });
244         } else {
245           configurationObject = factoryMethod.invoke(null);
246         }
247       } catch (final NoSuchMethodException ex) {
248         throw new ExecutorException("Cannot get Configuration as factory class ["
249                 + this.configurationFactory + "] is missing factory method of name ["
250                 + FACTORY_METHOD + "].", ex);
251       } catch (final PrivilegedActionException ex) {
252         throw new ExecutorException("Cannot get Configuration as factory method ["
253                 + this.configurationFactory + "]#["
254                 + FACTORY_METHOD + "] threw an exception.", ex.getCause());
255       } catch (final Exception ex) {
256         throw new ExecutorException("Cannot get Configuration as factory method ["
257                 + this.configurationFactory + "]#["
258                 + FACTORY_METHOD + "] threw an exception.", ex);
259       }
260 
261       if (!(configurationObject instanceof Configuration)) {
262         throw new ExecutorException("Cannot get Configuration as factory method ["
263                 + this.configurationFactory + "]#["
264                 + FACTORY_METHOD + "] didn't return [" + Configuration.class + "] but ["
265                 + (configurationObject == null ? "null" : configurationObject.getClass()) + "].");
266       }
267 
268       return Configuration.class.cast(configurationObject);
269     }
270 
271     private Log getLogger() {
272       if (this.log == null) {
273         this.log = LogFactory.getLog(this.getClass());
274       }
275       return this.log;
276     }
277   }

279   private static final class ClosedExecutor extends BaseExecutor {
280 
281     public ClosedExecutor() {
282       super(null, null);
283     }
284 
285     @Override
286     public boolean isClosed() {
287       return true;
288     }
289 
290     @Override

292       throw new UnsupportedOperationException("Not supported.");
293     }

297       throw new UnsupportedOperationException("Not supported.");
298     }

307       throw new UnsupportedOperationException("Not supported.");
308     }

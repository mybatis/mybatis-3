60    public String getPath() {
61      StringBuilder builder = new StringBuilder();
62      Node current = node;
63      while (current != null && current instanceof Element) {
64        if (current != node) {
65          builder.insert(0, "/");
66        }
67        builder.insert(0, current.getNodeName());
68        current = current.getParentNode();
69      }
70      return builder.toString();
71    }

73    public String getValueBasedIdentifier() {
74      StringBuilder builder = new StringBuilder();
75      XNode current = this;
76      while (current != null) {
77        if (current != this) {
78          builder.insert(0, "_");
79        }
80        String value = current.getStringAttribute("id",
81            current.getStringAttribute("value",
82                current.getStringAttribute("property", null)));
83        if (value != null) {
84          value = value.replace('.', '_');
85          builder.insert(0, "]");
86          builder.insert(0,
87              value);
88          builder.insert(0, "[");
89        }
90        builder.insert(0, current.getName());
91        current = current.getParent();
92      }
93      return builder.toString();
94    }

287   public List<XNode> getChildren() {
288     List<XNode> children = new ArrayList<XNode>();
289     NodeList nodeList = node.getChildNodes();
290     if (nodeList != null) {
291       for (int i = 0, n = nodeList.getLength(); i < n; i++) {
292         Node node = nodeList.item(i);
293         if (node.getNodeType() == Node.ELEMENT_NODE) {
294           children.add(new XNode(xpathParser, node, variables));
295         }
296       }
297     }
298     return children;
299   }

313   public String toString() {
314     StringBuilder builder = new StringBuilder();
315     builder.append("<");
316     builder.append(name);
317     for (Map.Entry<Object, Object> entry : attributes.entrySet()) {
318       builder.append(" ");
319       builder.append(entry.getKey());
320       builder.append("=\"");
321       builder.append(entry.getValue());
322       builder.append("\"");
323     }
324     List<XNode> children = getChildren();
325     if (children.size() > 0) {
326       builder.append(">\n");
327       for (XNode node : children) {
328         builder.append(node.toString());
329       }
330       builder.append("</");
331       builder.append(name);
332       builder.append(">");
333     } else if (body != null) {
334       builder.append(">");
335       builder.append(body);
336       builder.append("</");
337       builder.append(name);
338       builder.append(">");
339     } else {
340       builder.append("/>");
341     }
342     builder.append("\n");
343     return builder.toString();
344   }

346   private Properties parseAttributes(Node n) {
347     Properties attributes = new Properties();
348     NamedNodeMap attributeNodes = n.getAttributes();
349     if (attributeNodes != null) {
350       for (int i = 0; i < attributeNodes.getLength(); i++) {
351         Node attribute = attributeNodes.item(i);
352         String value = PropertyParser.parse(attribute.getNodeValue(), variables);
353         attributes.put(attribute.getNodeName(), value);
354       }
355     }
356     return attributes;
357   }
358 
359   private String parseBody(Node node) {
360     String data = getBodyData(node);
361     if (data == null) {
362       NodeList children = node.getChildNodes();
363       for (int i = 0; i < children.getLength(); i++) {
364         Node child = children.item(i);
365         data = getBodyData(child);
366         if (data != null) break;
367       }
368     }
369     return data;
370   }
371 
372   private String getBodyData(Node child) {
373     if (child.getNodeType() == Node.CDATA_SECTION_NODE
374         || child.getNodeType() == Node.TEXT_NODE) {
375       String data = ((CharacterData) child).getData();
376       data = PropertyParser.parse(data, variables);
377       return data;
378     }
379     return null;
380   }

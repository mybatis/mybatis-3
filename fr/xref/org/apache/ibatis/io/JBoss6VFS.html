18  import java.io.IOException;
19  import java.lang.reflect.Method;
20  import java.net.URL;
21  import java.util.ArrayList;
22  import java.util.Collections;
23  import java.util.List;

25  import org.apache.ibatis.logging.Log;
26  import org.apache.ibatis.logging.LogFactory;

28  /**
29   * A {@link VFS} implementation that works with the VFS API provided by JBoss 6.
30   * 
31   * @author Ben Gunter
32   */
33  public class JBoss6VFS extends VFS {
34    private static final Log log = LogFactory.getLog(ResolverUtil.class);
35  
36    /** A class that mimics a tiny subset of the JBoss VirtualFile class. */
37    static class VirtualFile {
38      static Class<?> VirtualFile;
39      static Method getPathNameRelativeTo, getChildrenRecursively;
40  
41      Object virtualFile;
42  
43      VirtualFile(Object virtualFile) {
44        this.virtualFile = virtualFile;
45      }
46  
47      String getPathNameRelativeTo(VirtualFile parent) {
48        try {
49          return invoke(getPathNameRelativeTo, virtualFile, parent.virtualFile);
50        } catch (IOException e) {
51          // This exception is not thrown by the called method
52          log.error("This should not be possible. VirtualFile.getPathNameRelativeTo() threw IOException.");
53          return null;
54        }
55      }
56  
57      List<VirtualFile> getChildren() throws IOException {
58        List<?> objects = invoke(getChildrenRecursively, virtualFile);
59        List<VirtualFile> children = new ArrayList<VirtualFile>(objects.size());
60        for (Object object : objects) {
61          children.add(new VirtualFile(object));
62        }
63        return children;
64      }
65    }
66  
67    /** A class that mimics a tiny subset of the JBoss VFS class. */
68    static class VFS {
69      static Class<?> VFS;
70      static Method getChild;
71  
72      static VirtualFile getChild(URL url) throws IOException {
73        Object o = invoke(getChild, VFS, url);
74        return o == null ? null : new VirtualFile(o);
75      }
76    }
77  
78    /** Flag that indicates if this VFS is valid for the current environment. */
79    private static Boolean valid;
80  
81    /** Find all the classes and methods that are required to access the JBoss 6 VFS. */
82    protected static synchronized void initialize() {
83      if (valid == null) {
84        // Assume valid. It will get flipped later if something goes wrong.
85        valid = true;
86  
87        // Look up and verify required classes
88        VFS.VFS = checkNotNull(getClass("org.jboss.vfs.VFS"));
89        VirtualFile.VirtualFile = checkNotNull(getClass("org.jboss.vfs.VirtualFile"));
90  
91        // Look up and verify required methods
92        VFS.getChild = checkNotNull(getMethod(VFS.VFS, "getChild", URL.class));
93        VirtualFile.getChildrenRecursively = checkNotNull(getMethod(VirtualFile.VirtualFile,
94            "getChildrenRecursively"));
95        VirtualFile.getPathNameRelativeTo = checkNotNull(getMethod(VirtualFile.VirtualFile,
96            "getPathNameRelativeTo", VirtualFile.VirtualFile));
97  
98        // Verify that the API has not changed
99        checkReturnType(VFS.getChild, VirtualFile.VirtualFile);
100       checkReturnType(VirtualFile.getChildrenRecursively, List.class);
101       checkReturnType(VirtualFile.getPathNameRelativeTo, String.class);
102     }
103   }
104 
105   /**
106    * Verifies that the provided object reference is null. If it is null, then this VFS is marked
107    * as invalid for the current environment.
108    * 
109    * @param object The object reference to check for null.
110    */
111   protected static <T> T checkNotNull(T object) {
112     if (object == null)
113       setInvalid();
114     return object;
115   }
116 
117   /**
118    * Verifies that the return type of a method is what it is expected to be. If it is not, then
119    * this VFS is marked as invalid for the current environment.
120    * 
121    * @param method The method whose return type is to be checked.
122    * @param expected A type to which the method's return type must be assignable.
123    * @see Class#isAssignableFrom(Class)
124    */
125   protected static void checkReturnType(Method method, Class<?> expected) {
126     if (method != null && !expected.isAssignableFrom(method.getReturnType())) {
127       log.error("Method " + method.getClass().getName() + "." + method.getName()
128           + "(..) should return " + expected.getName() + " but returns " //
129           + method.getReturnType().getName() + " instead.");
130       setInvalid();
131     }
132   }
133 
134   /** Mark this {@link VFS} as invalid for the current environment. */
135   protected static void setInvalid() {
136     if (JBoss6VFS.valid != null && JBoss6VFS.valid) {
137       log.debug("JBoss 6 VFS API is not available in this environment.");
138       JBoss6VFS.valid = false;
139     }
140   }
141 
142   static {
143     initialize();
144   }
145 
146   @Override
147   public boolean isValid() {
148     return valid;
149   }
150 
151   @Override
152   public List<String> list(URL url, String path) throws IOException {
153     VirtualFile directory;
154     directory = VFS.getChild(url);
155     if (directory == null)
156       return Collections.emptyList();
157 
158     if (!path.endsWith("/"))
159       path += "/";
160 
161     List<VirtualFile> children = directory.getChildren();
162     List<String> names = new ArrayList<String>(children.size());
163     for (VirtualFile vf : children) {
164       String relative = vf.getPathNameRelativeTo(directory);
165       names.add(path + relative);
166     }
167 
168     return names;
169   }

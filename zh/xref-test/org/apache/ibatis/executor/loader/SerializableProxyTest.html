178   protected byte[] serialize(Serializable value) throws Exception {
179     ByteArrayOutputStream bos = new ByteArrayOutputStream();
180     ObjectOutputStream oos = new ObjectOutputStream(bos);
181     oos.writeObject(value);
182     oos.flush();
183     oos.close();
184     return bos.toByteArray();
185   }
186 
187   protected Serializable deserialize(byte[] value) throws Exception {
188     ByteArrayInputStream bis = new ByteArrayInputStream(value);
189     ObjectInputStream ois = new ObjectInputStream(bis);
190     Serializable result = (Serializable) ois.readObject();
191     ois.close();
192     return result;
193   }

16  package org.apache.ibatis.domain.jpetstore;
17  
18  import java.io.Serializable;
19  import java.math.BigDecimal;
20  
21  
22  public class LineItem implements Serializable {
23  
24    private static final long serialVersionUID = 1L;
25  
26    private int orderId;
27    private int lineNumber;
28    private int quantity;
29    private String itemId;
30    private BigDecimal unitPrice;
31    private Item item;
32    private BigDecimal total;
33  
34    public LineItem() {
35    }
36  
37    public LineItem(int lineNumber, CartItem cartItem) {
38      this.lineNumber = lineNumber;
39      this.quantity = cartItem.getQuantity();
40      this.itemId = cartItem.getItem().getItemId();
41      this.unitPrice = cartItem.getItem().getListPrice();
42      this.item = cartItem.getItem();
43    }
44  
45    public int getOrderId() {
46      return orderId;
47    }
48  
49    public void setOrderId(int orderId) {
50      this.orderId = orderId;
51    }
52  
53    public int getLineNumber() {
54      return lineNumber;
55    }
56  
57    public void setLineNumber(int lineNumber) {
58      this.lineNumber = lineNumber;
59    }
60  
61    public String getItemId() {
62      return itemId;
63    }
64  
65    public void setItemId(String itemId) {
66      this.itemId = itemId;
67    }
68  
69    public BigDecimal getUnitPrice() {
70      return unitPrice;
71    }
72  
73    public void setUnitPrice(BigDecimal unitprice) {
74      this.unitPrice = unitprice;
75    }
76  
77    public BigDecimal getTotal() {
78      return total;
79    }
80  
81    public Item getItem() {
82      return item;
83    }
84  
85    public void setItem(Item item) {
86      this.item = item;
87      calculateTotal();
88    }
89  
90    public int getQuantity() {
91      return quantity;
92    }
93  
94    public void setQuantity(int quantity) {
95      this.quantity = quantity;
96      calculateTotal();
97    }
98  
99    private void calculateTotal() {
100     if (item != null && item.getListPrice() != null) {
101       total = item.getListPrice().multiply(new BigDecimal(quantity));
102     } else {
103       total = null;
104     }
105   }
106 
107 
108 }

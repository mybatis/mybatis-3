209   @Test
210   public void shouldDemonstrateDeeplyNestedMapProperties() {
211     HashMap<String, String> map = new HashMap<String, String>();
212     MetaObject metaMap = SystemMetaObject.forObject(map);
213 
214     assertTrue(metaMap.hasSetter("id"));
215     assertTrue(metaMap.hasSetter("name.first"));
216     assertTrue(metaMap.hasSetter("address.street"));
217 
218     assertFalse(metaMap.hasGetter("id"));
219     assertFalse(metaMap.hasGetter("name.first"));
220     assertFalse(metaMap.hasGetter("address.street"));
221 
222     metaMap.setValue("id", "100");
223     metaMap.setValue("name.first", "Clinton");
224     metaMap.setValue("name.last", "Begin");
225     metaMap.setValue("address.street", "1 Some Street");
226     metaMap.setValue("address.city", "This City");
227     metaMap.setValue("address.province", "A Province");
228     metaMap.setValue("address.postal_code", "1A3 4B6");
229 
230     assertTrue(metaMap.hasGetter("id"));
231     assertTrue(metaMap.hasGetter("name.first"));
232     assertTrue(metaMap.hasGetter("address.street"));
233 
234     assertEquals(3, metaMap.getGetterNames().length);
235     assertEquals(3, metaMap.getSetterNames().length);
236 
237     Map<String,String> name = (Map<String,String>) metaMap.getValue("name");
238     Map<String,String> address = (Map<String,String>) metaMap.getValue("address");
239 
240     assertEquals("Clinton", name.get("first"));
241     assertEquals("1 Some Street", address.get("street"));
242   }

244   @Test
245   public void shouldDemonstrateNullValueInMap() {
246     HashMap<String, String> map = new HashMap<String, String>();
247     MetaObject metaMap = SystemMetaObject.forObject(map);
248     assertFalse(metaMap.hasGetter("phone.home"));
249 
250     metaMap.setValue("phone", null);
251     assertTrue(metaMap.hasGetter("phone"));
252     // hasGetter returns true if the parent exists and is null.
253     assertTrue(metaMap.hasGetter("phone.home"));
254     assertTrue(metaMap.hasGetter("phone.home.ext"));
255     assertNull(metaMap.getValue("phone"));
256     assertNull(metaMap.getValue("phone.home"));
257     assertNull(metaMap.getValue("phone.home.ext"));
258 
259     metaMap.setValue("phone.office", "789");
260     assertFalse(metaMap.hasGetter("phone.home"));
261     assertFalse(metaMap.hasGetter("phone.home.ext"));
262     assertEquals("789", metaMap.getValue("phone.office"));
263     assertNotNull(metaMap.getValue("phone"));
264     assertNull(metaMap.getValue("phone.home"));
265   }

283   @Test
284   public void shouldMethodHasGetterReturnTrueWhenListElementSet() {
285     List<Object> param1 = new ArrayList<Object>();
286     param1.add("firstParam");
287     param1.add(222);
288     param1.add(new Date());
289 
290     Map<String, Object> parametersEmulation = new HashMap<String, Object>();
291     parametersEmulation.put("param1", param1);
292     parametersEmulation.put("filterParams", param1);
293 
294     MetaObject meta = SystemMetaObject.forObject(parametersEmulation);
295 
296     assertEquals(param1.get(0), meta.getValue("filterParams[0]"));
297     assertEquals(param1.get(1), meta.getValue("filterParams[1]"));
298     assertEquals(param1.get(2), meta.getValue("filterParams[2]"));
299 
300     assertTrue(meta.hasGetter("filterParams[0]"));
301     assertTrue(meta.hasGetter("filterParams[1]"));
302     assertTrue(meta.hasGetter("filterParams[2]"));
303   }

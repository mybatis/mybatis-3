16  package org.apache.ibatis.submitted.global_variables_defaults;
17  
18  import org.apache.ibatis.io.Resources;
19  import org.apache.ibatis.parsing.PropertyParser;
20  import org.apache.ibatis.session.Configuration;
21  import org.apache.ibatis.session.SqlSession;
22  import org.apache.ibatis.session.SqlSessionFactory;
23  import org.apache.ibatis.session.SqlSessionFactoryBuilder;
24  import org.hamcrest.core.Is;
25  import org.junit.Assert;
26  import org.junit.Test;
27  
28  import java.io.IOException;
29  import java.io.Reader;
30  import java.util.Properties;
31  
32  public class XmlMapperTest {
33  
34    @Test
35    public void applyDefaultValueOnXmlMapper() throws IOException {
36  
37      Properties props = new Properties();
38      props.setProperty(PropertyParser.KEY_ENABLE_DEFAULT_VALUE, "true");
39  
40      Reader reader = Resources.getResourceAsReader("org/apache/ibatis/submitted/global_variables_defaults/mybatis-config.xml");
41      SqlSessionFactory factory = new SqlSessionFactoryBuilder().build(reader, props);
42      Configuration configuration = factory.getConfiguration();
43      configuration.addMapper(XmlMapper.class);
44      SupportClasses.CustomCache cache = SupportClasses.Utils.unwrap(configuration.getCache(XmlMapper.class.getName()));
45  
46      Assert.assertThat(cache.getName(), Is.is("default"));
47  
48      SqlSession sqlSession = factory.openSession();
49      try {
50        XmlMapper mapper = sqlSession.getMapper(XmlMapper.class);
51  
52        Assert.assertThat(mapper.ping(), Is.is("Hello"));
53        Assert.assertThat(mapper.selectOne(), Is.is("1"));
54        Assert.assertThat(mapper.selectFromVariable(), Is.is("9999"));
55  
56      } finally {
57        sqlSession.close();
58      }
59  
60    }
61  
62    @Test
63    public void applyPropertyValueOnXmlMapper() throws IOException {
64  
65      Properties props = new Properties();
66      props.setProperty(PropertyParser.KEY_ENABLE_DEFAULT_VALUE, "true");
67      props.setProperty("ping.sql", "SELECT 'Hi' FROM INFORMATION_SCHEMA.SYSTEM_USERS");
68      props.setProperty("cache.name", "custom");
69      props.setProperty("select.columns", "'5555'");
70  
71      Reader reader = Resources.getResourceAsReader("org/apache/ibatis/submitted/global_variables_defaults/mybatis-config.xml");
72      SqlSessionFactory factory = new SqlSessionFactoryBuilder().build(reader, props);
73      Configuration configuration = factory.getConfiguration();
74      configuration.addMapper(XmlMapper.class);
75      SupportClasses.CustomCache cache = SupportClasses.Utils.unwrap(configuration.getCache(XmlMapper.class.getName()));
76  
77      Assert.assertThat(cache.getName(), Is.is("custom"));
78  
79      SqlSession sqlSession = factory.openSession();
80      try {
81        XmlMapper mapper = sqlSession.getMapper(XmlMapper.class);
82  
83        Assert.assertThat(mapper.ping(), Is.is("Hi"));
84        Assert.assertThat(mapper.selectOne(), Is.is("1"));
85        Assert.assertThat(mapper.selectFromVariable(), Is.is("5555"));
86  
87      } finally {
88        sqlSession.close();
89      }
90  
91    }
92  
93    public interface XmlMapper {
94  
95      String ping();
96  
97      String selectOne();
98  
99      String selectFromVariable();
100 
101   }

103 }

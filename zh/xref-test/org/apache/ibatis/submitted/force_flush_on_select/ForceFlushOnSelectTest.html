41    @Before
42    public void initDatabase() throws Exception {
43      Connection conn = null;
44  
45      try {
46        Class.forName("org.hsqldb.jdbcDriver");
47        conn = DriverManager.getConnection("jdbc:hsqldb:mem:force_flush_on_select", "sa", "");
48  
49        Reader reader = Resources.getResourceAsReader("org/apache/ibatis/submitted/force_flush_on_select/CreateDB.sql");
50        ScriptRunner runner = new ScriptRunner(conn);
51        runner.setLogWriter(null);
52        runner.setErrorLogWriter(null);
53        runner.runScript(reader);
54        conn.commit();
55        reader.close();
56  
57        reader = Resources.getResourceAsReader("org/apache/ibatis/submitted/force_flush_on_select/ibatisConfig.xml");
58        sqlSessionFactory = new SqlSessionFactoryBuilder().build(reader);
59        reader.close();
60      } finally {
61        if (conn != null) {
62          conn.close();
63        }
64      }

67    @Test
68    public void testShouldFlushLocalSessionCacheOnQuery() throws SQLException {
69      SqlSession sqlSession = sqlSessionFactory.openSession(ExecutorType.SIMPLE);
70      try {
71        PersonMapper personMapper = sqlSession.getMapper(PersonMapper.class);
72        personMapper.selectByIdFlush(1);
73        updateDatabase(sqlSession.getConnection());
74        Person updatedPerson = personMapper.selectByIdFlush(1);
75        assertEquals("Simone", updatedPerson.getFirstName());
76        sqlSession.commit();
77      } finally {
78        sqlSession.close();
79      }

133   @Test
134   public void testUpdateShouldFlushLocalCache() throws SQLException {
135     SqlSession sqlSession = sqlSessionFactory.openSession(ExecutorType.SIMPLE);
136     try {
137       PersonMapper personMapper = sqlSession.getMapper(PersonMapper.class);
138       Person person = personMapper.selectByIdNoFlush(1);
139       person.setLastName("Perez"); //it is ignored in update
140       personMapper.update(person);
141       Person updatedPerson = personMapper.selectByIdNoFlush(1);
142       assertEquals("Smith", updatedPerson.getLastName());
143       assertNotSame(person, updatedPerson);
144       sqlSession.commit();
145     } finally {
146       sqlSession.close();
147     }
148   }

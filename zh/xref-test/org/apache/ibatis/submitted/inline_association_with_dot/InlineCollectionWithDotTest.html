16  package org.apache.ibatis.submitted.inline_association_with_dot;
17  
18  import static org.junit.Assert.assertEquals;
19  
20  import java.io.Reader;
21  import java.sql.Connection;
22  
23  import org.apache.ibatis.io.Resources;
24  import org.apache.ibatis.jdbc.ScriptRunner;
25  import org.apache.ibatis.session.SqlSession;
26  import org.apache.ibatis.session.SqlSessionFactory;
27  import org.apache.ibatis.session.SqlSessionFactoryBuilder;
28  import org.junit.After;
29  import org.junit.Test;
30  
31  public class InlineCollectionWithDotTest {
32  
33    private SqlSession sqlSession;
34  
35    public void openSession(String aConfig) throws Exception {
36  
37      final String resource = "org/apache/ibatis/submitted/inline_association_with_dot/ibatis-" + aConfig + ".xml";
38      Reader batisConfigReader = Resources.getResourceAsReader(resource);
39  
40      SqlSessionFactory sqlSessionFactory;
41      try {
42        sqlSessionFactory = new SqlSessionFactoryBuilder().build(batisConfigReader);
43      } catch(Exception anException) {
44        throw new RuntimeException("Mapper configuration failed, expected this to work: " + anException.getMessage(), anException);
45      }
46  
47      SqlSession session = sqlSessionFactory.openSession();
48  
49      Connection conn = session.getConnection();
50      ScriptRunner runner = new ScriptRunner(conn);
51      runner.setLogWriter(null);
52      runner.setErrorLogWriter(null);
53      Reader createScriptReader = Resources.getResourceAsReader("org/apache/ibatis/submitted/inline_association_with_dot/create.sql");
54      runner.runScript(createScriptReader);
55  
56      sqlSession = sqlSessionFactory.openSession();
57    }
58  
59    @After
60    public void closeSession() {
61      if (sqlSession != null) {
62        sqlSession.close();
63      }
64    }
65  
66    /*
67     * Load an element with an element with and element with a value. Expect that this is
68     * possible bij using an inline 'association' map.
69     */
70    @Test
71    public void selectElementValueInContainerUsingInline()
72    throws Exception {
73      openSession("inline");
74  
75      Element myElement = sqlSession.getMapper(ElementMapperUsingInline.class).selectElement();
76  
77      assertEquals("value", myElement.getElement().getElement().getValue());
78    }
79  
80    /*
81     * Load an element with an element with and element with a value. Expect that this is
82     * possible bij using an sub-'association' map.
83     */
84    @Test
85    public void selectElementValueInContainerUsingSubMap() throws Exception {
86     openSession("submap");
87  
88     Element myElement = sqlSession.getMapper(ElementMapperUsingSubMap.class).selectElement();
89  
90     assertEquals("value", myElement.getElement().getElement().getValue());
91    }
92  }

34  /*
35   * This class contains tests for refcursors.  The tests require a
36   * local install of PostgreSQL and cannot be run as a part of the normal
37   * MyBatis build unless PostreSQL is setup on the build machine as 
38   * described in setupdb.txt
39   * 
40   * If PostgreSQL is setup as described in setupdb.txt, then remove
41   * the @Ignore annotation to enable the tests.
42   * 
43   * @author Jeff Butler
44   *
45   */
46  @Ignore("See setupdb.txt for instructions on how to run the tests in this class")
47  public class RefCursorTest {
48      @SuppressWarnings("unchecked")
49      @Test
50      public void testRefCursor1() throws IOException {
51          Reader reader = Resources.getResourceAsReader("org/apache/ibatis/submitted/refcursor/MapperConfig.xml");
52          SqlSessionFactory sqlSessionFactory = new SqlSessionFactoryBuilder().build(reader);
53          SqlSession sqlSession = sqlSessionFactory.openSession();
54          try {
55              OrdersMapper mapper = sqlSession.getMapper(OrdersMapper.class);
56              Map<String, Object> parameter = new HashMap<String, Object>();
57              parameter.put("orderId", 1);
58              mapper.getOrder1(parameter);
59              
60              assertNotNull(parameter.get("order"));
61              List<Order> orders = (List<Order>) parameter.get("order");
62              assertEquals(1, orders.size());
63              Order order = orders.get(0);
64              assertEquals(3, order.getDetailLines().size());
65          } finally {
66              sqlSession.close();
67          }
68      }

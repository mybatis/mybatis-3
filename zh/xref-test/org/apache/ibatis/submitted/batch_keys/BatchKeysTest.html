45    @Before
46    public void setUp() throws Exception {
47      Connection conn = null;
48  
49      try {
50        Class.forName("org.hsqldb.jdbcDriver");
51        conn = DriverManager.getConnection("jdbc:hsqldb:mem:batch_keys", "sa", "");
52  
53        Reader reader = Resources.getResourceAsReader("org/apache/ibatis/submitted/batch_keys/CreateDB.sql");
54  
55        ScriptRunner runner = new ScriptRunner(conn);
56        runner.setLogWriter(null);
57        runner.setErrorLogWriter(new PrintWriter(System.err));
58        runner.runScript(reader);
59        conn.commit();
60        reader.close();
61  
62        reader = Resources.getResourceAsReader("org/apache/ibatis/submitted/batch_keys/Config.xml");
63        sqlSessionFactory = new SqlSessionFactoryBuilder().build(reader);
64        reader.close();
65      } finally {
66        if (conn != null) {
67          conn.close();
68        }
69      }
70    }

72    public void testJdbc3Support() throws Exception {
73      Connection conn = sqlSessionFactory.getConfiguration().getEnvironment().getDataSource().getConnection();
74      PreparedStatement stmt = conn.prepareStatement("insert into users2 values(null, 'Pocoyo')", Statement.RETURN_GENERATED_KEYS);
75      stmt.addBatch();
76      stmt.executeBatch();
77      ResultSet rs = stmt.getGeneratedKeys();
78      if (rs.next()) {
79        ResultSetMetaData rsmd = rs.getMetaData();
80        int colCount = rsmd.getColumnCount();
81        do {
82          for (int i = 1; i <= colCount; i++) {
83            String key = rs.getString(i);
84            System.out.println("key " + i + " is " + key);
85          }
86        } while (rs.next());
87      } else {
88        System.out.println("There are no generated keys.");
89      }
90    }

92    @Test
93    public void testInsert() throws Exception {
94      SqlSession sqlSession = sqlSessionFactory.openSession(ExecutorType.BATCH);
95      try {
96        User user1 = new User(null, "Pocoyo");
97        sqlSession.insert("insert", user1);
98        User user2 = new User(null, "Valentina");
99        sqlSession.insert("insert", user2);
100       sqlSession.flushStatements();
101       assertEquals(new Integer(50), user1.getId());
102       assertEquals(new Integer(50), user2.getId());
103       sqlSession.commit();
104     } finally {
105       sqlSession.close();
106     }
107     try {
108       sqlSession = sqlSessionFactory.openSession();
109       List<User> users = sqlSession.selectList("select");
110       Assert.assertTrue(users.size() == 2);
111     } finally {
112       sqlSession.close();
113     }
114   }

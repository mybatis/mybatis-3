42  /**
43   * DefaultParameterHandlerTest
44   *
45   * @author Ryan Lamore
46   */
47  public class DefaultParameterHandlerTest {
48  
49    @Test
50    public void setParametersThrowsProperException() throws SQLException {
51      final MappedStatement mappedStatement = getMappedStatement();
52      final Object parameterObject = null;
53      final BoundSql boundSql = mock(BoundSql.class);
54  
55      TypeHandler<Object> typeHandler = mock(TypeHandler.class);
56      doThrow(new SQLException("foo")).when(typeHandler).setParameter(any(PreparedStatement.class), anyInt(), any(), any(JdbcType.class));
57      ParameterMapping parameterMapping = new ParameterMapping.Builder(mappedStatement.getConfiguration(), "prop", typeHandler).build();
58      List<ParameterMapping> parameterMappings = Collections.singletonList(parameterMapping);
59      when(boundSql.getParameterMappings()).thenReturn(parameterMappings);
60  
61      DefaultParameterHandler defaultParameterHandler = new DefaultParameterHandler(mappedStatement, parameterObject, boundSql);
62  
63      PreparedStatement ps = mock(PreparedStatement.class);
64      try {
65        defaultParameterHandler.setParameters(ps);
66        Assert.fail("Should have thrown TypeException");
67      } catch (Exception e) {
68        Assert.assertTrue("expected TypeException", e instanceof TypeException);
69        Assert.assertTrue("", e.getMessage().contains("mapping: ParameterMapping"));
70      }
71  
72    }
73  
74    MappedStatement getMappedStatement() {
75      final Configuration config = new Configuration();
76      final TypeHandlerRegistry registry = config.getTypeHandlerRegistry();
77      return new MappedStatement.Builder(config, "testSelect", new StaticSqlSource(config, "some select statement"), SqlCommandType.SELECT).resultMaps(
78          new ArrayList<ResultMap>() {
79            {
80              add(new ResultMap.Builder(config, "testMap", HashMap.class, new ArrayList<ResultMapping>() {
81                {
82                  add(new ResultMapping.Builder(config, "cOlUmN1", "CoLuMn1", registry.getTypeHandler(Integer.class)).build());
83                }
84              }).build());
85            }
86          }).build();
87    }

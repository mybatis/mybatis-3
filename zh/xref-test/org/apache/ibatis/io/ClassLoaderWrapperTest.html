16  package org.apache.ibatis.io;
17  
18  import org.apache.ibatis.BaseDataTest;
19  import static org.junit.Assert.assertNotNull;
20  import static org.junit.Assert.assertNull;
21  import org.junit.Before;
22  import org.junit.Test;
23  
24  public class ClassLoaderWrapperTest extends BaseDataTest {
25  
26    ClassLoaderWrapper wrapper;
27    ClassLoader loader;
28    private final String RESOURCE_NOT_FOUND = "some_resource_that_does_not_exist.properties";
29    private final String CLASS_NOT_FOUND = "some.random.class.that.does.not.Exist";
30    private final String CLASS_FOUND = "java.lang.Object";
31  
32  
33    @Before
34    public void beforeClassLoaderWrapperTest() {
35      wrapper = new ClassLoaderWrapper();
36      loader = getClass().getClassLoader();
37    }
38  
39    @Test
40    public void classForName() throws ClassNotFoundException {
41      assertNotNull(wrapper.classForName(CLASS_FOUND));
42    }
43  
44    @Test(expected = ClassNotFoundException.class)
45    public void classForNameNotFound() throws ClassNotFoundException {
46      assertNotNull(wrapper.classForName(CLASS_NOT_FOUND));
47    }
48  
49    @Test
50    public void classForNameWithClassLoader() throws ClassNotFoundException {
51      assertNotNull(wrapper.classForName(CLASS_FOUND, loader));
52    }
53  
54    @Test
55    public void getResourceAsURL() {
56      assertNotNull(wrapper.getResourceAsURL(JPETSTORE_PROPERTIES));
57    }
58  
59    @Test
60    public void getResourceAsURLNotFound() {
61      assertNull(wrapper.getResourceAsURL(RESOURCE_NOT_FOUND));
62    }
63  
64    @Test
65    public void getResourceAsURLWithClassLoader() {
66      assertNotNull(wrapper.getResourceAsURL(JPETSTORE_PROPERTIES, loader));
67    }
68  
69    @Test
70    public void getResourceAsStream() {
71      assertNotNull(wrapper.getResourceAsStream(JPETSTORE_PROPERTIES));
72    }
73  
74    @Test
75    public void getResourceAsStreamNotFound() {
76      assertNull(wrapper.getResourceAsStream(RESOURCE_NOT_FOUND));
77    }
78  
79    @Test
80    public void getResourceAsStreamWithClassLoader() {
81      assertNotNull(wrapper.getResourceAsStream(JPETSTORE_PROPERTIES, loader));
82    }
83  
84  }

16  package org.apache.ibatis.type;
17  
18  import static org.junit.jupiter.api.Assertions.*;
19  import static org.mockito.Mockito.*;
20  
21  import java.time.LocalDateTime;
22  
23  import org.junit.jupiter.api.Test;
24  
25  class LocalDateTimeTypeHandlerTest extends BaseTypeHandlerTest {
26  
27    private static final TypeHandler<LocalDateTime> TYPE_HANDLER = new LocalDateTimeTypeHandler();
28    private static final LocalDateTime LOCAL_DATE_TIME = LocalDateTime.now();
29  
30    @Override
31    @Test
32    public void shouldSetParameter() throws Exception {
33      TYPE_HANDLER.setParameter(ps, 1, LOCAL_DATE_TIME, null);
34      verify(ps).setObject(1, LOCAL_DATE_TIME);
35    }
36  
37    @Override
38    @Test
39    public void shouldGetResultFromResultSetByName() throws Exception {
40      when(rs.getObject("column", LocalDateTime.class)).thenReturn(LOCAL_DATE_TIME);
41      assertEquals(LOCAL_DATE_TIME, TYPE_HANDLER.getResult(rs, "column"));
42      verify(rs, never()).wasNull();
43    }
44  
45    @Override
46    @Test
47    public void shouldGetResultNullFromResultSetByName() throws Exception {
48      when(rs.getObject("column", LocalDateTime.class)).thenReturn(null);
49      assertNull(TYPE_HANDLER.getResult(rs, "column"));
50      verify(rs, never()).wasNull();
51    }
52  
53    @Override
54    @Test
55    public void shouldGetResultFromResultSetByPosition() throws Exception {
56      when(rs.getObject(1, LocalDateTime.class)).thenReturn(LOCAL_DATE_TIME);
57      assertEquals(LOCAL_DATE_TIME, TYPE_HANDLER.getResult(rs, 1));
58      verify(rs, never()).wasNull();
59    }
60  
61    @Override
62    @Test
63    public void shouldGetResultNullFromResultSetByPosition() throws Exception {
64      when(rs.getObject(1, LocalDateTime.class)).thenReturn(null);
65      assertNull(TYPE_HANDLER.getResult(rs, 1));
66      verify(rs, never()).wasNull();
67    }
68  
69    @Override
70    @Test
71    public void shouldGetResultFromCallableStatement() throws Exception {
72      when(cs.getObject(1, LocalDateTime.class)).thenReturn(LOCAL_DATE_TIME);
73      assertEquals(LOCAL_DATE_TIME, TYPE_HANDLER.getResult(cs, 1));
74      verify(cs, never()).wasNull();
75    }
76  
77    @Override
78    @Test
79    public void shouldGetResultNullFromCallableStatement() throws Exception {
80      when(cs.getObject(1, LocalDateTime.class)).thenReturn(null);
81      assertNull(TYPE_HANDLER.getResult(cs, 1));
82      verify(cs, never()).wasNull();
83    }
84  }

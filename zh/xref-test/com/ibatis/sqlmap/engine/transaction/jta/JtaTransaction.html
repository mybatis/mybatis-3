16  package com.ibatis.sqlmap.engine.transaction.jta;
17  
18  import com.ibatis.sqlmap.engine.transaction.BaseTransaction;
19  import com.ibatis.sqlmap.engine.transaction.IsolationLevel;
20  import com.ibatis.sqlmap.engine.transaction.TransactionException;
21  import org.apache.ibatis.executor.Executor;
22  import org.apache.ibatis.session.Configuration;
23  import org.apache.ibatis.transaction.jdbc.JdbcTransaction;
24  
25  import javax.sql.DataSource;
26  import javax.transaction.Status;
27  import javax.transaction.UserTransaction;
28  import java.sql.Connection;
29  import java.sql.SQLException;
30  
31  public class JtaTransaction extends BaseTransaction {
32  
33    private UserTransaction userTransaction;
34    private DataSource dataSource;
35    private Configuration configuration;
36    private Executor executor;
37    private IsolationLevel isolationLevel = new IsolationLevel();
38  
39    private boolean commmitted = false;
40    private boolean newTransaction = false;
41  
42    public JtaTransaction(Configuration configuration, UserTransaction utx, DataSource ds, int isolationLevel) throws TransactionException {
43      // Check parameters
44      this.configuration = configuration;
45      userTransaction = utx;
46      dataSource = ds;
47      if (userTransaction == null) {
48        throw new TransactionException("JtaTransaction initialization failed.  UserTransaction was null.");
49      }
50      if (dataSource == null) {
51        throw new TransactionException("JtaTransaction initialization failed.  DataSource was null.");
52      }
53      this.isolationLevel.setIsolationLevel(isolationLevel);
54    }
55  
56    private void init() throws TransactionException, SQLException {
57      // Start JTA Transaction
58      try {
59        newTransaction = userTransaction.getStatus() == Status.STATUS_NO_TRANSACTION;
60        if (newTransaction) {
61          userTransaction.begin();
62        }
63      } catch (Exception e) {
64        throw new TransactionException("JtaTransaction could not start transaction.  Cause: ", e);
65      }
66  
67      // Open JDBC Connection
68      Connection connection = dataSource.getConnection();
69      if (connection == null) {
70        throw new TransactionException("JtaTransaction could not start transaction.  Cause: The DataSource returned a null connection.");
71      }
72      // Isolation Level
73      isolationLevel.applyIsolationLevel(connection);
74      // AutoCommit
75      if (connection.getAutoCommit()) {
76        connection.setAutoCommit(false);
77      }
78      executor = configuration.newExecutor(new JdbcTransaction(connection));
79    }
80  
81    public void commit(boolean required) throws SQLException, TransactionException {
82      if (required) {
83        if (executor != null) {
84          if (commmitted) {
85            throw new TransactionException("JtaTransaction could not commit because this transaction has already been committed.");
86          }
87          try {
88            if (newTransaction) {
89              userTransaction.commit();
90            }
91          } catch (Exception e) {
92            throw new TransactionException("JtaTransaction could not commit.  Cause: ", e);
93          }
94          commmitted = true;
95        }
96      }
97    }
98  
99    public void rollback(boolean required) throws SQLException, TransactionException {
100     if (required) {
101       if (executor != null) {
102         if (!commmitted) {
103           try {
104             if (userTransaction != null) {
105               if (newTransaction) {
106                 userTransaction.rollback();
107               } else {
108                 userTransaction.setRollbackOnly();
109               }
110             }
111           } catch (Exception e) {
112             throw new TransactionException("JtaTransaction could not rollback.  Cause: ", e);
113           }
114         }
115       }
116     }
117   }
118 
119   public void close() throws SQLException, TransactionException {
120     if (executor != null) {
121       try {
122         isolationLevel.restoreIsolationLevel(executor.getTransaction().getConnection());
123       } finally {
124         executor.close(false);
125         executor = null;
126       }
127     }
128   }
129 
130   public Executor getExecutor() throws SQLException, TransactionException {
131     if (executor == null) {
132       init();
133     }
134     return executor;
135   }

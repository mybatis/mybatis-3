44    public void parseGeneralStatement(XNode context) {
45      // get attributes
46      String id = context.getStringAttribute("id");
47      String parameterMapName = context.getStringAttribute("parameterMap");
48      String parameterClassName = context.getStringAttribute("parameterClass");
49      String resultMapName = context.getStringAttribute("resultMap");
50      String resultClassName = context.getStringAttribute("resultClass");
51      String cacheModelName = context.getStringAttribute("cacheModel");
52      String resultSetType = context.getStringAttribute("resultSetType");
53      String fetchSize = context.getStringAttribute("fetchSize");
54      String timeout = context.getStringAttribute("timeout");
55      // 2.x -- String allowRemapping = context.getStringAttribute("remapResults");
56  
57      if (context.getStringAttribute("xmlResultName") != null) {
58        throw new UnsupportedOperationException("xmlResultName is not supported by iBATIS 3");
59      }
60  
61      if (mapParser.getConfigParser().isUseStatementNamespaces()) {
62        id = mapParser.applyNamespace(id);
63      }
64  
65      String[] additionalResultMapNames = null;
66      if (resultMapName != null) {
67        additionalResultMapNames = getAllButFirstToken(resultMapName);
68        resultMapName = getFirstToken(resultMapName);
69        resultMapName = mapParser.applyNamespace(resultMapName);
70        for (int i = 0; i < additionalResultMapNames.length; i++) {
71          additionalResultMapNames[i] = mapParser.applyNamespace(additionalResultMapNames[i]);
72        }
73      }
74  
75      String[] additionalResultClassNames = null;
76      if (resultClassName != null) {
77        additionalResultClassNames = getAllButFirstToken(resultClassName);
78        resultClassName = getFirstToken(resultClassName);
79      }
80      Class[] additionalResultClasses = null;
81      if (additionalResultClassNames != null) {
82        additionalResultClasses = new Class[additionalResultClassNames.length];
83        for (int i = 0; i < additionalResultClassNames.length; i++) {
84          additionalResultClasses[i] = resolveClass(additionalResultClassNames[i]);
85        }
86      }
87  
88      Integer timeoutInt = timeout == null ? null : new Integer(timeout);
89      Integer fetchSizeInt = fetchSize == null ? null : new Integer(fetchSize);
90  
91      // 2.x -- boolean allowRemappingBool = "true".equals(allowRemapping);
92  
93      SqlSource sqlSource = new SqlSourceFactory(mapParser).newSqlSourceIntance(mapParser, context);
94  
95      String nodeName = context.getNode().getNodeName();
96      SqlCommandType sqlCommandType;
97      try {
98        sqlCommandType = SqlCommandType.valueOf(nodeName.toUpperCase());
99      } catch (Exception e) {
100       sqlCommandType = SqlCommandType.UNKNOWN; 
101     }
102 
103 
104     MappedStatement.Builder builder = new MappedStatement.Builder(configuration, id, sqlSource,sqlCommandType);
105 
106     builder.useCache(true);
107     if (!"select".equals(context.getNode().getNodeName())) {
108       builder.flushCacheRequired(true);
109     }
110 
111     if (parameterMapName != null) {
112       parameterMapName = mapParser.applyNamespace(parameterMapName);
113       builder.parameterMap(configuration.getParameterMap(parameterMapName));
114     } else if (parameterClassName != null) {
115       Class parameterClass = resolveClass(parameterClassName);
116       List<ParameterMapping> parameterMappings = new ArrayList<ParameterMapping>();
117       if (sqlSource instanceof SimpleSqlSource) {
118         parameterMappings = sqlSource.getBoundSql(null).getParameterMappings();
119       }
120       ParameterMap.Builder parameterMapBuilder = new ParameterMap.Builder(configuration, id + "-ParameterMap", parameterClass, parameterMappings);
121       builder.parameterMap(parameterMapBuilder.build());
122     }
123 
124     List<ResultMap> resultMaps = new ArrayList<ResultMap>();
125     if (resultMapName != null) {
126       resultMaps.add(configuration.getResultMap(resultMapName));
127       if (additionalResultMapNames != null) {
128         for (String additionalResultMapName : additionalResultMapNames) {
129           resultMaps.add(configuration.getResultMap(additionalResultMapName));
130         }
131       }
132     } else if (resultClassName != null) {
133       Class resultClass = resolveClass(resultClassName);
134       ResultMap.Builder resultMapBuilder = new ResultMap.Builder(configuration, id + "-ResultMap", resultClass, new ArrayList<ResultMapping>());
135       resultMaps.add(resultMapBuilder.build());
136       if (additionalResultClasses != null) {
137         for (Class additionalResultClass : additionalResultClasses) {
138           resultMapBuilder = new ResultMap.Builder(configuration, id + "-ResultMap", additionalResultClass, new ArrayList<ResultMapping>());
139           resultMaps.add(resultMapBuilder.build());
140         }
141       }
142     }
143     builder.resultMaps(resultMaps);
144 
145     builder.fetchSize(fetchSizeInt);
146 
147     builder.timeout(timeoutInt);
148 
149     if (cacheModelName != null) {
150       cacheModelName = mapParser.applyNamespace(cacheModelName);
151       Cache cache = configuration.getCache(cacheModelName);
152       builder.cache(cache);
153     }
154 
155     if (resultSetType != null) {
156       builder.resultSetType(ResultSetType.valueOf(resultSetType));
157     }
158 
159     // allowRemappingBool -- silently ignored
160 
161     findAndParseSelectKey(id, context);
162 
163     configuration.addMappedStatement(builder.build());
164   }

166   private Class resolveClass(String resultClassName) {
167     try {
168       if (resultClassName != null) {
169         return configuration.getTypeAliasRegistry().resolveAlias(resultClassName);
170       } else {
171         return null;
172       }
173     } catch (Exception e) {
174       throw new SqlMapException("Error.  Could not initialize class.  Cause: " + e, e);
175     }
176   }

178   private void findAndParseSelectKey(String parentId, XNode context) {
179     try {
180       boolean runStatementFirst = false;
181       NodeList children = context.getNode().getChildNodes();
182       for (int i = 0; i < children.getLength(); i++) {
183         Node child = children.item(i);
184         if (child.getNodeType() == Node.CDATA_SECTION_NODE
185             || child.getNodeType() == Node.TEXT_NODE) {
186           String data = ((CharacterData) child).getData();
187           if (data.trim().length() > 0) {
188             runStatementFirst = true;
189           }
190         } else if (child.getNodeType() == Node.ELEMENT_NODE
191             && "selectKey".equals(child.getNodeName())) {
192           buildSelectKeyStatement(parentId, context.newXNode(child), runStatementFirst);
193 
194           break;
195         }
196       }
197     } catch (ClassNotFoundException e) {
198       throw new RuntimeException("Error loading result class.  Cause: " + e, e);
199     }
200   }

202   public String getFirstToken(String s) {
203     return new StringTokenizer(s, ", ", false).nextToken();
204   }
205 
206   public String[] getAllButFirstToken(String s) {
207     List strings = new ArrayList();
208     StringTokenizer parser = new StringTokenizer(s, ", ", false);
209     parser.nextToken();
210     while (parser.hasMoreTokens()) {
211       strings.add(parser.nextToken());
212     }
213     return (String[]) strings.toArray(new String[strings.size()]);
214   }

216   private void buildSelectKeyStatement(String parentId, XNode context, boolean runStatementFirstParam) throws ClassNotFoundException {
217     final String keyPropName = context.getStringAttribute("keyProperty");
218     String resultClassName = context.getStringAttribute("resultClass");
219     final SimpleSqlSource source = new SimpleSqlSource(mapParser, context);
220 
221     final boolean runStatementFirst = "post".equalsIgnoreCase(context.getStringAttribute("type", runStatementFirstParam ? "post" : "pre"));
222     final String keyStatementId = SqlMapSessionImpl.selectKeyIdFor(parentId);
223     TypeHandler typeHandler = configuration.getTypeHandlerRegistry().getUnknownTypeHandler();
224     if (resultClassName != null) {
225       final Class resultClass = configuration.getTypeAliasRegistry().resolveAlias(resultClassName);
226       typeHandler = configuration.getTypeHandlerRegistry().getTypeHandler(resultClass);
227     }
228 
229     ResultMapping.Builder mappingBuilder = new ResultMapping.Builder(configuration, keyPropName, keyPropName, typeHandler);
230     ArrayList<ResultMapping> resultMappingArrayList = new ArrayList<ResultMapping>();
231     resultMappingArrayList.add(mappingBuilder.build());
232 
233     ResultMap.Builder resultMapBuilder = new ResultMap.Builder(configuration, keyStatementId + "ResultMap", HashMap.class, resultMappingArrayList);
234     ArrayList<ResultMap> resultMapList = new ArrayList<ResultMap>();
235     resultMapList.add(resultMapBuilder.build());
236 
237     MappedStatement.Builder builder = new MappedStatement.Builder(configuration, keyStatementId, source, SqlCommandType.SELECT);
238     builder.resultMaps(resultMapList);
239 
240     configuration.setPostSelectKey(keyStatementId, runStatementFirst);
241     configuration.addMappedStatement(builder.build());
242   }

198   @SuppressWarnings("unchecked")
199   private <T> TypeHandler<T> getTypeHandler(Type type, JdbcType jdbcType) {
200     Map<JdbcType, TypeHandler<?>> jdbcHandlerMap = getJdbcHandlerMap(type);
201     TypeHandler<?> handler = null;
202     if (jdbcHandlerMap != null) {
203       handler = jdbcHandlerMap.get(jdbcType);
204       if (handler == null) {
205         handler = jdbcHandlerMap.get(null);
206       }
207       if (handler == null) {
208         // #591
209         handler = pickSoleHandler(jdbcHandlerMap);
210       }
211     }
212     // type drives generics here
213     return (TypeHandler<T>) handler;
214   }
215 
216   @SuppressWarnings({ "rawtypes", "unchecked" })
217   private Map<JdbcType, TypeHandler<?>> getJdbcHandlerMap(Type type) {
218     Map<JdbcType, TypeHandler<?>> jdbcHandlerMap = TYPE_HANDLER_MAP.get(type);
219     if (NULL_TYPE_HANDLER_MAP.equals(jdbcHandlerMap)) {
220       return null;
221     }
222     if (jdbcHandlerMap == null && type instanceof Class) {
223       Class<?> clazz = (Class<?>) type;
224       jdbcHandlerMap = getJdbcHandlerMapForSuperclass(clazz);
225       if (jdbcHandlerMap != null) {
226         TYPE_HANDLER_MAP.put(type, jdbcHandlerMap);
227       } else if (clazz.isEnum()) {
228         register(clazz, new EnumTypeHandler(clazz));
229         return TYPE_HANDLER_MAP.get(clazz);
230       }
231     }
232     if (jdbcHandlerMap == null) {
233       TYPE_HANDLER_MAP.put(type, NULL_TYPE_HANDLER_MAP);
234     }
235     return jdbcHandlerMap;
236   }
237 
238   private Map<JdbcType, TypeHandler<?>> getJdbcHandlerMapForSuperclass(Class<?> clazz) {
239     Class<?> superclass =  clazz.getSuperclass();
240     if (superclass == null || Object.class.equals(superclass)) {
241       return null;
242     }
243     Map<JdbcType, TypeHandler<?>> jdbcHandlerMap = TYPE_HANDLER_MAP.get(superclass);
244     if (jdbcHandlerMap != null) {
245       return jdbcHandlerMap;
246     } else {
247       return getJdbcHandlerMapForSuperclass(superclass);
248     }
249   }
250 
251   private TypeHandler<?> pickSoleHandler(Map<JdbcType, TypeHandler<?>> jdbcHandlerMap) {
252     TypeHandler<?> soleHandler = null;
253     for (TypeHandler<?> handler : jdbcHandlerMap.values()) {
254       if (soleHandler == null) {
255         soleHandler = handler;
256       } else if (!handler.getClass().equals(soleHandler.getClass())) {
257         // More than one type handlers registered.
258         return null;
259       }
260     }
261     return soleHandler;
262   }

278   @SuppressWarnings("unchecked")
279   public <T> void register(TypeHandler<T> typeHandler) {
280     boolean mappedTypeFound = false;
281     MappedTypes mappedTypes = typeHandler.getClass().getAnnotation(MappedTypes.class);
282     if (mappedTypes != null) {
283       for (Class<?> handledType : mappedTypes.value()) {
284         register(handledType, typeHandler);
285         mappedTypeFound = true;
286       }
287     }
288     // @since 3.1.0 - try to auto-discover the mapped type
289     if (!mappedTypeFound && typeHandler instanceof TypeReference) {
290       try {
291         TypeReference<T> typeReference = (TypeReference<T>) typeHandler;
292         register(typeReference.getRawType(), typeHandler);
293         mappedTypeFound = true;
294       } catch (Throwable t) {
295         // maybe users define the TypeReference with a different type and are not assignable, so just ignore it
296       }
297     }
298     if (!mappedTypeFound) {
299       register((Class<T>) null, typeHandler);
300     }
301   }

309   private <T> void register(Type javaType, TypeHandler<? extends T> typeHandler) {
310     MappedJdbcTypes mappedJdbcTypes = typeHandler.getClass().getAnnotation(MappedJdbcTypes.class);
311     if (mappedJdbcTypes != null) {
312       for (JdbcType handledJdbcType : mappedJdbcTypes.value()) {
313         register(javaType, handledJdbcType, typeHandler);
314       }
315       if (mappedJdbcTypes.includeNullJdbcType()) {
316         register(javaType, null, typeHandler);
317       }
318     } else {
319       register(javaType, null, typeHandler);
320     }
321   }

333   private void register(Type javaType, JdbcType jdbcType, TypeHandler<?> handler) {
334     if (javaType != null) {
335       Map<JdbcType, TypeHandler<?>> map = TYPE_HANDLER_MAP.get(javaType);
336       if (map == null) {
337         map = new HashMap<JdbcType, TypeHandler<?>>();
338         TYPE_HANDLER_MAP.put(javaType, map);
339       }
340       map.put(jdbcType, handler);
341     }
342     ALL_TYPE_HANDLERS_MAP.put(handler.getClass(), handler);
343   }

351   public void register(Class<?> typeHandlerClass) {
352     boolean mappedTypeFound = false;
353     MappedTypes mappedTypes = typeHandlerClass.getAnnotation(MappedTypes.class);
354     if (mappedTypes != null) {
355       for (Class<?> javaTypeClass : mappedTypes.value()) {
356         register(javaTypeClass, typeHandlerClass);
357         mappedTypeFound = true;
358       }
359     }
360     if (!mappedTypeFound) {
361       register(getInstance(null, typeHandlerClass));
362     }
363   }

383   @SuppressWarnings("unchecked")
384   public <T> TypeHandler<T> getInstance(Class<?> javaTypeClass, Class<?> typeHandlerClass) {
385     if (javaTypeClass != null) {
386       try {
387         Constructor<?> c = typeHandlerClass.getConstructor(Class.class);
388         return (TypeHandler<T>) c.newInstance(javaTypeClass);
389       } catch (NoSuchMethodException ignored) {
390         // ignored
391       } catch (Exception e) {
392         throw new TypeException("Failed invoking constructor for handler " + typeHandlerClass, e);
393       }
394     }
395     try {
396       Constructor<?> c = typeHandlerClass.getConstructor();
397       return (TypeHandler<T>) c.newInstance();
398     } catch (Exception e) {
399       throw new TypeException("Unable to find a usable constructor for " + typeHandlerClass, e);
400     }
401   }

405   public void register(String packageName) {
406     ResolverUtil<Class<?>> resolverUtil = new ResolverUtil<Class<?>>();
407     resolverUtil.find(new ResolverUtil.IsA(TypeHandler.class), packageName);
408     Set<Class<? extends Class<?>>> handlerSet = resolverUtil.getClasses();
409     for (Class<?> type : handlerSet) {
410       //Ignore inner classes and interfaces (including package-info.java) and abstract classes
411       if (!type.isAnonymousClass() && !type.isInterface() && !Modifier.isAbstract(type.getModifiers())) {
412         register(type);
413       }
414     }
415   }

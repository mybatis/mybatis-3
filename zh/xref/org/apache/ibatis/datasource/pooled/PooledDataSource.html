270   /*
271    * Closes all active and idle connections in the pool
272    */
273   public void forceCloseAll() {
274     synchronized (state) {
275       expectedConnectionTypeCode = assembleConnectionTypeCode(dataSource.getUrl(), dataSource.getUsername(), dataSource.getPassword());
276       for (int i = state.activeConnections.size(); i > 0; i--) {
277         try {
278           PooledConnection conn = state.activeConnections.remove(i - 1);
279           conn.invalidate();
280 
281           Connection realConn = conn.getRealConnection();
282           if (!realConn.getAutoCommit()) {
283             realConn.rollback();
284           }
285           realConn.close();
286         } catch (Exception e) {
287           // ignore
288         }
289       }
290       for (int i = state.idleConnections.size(); i > 0; i--) {
291         try {
292           PooledConnection conn = state.idleConnections.remove(i - 1);
293           conn.invalidate();
294 
295           Connection realConn = conn.getRealConnection();
296           if (!realConn.getAutoCommit()) {
297             realConn.rollback();
298           }
299           realConn.close();
300         } catch (Exception e) {
301           // ignore
302         }
303       }
304     }
305     if (log.isDebugEnabled()) {
306       log.debug("PooledDataSource forcefully closed/removed all connections.");
307     }
308   }

314   private int assembleConnectionTypeCode(String url, String username, String password) {
315     return ("" + url + username + password).hashCode();

318   protected void pushConnection(PooledConnection conn) throws SQLException {
319 
320     synchronized (state) {
321       state.activeConnections.remove(conn);
322       if (conn.isValid()) {
323         if (state.idleConnections.size() < poolMaximumIdleConnections && conn.getConnectionTypeCode() == expectedConnectionTypeCode) {
324           state.accumulatedCheckoutTime += conn.getCheckoutTime();
325           if (!conn.getRealConnection().getAutoCommit()) {
326             conn.getRealConnection().rollback();
327           }
328           PooledConnection newConn = new PooledConnection(conn.getRealConnection(), this);
329           state.idleConnections.add(newConn);
330           newConn.setCreatedTimestamp(conn.getCreatedTimestamp());
331           newConn.setLastUsedTimestamp(conn.getLastUsedTimestamp());
332           conn.invalidate();
333           if (log.isDebugEnabled()) {
334             log.debug("Returned connection " + newConn.getRealHashCode() + " to pool.");
335           }
336           state.notifyAll();
337         } else {
338           state.accumulatedCheckoutTime += conn.getCheckoutTime();
339           if (!conn.getRealConnection().getAutoCommit()) {
340             conn.getRealConnection().rollback();
341           }
342           conn.getRealConnection().close();
343           if (log.isDebugEnabled()) {
344             log.debug("Closed connection " + conn.getRealHashCode() + ".");
345           }
346           conn.invalidate();
347         }
348       } else {
349         if (log.isDebugEnabled()) {
350           log.debug("A bad connection (" + conn.getRealHashCode() + ") attempted to return to the pool, discarding connection.");
351         }
352         state.badConnectionCount++;
353       }
354     }
355   }

357   private PooledConnection popConnection(String username, String password) throws SQLException {
358     boolean countedWait = false;
359     PooledConnection conn = null;
360     long t = System.currentTimeMillis();
361     int localBadConnectionCount = 0;
362 
363     while (conn == null) {
364       synchronized (state) {
365         if (state.idleConnections.size() > 0) {
366           // Pool has available connection
367           conn = state.idleConnections.remove(0);
368           if (log.isDebugEnabled()) {
369             log.debug("Checked out connection " + conn.getRealHashCode() + " from pool.");
370           }
371         } else {
372           // Pool does not have available connection
373           if (state.activeConnections.size() < poolMaximumActiveConnections) {
374             // Can create new connection
375             conn = new PooledConnection(dataSource.getConnection(), this);
376             @SuppressWarnings("unused")
377             //used in logging, if enabled
378             Connection realConn = conn.getRealConnection();
379             if (log.isDebugEnabled()) {
380               log.debug("Created connection " + conn.getRealHashCode() + ".");
381             }
382           } else {
383             // Cannot create new connection
384             PooledConnection oldestActiveConnection = state.activeConnections.get(0);
385             long longestCheckoutTime = oldestActiveConnection.getCheckoutTime();
386             if (longestCheckoutTime > poolMaximumCheckoutTime) {
387               // Can claim overdue connection
388               state.claimedOverdueConnectionCount++;
389               state.accumulatedCheckoutTimeOfOverdueConnections += longestCheckoutTime;
390               state.accumulatedCheckoutTime += longestCheckoutTime;
391               state.activeConnections.remove(oldestActiveConnection);
392               if (!oldestActiveConnection.getRealConnection().getAutoCommit()) {
393                 oldestActiveConnection.getRealConnection().rollback();
394               }
395               conn = new PooledConnection(oldestActiveConnection.getRealConnection(), this);
396               oldestActiveConnection.invalidate();
397               if (log.isDebugEnabled()) {
398                 log.debug("Claimed overdue connection " + conn.getRealHashCode() + ".");
399               }
400             } else {
401               // Must wait
402               try {
403                 if (!countedWait) {
404                   state.hadToWaitCount++;
405                   countedWait = true;
406                 }
407                 if (log.isDebugEnabled()) {
408                   log.debug("Waiting as long as " + poolTimeToWait + " milliseconds for connection.");
409                 }
410                 long wt = System.currentTimeMillis();
411                 state.wait(poolTimeToWait);
412                 state.accumulatedWaitTime += System.currentTimeMillis() - wt;
413               } catch (InterruptedException e) {
414                 break;
415               }
416             }
417           }
418         }
419         if (conn != null) {
420           if (conn.isValid()) {
421             if (!conn.getRealConnection().getAutoCommit()) {
422               conn.getRealConnection().rollback();
423             }
424             conn.setConnectionTypeCode(assembleConnectionTypeCode(dataSource.getUrl(), username, password));
425             conn.setCheckoutTimestamp(System.currentTimeMillis());
426             conn.setLastUsedTimestamp(System.currentTimeMillis());
427             state.activeConnections.add(conn);
428             state.requestCount++;
429             state.accumulatedRequestTime += System.currentTimeMillis() - t;
430           } else {
431             if (log.isDebugEnabled()) {
432               log.debug("A bad connection (" + conn.getRealHashCode() + ") was returned from the pool, getting another connection.");
433             }
434             state.badConnectionCount++;
435             localBadConnectionCount++;
436             conn = null;
437             if (localBadConnectionCount > (poolMaximumIdleConnections + 3)) {
438               if (log.isDebugEnabled()) {
439                 log.debug("PooledDataSource: Could not get a good connection to the database.");
440               }
441               throw new SQLException("PooledDataSource: Could not get a good connection to the database.");
442             }
443           }
444         }
445       }
446 
447     }
448 
449     if (conn == null) {
450       if (log.isDebugEnabled()) {
451         log.debug("PooledDataSource: Unknown severe error condition.  The connection pool returned a null connection.");
452       }
453       throw new SQLException("PooledDataSource: Unknown severe error condition.  The connection pool returned a null connection.");
454     }
455 
456     return conn;
457   }

459   /*
460    * Method to check to see if a connection is still usable
461    *
462    * @param conn - the connection to check
463    * @return True if the connection is still usable
464    */
465   protected boolean pingConnection(PooledConnection conn) {
466     boolean result = true;
467 
468     try {
469       result = !conn.getRealConnection().isClosed();
470     } catch (SQLException e) {
471       if (log.isDebugEnabled()) {
472         log.debug("Connection " + conn.getRealHashCode() + " is BAD: " + e.getMessage());
473       }
474       result = false;
475     }
476 
477     if (result) {
478       if (poolPingEnabled) {
479         if (poolPingConnectionsNotUsedFor >= 0 && conn.getTimeElapsedSinceLastUse() > poolPingConnectionsNotUsedFor) {
480           try {
481             if (log.isDebugEnabled()) {
482               log.debug("Testing connection " + conn.getRealHashCode() + " ...");
483             }
484             Connection realConn = conn.getRealConnection();
485             Statement statement = realConn.createStatement();
486             ResultSet rs = statement.executeQuery(poolPingQuery);
487             rs.close();
488             statement.close();
489             if (!realConn.getAutoCommit()) {
490               realConn.rollback();
491             }
492             result = true;
493             if (log.isDebugEnabled()) {
494               log.debug("Connection " + conn.getRealHashCode() + " is GOOD!");
495             }
496           } catch (Exception e) {
497             log.warn("Execution of ping query '" + poolPingQuery + "' failed: " + e.getMessage());
498             try {
499               conn.getRealConnection().close();
500             } catch (Exception e2) {
501               //ignore
502             }
503             result = false;
504             if (log.isDebugEnabled()) {
505               log.debug("Connection " + conn.getRealHashCode() + " is BAD: " + e.getMessage());
506             }
507           }
508         }
509       }
510     }
511     return result;
512   }

514   /*
515    * Unwraps a pooled connection to get to the 'real' connection
516    *
517    * @param conn - the pooled connection to unwrap
518    * @return The 'real' connection
519    */
520   public static Connection unwrapConnection(Connection conn) {
521     if (Proxy.isProxyClass(conn.getClass())) {
522       InvocationHandler handler = Proxy.getInvocationHandler(conn);
523       if (handler instanceof PooledConnection) {
524         return ((PooledConnection) handler).getRealConnection();
525       }
526     }
527     return conn;
528   }

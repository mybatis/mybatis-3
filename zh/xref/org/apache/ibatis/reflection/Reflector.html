51    private Class<?> type;
52    private String[] readablePropertyNames = EMPTY_STRING_ARRAY;
53    private String[] writeablePropertyNames = EMPTY_STRING_ARRAY;
54    private Map<String, Invoker> setMethods = new HashMap<String, Invoker>();
55    private Map<String, Invoker> getMethods = new HashMap<String, Invoker>();
56    private Map<String, Class<?>> setTypes = new HashMap<String, Class<?>>();
57    private Map<String, Class<?>> getTypes = new HashMap<String, Class<?>>();
58    private Constructor<?> defaultConstructor;

62    public Reflector(Class<?> clazz) {
63      type = clazz;
64      addDefaultConstructor(clazz);
65      addGetMethods(clazz);
66      addSetMethods(clazz);
67      addFields(clazz);
68      readablePropertyNames = getMethods.keySet().toArray(new String[getMethods.keySet().size()]);
69      writeablePropertyNames = setMethods.keySet().toArray(new String[setMethods.keySet().size()]);
70      for (String propName : readablePropertyNames) {
71        caseInsensitivePropertyMap.put(propName.toUpperCase(Locale.ENGLISH), propName);
72      }
73      for (String propName : writeablePropertyNames) {
74        caseInsensitivePropertyMap.put(propName.toUpperCase(Locale.ENGLISH), propName);
75      }
76    }

78    private void addDefaultConstructor(Class<?> clazz) {
79      Constructor<?>[] consts = clazz.getDeclaredConstructors();
80      for (Constructor<?> constructor : consts) {
81        if (constructor.getParameterTypes().length == 0) {
82          if (canAccessPrivateMethods()) {
83            try {
84              constructor.setAccessible(true);
85            } catch (Exception e) {
86              // Ignored. This is only a final precaution, nothing we can do.
87            }
88          }
89          if (constructor.isAccessible()) {
90            this.defaultConstructor = constructor;
91          }
92        }
93      }
94    }

96    private void addGetMethods(Class<?> cls) {
97      Map<String, List<Method>> conflictingGetters = new HashMap<String, List<Method>>();
98      Method[] methods = getClassMethods(cls);
99      for (Method method : methods) {
100       String name = method.getName();
101       if (name.startsWith("get") && name.length() > 3) {
102         if (method.getParameterTypes().length == 0) {
103           name = PropertyNamer.methodToProperty(name);
104           addMethodConflict(conflictingGetters, name, method);
105         }
106       } else if (name.startsWith("is") && name.length() > 2) {
107         if (method.getParameterTypes().length == 0) {
108           name = PropertyNamer.methodToProperty(name);
109           addMethodConflict(conflictingGetters, name, method);
110         }
111       }
112     }
113     resolveGetterConflicts(conflictingGetters);
114   }

116   private void resolveGetterConflicts(Map<String, List<Method>> conflictingGetters) {
117     for (String propName : conflictingGetters.keySet()) {
118       List<Method> getters = conflictingGetters.get(propName);
119       Iterator<Method> iterator = getters.iterator();
120       Method firstMethod = iterator.next();
121       if (getters.size() == 1) {
122         addGetMethod(propName, firstMethod);
123       } else {
124         Method getter = firstMethod;
125         Class<?> getterType = firstMethod.getReturnType();
126         while (iterator.hasNext()) {
127           Method method = iterator.next();
128           Class<?> methodType = method.getReturnType();
129           if (methodType.equals(getterType)) {
130             throw new ReflectionException("Illegal overloaded getter method with ambiguous type for property "
131                 + propName + " in class " + firstMethod.getDeclaringClass()
132                 + ".  This breaks the JavaBeans " + "specification and can cause unpredictable results.");
133           } else if (methodType.isAssignableFrom(getterType)) {
134             // OK getter type is descendant
135           } else if (getterType.isAssignableFrom(methodType)) {
136             getter = method;
137             getterType = methodType;
138           } else {
139             throw new ReflectionException("Illegal overloaded getter method with ambiguous type for property "
140                 + propName + " in class " + firstMethod.getDeclaringClass()
141                 + ".  This breaks the JavaBeans " + "specification and can cause unpredictable results.");
142           }
143         }
144         addGetMethod(propName, getter);
145       }
146     }
147   }

149   private void addGetMethod(String name, Method method) {
150     if (isValidPropertyName(name)) {
151       getMethods.put(name, new MethodInvoker(method));
152       Type returnType = TypeParameterResolver.resolveReturnType(method, type);
153       getTypes.put(name, typeToClass(returnType));
154     }
155   }

157   private void addSetMethods(Class<?> cls) {
158     Map<String, List<Method>> conflictingSetters = new HashMap<String, List<Method>>();
159     Method[] methods = getClassMethods(cls);
160     for (Method method : methods) {
161       String name = method.getName();
162       if (name.startsWith("set") && name.length() > 3) {
163         if (method.getParameterTypes().length == 1) {
164           name = PropertyNamer.methodToProperty(name);
165           addMethodConflict(conflictingSetters, name, method);
166         }
167       }
168     }
169     resolveSetterConflicts(conflictingSetters);
170   }
171 
172   private void addMethodConflict(Map<String, List<Method>> conflictingMethods, String name, Method method) {
173     List<Method> list = conflictingMethods.get(name);
174     if (list == null) {
175       list = new ArrayList<Method>();
176       conflictingMethods.put(name, list);
177     }
178     list.add(method);
179   }

181   private void resolveSetterConflicts(Map<String, List<Method>> conflictingSetters) {
182     for (String propName : conflictingSetters.keySet()) {
183       List<Method> setters = conflictingSetters.get(propName);
184       Class<?> getterType = getTypes.get(propName);
185       Method match = null;
186       ReflectionException exception = null;
187       for (Method setter : setters) {
188         Class<?> paramType = setter.getParameterTypes()[0];
189         if (paramType.equals(getterType)) {
190           // should be the best match
191           match = setter;
192           break;
193         }
194         if (exception == null) {
195           try {
196             match = pickBetterSetter(match, setter, propName);
197           } catch (ReflectionException e) {
198             // there could still be the 'best match'
199             match = null;
200             exception = e;
201           }
202         }
203       }
204       if (match == null) {
205         throw exception;
206       } else {
207         addSetMethod(propName, match);
208       }
209     }
210   }
211 
212   private Method pickBetterSetter(Method setter1, Method setter2, String property) {
213     if (setter1 == null) {
214       return setter2;
215     }
216     Class<?> paramType1 = setter1.getParameterTypes()[0];
217     Class<?> paramType2 = setter2.getParameterTypes()[0];
218     if (paramType1.isAssignableFrom(paramType2)) {
219       return setter2;
220     } else if (paramType2.isAssignableFrom(paramType1)) {
221       return setter1;
222     }
223     throw new ReflectionException("Ambiguous setters defined for property '" + property + "' in class '"
224         + setter2.getDeclaringClass() + "' with types '" + paramType1.getName() + "' and '"
225         + paramType2.getName() + "'.");
226   }

228   private void addSetMethod(String name, Method method) {
229     if (isValidPropertyName(name)) {
230       setMethods.put(name, new MethodInvoker(method));
231       Type[] paramTypes = TypeParameterResolver.resolveParamTypes(method, type);
232       setTypes.put(name, typeToClass(paramTypes[0]));
233     }
234   }

236   private Class<?> typeToClass(Type src) {
237     Class<?> result = null;
238     if (src instanceof Class) {
239       result = (Class<?>) src;
240     } else if (src instanceof ParameterizedType) {
241       result = (Class<?>) ((ParameterizedType) src).getRawType();
242     } else if (src instanceof GenericArrayType) {
243       Type componentType = ((GenericArrayType) src).getGenericComponentType();
244       if (componentType instanceof Class) {
245         result = Array.newInstance((Class<?>) componentType, 0).getClass();
246       } else {
247         Class<?> componentClass = typeToClass(componentType);
248         result = Array.newInstance((Class<?>) componentClass, 0).getClass();
249       }
250     }
251     if (result == null) {
252       result = Object.class;
253     }
254     return result;
255   }

257   private void addFields(Class<?> clazz) {
258     Field[] fields = clazz.getDeclaredFields();
259     for (Field field : fields) {
260       if (canAccessPrivateMethods()) {
261         try {
262           field.setAccessible(true);
263         } catch (Exception e) {
264           // Ignored. This is only a final precaution, nothing we can do.
265         }
266       }
267       if (field.isAccessible()) {
268         if (!setMethods.containsKey(field.getName())) {
269           // issue #379 - removed the check for final because JDK 1.5 allows
270           // modification of final fields through reflection (JSR-133). (JGB)
271           // pr #16 - final static can only be set by the classloader
272           int modifiers = field.getModifiers();
273           if (!(Modifier.isFinal(modifiers) && Modifier.isStatic(modifiers))) {
274             addSetField(field);
275           }
276         }
277         if (!getMethods.containsKey(field.getName())) {
278           addGetField(field);
279         }
280       }
281     }
282     if (clazz.getSuperclass() != null) {
283       addFields(clazz.getSuperclass());
284     }
285   }
286 
287   private void addSetField(Field field) {
288     if (isValidPropertyName(field.getName())) {
289       setMethods.put(field.getName(), new SetFieldInvoker(field));
290       Type fieldType = TypeParameterResolver.resolveFieldType(field, type);
291       setTypes.put(field.getName(), typeToClass(fieldType));
292     }
293   }
294 
295   private void addGetField(Field field) {
296     if (isValidPropertyName(field.getName())) {
297       getMethods.put(field.getName(), new GetFieldInvoker(field));
298       Type fieldType = TypeParameterResolver.resolveFieldType(field, type);
299       getTypes.put(field.getName(), typeToClass(fieldType));
300     }
301   }

303   private boolean isValidPropertyName(String name) {
304     return !(name.startsWith("$") || "serialVersionUID".equals(name) || "class".equals(name));
305   }

307   /*
308    * This method returns an array containing all methods
309    * declared in this class and any superclass.
310    * We use this method, instead of the simpler Class.getMethods(),
311    * because we want to look for private methods as well.
312    *
313    * @param cls The class
314    * @return An array containing all methods in this class
315    */
316   private Method[] getClassMethods(Class<?> cls) {
317     Map<String, Method> uniqueMethods = new HashMap<String, Method>();
318     Class<?> currentClass = cls;
319     while (currentClass != null) {
320       addUniqueMethods(uniqueMethods, currentClass.getDeclaredMethods());
321 
322       // we also need to look for interface methods -
323       // because the class may be abstract
324       Class<?>[] interfaces = currentClass.getInterfaces();
325       for (Class<?> anInterface : interfaces) {
326         addUniqueMethods(uniqueMethods, anInterface.getMethods());
327       }
328 
329       currentClass = currentClass.getSuperclass();
330     }
331 
332     Collection<Method> methods = uniqueMethods.values();
333 
334     return methods.toArray(new Method[methods.size()]);
335   }
336 
337   private void addUniqueMethods(Map<String, Method> uniqueMethods, Method[] methods) {
338     for (Method currentMethod : methods) {
339       if (!currentMethod.isBridge()) {
340         String signature = getSignature(currentMethod);
341         // check to see if the method is already known
342         // if it is known, then an extended class must have
343         // overridden a method
344         if (!uniqueMethods.containsKey(signature)) {
345           if (canAccessPrivateMethods()) {
346             try {
347               currentMethod.setAccessible(true);
348             } catch (Exception e) {
349               // Ignored. This is only a final precaution, nothing we can do.
350             }
351           }
352 
353           uniqueMethods.put(signature, currentMethod);
354         }
355       }
356     }
357   }

359   private String getSignature(Method method) {
360     StringBuilder sb = new StringBuilder();
361     Class<?> returnType = method.getReturnType();
362     if (returnType != null) {
363       sb.append(returnType.getName()).append('#');
364     }
365     sb.append(method.getName());
366     Class<?>[] parameters = method.getParameterTypes();
367     for (int i = 0; i < parameters.length; i++) {
368       if (i == 0) {
369         sb.append(':');
370       } else {
371         sb.append(',');
372       }
373       sb.append(parameters[i].getName());
374     }
375     return sb.toString();
376   }

378   private static boolean canAccessPrivateMethods() {
379     try {
380       SecurityManager securityManager = System.getSecurityManager();
381       if (null != securityManager) {
382         securityManager.checkPermission(new ReflectPermission("suppressAccessChecks"));
383       }
384     } catch (SecurityException e) {
385       return false;
386     }
387     return true;
388   }

411   public Invoker getSetInvoker(String propertyName) {
412     Invoker method = setMethods.get(propertyName);
413     if (method == null) {
414       throw new ReflectionException("There is no setter for property named '" + propertyName + "' in '" + type + "'");
415     }
416     return method;
417   }

421     if (method == null) {
422       throw new ReflectionException("There is no getter for property named '" + propertyName + "' in '" + type + "'");
423     }
424     return method;
425   }

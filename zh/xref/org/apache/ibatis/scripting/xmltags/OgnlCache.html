16  package org.apache.ibatis.scripting.xmltags;
17  
18  import java.util.Map;
19  import java.util.concurrent.ConcurrentHashMap;
20  
21  import ognl.Ognl;
22  import ognl.OgnlException;
23  
24  import org.apache.ibatis.builder.BuilderException;
25  
26  /**
27   * Caches OGNL parsed expressions.
28   *
29   * @author Eduardo Macarron
30   *
31   * @see <a href='http://code.google.com/p/mybatis/issues/detail?id=342'>Issue 342</a>
32   */
33  public final class OgnlCache {
34  
35    private static final Map<String, Object> expressionCache = new ConcurrentHashMap<String, Object>();
36  
37    private OgnlCache() {
38      // Prevent Instantiation of Static Class
39    }
40  
41    public static Object getValue(String expression, Object root) {
42      try {
43        Map<Object, OgnlClassResolver> context = Ognl.createDefaultContext(root, new OgnlClassResolver());
44        return Ognl.getValue(parseExpression(expression), context, root);
45      } catch (OgnlException e) {
46        throw new BuilderException("Error evaluating expression '" + expression + "'. Cause: " + e, e);
47      }
48    }
49  
50    private static Object parseExpression(String expression) throws OgnlException {
51      Object node = expressionCache.get(expression);
52      if (node == null) {
53        node = Ognl.parseExpression(expression);
54        expressionCache.put(expression, node);
55      }
56      return node;
57    }
58  
59  }

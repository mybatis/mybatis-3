16  package org.apache.ibatis.scripting.xmltags;
17  
18  import java.util.ArrayList;
19  import java.util.Collections;
20  import java.util.List;
21  import java.util.Locale;
22  import java.util.Map;
23  import java.util.StringTokenizer;
24  
25  import org.apache.ibatis.session.Configuration;
26  
27  public class TrimSqlNode implements SqlNode {
28  
29    private SqlNode contents;
30    private String prefix;
31    private String suffix;
32    private List<String> prefixesToOverride = new ArrayList<String>();
33    private List<String> suffixesToOverride = new ArrayList<String>();
34    private Configuration configuration;
35  
36    public TrimSqlNode(Configuration configuration, SqlNode contents, String prefix, String prefixesToOverride, String suffix, String suffixesToOverride) {
37      this.contents = contents;
38      this.prefix = prefix;
39      this.prefixesToOverride = parseOverrides(prefixesToOverride);
40      this.suffix = suffix;
41      this.suffixesToOverride = parseOverrides(suffixesToOverride);
42      this.configuration = configuration;
43    }
44  
45    public boolean apply(DynamicContext context) {
46      FilteredDynamicContext filteredDynamicContext = new FilteredDynamicContext(context);
47      boolean result = contents.apply(filteredDynamicContext);
48      filteredDynamicContext.applyAll();
49      return result;
50    }
51  
52    private List<String> parseOverrides(String overrides) {
53      if (overrides != null) {
54        final StringTokenizer parser = new StringTokenizer(overrides, "|", false);
55        return new ArrayList<String>() {
56          private static final long serialVersionUID = -2504816393625384165L;
57  
58          {
59            while (parser.hasMoreTokens()) {
60              add(parser.nextToken().toUpperCase(Locale.ENGLISH));
61            }
62          }
63        };
64      }
65      return Collections.emptyList();
66    }
67  
68    private class FilteredDynamicContext extends DynamicContext {
69      private DynamicContext delegate;
70      private boolean prefixApplied;
71      private boolean suffixApplied;
72      private StringBuilder sqlBuffer;
73  
74      public FilteredDynamicContext(DynamicContext delegate) {
75        super(configuration, null);
76        this.delegate = delegate;
77        this.prefixApplied = false;
78        this.suffixApplied = false;
79        this.sqlBuffer = new StringBuilder();
80      }
81  
82      public void applyAll() {
83        sqlBuffer = new StringBuilder(sqlBuffer.toString().trim());
84        String trimmedUppercaseSql = sqlBuffer.toString().toUpperCase(Locale.ENGLISH);
85        if (trimmedUppercaseSql.length() > 0) {
86          applyPrefix(sqlBuffer, trimmedUppercaseSql);
87          applySuffix(sqlBuffer, trimmedUppercaseSql);
88        }
89        delegate.appendSql(sqlBuffer.toString());
90      }
91  
92      @Override
93      public Map<String, Object> getBindings() {
94        return delegate.getBindings();
95      }
96  
97      @Override
98      public void bind(String name, Object value) {
99        delegate.bind(name, value);
100     }
101 
102     @Override
103     public int getUniqueNumber() {
104       return delegate.getUniqueNumber();
105     }
106 
107     @Override
108     public void appendSql(String sql) {
109       sqlBuffer.append(sql);
110     }
111 
112     @Override
113     public String getSql() {
114       return delegate.getSql();
115     }
116 
117     private void applyPrefix(StringBuilder sql, String trimmedUppercaseSql) {
118       if (!prefixApplied) {
119         prefixApplied = true;
120         for (String toRemove : prefixesToOverride) {
121           if (trimmedUppercaseSql.startsWith(toRemove)) {
122             sql.delete(0, toRemove.trim().length());
123             break;
124           }
125         }
126         if (prefix != null) {
127           sql.insert(0, " ");
128           sql.insert(0, prefix);
129         }
130       }
131     }
132 
133     private void applySuffix(StringBuilder sql, String trimmedUppercaseSql) {
134       if (!suffixApplied) {
135         suffixApplied = true;
136         for (String toRemove : suffixesToOverride) {
137           if (trimmedUppercaseSql.endsWith(toRemove) || trimmedUppercaseSql.endsWith(toRemove.trim())) {
138             int start = sql.length() - toRemove.trim().length();
139             int end = sql.length();
140             sql.delete(start, end);
141             break;
142           }
143         }
144         if (suffix != null) {
145           sql.append(" ");
146           sql.append(suffix);
147         }
148       }
149     }
150 
151   }

16  package org.apache.ibatis.logging.jdbc;
17  
18  import java.lang.reflect.InvocationHandler;
19  import java.lang.reflect.Method;
20  import java.lang.reflect.Proxy;
21  import java.sql.ResultSet;
22  import java.sql.Statement;
23  
24  import org.apache.ibatis.logging.Log;
25  import org.apache.ibatis.reflection.ExceptionUtil;
26  
27  /*
28   * Statement proxy to add logging
29   */
30  public final class StatementLogger extends BaseJdbcLogger implements InvocationHandler {
31  
32    private Statement statement;
33  
34    private StatementLogger(Statement stmt, Log statementLog) {
35      super(statementLog);
36      this.statement = stmt;
37    }
38  
39    public Object invoke(Object proxy, Method method, Object[] params) throws Throwable {
40      try {
41        if (EXECUTE_METHODS.contains(method.getName())) {
42          if (isDebugEnabled()) {
43            debug("==>  Executing: " + removeBreakingWhitespace((String) params[0]));
44          }
45          if ("executeQuery".equals(method.getName())) {
46            ResultSet rs = (ResultSet) method.invoke(statement, params);
47            if (rs != null) {
48              return ResultSetLogger.newInstance(rs, getStatementLog());
49            } else {
50              return null;
51            }
52          } else {
53            return method.invoke(statement, params);
54          }
55        } else if ("getResultSet".equals(method.getName())) {
56          ResultSet rs = (ResultSet) method.invoke(statement, params);
57          if (rs != null) {
58            return ResultSetLogger.newInstance(rs, getStatementLog());
59          } else {
60            return null;
61          }
62        } else if ("equals".equals(method.getName())) {
63          Object ps = params[0];
64          return ps instanceof Proxy && proxy == ps;
65        } else if ("hashCode".equals(method.getName())) {
66          return proxy.hashCode();
67        } else {
68          return method.invoke(statement, params);
69        }
70      } catch (Throwable t) {
71        throw ExceptionUtil.unwrapThrowable(t);
72      }
73    }
74  
75    /*
76     * Creates a logging version of a Statement
77     *
78     * @param stmt - the statement
79     * @return - the proxy
80     */
81    public static Statement newInstance(Statement stmt, Log statementLog) {
82      InvocationHandler handler = new StatementLogger(stmt, statementLog);
83      ClassLoader cl = Statement.class.getClassLoader();
84      return (Statement) Proxy.newProxyInstance(cl, new Class[]{Statement.class}, handler);
85    }
86  
87    /*
88     * return the wrapped statement
89     *
90     * @return the statement
91     */
92    public Statement getStatement() {
93      return statement;
94    }
95  
96  }

16  package org.apache.ibatis.logging.jdbc;
17  
18  import java.sql.Array;
19  import java.sql.SQLException;
20  import java.util.ArrayList;
21  import java.util.Arrays;
22  import java.util.HashMap;
23  import java.util.HashSet;
24  import java.util.List;
25  import java.util.Map;
26  import java.util.Set;
27  import java.util.StringTokenizer;
28  
29  import org.apache.ibatis.logging.Log;
30  
31  /**
32   * Base class for proxies to do logging
33   * 
34   * @author Clinton Begin
35   * @author Eduardo Macarron
36   */
37  public abstract class BaseJdbcLogger {
38  
39    protected static final Set<String> SET_METHODS = new HashSet<String>();
40    protected static final Set<String> EXECUTE_METHODS = new HashSet<String>();
41  
42    private Map<Object, Object> columnMap = new HashMap<Object, Object>();
43  
44    private List<Object> columnNames = new ArrayList<Object>();
45    private List<Object> columnValues = new ArrayList<Object>();
46  
47    protected Log statementLog;
48    protected int queryStack;
49  
50    /*
51     * Default constructor
52     */
53    public BaseJdbcLogger(Log log, int queryStack) {
54      this.statementLog = log;
55      if (queryStack == 0) {
56        this.queryStack = 1;
57      } else {
58        this.queryStack = queryStack;
59      }
60    }
61  
62    static {
63      SET_METHODS.add("setString");
64      SET_METHODS.add("setNString");
65      SET_METHODS.add("setInt");
66      SET_METHODS.add("setByte");
67      SET_METHODS.add("setShort");
68      SET_METHODS.add("setLong");
69      SET_METHODS.add("setDouble");
70      SET_METHODS.add("setFloat");
71      SET_METHODS.add("setTimestamp");
72      SET_METHODS.add("setDate");
73      SET_METHODS.add("setTime");
74      SET_METHODS.add("setArray");
75      SET_METHODS.add("setBigDecimal");
76      SET_METHODS.add("setAsciiStream");
77      SET_METHODS.add("setBinaryStream");
78      SET_METHODS.add("setBlob");
79      SET_METHODS.add("setBoolean");
80      SET_METHODS.add("setBytes");
81      SET_METHODS.add("setCharacterStream");
82      SET_METHODS.add("setNCharacterStream");
83      SET_METHODS.add("setClob");
84      SET_METHODS.add("setNClob");
85      SET_METHODS.add("setObject");
86      SET_METHODS.add("setNull");
87  
88      EXECUTE_METHODS.add("execute");
89      EXECUTE_METHODS.add("executeUpdate");
90      EXECUTE_METHODS.add("executeQuery");
91      EXECUTE_METHODS.add("addBatch");
92    }
93  
94    protected void setColumn(Object key, Object value) {
95      columnMap.put(key, value);
96      columnNames.add(key);
97      columnValues.add(value);
98    }
99  
100   protected Object getColumn(Object key) {
101     return columnMap.get(key);
102   }
103 
104   protected String getParameterValueString() {
105     List<Object> typeList = new ArrayList<Object>(columnValues.size());
106     for (Object value : columnValues) {
107       if (value == null) {
108         typeList.add("null");
109       } else {
110         typeList.add(objectValueString(value) + "(" + value.getClass().getSimpleName() + ")");
111       }
112     }
113     final String parameters = typeList.toString();
114     return parameters.substring(1, parameters.length() - 1);
115   }
116 
117   protected String objectValueString(Object value) {
118     if (value instanceof Array) {
119       return arrayValueString((Array) value);
120     }
121     return value.toString();
122   }
123 
124   protected String arrayValueString(Array array) {
125     try {
126       Object value = array.getArray();
127       if (value instanceof Object[]) {
128         return Arrays.toString((Object[]) value);
129       } else if (value instanceof long[]) {
130         return Arrays.toString((long[]) value);
131       } else if (value instanceof int[]) {
132         return Arrays.toString((int[]) value);
133       } else if (value instanceof short[]) {
134         return Arrays.toString((short[]) value);
135       } else if (value instanceof char[]) {
136         return Arrays.toString((char[]) value);
137       } else if (value instanceof byte[]) {
138         return Arrays.toString((byte[]) value);
139       } else if (value instanceof boolean[]) {
140         return Arrays.toString((boolean[]) value);
141       } else if (value instanceof float[]) {
142         return Arrays.toString((float[]) value);
143       } else {
144         return Arrays.toString((double[]) value);
145       }
146     } catch (SQLException e) {
147       return array.toString();
148     }
149   }
150 
151   protected String getColumnString() {
152     return columnNames.toString();
153   }
154 
155   protected void clearColumnInfo() {
156     columnMap.clear();
157     columnNames.clear();
158     columnValues.clear();
159   }
160 
161   protected String removeBreakingWhitespace(String original) {
162     StringTokenizer whitespaceStripper = new StringTokenizer(original);
163     StringBuilder builder = new StringBuilder();
164     while (whitespaceStripper.hasMoreTokens()) {
165       builder.append(whitespaceStripper.nextToken());
166       builder.append(" ");
167     }
168     return builder.toString();
169   }
170 
171   protected boolean isDebugEnabled() {
172     return statementLog.isDebugEnabled();
173   }
174 
175   protected boolean isTraceEnabled() {
176     return statementLog.isTraceEnabled();
177   }
178 
179   protected void debug(String text, boolean input) {
180     if (statementLog.isDebugEnabled()) {
181       statementLog.debug(prefix(input) + text);
182     }
183   }
184 
185   protected void trace(String text, boolean input) {
186     if (statementLog.isTraceEnabled()) {
187       statementLog.trace(prefix(input) + text);
188     }
189   }
190 
191   private String prefix(boolean isInput) {
192     char[] buffer = new char[queryStack * 2 + 2];
193     Arrays.fill(buffer, '=');
194     buffer[queryStack * 2 + 1] = ' ';
195     if (isInput) {
196       buffer[queryStack * 2] = '>';
197     } else {
198       buffer[0] = '<';
199     }
200     return new String(buffer);
201   }
202 
203 }

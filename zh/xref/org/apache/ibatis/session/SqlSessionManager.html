263   public void close() {
264     final SqlSession sqlSession = localSqlSession.get();
265     if (sqlSession == null) throw new SqlSessionException("Error:  Cannot close.  No managed session is started.");

273   private class SqlSessionInterceptor implements InvocationHandler {
274     public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
275       final SqlSession sqlSession = SqlSessionManager.this.localSqlSession.get();
276       if (sqlSession != null) {
277         try {
278           return method.invoke(sqlSession, args);
279         } catch (Throwable t) {
280           throw ExceptionUtil.unwrapThrowable(t);
281         }
282       } else {
283         final SqlSession autoSqlSession = openSession();
284         try {
285           final Object result = method.invoke(autoSqlSession, args);
286           autoSqlSession.commit();
287           return result;
288         } catch (Throwable t) {
289           autoSqlSession.rollback();
290           throw ExceptionUtil.unwrapThrowable(t);
291         } finally {
292           autoSqlSession.close();
293         }
294       }
295     }
296   }

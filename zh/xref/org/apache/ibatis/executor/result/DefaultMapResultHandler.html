16  package org.apache.ibatis.executor.result;
17  
18  import java.util.Map;
19  
20  import org.apache.ibatis.reflection.MetaObject;
21  import org.apache.ibatis.reflection.factory.ObjectFactory;
22  import org.apache.ibatis.reflection.wrapper.ObjectWrapperFactory;
23  import org.apache.ibatis.session.ResultContext;
24  import org.apache.ibatis.session.ResultHandler;
25  
26  public class DefaultMapResultHandler<K, V> implements ResultHandler {
27  
28    private final Map<K, V> mappedResults;
29    private final String mapKey;
30    private final ObjectFactory objectFactory;
31    private final ObjectWrapperFactory objectWrapperFactory;
32  
33    @SuppressWarnings("unchecked")
34    public DefaultMapResultHandler(String mapKey, ObjectFactory objectFactory, ObjectWrapperFactory objectWrapperFactory) {
35      this.objectFactory = objectFactory;
36      this.objectWrapperFactory = objectWrapperFactory;
37      this.mappedResults = objectFactory.create(Map.class);
38      this.mapKey = mapKey;
39    }
40  
41    public void handleResult(ResultContext context) {
42      // TODO is that assignment always true?
43      final V value = (V) context.getResultObject();
44      final MetaObject mo = MetaObject.forObject(value, objectFactory, objectWrapperFactory);
45      // TODO is that assignment always true?
46      final K key = (K) mo.getValue(mapKey);
47      mappedResults.put(key, value);
48    }
49  
50    public Map<K, V> getMappedResults() {
51      return mappedResults;
52    }
53  }

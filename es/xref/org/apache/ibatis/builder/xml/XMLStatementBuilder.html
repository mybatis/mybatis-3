46    public XMLStatementBuilder(Configuration configuration, MapperBuilderAssistant builderAssistant, XNode context, String databaseId) {

53    public void parseStatementNode() {
54      String id = context.getStringAttribute("id");
55      String databaseId = context.getStringAttribute("databaseId");
56  
57      if (!databaseIdMatchesCurrent(id, databaseId, this.requiredDatabaseId)) {
58        return;
59      }
60  
61      Integer fetchSize = context.getIntAttribute("fetchSize", null);
62      Integer timeout = context.getIntAttribute("timeout", null);
63      String parameterMap = context.getStringAttribute("parameterMap");
64      String parameterType = context.getStringAttribute("parameterType");
65      Class<?> parameterTypeClass = resolveClass(parameterType);
66      String resultMap = context.getStringAttribute("resultMap");
67      String resultType = context.getStringAttribute("resultType");
68      String lang = context.getStringAttribute("lang");
69      LanguageDriver langDriver = getLanguageDriver(lang);
70  
71      Class<?> resultTypeClass = resolveClass(resultType);
72      String resultSetType = context.getStringAttribute("resultSetType");
73      StatementType statementType = StatementType.valueOf(context.getStringAttribute("statementType", StatementType.PREPARED.toString()));
74      ResultSetType resultSetTypeEnum = resolveResultSetType(resultSetType);
75  
76      String nodeName = context.getNode().getNodeName();
77      SqlCommandType sqlCommandType = SqlCommandType.valueOf(nodeName.toUpperCase(Locale.ENGLISH));
78      boolean isSelect = sqlCommandType == SqlCommandType.SELECT;
79      boolean flushCache = context.getBooleanAttribute("flushCache", !isSelect);
80      boolean useCache = context.getBooleanAttribute("useCache", isSelect);
81      boolean resultOrdered = context.getBooleanAttribute("resultOrdered", false);
82  
83      // Include Fragments before parsing
84      XMLIncludeTransformer includeParser = new XMLIncludeTransformer(configuration, builderAssistant);
85      includeParser.applyIncludes(context.getNode());
86  
87      // Parse selectKey after includes,
88      // in case if IncompleteElementException (issue #291)
89      List<XNode> selectKeyNodes = context.evalNodes("selectKey");
90      if (configuration.getDatabaseId() != null) {
91        parseSelectKeyNodes(id, selectKeyNodes, parameterTypeClass, langDriver, configuration.getDatabaseId());
92      }
93      parseSelectKeyNodes(id, selectKeyNodes, parameterTypeClass, langDriver, null);
94  
95      // Parse the SQL (pre: <selectKey> and <include> were parsed and removed)
96      SqlSource sqlSource = langDriver.createSqlSource(configuration, context, parameterTypeClass);
97  
98      String keyProperty = context.getStringAttribute("keyProperty");
99      String keyColumn = context.getStringAttribute("keyColumn");
100     KeyGenerator keyGenerator;
101     String keyStatementId = id + SelectKeyGenerator.SELECT_KEY_SUFFIX;
102     keyStatementId = builderAssistant.applyCurrentNamespace(keyStatementId, true);
103     if (configuration.hasKeyGenerator(keyStatementId)) {
104       keyGenerator = configuration.getKeyGenerator(keyStatementId);
105     } else {
106       keyGenerator = context.getBooleanAttribute("useGeneratedKeys",
107           configuration.isUseGeneratedKeys() && SqlCommandType.INSERT.equals(sqlCommandType))
108           ? new Jdbc3KeyGenerator() : new NoKeyGenerator();
109     }
110 
111     builderAssistant.addMappedStatement(id, sqlSource, statementType, sqlCommandType,
112         fetchSize, timeout, parameterMap, parameterTypeClass, resultMap, resultTypeClass,
113         resultSetTypeEnum, flushCache, useCache, resultOrdered, 
114         keyGenerator, keyProperty, keyColumn, databaseId, langDriver);
115   }

117   public void parseSelectKeyNodes(String parentId, List<XNode> list, Class<?> parameterTypeClass, LanguageDriver langDriver, String skRequiredDatabaseId) {
118     for (XNode nodeToHandle : list) {
119       String id = parentId + SelectKeyGenerator.SELECT_KEY_SUFFIX;
120       String databaseId = nodeToHandle.getStringAttribute("databaseId");
121       if (databaseIdMatchesCurrent(id, databaseId, skRequiredDatabaseId)) {
122         parseSelectKeyNode(id, nodeToHandle, parameterTypeClass, langDriver, databaseId);
123       }
124     }
125   }

127   public void parseSelectKeyNode(String id, XNode nodeToHandle, Class<?> parameterTypeClass, LanguageDriver langDriver, String databaseId) {
128     String resultType = nodeToHandle.getStringAttribute("resultType");
129     Class<?> resultTypeClass = resolveClass(resultType);
130     StatementType statementType = StatementType.valueOf(nodeToHandle.getStringAttribute("statementType", StatementType.PREPARED.toString()));
131     String keyProperty = nodeToHandle.getStringAttribute("keyProperty");
132     boolean executeBefore = "BEFORE".equals(nodeToHandle.getStringAttribute("order", "AFTER"));
133 
134     //defaults
135     boolean useCache = false;
136     boolean resultOrdered = false;
137     KeyGenerator keyGenerator = new NoKeyGenerator();
138     Integer fetchSize = null;
139     Integer timeout = null;
140     boolean flushCache = false;
141     String parameterMap = null;
142     String resultMap = null;
143     ResultSetType resultSetTypeEnum = null;
144 
145     SqlSource sqlSource = langDriver.createSqlSource(configuration, nodeToHandle, parameterTypeClass);
146     SqlCommandType sqlCommandType = SqlCommandType.SELECT;
147 
148     builderAssistant.addMappedStatement(id, sqlSource, statementType, sqlCommandType,
149         fetchSize, timeout, parameterMap, parameterTypeClass, resultMap, resultTypeClass,
150         resultSetTypeEnum, flushCache, useCache, resultOrdered,
151         keyGenerator, keyProperty, null, databaseId, langDriver);
152 
153     id = builderAssistant.applyCurrentNamespace(id, false);
154 
155     MappedStatement keyStatement = configuration.getMappedStatement(id, false);
156     configuration.addKeyGenerator(id, new SelectKeyGenerator(keyStatement, executeBefore));
157     nodeToHandle.getParent().getNode().removeChild(nodeToHandle.getNode());
158   }

160   private boolean databaseIdMatchesCurrent(String id, String databaseId, String requiredDatabaseId) {
161     if (requiredDatabaseId != null) {
162       if (!requiredDatabaseId.equals(databaseId)) {
163         return false;
164       }
165     } else {
166       if (databaseId != null) {
167         return false;
168       }
169       // skip this statement if there is a previous one with a not null databaseId
170       id = builderAssistant.applyCurrentNamespace(id, false);
171       if (this.configuration.hasStatement(id, false)) {
172         MappedStatement previous = this.configuration.getMappedStatement(id, false); // issue #2
173         if (previous.getDatabaseId() != null) {
174           return false;
175         }
176       }
177     }
178     return true;
179   }

181   private LanguageDriver getLanguageDriver(String lang) {
182     Class<?> langClass;
183     if (lang == null) {
184       langClass = configuration.getLanguageRegistry().getDefaultDriverClass();
185     } else {
186       langClass = resolveClass(lang);
187       configuration.getLanguageRegistry().register(langClass);
188     }
189     if (langClass == null) {
190       langClass = configuration.getLanguageRegistry().getDefaultDriverClass();
191     }
192     return configuration.getLanguageRegistry().getDriver(langClass);
193   }

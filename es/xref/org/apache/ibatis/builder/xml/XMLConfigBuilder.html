75    private XMLConfigBuilder(XPathParser parser, String environment, Properties props) {
76      super(new Configuration());
77      ErrorContext.instance().resource("SQL Mapper Configuration");
78      this.configuration.setVariables(props);
79      this.parsed = false;
80      this.environment = environment;
81      this.parser = parser;
82    }
83  
84    public Configuration parse() {
85      if (parsed) {
86        throw new BuilderException("Each XMLConfigBuilder can only be used once.");
87      }
88      parsed = true;
89      parseConfiguration(parser.evalNode("/configuration"));
90      return configuration;
91    }

93    private void parseConfiguration(XNode root) {
94      try {
95        propertiesElement(root.evalNode("properties")); //issue #117 read properties first
96        typeAliasesElement(root.evalNode("typeAliases"));
97        pluginElement(root.evalNode("plugins"));
98        objectFactoryElement(root.evalNode("objectFactory"));
99        objectWrapperFactoryElement(root.evalNode("objectWrapperFactory"));
100       settingsElement(root.evalNode("settings"));
101       environmentsElement(root.evalNode("environments")); // read it after objectFactory and objectWrapperFactory issue #631
102       databaseIdProviderElement(root.evalNode("databaseIdProvider"));
103       typeHandlerElement(root.evalNode("typeHandlers"));
104       mapperElement(root.evalNode("mappers"));
105     } catch (Exception e) {
106       throw new BuilderException("Error parsing SQL Mapper Configuration. Cause: " + e, e);
107     }
108   }

110   private void typeAliasesElement(XNode parent) {
111     if (parent != null) {
112       for (XNode child : parent.getChildren()) {
113         if ("package".equals(child.getName())) {
114           String typeAliasPackage = child.getStringAttribute("name");
115           configuration.getTypeAliasRegistry().registerAliases(typeAliasPackage);
116         } else {
117           String alias = child.getStringAttribute("alias");
118           String type = child.getStringAttribute("type");
119           try {
120             Class<?> clazz = Resources.classForName(type);
121             if (alias == null) {
122               typeAliasRegistry.registerAlias(clazz);
123             } else {
124               typeAliasRegistry.registerAlias(alias, clazz);
125             }
126           } catch (ClassNotFoundException e) {
127             throw new BuilderException("Error registering typeAlias for '" + alias + "'. Cause: " + e, e);
128           }
129         }
130       }
131     }
132   }

164   private void propertiesElement(XNode context) throws Exception {
165     if (context != null) {
166       Properties defaults = context.getChildrenAsProperties();
167       String resource = context.getStringAttribute("resource");
168       String url = context.getStringAttribute("url");
169       if (resource != null && url != null) {
170         throw new BuilderException("The properties element cannot specify both a URL and a resource based property file reference.  Please specify one or the other.");
171       }
172       if (resource != null) {
173         defaults.putAll(Resources.getResourceAsProperties(resource));
174       } else if (url != null) {
175         defaults.putAll(Resources.getUrlAsProperties(url));
176       }
177       Properties vars = configuration.getVariables();
178       if (vars != null) {
179         defaults.putAll(vars);
180       }
181       parser.setVariables(defaults);
182       configuration.setVariables(defaults);
183     }
184   }

186   private void settingsElement(XNode context) throws Exception {
187     if (context != null) {
188       Properties props = context.getChildrenAsProperties();
189       // Check that all settings are known to the configuration class
190       MetaClass metaConfig = MetaClass.forClass(Configuration.class);
191       for (Object key : props.keySet()) {
192         if (!metaConfig.hasSetter(String.valueOf(key))) {
193           throw new BuilderException("The setting " + key + " is not known.  Make sure you spelled it correctly (case sensitive).");
194         }
195       }
196       configuration.setAutoMappingBehavior(AutoMappingBehavior.valueOf(props.getProperty("autoMappingBehavior", "PARTIAL")));
197       configuration.setCacheEnabled(booleanValueOf(props.getProperty("cacheEnabled"), true));
198       configuration.setProxyFactory((ProxyFactory) createInstance(props.getProperty("proxyFactory")));
199       configuration.setLazyLoadingEnabled(booleanValueOf(props.getProperty("lazyLoadingEnabled"), false));
200       configuration.setAggressiveLazyLoading(booleanValueOf(props.getProperty("aggressiveLazyLoading"), true));
201       configuration.setMultipleResultSetsEnabled(booleanValueOf(props.getProperty("multipleResultSetsEnabled"), true));
202       configuration.setUseColumnLabel(booleanValueOf(props.getProperty("useColumnLabel"), true));
203       configuration.setUseGeneratedKeys(booleanValueOf(props.getProperty("useGeneratedKeys"), false));
204       configuration.setDefaultExecutorType(ExecutorType.valueOf(props.getProperty("defaultExecutorType", "SIMPLE")));
205       configuration.setDefaultStatementTimeout(integerValueOf(props.getProperty("defaultStatementTimeout"), null));
206       configuration.setMapUnderscoreToCamelCase(booleanValueOf(props.getProperty("mapUnderscoreToCamelCase"), false));
207       configuration.setSafeRowBoundsEnabled(booleanValueOf(props.getProperty("safeRowBoundsEnabled"), false));
208       configuration.setLocalCacheScope(LocalCacheScope.valueOf(props.getProperty("localCacheScope", "SESSION")));
209       configuration.setJdbcTypeForNull(JdbcType.valueOf(props.getProperty("jdbcTypeForNull", "OTHER")));
210       configuration.setLazyLoadTriggerMethods(stringSetValueOf(props.getProperty("lazyLoadTriggerMethods"), "equals,clone,hashCode,toString"));
211       configuration.setSafeResultHandlerEnabled(booleanValueOf(props.getProperty("safeResultHandlerEnabled"), true));
212       configuration.setDefaultScriptingLanguage(resolveClass(props.getProperty("defaultScriptingLanguage")));
213       configuration.setCallSettersOnNulls(booleanValueOf(props.getProperty("callSettersOnNulls"), false));
214       configuration.setLogPrefix(props.getProperty("logPrefix"));
215       configuration.setLogImpl(resolveClass(props.getProperty("logImpl")));
216       configuration.setConfigurationFactory(resolveClass(props.getProperty("configurationFactory")));
217     }

220   private void environmentsElement(XNode context) throws Exception {
221     if (context != null) {
222       if (environment == null) {
223         environment = context.getStringAttribute("default");
224       }
225       for (XNode child : context.getChildren()) {
226         String id = child.getStringAttribute("id");
227         if (isSpecifiedEnvironment(id)) {
228           TransactionFactory txFactory = transactionManagerElement(child.evalNode("transactionManager"));
229           DataSourceFactory dsFactory = dataSourceElement(child.evalNode("dataSource"));
230           DataSource dataSource = dsFactory.getDataSource();
231           Environment.Builder environmentBuilder = new Environment.Builder(id)
232               .transactionFactory(txFactory)
233               .dataSource(dataSource);
234           configuration.setEnvironment(environmentBuilder.build());
235         }
236       }
237     }
238   }

277   private void typeHandlerElement(XNode parent) throws Exception {
278     if (parent != null) {
279       for (XNode child : parent.getChildren()) {
280         if ("package".equals(child.getName())) {
281           String typeHandlerPackage = child.getStringAttribute("name");
282           typeHandlerRegistry.register(typeHandlerPackage);
283         } else {
284           String javaTypeName = child.getStringAttribute("javaType");
285           String jdbcTypeName = child.getStringAttribute("jdbcType");
286           String handlerTypeName = child.getStringAttribute("handler");
287           Class<?> javaTypeClass = resolveClass(javaTypeName);
288           JdbcType jdbcType = resolveJdbcType(jdbcTypeName);
289           Class<?> typeHandlerClass = resolveClass(handlerTypeName);
290           if (javaTypeClass != null) {
291             if (jdbcType == null) {
292               typeHandlerRegistry.register(javaTypeClass, typeHandlerClass);
293             } else {
294               typeHandlerRegistry.register(javaTypeClass, jdbcType, typeHandlerClass);
295             }
296           } else {
297             typeHandlerRegistry.register(typeHandlerClass);
298           }
299         }
300       }
301     }
302   }

304   private void mapperElement(XNode parent) throws Exception {
305     if (parent != null) {
306       for (XNode child : parent.getChildren()) {
307         if ("package".equals(child.getName())) {
308           String mapperPackage = child.getStringAttribute("name");
309           configuration.addMappers(mapperPackage);
310         } else {
311           String resource = child.getStringAttribute("resource");
312           String url = child.getStringAttribute("url");
313           String mapperClass = child.getStringAttribute("class");
314           if (resource != null && url == null && mapperClass == null) {
315             ErrorContext.instance().resource(resource);
316             InputStream inputStream = Resources.getResourceAsStream(resource);
317             XMLMapperBuilder mapperParser = new XMLMapperBuilder(inputStream, configuration, resource, configuration.getSqlFragments());
318             mapperParser.parse();
319           } else if (resource == null && url != null && mapperClass == null) {
320             ErrorContext.instance().resource(url);
321             InputStream inputStream = Resources.getUrlAsStream(url);
322             XMLMapperBuilder mapperParser = new XMLMapperBuilder(inputStream, configuration, url, configuration.getSqlFragments());
323             mapperParser.parse();
324           } else if (resource == null && url == null && mapperClass != null) {
325             Class<?> mapperInterface = Resources.classForName(mapperClass);
326             configuration.addMapper(mapperInterface);
327           } else {
328             throw new BuilderException("A mapper element may only specify a url, resource or class, but not more than one.");
329           }
330         }
331       }
332     }
333   }

335   private boolean isSpecifiedEnvironment(String id) {
336     if (environment == null) {
337       throw new BuilderException("No environment specified.");
338     } else if (id == null) {
339       throw new BuilderException("Environment requires an id attribute.");
340     } else if (environment.equals(id)) {
341       return true;
342     }
343     return false;
344   }

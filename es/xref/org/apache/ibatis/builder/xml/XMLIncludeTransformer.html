16  package org.apache.ibatis.builder.xml;
17  
18  import org.apache.ibatis.builder.IncompleteElementException;
19  import org.apache.ibatis.builder.MapperBuilderAssistant;
20  import org.apache.ibatis.parsing.XNode;
21  import org.apache.ibatis.session.Configuration;
22  import org.w3c.dom.Node;
23  import org.w3c.dom.NodeList;
24  
25  public class XMLIncludeTransformer {
26  
27    private final Configuration configuration;
28    private final MapperBuilderAssistant builderAssistant;
29  
30    public XMLIncludeTransformer(Configuration configuration, MapperBuilderAssistant builderAssistant) {
31      this.configuration = configuration;
32      this.builderAssistant = builderAssistant;
33    }
34  
35    public void applyIncludes(Node source) {
36      if (source.getNodeName().equals("include")) {
37        Node toInclude = findSqlFragment(getStringAttribute(source, "refid"));
38        applyIncludes(toInclude);
39        if (toInclude.getOwnerDocument() != source.getOwnerDocument()) {
40          toInclude = source.getOwnerDocument().importNode(toInclude, true);
41        }
42        source.getParentNode().replaceChild(toInclude, source);
43        while (toInclude.hasChildNodes()) {
44          toInclude.getParentNode().insertBefore(toInclude.getFirstChild(), toInclude);
45        }
46        toInclude.getParentNode().removeChild(toInclude);
47      } else if (source.getNodeType() == Node.ELEMENT_NODE) {
48        NodeList children = source.getChildNodes();
49        for (int i=0; i<children.getLength(); i++) {
50          applyIncludes(children.item(i));
51        }
52      }
53    }
54  
55    private Node findSqlFragment(String refid) {
56      refid = builderAssistant.applyCurrentNamespace(refid, true);
57      try {
58        XNode nodeToInclude = configuration.getSqlFragments().get(refid);
59        Node result = nodeToInclude.getNode().cloneNode(true);
60        return result;
61      } catch (IllegalArgumentException e) {
62        throw new IncompleteElementException("Could not find SQL statement to include with refid '" + refid + "'", e);
63      }
64    }
65  
66    private String getStringAttribute(Node node, String name) {
67      return node.getAttributes().getNamedItem(name).getNodeValue();
68    }
69  }

196   public List<XNode> evalNodes(Object root, String expression) {
197     List<XNode> xnodes = new ArrayList<XNode>();
198     NodeList nodes = (NodeList) evaluate(expression, root, XPathConstants.NODESET);
199     for (int i = 0; i < nodes.getLength(); i++) {
200       xnodes.add(new XNode(this, nodes.item(i), variables));
201     }
202     return xnodes;
203   }

209   public XNode evalNode(Object root, String expression) {
210     Node node = (Node) evaluate(expression, root, XPathConstants.NODE);
211     if (node == null) {
212       return null;
213     }
214     return new XNode(this, node, variables);
215   }

217   private Object evaluate(String expression, Object root, QName returnType) {
218     try {
219       return xpath.evaluate(expression, root, returnType);
220     } catch (Exception e) {
221       throw new BuilderException("Error evaluating XPath.  Cause: " + e, e);
222     }
223   }

225   private Document createDocument(InputSource inputSource) {
226     // important: this must only be called AFTER common constructor
227     try {
228       DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
229       factory.setValidating(validation);
230 
231       factory.setNamespaceAware(false);
232       factory.setIgnoringComments(true);
233       factory.setIgnoringElementContentWhitespace(false);
234       factory.setCoalescing(false);
235       factory.setExpandEntityReferences(true);
236 
237       DocumentBuilder builder = factory.newDocumentBuilder();
238       builder.setEntityResolver(entityResolver);
239       builder.setErrorHandler(new ErrorHandler() {
240         public void error(SAXParseException exception) throws SAXException {
241           throw exception;
242         }
243 
244         public void fatalError(SAXParseException exception) throws SAXException {
245           throw exception;
246         }
247 
248         public void warning(SAXParseException exception) throws SAXException {
249         }
250       });
251       return builder.parse(inputSource);
252     } catch (Exception e) {
253       throw new BuilderException("Error creating document instance.  Cause: " + e, e);
254     }
255   }

257   private void commonConstructor(boolean validation, Properties variables, EntityResolver entityResolver) {
258     this.validation = validation;
259     this.entityResolver = entityResolver;
260     this.variables = variables;
261     XPathFactory factory = XPathFactory.newInstance();
262     this.xpath = factory.newXPath();
263   }

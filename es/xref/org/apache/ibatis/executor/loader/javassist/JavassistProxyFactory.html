67    private static Object crateProxy(Class<?> type, MethodHandler callback, List<Class<?>> constructorArgTypes, List<Object> constructorArgs) {
68  
69      ProxyFactory enhancer = new ProxyFactory();
70      enhancer.setSuperclass(type);
71  
72      try {
73        type.getDeclaredMethod(WRITE_REPLACE_METHOD);
74        // ObjectOutputStream will call writeReplace of objects returned by writeReplace
75        log.debug(WRITE_REPLACE_METHOD + " method was found on bean " + type + ", make sure it returns this");
76      } catch (NoSuchMethodException e) {
77        enhancer.setInterfaces(new Class[]{WriteReplaceInterface.class});
78      } catch (SecurityException e) {
79        // nothing to do here
80      }
81  
82      Object enhanced = null;
83      Class<?>[] typesArray = constructorArgTypes.toArray(new Class[constructorArgTypes.size()]);
84      Object[] valuesArray = constructorArgs.toArray(new Object[constructorArgs.size()]);
85      try {
86        enhanced = enhancer.create(typesArray, valuesArray);
87      } catch (Exception e) {
88        throw new ExecutorException("Error creating lazy proxy.  Cause: " + e, e);
89      }
90      ((Proxy) enhanced).setHandler(callback);
91      return enhanced;
92    }

94    private static class EnhancedResultObjectProxyImpl implements MethodHandler {
95  
96      private Class<?> type;
97      private ResultLoaderMap lazyLoader;
98      private boolean aggressive;
99      private Set<String> lazyLoadTriggerMethods;
100     private ObjectFactory objectFactory;
101     private List<Class<?>> constructorArgTypes;
102     private List<Object> constructorArgs;
103 
104     private EnhancedResultObjectProxyImpl(Class<?> type, ResultLoaderMap lazyLoader, Configuration configuration, ObjectFactory objectFactory, List<Class<?>> constructorArgTypes, List<Object> constructorArgs) {
105       this.type = type;
106       this.lazyLoader = lazyLoader;
107       this.aggressive = configuration.isAggressiveLazyLoading();
108       this.lazyLoadTriggerMethods = configuration.getLazyLoadTriggerMethods();
109       this.objectFactory = objectFactory;
110       this.constructorArgTypes = constructorArgTypes;
111       this.constructorArgs = constructorArgs;
112     }
113 
114     public static Object createProxy(Object target, ResultLoaderMap lazyLoader, Configuration configuration, ObjectFactory objectFactory, List<Class<?>> constructorArgTypes, List<Object> constructorArgs) {
115       final Class<?> type = target.getClass();
116       EnhancedResultObjectProxyImpl callback = new EnhancedResultObjectProxyImpl(type, lazyLoader, configuration, objectFactory, constructorArgTypes, constructorArgs);
117       Object enhanced = crateProxy(type, callback, constructorArgTypes, constructorArgs);
118       PropertyCopier.copyBeanProperties(type, target, enhanced);
119       return enhanced;
120     }
121 
122     public Object invoke(Object enhanced, Method method, Method methodProxy, Object[] args) throws Throwable {
123       final String methodName = method.getName();
124       try {
125         synchronized (lazyLoader) {
126           if (WRITE_REPLACE_METHOD.equals(methodName)) {
127             Object original = null;
128             if (constructorArgTypes.isEmpty()) {
129               original = objectFactory.create(type);
130             } else {
131               original = objectFactory.create(type, constructorArgTypes, constructorArgs);
132             }
133             PropertyCopier.copyBeanProperties(type, enhanced, original);
134             if (lazyLoader.size() > 0) {
135               return new JavassistSerialStateHolder(original, lazyLoader.getProperties(), objectFactory, constructorArgTypes, constructorArgs);
136             } else {
137               return original;
138             }
139           } else {
140             if (lazyLoader.size() > 0 && !FINALIZE_METHOD.equals(methodName)) {
141               if (aggressive || lazyLoadTriggerMethods.contains(methodName)) {
142                 lazyLoader.loadAll();
143               } else if (PropertyNamer.isProperty(methodName)) {
144                 final String property = PropertyNamer.methodToProperty(methodName);
145                 if (lazyLoader.hasLoader(property)) {
146                   lazyLoader.load(property);
147                 }
148               }
149             }
150           }
151         }
152         return methodProxy.invoke(enhanced, args);
153       } catch (Throwable t) {
154         throw ExceptionUtil.unwrapThrowable(t);
155       }
156     }
157   }

161     private EnhancedDeserializationProxyImpl(Class<?> type, Map<String, ResultLoaderMap.LoadPair> unloadedProperties, ObjectFactory objectFactory,
162             List<Class<?>> constructorArgTypes, List<Object> constructorArgs) {
163       super(type, unloadedProperties, objectFactory, constructorArgTypes, constructorArgs);

166     public static Object createProxy(Object target, Map<String, ResultLoaderMap.LoadPair> unloadedProperties, ObjectFactory objectFactory,
167             List<Class<?>> constructorArgTypes, List<Object> constructorArgs) {
168       final Class<?> type = target.getClass();
169       EnhancedDeserializationProxyImpl callback = new EnhancedDeserializationProxyImpl(type, unloadedProperties, objectFactory, constructorArgTypes, constructorArgs);
170       Object enhanced = crateProxy(type, callback, constructorArgTypes, constructorArgs);
171       PropertyCopier.copyBeanProperties(type, target, enhanced);
172       return enhanced;
173     }

175     @Override
176     public Object invoke(Object enhanced, Method method, Method methodProxy, Object[] args) throws Throwable {
177       final Object o = super.invoke(enhanced, method, args);
178       return (o instanceof AbstractSerialStateHolder) ? o : methodProxy.invoke(o, args);
179     }

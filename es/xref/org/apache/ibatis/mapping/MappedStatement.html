64    public static class Builder {
65      private MappedStatement mappedStatement = new MappedStatement();
66  
67      public Builder(Configuration configuration, String id, SqlSource sqlSource, SqlCommandType sqlCommandType) {
68        mappedStatement.configuration = configuration;
69        mappedStatement.id = id;
70        mappedStatement.sqlSource = sqlSource;
71        mappedStatement.statementType = StatementType.PREPARED;
72        mappedStatement.parameterMap = new ParameterMap.Builder(configuration, "defaultParameterMap", null, new ArrayList<ParameterMapping>()).build();
73        mappedStatement.resultMaps = new ArrayList<ResultMap>();
74        mappedStatement.sqlCommandType = sqlCommandType;
75        mappedStatement.keyGenerator = configuration.isUseGeneratedKeys() && SqlCommandType.INSERT.equals(sqlCommandType) ? new Jdbc3KeyGenerator() : new NoKeyGenerator();
76        String logId = id;
77        if (configuration.getLogPrefix() != null) {
78          logId = configuration.getLogPrefix() + id;
79        }
80        mappedStatement.statementLog = LogFactory.getLog(logId);
81        mappedStatement.lang = configuration.getDefaultScriptingLanguageInstance();
82      }

291   public BoundSql getBoundSql(Object parameterObject) {
292     BoundSql boundSql = sqlSource.getBoundSql(parameterObject);
293     List<ParameterMapping> parameterMappings = boundSql.getParameterMappings();
294     if (parameterMappings == null || parameterMappings.isEmpty()) {
295       boundSql = new BoundSql(configuration, boundSql.getSql(), parameterMap.getParameterMappings(), parameterObject);
296     }
297 
298     // check for nested result maps in parameter mappings (issue #30)
299     for (ParameterMapping pm : boundSql.getParameterMappings()) {
300       String rmId = pm.getResultMapId();
301       if (rmId != null) {
302         ResultMap rm = configuration.getResultMap(rmId);
303         if (rm != null) {
304           hasNestedResultMaps |= rm.hasNestedResultMaps();
305         }
306       }
307     }
308 
309     return boundSql;
310   }

312   private static String[] delimitedStringToArray(String in) {
313     if (in == null || in.trim().length() == 0) {
314       return null;
315     } else {
316       return in.split(",");
317     }

31  public class TypeParameterResolver {
32  
33    /**
34     * @return The field type as {@link Type}. If it has type parameters in the declaration,<br>
35     *         they will be resolved to the actual runtime {@link Type}s.
36     */
37    public static Type resolveFieldType(Field field, Type srcType) {
38      Type fieldType = field.getGenericType();
39      Class<?> declaringClass = field.getDeclaringClass();
40      return resolveType(fieldType, srcType, declaringClass);
41    }
42  
43    /**
44     * @return The return type of the method as {@link Type}. If it has type parameters in the declaration,<br>
45     *         they will be resolved to the actual runtime {@link Type}s.
46     */
47    public static Type resolveReturnType(Method method, Type srcType) {
48      Type returnType = method.getGenericReturnType();
49      Class<?> declaringClass = method.getDeclaringClass();
50      return resolveType(returnType, srcType, declaringClass);
51    }
52  
53    /**
54     * @return The parameter types of the method as an array of {@link Type}s. If they have type parameters in the declaration,<br>
55     *         they will be resolved to the actual runtime {@link Type}s.
56     */
57    public static Type[] resolveParamTypes(Method method, Type srcType) {
58      Type[] paramTypes = method.getGenericParameterTypes();
59      Class<?> declaringClass = method.getDeclaringClass();
60      Type[] result = new Type[paramTypes.length];
61      for (int i = 0; i < paramTypes.length; i++) {
62        result[i] = resolveType(paramTypes[i], srcType, declaringClass);
63      }
64      return result;
65    }
66  
67    private static Type resolveType(Type type, Type srcType, Class<?> declaringClass) {
68      if (type instanceof TypeVariable) {
69        return resolveTypeVar((TypeVariable<?>) type, srcType, declaringClass);
70      } else if (type instanceof ParameterizedType) {
71        return resolveParameterizedType((ParameterizedType) type, srcType, declaringClass);
72      } else if (type instanceof GenericArrayType) {
73        return resolveGenericArrayType((GenericArrayType) type, srcType, declaringClass);
74      } else {
75        return type;
76      }
77    }
78  
79    private static Type resolveGenericArrayType(GenericArrayType genericArrayType, Type srcType, Class<?> declaringClass) {
80      Type componentType = genericArrayType.getGenericComponentType();
81      Type resolvedComponentType = null;
82      if (componentType instanceof TypeVariable) {
83        resolvedComponentType = resolveTypeVar((TypeVariable<?>) componentType, srcType, declaringClass);
84      } else if (componentType instanceof GenericArrayType) {
85        resolvedComponentType = resolveGenericArrayType((GenericArrayType) componentType, srcType, declaringClass);
86      } else if (componentType instanceof ParameterizedType) {
87        resolvedComponentType = resolveParameterizedType((ParameterizedType) componentType, srcType, declaringClass);
88      }
89      if (resolvedComponentType instanceof Class) {
90        return Array.newInstance((Class<?>) resolvedComponentType, 0).getClass();
91      } else {
92        return new GenericArrayTypeImpl(resolvedComponentType);
93      }
94    }
95  
96    private static ParameterizedType resolveParameterizedType(ParameterizedType parameterizedType, Type srcType, Class<?> declaringClass) {
97      Class<?> rawType = (Class<?>) parameterizedType.getRawType();
98      Type[] typeArgs = parameterizedType.getActualTypeArguments();
99      Type[] args = new Type[typeArgs.length];
100     for (int i = 0; i < typeArgs.length; i++) {
101       if (typeArgs[i] instanceof TypeVariable) {
102         args[i] = resolveTypeVar((TypeVariable<?>) typeArgs[i], srcType, declaringClass);
103       } else if (typeArgs[i] instanceof ParameterizedType) {
104         args[i] = resolveParameterizedType((ParameterizedType) typeArgs[i], srcType, declaringClass);
105       } else if (typeArgs[i] instanceof WildcardType) {
106         args[i] = resolveWildcardType((WildcardType) typeArgs[i], srcType, declaringClass);
107       } else {
108         args[i] = typeArgs[i];
109       }
110     }
111     return new ParameterizedTypeImpl(rawType, null, args);
112   }
113 
114   private static Type resolveWildcardType(WildcardType wildcardType, Type srcType, Class<?> declaringClass) {
115     Type[] lowerBounds = resolveWildcardTypeBounds(wildcardType.getLowerBounds(), srcType, declaringClass);
116     Type[] upperBounds = resolveWildcardTypeBounds(wildcardType.getUpperBounds(), srcType, declaringClass);
117     return new WildcardTypeImpl(lowerBounds, upperBounds);
118   }
119 
120   private static Type[] resolveWildcardTypeBounds(Type[] bounds, Type srcType, Class<?> declaringClass) {
121     Type[] result = new Type[bounds.length];
122     for (int i = 0; i < bounds.length; i++) {
123       if (bounds[i] instanceof TypeVariable) {
124         result[i] = resolveTypeVar((TypeVariable<?>) bounds[i], srcType, declaringClass);
125       } else if (bounds[i] instanceof ParameterizedType) {
126         result[i] = resolveParameterizedType((ParameterizedType) bounds[i], srcType, declaringClass);
127       } else if (bounds[i] instanceof WildcardType) {
128         result[i] = resolveWildcardType((WildcardType) bounds[i], srcType, declaringClass);
129       } else {
130         result[i] = bounds[i];
131       }
132     }
133     return result;
134   }
135 
136   private static Type resolveTypeVar(TypeVariable<?> typeVar, Type srcType, Class<?> declaringClass) {
137     Type result = null;
138     Class<?> clazz = null;
139     if (srcType instanceof Class) {
140       clazz = (Class<?>) srcType;
141     } else if (srcType instanceof ParameterizedType) {
142       ParameterizedType parameterizedType = (ParameterizedType) srcType;
143       clazz = (Class<?>) parameterizedType.getRawType();
144     } else {
145       throw new IllegalArgumentException("The 2nd arg must be Class or ParameterizedType, but was: " + srcType.getClass());
146     }
147 
148     if (clazz == declaringClass) {
149       Type[] bounds = typeVar.getBounds();
150       if(bounds.length > 0) {
151         return bounds[0];
152       }
153       return Object.class;
154     }
155 
156     Type superclass = clazz.getGenericSuperclass();
157     result = scanSuperTypes(typeVar, srcType, declaringClass, clazz, superclass);
158     if (result != null) {
159       return result;
160     }
161 
162     Type[] superInterfaces = clazz.getGenericInterfaces();
163     for (Type superInterface : superInterfaces) {
164       result = scanSuperTypes(typeVar, srcType, declaringClass, clazz, superInterface);
165       if (result != null) {
166         return result;
167       }
168     }
169     return Object.class;
170   }
171 
172   private static Type scanSuperTypes(TypeVariable<?> typeVar, Type srcType, Class<?> declaringClass, Class<?> clazz, Type superclass) {
173     Type result = null;
174     if (superclass instanceof ParameterizedType) {
175       ParameterizedType parentAsType = (ParameterizedType) superclass;
176       Class<?> parentAsClass = (Class<?>) parentAsType.getRawType();
177       if (declaringClass == parentAsClass) {
178         Type[] typeArgs = parentAsType.getActualTypeArguments();
179         TypeVariable<?>[] declaredTypeVars = declaringClass.getTypeParameters();
180         for (int i = 0; i < declaredTypeVars.length; i++) {
181           if (declaredTypeVars[i] == typeVar) {
182             if (typeArgs[i] instanceof TypeVariable) {
183               TypeVariable<?>[] typeParams = clazz.getTypeParameters();
184               for (int j = 0; j < typeParams.length; j++) {
185                 if (typeParams[j] == typeArgs[i]) {
186                   if (srcType instanceof ParameterizedType) {
187                     result = ((ParameterizedType) srcType).getActualTypeArguments()[j];
188                   }
189                   break;
190                 }
191               }
192             } else {
193               result = typeArgs[i];
194             }
195           }
196         }
197       } else if (declaringClass.isAssignableFrom(parentAsClass)) {
198         result = resolveTypeVar(typeVar, parentAsType, declaringClass);
199       }
200     } else if (superclass instanceof Class) {
201       if (declaringClass.isAssignableFrom((Class<?>) superclass)) {
202         result = resolveTypeVar(typeVar, superclass, declaringClass);
203       }
204     }
205     return result;
206   }

208   private TypeParameterResolver() {
209     super();
210   }

212   static class ParameterizedTypeImpl implements ParameterizedType {
213     private Class<?> rawType;
214 
215     private Type ownerType;
216 
217     private Type[] actualTypeArguments;
218 
219     public ParameterizedTypeImpl(Class<?> rawType, Type ownerType, Type[] actualTypeArguments) {
220       super();
221       this.rawType = rawType;
222       this.ownerType = ownerType;
223       this.actualTypeArguments = actualTypeArguments;
224     }
225 
226     @Override
227     public Type[] getActualTypeArguments() {
228       return actualTypeArguments;
229     }
230 
231     @Override
232     public Type getOwnerType() {
233       return ownerType;
234     }
235 
236     @Override
237     public Type getRawType() {
238       return rawType;
239     }
240 
241     @Override
242     public String toString() {
243       return "ParameterizedTypeImpl [rawType=" + rawType + ", ownerType=" + ownerType + ", actualTypeArguments=" + Arrays.toString(actualTypeArguments) + "]";
244     }
245   }

247   static class WildcardTypeImpl implements WildcardType {
248     private Type[] lowerBounds;
249 
250     private Type[] upperBounds;
251 
252     private WildcardTypeImpl(Type[] lowerBounds, Type[] upperBounds) {
253       super();
254       this.lowerBounds = lowerBounds;
255       this.upperBounds = upperBounds;
256     }
257 
258     @Override
259     public Type[] getLowerBounds() {
260       return lowerBounds;
261     }
262 
263     @Override
264     public Type[] getUpperBounds() {
265       return upperBounds;
266     }
267   }

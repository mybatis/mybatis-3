47    public Object execute(SqlSession sqlSession, Object[] args) {
48      Object result;
49      if (SqlCommandType.INSERT == command.getType()) {
50        Object param = method.convertArgsToSqlCommandParam(args);
51        result = rowCountResult(sqlSession.insert(command.getName(), param));
52      } else if (SqlCommandType.UPDATE == command.getType()) {
53        Object param = method.convertArgsToSqlCommandParam(args);
54        result = rowCountResult(sqlSession.update(command.getName(), param));
55      } else if (SqlCommandType.DELETE == command.getType()) {
56        Object param = method.convertArgsToSqlCommandParam(args);
57        result = rowCountResult(sqlSession.delete(command.getName(), param));
58      } else if (SqlCommandType.SELECT == command.getType()) {
59        if (method.returnsVoid() && method.hasResultHandler()) {
60          executeWithResultHandler(sqlSession, args);
61          result = null;
62        } else if (method.returnsMany()) {
63          result = executeForMany(sqlSession, args);
64        } else if (method.returnsMap()) {
65          result = executeForMap(sqlSession, args);
66        } else {
67          Object param = method.convertArgsToSqlCommandParam(args);
68          result = sqlSession.selectOne(command.getName(), param);
69        }
70      } else {
71        throw new BindingException("Unknown execution method for: " + command.getName());
72      }
73      if (result == null && method.getReturnType().isPrimitive() && !method.returnsVoid()) {
74        throw new BindingException("Mapper method '" + command.getName() 
75            + " attempted to return null from a method with a primitive return type (" + method.getReturnType() + ").");
76      }
77      return result;
78    }

80    private Object rowCountResult(int rowCount) {
81      final Object result;
82      if (method.returnsVoid()) {
83        result = null;
84      } else if (Integer.class.equals(method.getReturnType()) || Integer.TYPE.equals(method.getReturnType())) {
85        result = rowCount;
86      } else if (Long.class.equals(method.getReturnType()) || Long.TYPE.equals(method.getReturnType())) {
87        result = (long) rowCount;
88      } else if (Boolean.class.equals(method.getReturnType()) || Boolean.TYPE.equals(method.getReturnType())) {
89        result = (rowCount > 0);
90      } else {
91        throw new BindingException("Mapper method '" + command.getName() + "' has an unsupported return type: " + method.getReturnType());
92      }
93      return result;
94    }

96    private void executeWithResultHandler(SqlSession sqlSession, Object[] args) {
97      MappedStatement ms = sqlSession.getConfiguration().getMappedStatement(command.getName());
98      if (void.class.equals(ms.getResultMaps().get(0).getType())) {
99        throw new BindingException("method " + command.getName() 
100           + " needs either a @ResultMap annotation, a @ResultType annotation," 
101           + " or a resultType attribute in XML so a ResultHandler can be used as a parameter.");
102     }
103     Object param = method.convertArgsToSqlCommandParam(args);
104     if (method.hasRowBounds()) {
105       RowBounds rowBounds = method.extractRowBounds(args);
106       sqlSession.select(command.getName(), param, rowBounds, method.extractResultHandler(args));
107     } else {
108       sqlSession.select(command.getName(), param, method.extractResultHandler(args));
109     }
110   }

112   private <E> Object executeForMany(SqlSession sqlSession, Object[] args) {
113     List<E> result;
114     Object param = method.convertArgsToSqlCommandParam(args);
115     if (method.hasRowBounds()) {
116       RowBounds rowBounds = method.extractRowBounds(args);
117       result = sqlSession.<E>selectList(command.getName(), param, rowBounds);
118     } else {
119       result = sqlSession.<E>selectList(command.getName(), param);
120     }
121     // issue #510 Collections & arrays support
122     if (!method.getReturnType().isAssignableFrom(result.getClass())) {
123       if (method.getReturnType().isArray()) {
124         return convertToArray(result);
125       } else {
126         return convertToDeclaredCollection(sqlSession.getConfiguration(), result);
127       }
128     }
129     return result;
130   }

132   private <E> Object convertToDeclaredCollection(Configuration config, List<E> list) {
133     Object collection = config.getObjectFactory().create(method.getReturnType());
134     MetaObject metaObject = config.newMetaObject(collection);
135     metaObject.addAll(list);
136     return collection;
137   }

139   @SuppressWarnings("unchecked")
140   private <E> E[] convertToArray(List<E> list) {
141     E[] array = (E[]) Array.newInstance(method.getReturnType().getComponentType(), list.size());
142     array = list.toArray(array);
143     return array;
144   }

146   private <K, V> Map<K, V> executeForMap(SqlSession sqlSession, Object[] args) {
147     Map<K, V> result;
148     Object param = method.convertArgsToSqlCommandParam(args);
149     if (method.hasRowBounds()) {
150       RowBounds rowBounds = method.extractRowBounds(args);
151       result = sqlSession.<K, V>selectMap(command.getName(), param, method.getMapKey(), rowBounds);
152     } else {
153       result = sqlSession.<K, V>selectMap(command.getName(), param, method.getMapKey());
154     }
155     return result;

172   public static class SqlCommand {
173 
174     private final String name;
175     private final SqlCommandType type;
176 
177     public SqlCommand(Configuration configuration, Class<?> mapperInterface, Method method) throws BindingException {
178       String statementName = mapperInterface.getName() + "." + method.getName();
179       MappedStatement ms = null;
180       if (configuration.hasStatement(statementName)) {
181         ms = configuration.getMappedStatement(statementName);
182       } else if (!mapperInterface.equals(method.getDeclaringClass().getName())) { // issue #35
183         String parentStatementName = method.getDeclaringClass().getName() + "." + method.getName();
184         if (configuration.hasStatement(parentStatementName)) {
185           ms = configuration.getMappedStatement(parentStatementName);
186         }
187       }
188       if (ms == null) {
189         throw new BindingException("Invalid bound statement (not found): " + statementName);
190       }
191       name = ms.getId();
192       type = ms.getSqlCommandType();
193       if (type == SqlCommandType.UNKNOWN) {
194         throw new BindingException("Unknown execution method for: " + name);
195       }
196     }
197 
198     public String getName() {
199       return name;
200     }
201 
202     public SqlCommandType getType() {
203       return type;
204     }

207   public static class MethodSignature {
208 
209     private final boolean returnsMany;
210     private final boolean returnsMap;
211     private final boolean returnsVoid;
212     private final Class<?> returnType;
213     private final String mapKey;
214     private final Integer resultHandlerIndex;
215     private final Integer rowBoundsIndex;
216     private final SortedMap<Integer, String> params;
217     private final boolean hasNamedParameters;
218 
219     public MethodSignature(Configuration configuration, Method method) throws BindingException {
220       this.returnType = method.getReturnType();
221       this.returnsVoid = void.class.equals(this.returnType);
222       this.returnsMany = (configuration.getObjectFactory().isCollection(this.returnType) || this.returnType.isArray());
223       this.mapKey = getMapKey(method);
224       this.returnsMap = (this.mapKey != null);
225       this.hasNamedParameters = hasNamedParams(method);
226       this.rowBoundsIndex = getUniqueParamIndex(method, RowBounds.class);
227       this.resultHandlerIndex = getUniqueParamIndex(method, ResultHandler.class);
228       this.params = Collections.unmodifiableSortedMap(getParams(method, this.hasNamedParameters));
229     }
230 
231     public Object convertArgsToSqlCommandParam(Object[] args) {
232       final int paramCount = params.size();
233       if (args == null || paramCount == 0) {
234         return null;
235       } else if (!hasNamedParameters && paramCount == 1) {
236         return args[params.keySet().iterator().next()];
237       } else {
238         final Map<String, Object> param = new ParamMap<Object>();
239         int i = 0;
240         for (Map.Entry<Integer, String> entry : params.entrySet()) {
241           param.put(entry.getValue(), args[entry.getKey()]);
242           // issue #71, add param names as param1, param2...but ensure backward compatibility
243           final String genericParamName = "param" + String.valueOf(i + 1);
244           if (!param.containsKey(genericParamName)) {
245             param.put(genericParamName, args[entry.getKey()]);
246           }
247           i++;
248         }
249         return param;
250       }
251     }
252 
253     public boolean hasRowBounds() {
254       return (rowBoundsIndex != null);
255     }
256 
257     public RowBounds extractRowBounds(Object[] args) {
258       return (hasRowBounds() ? (RowBounds) args[rowBoundsIndex] : null);
259     }
260 
261     public boolean hasResultHandler() {
262       return (resultHandlerIndex != null);
263     }
264 
265     public ResultHandler extractResultHandler(Object[] args) {
266       return (hasResultHandler() ? (ResultHandler) args[resultHandlerIndex] : null);
267     }

289     private Integer getUniqueParamIndex(Method method, Class<?> paramType) {
290       Integer index = null;
291       final Class<?>[] argTypes = method.getParameterTypes();
292       for (int i = 0; i < argTypes.length; i++) {
293         if (paramType.isAssignableFrom(argTypes[i])) {
294           if (index == null) {
295             index = i;
296           } else {
297             throw new BindingException(method.getName() + " cannot have multiple " + paramType.getSimpleName() + " parameters");
298           }
299         }
300       }
301       return index;
302     }
303 
304     private String getMapKey(Method method) {
305       String mapKey = null;
306       if (Map.class.isAssignableFrom(method.getReturnType())) {
307         final MapKey mapKeyAnnotation = method.getAnnotation(MapKey.class);
308         if (mapKeyAnnotation != null) {
309           mapKey = mapKeyAnnotation.value();
310         }
311       }
312       return mapKey;
313     }
314 
315     private SortedMap<Integer, String> getParams(Method method, boolean hasNamedParameters) {
316       final SortedMap<Integer, String> params = new TreeMap<Integer, String>();
317       final Class<?>[] argTypes = method.getParameterTypes();
318       for (int i = 0; i < argTypes.length; i++) {
319         if (!RowBounds.class.isAssignableFrom(argTypes[i]) && !ResultHandler.class.isAssignableFrom(argTypes[i])) {
320           String paramName = String.valueOf(params.size());
321           if (hasNamedParameters) {
322             paramName = getParamNameFromAnnotation(method, i, paramName);
323           }
324           params.put(i, paramName);
325         }
326       }
327       return params;
328     }
329 
330     private String getParamNameFromAnnotation(Method method, int i, String paramName) {
331       final Object[] paramAnnos = method.getParameterAnnotations()[i];
332       for (Object paramAnno : paramAnnos) {
333         if (paramAnno instanceof Param) {
334           paramName = ((Param) paramAnno).value();
335         }
336       }
337       return paramName;
338     }
339 
340     private boolean hasNamedParams(Method method) {
341       boolean hasNamedParams = false;
342       final Object[][] paramAnnos = method.getParameterAnnotations();
343       for (Object[] paramAnno : paramAnnos) {
344         for (Object aParamAnno : paramAnno) {
345           if (aParamAnno instanceof Param) {
346             hasNamedParams = true;
347             break;
348           }
349         }
350       }
351       return hasNamedParams;
352     }

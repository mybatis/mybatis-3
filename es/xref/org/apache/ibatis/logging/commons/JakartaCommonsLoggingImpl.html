16  package org.apache.ibatis.logging.commons;
17  
18  import org.apache.commons.logging.Log;
19  import org.apache.commons.logging.LogFactory;
20  
21  /**
22   * @author Clinton Begin
23   */
24  public class JakartaCommonsLoggingImpl implements org.apache.ibatis.logging.Log {
25  
26    private Log log;
27  
28    public JakartaCommonsLoggingImpl(String clazz) {
29      log = LogFactory.getLog(clazz);
30    }
31  
32    @Override
33    public boolean isDebugEnabled() {
34      return log.isDebugEnabled();
35    }
36  
37    @Override
38    public boolean isTraceEnabled() {
39      return log.isTraceEnabled();
40    }
41  
42    @Override
43    public void error(String s, Throwable e) {
44      log.error(s, e);
45    }
46  
47    @Override
48    public void error(String s) {
49      log.error(s);
50    }
51  
52    @Override
53    public void debug(String s) {
54      log.debug(s);
55    }
56  
57    @Override
58    public void trace(String s) {
59      log.trace(s);
60    }
61  
62    @Override
63    public void warn(String s) {
64      log.warn(s);
65    }
66  
67  }

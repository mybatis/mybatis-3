16  package org.apache.ibatis.scripting.defaults;
17  
18  import java.util.HashMap;
19  
20  import org.apache.ibatis.builder.SqlSourceBuilder;
21  import org.apache.ibatis.mapping.BoundSql;
22  import org.apache.ibatis.mapping.SqlSource;
23  import org.apache.ibatis.scripting.xmltags.DynamicContext;
24  import org.apache.ibatis.scripting.xmltags.DynamicSqlSource;
25  import org.apache.ibatis.scripting.xmltags.SqlNode;
26  import org.apache.ibatis.session.Configuration;
27  
28  /**
29   * Static SqlSource. It is faster than {@link DynamicSqlSource} because mappings are 
30   * calculated during startup.
31   * 
32   * @since 3.2.0
33   */
34  public class RawSqlSource implements SqlSource {
35  
36    private final SqlSource sqlSource;
37  
38    public RawSqlSource(Configuration configuration, SqlNode rootSqlNode, Class<?> parameterType) {
39      this(configuration, getSql(configuration, rootSqlNode), parameterType);
40    }
41  
42    public RawSqlSource(Configuration configuration, String sql, Class<?> parameterType) {
43      SqlSourceBuilder sqlSourceParser = new SqlSourceBuilder(configuration);
44      Class<?> clazz = parameterType == null ? Object.class : parameterType;
45      sqlSource = sqlSourceParser.parse(sql, clazz, new HashMap<String, Object>());
46    }
47  
48    private static String getSql(Configuration configuration, SqlNode rootSqlNode) {
49      DynamicContext context = new DynamicContext(configuration, null);
50      rootSqlNode.apply(context);
51      return context.getSql();
52    }
53  
54    public BoundSql getBoundSql(Object parameterObject) {
55      return sqlSource.getBoundSql(parameterObject);
56    }
57  
58  }

37    private final XNode context;
38    private boolean isDynamic;
39    private final Class<?> parameterType;

42    public XMLScriptBuilder(Configuration configuration, XNode context) {
43      this(configuration, context, null);

46    public XMLScriptBuilder(Configuration configuration, XNode context, Class<?> parameterType) {
47      super(configuration);
48      this.context = context;
49      this.parameterType = parameterType;
50      initNodeHandlerMap();
51    }

54    private void initNodeHandlerMap() {
55      nodeHandlerMap.put("trim", new TrimHandler());
56      nodeHandlerMap.put("where", new WhereHandler());
57      nodeHandlerMap.put("set", new SetHandler());
58      nodeHandlerMap.put("foreach", new ForEachHandler());
59      nodeHandlerMap.put("if", new IfHandler());
60      nodeHandlerMap.put("choose", new ChooseHandler());
61      nodeHandlerMap.put("when", new IfHandler());
62      nodeHandlerMap.put("otherwise", new OtherwiseHandler());
63      nodeHandlerMap.put("bind", new BindHandler());
64    }
65  
66    public SqlSource parseScriptNode() {
67      MixedSqlNode rootSqlNode = parseDynamicTags(context);
68      SqlSource sqlSource;
69      if (isDynamic) {
70        sqlSource = new DynamicSqlSource(configuration, rootSqlNode);
71      } else {
72        sqlSource = new RawSqlSource(configuration, rootSqlNode, parameterType);
73      }
74      return sqlSource;
75    }
76  
77    protected MixedSqlNode parseDynamicTags(XNode node) {
78      List<SqlNode> contents = new ArrayList<>();
79      NodeList children = node.getNode().getChildNodes();
80      for (int i = 0; i < children.getLength(); i++) {
81        XNode child = node.newXNode(children.item(i));
82        if (child.getNode().getNodeType() == Node.CDATA_SECTION_NODE || child.getNode().getNodeType() == Node.TEXT_NODE) {
83          String data = child.getStringBody("");
84          TextSqlNode textSqlNode = new TextSqlNode(data);
85          if (textSqlNode.isDynamic()) {
86            contents.add(textSqlNode);
87            isDynamic = true;
88          } else {
89            contents.add(new StaticTextSqlNode(data));
90          }
91        } else if (child.getNode().getNodeType() == Node.ELEMENT_NODE) { // issue #628
92          String nodeName = child.getNode().getNodeName();
93          NodeHandler handler = nodeHandlerMap.get(nodeName);
94          if (handler == null) {
95            throw new BuilderException("Unknown element <" + nodeName + "> in SQL statement.");
96          }
97          handler.handleNode(child, contents);
98          isDynamic = true;
99        }
100     }
101     return new MixedSqlNode(contents);
102   }

104   private interface NodeHandler {
105     void handleNode(XNode nodeToHandle, List<SqlNode> targetContents);
106   }
107 
108   private class BindHandler implements NodeHandler {
109     public BindHandler() {
110       // Prevent Synthetic Access
111     }
112 
113     @Override
114     public void handleNode(XNode nodeToHandle, List<SqlNode> targetContents) {
115       final String name = nodeToHandle.getStringAttribute("name");
116       final String expression = nodeToHandle.getStringAttribute("value");
117       final VarDeclSqlNode node = new VarDeclSqlNode(name, expression);
118       targetContents.add(node);
119     }
120   }
121 
122   private class TrimHandler implements NodeHandler {
123     public TrimHandler() {
124       // Prevent Synthetic Access
125     }
126 
127     @Override
128     public void handleNode(XNode nodeToHandle, List<SqlNode> targetContents) {
129       MixedSqlNode mixedSqlNode = parseDynamicTags(nodeToHandle);
130       String prefix = nodeToHandle.getStringAttribute("prefix");
131       String prefixOverrides = nodeToHandle.getStringAttribute("prefixOverrides");
132       String suffix = nodeToHandle.getStringAttribute("suffix");
133       String suffixOverrides = nodeToHandle.getStringAttribute("suffixOverrides");
134       TrimSqlNode trim = new TrimSqlNode(configuration, mixedSqlNode, prefix, prefixOverrides, suffix, suffixOverrides);
135       targetContents.add(trim);
136     }
137   }

211   private class ChooseHandler implements NodeHandler {
212     public ChooseHandler() {
213       // Prevent Synthetic Access
214     }
215 
216     @Override
217     public void handleNode(XNode nodeToHandle, List<SqlNode> targetContents) {
218       List<SqlNode> whenSqlNodes = new ArrayList<>();
219       List<SqlNode> otherwiseSqlNodes = new ArrayList<>();
220       handleWhenOtherwiseNodes(nodeToHandle, whenSqlNodes, otherwiseSqlNodes);
221       SqlNode defaultSqlNode = getDefaultSqlNode(otherwiseSqlNodes);
222       ChooseSqlNode chooseSqlNode = new ChooseSqlNode(whenSqlNodes, defaultSqlNode);
223       targetContents.add(chooseSqlNode);
224     }
225 
226     private void handleWhenOtherwiseNodes(XNode chooseSqlNode, List<SqlNode> ifSqlNodes, List<SqlNode> defaultSqlNodes) {
227       List<XNode> children = chooseSqlNode.getChildren();
228       for (XNode child : children) {
229         String nodeName = child.getNode().getNodeName();
230         NodeHandler handler = nodeHandlerMap.get(nodeName);
231         if (handler instanceof IfHandler) {
232           handler.handleNode(child, ifSqlNodes);
233         } else if (handler instanceof OtherwiseHandler) {
234           handler.handleNode(child, defaultSqlNodes);
235         }
236       }
237     }
238 
239     private SqlNode getDefaultSqlNode(List<SqlNode> defaultSqlNodes) {
240       SqlNode defaultSqlNode = null;
241       if (defaultSqlNodes.size() == 1) {
242         defaultSqlNode = defaultSqlNodes.get(0);
243       } else if (defaultSqlNodes.size() > 1) {
244         throw new BuilderException("Too many default (otherwise) elements in choose statement.");
245       }
246       return defaultSqlNode;
247     }
248   }

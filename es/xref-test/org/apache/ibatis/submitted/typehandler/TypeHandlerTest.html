16  package org.apache.ibatis.submitted.typehandler;
17  
18  import static org.junit.Assert.*;
19  
20  import java.io.Reader;
21  import java.sql.Connection;
22  
23  import org.apache.ibatis.builder.BuilderException;
24  import org.apache.ibatis.io.Resources;
25  import org.apache.ibatis.jdbc.ScriptRunner;
26  import org.apache.ibatis.session.SqlSession;
27  import org.apache.ibatis.session.SqlSessionFactory;
28  import org.apache.ibatis.session.SqlSessionFactoryBuilder;
29  import org.apache.ibatis.submitted.typehandler.Product.ConstantProductIdTypeHandler;
30  import org.apache.ibatis.submitted.typehandler.Product.ProductId;
31  import org.apache.ibatis.submitted.typehandler.Product.ProductIdTypeHandler;
32  import org.apache.ibatis.type.JdbcType;
33  import org.junit.Before;
34  import org.junit.Test;
35  
36  public class TypeHandlerTest {
37  
38    private SqlSessionFactory sqlSessionFactory;
39  
40    @Before
41    public void setUp() throws Exception {
42      // create a SqlSessionFactory
43      Reader reader = Resources.getResourceAsReader("org/apache/ibatis/submitted/typehandler/mybatis-config.xml");
44      sqlSessionFactory = new SqlSessionFactoryBuilder().build(reader);
45      reader.close();
46      sqlSessionFactory.getConfiguration().getTypeHandlerRegistry().register(StringTrimmingTypeHandler.class);
47  
48      // populate in-memory database
49      SqlSession session = sqlSessionFactory.openSession();
50      Connection conn = session.getConnection();
51      reader = Resources.getResourceAsReader("org/apache/ibatis/submitted/typehandler/CreateDB.sql");
52      ScriptRunner runner = new ScriptRunner(conn);
53      runner.setLogWriter(null);
54      runner.runScript(reader);
55      reader.close();
56      session.close();
57    }
58  
59    // Some tests need to register additional type handler
60    // before adding mapper.
61    private void addMapper() {
62      sqlSessionFactory.getConfiguration().addMapper(Mapper.class);
63    }
64  
65    @Test
66    public void shouldGetAUser() {
67      addMapper();
68      SqlSession sqlSession = sqlSessionFactory.openSession();
69      try {
70        Mapper mapper = sqlSession.getMapper(Mapper.class);
71        User user = mapper.getUser(1);
72        assertEquals("User1", user.getName());
73        assertEquals("Carmel", user.getCity());
74        assertEquals("IN", user.getState());
75      } finally {
76        sqlSession.close();
77      }
78    }
79  
80    @Test
81    public void shouldApplyTypeHandlerOnGeneratedKey() {
82      addMapper();
83      SqlSession sqlSession = sqlSessionFactory.openSession();
84      try {
85        Mapper mapper = sqlSession.getMapper(Mapper.class);
86        Product product = new Product();
87        product.setName("new product");
88        mapper.insertProduct(product);
89        assertNotNull(product.getId());
90        assertNotNull(product.getId().getValue());
91      } finally {
92        sqlSession.close();
93      }
94    }
95  
96    @Test
97    public void shouldApplyTypeHandlerWithJdbcTypeSpecified() throws Exception {
98      addMapper();
99      SqlSession sqlSession = sqlSessionFactory.openSession();
100     try {
101       Mapper mapper = sqlSession.getMapper(Mapper.class);
102       Product product = mapper.getProductByName("iPad");
103       assertEquals(Integer.valueOf(2), product.getId().getValue());
104     } finally {
105       sqlSession.close();
106     }
107   }
108

135   @Test
136   public void shouldPickSoleTypeHandlerOnXmlResultMap() throws Exception {
137     addMapper();

146   }
147 
148   @Test
149   public void shouldPickSameTypeHandlerMappedToDifferentJdbcTypes() throws Exception {
150     sqlSessionFactory.getConfiguration().getTypeHandlerRegistry().register(ProductId.class, JdbcType.BIGINT, ProductIdTypeHandler.class);
151     addMapper();
152     SqlSession sqlSession = sqlSessionFactory.openSession();
153     try {
154       Mapper mapper = sqlSession.getMapper(Mapper.class);
155       Product product = mapper.getProductByNameXml("iPad");
156       assertEquals(Integer.valueOf(2), product.getId().getValue());
157     } finally {
158       sqlSession.close();
159     }
160   }
161 
162   @Test(expected = BuilderException.class)
163   public void shouldFailIfMultipleHandlerMappedToAType() throws Exception {
164     sqlSessionFactory.getConfiguration().getTypeHandlerRegistry().register(ProductId.class, JdbcType.BIGINT, ConstantProductIdTypeHandler.class);
165     // multiple type handlers are mapped to ProductId and
166     // none of them are mapped to null jdbcType.
167     addMapper();
168   }
169 
170   @Test
171   public void shouldPickHandlerForNull() throws Exception {
172     sqlSessionFactory.getConfiguration().getTypeHandlerRegistry().register(ProductId.class, null, ConstantProductIdTypeHandler.class);
173     // multiple type handlers are mapped to ProductId and
174     // one of them are mapped to null jdbcType.
175     addMapper();
176     SqlSession sqlSession = sqlSessionFactory.openSession();
177     try {
178       Mapper mapper = sqlSession.getMapper(Mapper.class);
179       Product product = mapper.getProductByNameXml("iPad");
180       assertEquals(Integer.valueOf(999), product.getId().getValue());
181     } finally {
182       sqlSession.close();
183     }
184   }

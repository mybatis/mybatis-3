119     public static boolean blobsAreEqual(byte[] blob1, byte[] blob2) {
120         if (blob1 == null) {
121             return blob2 == null;
122         }
123 
124         if (blob2 == null) {
125             return blob1 == null;
126         }
127 
128         boolean rc = blob1.length == blob2.length;
129 
130         if (rc) {
131             for (int i = 0; i < blob1.length; i++) {
132                 if (blob1[i] != blob2[i]) {
133                     rc = false;
134                     break;
135                 }
136             }
137         }
138 
139         return rc;
140     }

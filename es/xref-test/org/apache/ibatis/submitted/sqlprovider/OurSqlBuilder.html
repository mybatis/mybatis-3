26    public String buildGetUsersQuery(Map<String, Object> parameter) {
27      // MyBatis wraps a single List parameter in a Map with the key="list",
28      // so need to pull it out
29      @SuppressWarnings("unchecked")
30      List<Integer> ids = (List<Integer>) parameter.get("list");
31      StringBuilder sb = new StringBuilder();
32      sb.append("select * from users where id in (");
33      for (int i = 0; i < ids.size(); i++) {
34        if (i > 0) {
35          sb.append(",");
36        }
37        sb.append("#{list[");
38        sb.append(i);
39        sb.append("]}");
40      }
41      sb.append(") order by id");
42      return sb.toString();
43    }

54    public String buildGetUsersByCriteriaQuery(final User criteria) {
55      return new SQL(){{
56        SELECT("*");
57        FROM("users");
58        if (criteria.getId() != null) {
59          WHERE("id = #{id}");
60        }
61        if (criteria.getName() != null) {
62          WHERE("name like #{name} || '%'");
63        }
64      }}.toString();
65    }

81      return new SQL(){{

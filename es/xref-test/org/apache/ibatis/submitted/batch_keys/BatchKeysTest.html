73    @Test
74    public void testInsert() throws Exception {
75      SqlSession sqlSession = sqlSessionFactory.openSession(ExecutorType.BATCH);
76      try {
77        User user1 = new User(null, "Pocoyo");
78        sqlSession.insert("insert", user1);
79        User user2 = new User(null, "Valentina");
80        sqlSession.insert("insert", user2);
81        sqlSession.flushStatements();
82        assertEquals(new Integer(50), user1.getId());
83        assertEquals(new Integer(50), user2.getId());
84        sqlSession.commit();
85      } finally {
86        sqlSession.close();
87      }
88  
89      sqlSession = sqlSessionFactory.openSession();

94    public void testJdbc3Support() throws Exception {
95      Connection conn = sqlSessionFactory.getConfiguration().getEnvironment().getDataSource().getConnection();
96      PreparedStatement stmt = conn.prepareStatement("insert into users2 values(null, 'Pocoyo')", Statement.RETURN_GENERATED_KEYS);
97      stmt.addBatch();
98      stmt.executeBatch();
99      ResultSet rs = stmt.getGeneratedKeys();
100     if (rs.next()) {
101       ResultSetMetaData rsmd = rs.getMetaData();
102       int colCount = rsmd.getColumnCount();
103       do {
104         for (int i = 1; i <= colCount; i++) {
105           String key = rs.getString(i);
106           System.out.println("key " + i + " is " + key);
107         }
108       } while (rs.next());
109     } else {
110       System.out.println("There are no generated keys.");

113   public static class EnumOrderTypeHandler<E extends Enum<E>> extends BaseTypeHandler<E> {
114 
115     private E[] constants;
116 
117     public EnumOrderTypeHandler(Class<E> javaType) {
118       constants = javaType.getEnumConstants();
119     }
120 
121     @Override
122     public void setNonNullParameter(PreparedStatement ps, int i, E parameter, JdbcType jdbcType) throws SQLException {
123       ps.setInt(i, parameter.ordinal() + 1); // 0 means NULL so add +1
124     }
125 
126     @Override
127     public E getNullableResult(ResultSet rs, String columnName) throws SQLException {
128       int index = rs.getInt(columnName) - 1;
129       return index < 0 ? null : constants[index];
130     }
131 
132     @Override
133     public E getNullableResult(ResultSet rs, int columnIndex) throws SQLException {
134       int index = rs.getInt(rs.getInt(columnIndex)) - 1;
135       return index < 0 ? null : constants[index];
136     }
137 
138     @Override
139     public E getNullableResult(CallableStatement cs, int columnIndex) throws SQLException {
140       int index = cs.getInt(columnIndex) - 1;
141       return index < 0 ? null : constants[index];
142     }
143   }

145   @Test
146   void registerJavaTypeInitializingTypeHandler() {
147     final String MAPPER_CONFIG = "<?xml version=\"1.0\" encoding=\"UTF-8\" ?>\n"
148         + "<!DOCTYPE configuration PUBLIC \"-//mybatis.org//DTD Config 3.0//EN\" \"http://mybatis.org/dtd/mybatis-3-config.dtd\">\n"
149         + "<configuration>\n"
150         + "  <typeHandlers>\n"
151         + "    <typeHandler javaType=\"org.apache.ibatis.builder.XmlConfigBuilderTest$MyEnum\"\n"
152         + "      handler=\"org.apache.ibatis.builder.XmlConfigBuilderTest$EnumOrderTypeHandler\"/>\n"
153         + "  </typeHandlers>\n"
154         + "</configuration>\n";
155 
156     XMLConfigBuilder builder = new XMLConfigBuilder(new StringReader(MAPPER_CONFIG));
157     builder.parse();
158 
159     TypeHandlerRegistry typeHandlerRegistry = builder.getConfiguration().getTypeHandlerRegistry();
160     TypeHandler<MyEnum> typeHandler = typeHandlerRegistry.getTypeHandler(MyEnum.class);
161 
162     assertTrue(typeHandler instanceof EnumOrderTypeHandler);
163     assertArrayEquals(MyEnum.values(), ((EnumOrderTypeHandler<MyEnum>) typeHandler).constants);
164   }

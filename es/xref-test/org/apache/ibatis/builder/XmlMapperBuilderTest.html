41    public void shouldSuccessfullyLoadXMLMapperFile() throws Exception {
42      Configuration configuration = new Configuration();
43      String resource = "org/apache/ibatis/builder/AuthorMapper.xml";
44      InputStream inputStream = Resources.getResourceAsStream(resource);
45      XMLMapperBuilder builder = new XMLMapperBuilder(inputStream, configuration, resource, configuration.getSqlFragments());
46      builder.parse();
47    }

49    @Test
50    public void mappedStatementWithOptions() throws Exception {
51      Configuration configuration = new Configuration();
52      String resource = "org/apache/ibatis/builder/AuthorMapper.xml";
53      InputStream inputStream = Resources.getResourceAsStream(resource);
54      XMLMapperBuilder builder = new XMLMapperBuilder(inputStream, configuration, resource, configuration.getSqlFragments());
55      builder.parse();
56  
57      MappedStatement mappedStatement = configuration.getMappedStatement("selectWithOptions");
58      assertThat(mappedStatement.getFetchSize(), is(200));
59      assertThat(mappedStatement.getTimeout(), is(10));
60      assertThat(mappedStatement.getStatementType(), is(StatementType.PREPARED));
61      assertThat(mappedStatement.getResultSetType(), is(ResultSetType.SCROLL_SENSITIVE));
62      assertThat(mappedStatement.isFlushCacheRequired(), is(false));
63      assertThat(mappedStatement.isUseCache(), is(false));
64  
65    }

67    @Test
68    public void parseExpression() {
69      BaseBuilder builder = new BaseBuilder(new Configuration()){{}};
70      {
71        Pattern pattern = builder.parseExpression("[0-9]", "[a-z]");
72        assertThat(pattern.matcher("0").find(), is(true));
73        assertThat(pattern.matcher("a").find(), is(false));
74      }
75      {
76        Pattern pattern = builder.parseExpression(null, "[a-z]");
77        assertThat(pattern.matcher("0").find(), is(false));
78        assertThat(pattern.matcher("a").find(), is(true));
79      }
80    }

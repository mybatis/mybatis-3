138   static abstract class Parent<T extends Serializable> {
139     protected T id;
140     protected List<T> list;
141     protected T[] array;
142     private T fld;
143     public T pubFld;
144     public T getId() {
145       return id;
146     }
147     public void setId(T id) {
148       this.id = id;
149     }
150     public List<T> getList() {
151       return list;
152     }
153     public void setList(List<T> list) {
154       this.list = list;
155     }
156     public T[] getArray() {
157       return array;
158     }
159     public void setArray(T[] array) {
160       this.array = array;
161     }
162     public T getFld() {
163       return fld;
164     }
165   }

170   @Test
171   public void shouldResoleveReadonlySetterWithOverload() throws Exception {
172     class BeanClass implements BeanInterface<String> {
173       public void setId(String id) {}
174     }
175     ReflectorFactory reflectorFactory = new DefaultReflectorFactory();
176     Reflector reflector = reflectorFactory.findForClass(BeanClass.class);
177     assertEquals(String.class, reflector.getSetterType("id"));
178   }

184   @Test
185   public void shouldSettersWithUnrelatedArgTypesThrowException() throws Exception {
186     @SuppressWarnings("unused")
187     class BeanClass {
188       public void setTheProp(String arg) {}
189       public void setTheProp(Integer arg) {}
190     }
191     expectedException.expect(ReflectionException.class);
192     expectedException.expectMessage(allOf(
193         containsString("theProp"),
194         containsString("BeanClass"),
195         containsString("java.lang.String"),
196         containsString("java.lang.Integer")));
197     ReflectorFactory reflectorFactory = new DefaultReflectorFactory();
198     Reflector reflector = reflectorFactory.findForClass(BeanClass.class);
199     reflector.getSetterType("theProp");
200   }

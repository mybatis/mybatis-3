16  package org.apache.ibatis.binding;
17  
18  import domain.blog.Author;
19  import domain.blog.Blog;
20  import domain.blog.Post;
21  import org.apache.ibatis.BaseDataTest;
22  import org.apache.ibatis.mapping.Environment;
23  import org.apache.ibatis.session.Configuration;
24  import org.apache.ibatis.session.SqlSessionFactory;
25  import org.apache.ibatis.session.SqlSessionFactoryBuilder;
26  import org.apache.ibatis.transaction.TransactionFactory;
27  import org.apache.ibatis.transaction.jdbc.JdbcTransactionFactory;
28  
29  import javax.sql.DataSource;
30  
31  public class IbatisConfig {
32  
33    public SqlSessionFactory getSqlSessionFactory() {
34      try {
35        DataSource dataSource = BaseDataTest.createBlogDataSource();
36        BaseDataTest.runScript(dataSource, BaseDataTest.BLOG_DDL);
37        BaseDataTest.runScript(dataSource, BaseDataTest.BLOG_DATA);
38        TransactionFactory transactionFactory = new JdbcTransactionFactory();
39        Environment environment = new Environment("Production", transactionFactory, dataSource);
40        Configuration configuration = new Configuration(environment);
41        configuration.setLazyLoadingEnabled(true);
42        configuration.getTypeAliasRegistry().registerAlias(Blog.class);
43        configuration.getTypeAliasRegistry().registerAlias(Post.class);
44        configuration.getTypeAliasRegistry().registerAlias(Author.class);
45        configuration.addMapper(BoundBlogMapper.class);
46        configuration.addMapper(BoundAuthorMapper.class);
47        return new SqlSessionFactoryBuilder().build(configuration);
48      } catch (Exception e) {
49        throw new RuntimeException("Error initializing SqlSessionFactory. Cause: " + e, e);
50      }
51    }
52  
53  }

16  package org.apache.ibatis.type;
17  
18  import static org.assertj.core.api.Assertions.assertThat;
19  
20  import java.time.Instant;
21  import java.time.LocalDate;
22  import java.time.LocalDateTime;
23  import java.time.LocalTime;
24  import java.time.Month;
25  import java.time.OffsetDateTime;
26  import java.time.OffsetTime;
27  import java.time.Year;
28  import java.time.YearMonth;
29  import java.time.ZonedDateTime;
30  import java.time.chrono.JapaneseDate;
31  
32  import org.junit.jupiter.api.BeforeEach;
33  import org.junit.jupiter.api.Test;
34  
35  /**
36   * @author Kazuki Shimizu
37   */
38  class Jsr310TypeHandlerRegistryTest {
39  
40    private TypeHandlerRegistry typeHandlerRegistry;
41  
42    @BeforeEach
43    void setup() {
44      typeHandlerRegistry = new TypeHandlerRegistry();
45    }
46  
47    @Test
48    void shouldRegisterJsr310TypeHandlers() {
49      assertThat(typeHandlerRegistry.getTypeHandler(Instant.class)).isInstanceOf(InstantTypeHandler.class);
50      assertThat(typeHandlerRegistry.getTypeHandler(LocalDateTime.class)).isInstanceOf(LocalDateTimeTypeHandler.class);
51      assertThat(typeHandlerRegistry.getTypeHandler(LocalDate.class)).isInstanceOf(LocalDateTypeHandler.class);
52      assertThat(typeHandlerRegistry.getTypeHandler(LocalTime.class)).isInstanceOf(LocalTimeTypeHandler.class);
53      assertThat(typeHandlerRegistry.getTypeHandler(OffsetDateTime.class)).isInstanceOf(OffsetDateTimeTypeHandler.class);
54      assertThat(typeHandlerRegistry.getTypeHandler(OffsetTime.class)).isInstanceOf(OffsetTimeTypeHandler.class);
55      assertThat(typeHandlerRegistry.getTypeHandler(ZonedDateTime.class)).isInstanceOf(ZonedDateTimeTypeHandler.class);
56      assertThat(typeHandlerRegistry.getTypeHandler(Month.class)).isInstanceOf(MonthTypeHandler.class);
57      assertThat(typeHandlerRegistry.getTypeHandler(Year.class)).isInstanceOf(YearTypeHandler.class);
58      assertThat(typeHandlerRegistry.getTypeHandler(YearMonth.class)).isInstanceOf(YearMonthTypeHandler.class);
59      assertThat(typeHandlerRegistry.getTypeHandler(JapaneseDate.class)).isInstanceOf(JapaneseDateTypeHandler.class);
60    }
61  }

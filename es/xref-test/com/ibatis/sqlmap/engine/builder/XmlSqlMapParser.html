67    public XmlSqlMapParser(XmlSqlMapConfigParser configParser, Reader reader) {
68      this.configParser = configParser;
69      this.config = configParser.getConfiguration();
70      this.reader = reader;
71      this.parser = new NodeEventParser();
72      this.parser.addNodeletHandler(this);

75      this.parser.setEntityResolver(new SqlMapEntityResolver());

107   @NodeEvent("/sqlMap/cacheModel")
108   public void sqlMapcacheModel(XNode context) throws Exception {
109     String id = applyNamespace(context.getStringAttribute("id"));
110     String type = context.getStringAttribute("type");
111     Boolean readOnly = context.getBooleanAttribute("readOnly", true);
112     Boolean serialize = context.getBooleanAttribute("serialize", true);
113     Class clazz = config.getTypeAliasRegistry().resolveAlias(type);
114     cacheBuilder = new CacheBuilder(id);
115     cacheBuilder.addDecorator(clazz);
116 
117     //LOCAL_READ_WRITE (serializable=false, readOnly=false)
118     //SHARED_READ_ONLY (serializable=false, readOnly=true)
119     //SHARED_READ_WRITE (serializable=true, readOnly=false)
120     if (serialize) {
121       if (readOnly) {
122         cacheBuilder.readWrite(false);
123       } else {
124         cacheBuilder.readWrite(true);
125       }
126     } else {
127       if (readOnly) {
128         cacheBuilder.readWrite(false);
129       } else {
130         cacheBuilder = null;
131       }
132     }
133   }

146   @NodeEvent("/sqlMap/cacheModel/flushInterval")
147   public void sqlMapcacheModelflushInterval(XNode context) throws Exception {
148     if (cacheBuilder != null) {
149       long clearInterval = 0L;
150       clearInterval += context.getIntAttribute("milliseconds", 0);
151       clearInterval += context.getIntAttribute("seconds", 0) * 1000L;
152       clearInterval += context.getIntAttribute("minutes", 0) * 60L * 1000L;
153       clearInterval += context.getIntAttribute("hours", 0) * 60L * 60L * 1000L;
154       if (clearInterval < 1L) {
155         throw new RuntimeException("A flush interval must specify one or more of milliseconds, seconds, minutes or hours.");
156       }
157       cacheBuilder.clearInterval(clearInterval);
158     }
159   }

284   @NodeEvent("/sqlMap/resultMap/result")
285   public void sqlMapresultMapresult(XNode context) throws Exception {
286     String nullValue = context.getStringAttribute("nullValue");
287     if (nullValue != null) {
288       throw new UnsupportedOperationException("Null value subsitution is not supported by iBATIS 3.");
289     }
290     String columnIndexProp = context.getStringAttribute("columnIndex");
291     if (columnIndexProp != null) {
292       throw new UnsupportedOperationException("Numerical column indices are not supported.  Use the column name instead.");
293     }
294 
295     String propertyName = context.getStringAttribute("property");
296     String jdbcType = context.getStringAttribute("jdbcType");
297     String javaType = context.getStringAttribute("javaType");
298     String columnName = context.getStringAttribute("column");
299 
300     String statementName = context.getStringAttribute("select");
301     String resultMapName = context.getStringAttribute("resultMap");
302     String callback = context.getStringAttribute("typeHandler");
303 
304     Class javaClass = null;
305     try {
306       if (javaType != null && javaType.length() > 0) {
307         javaClass = config.getTypeAliasRegistry().resolveAlias(javaType);
308       }
309     } catch (Exception e) {
310       throw new RuntimeException("Error setting java type on result discriminator mapping.  Cause: " + e);
311     }
312     if (javaClass == null
313         && !Map.class.isAssignableFrom(resultMapBuilder.type())
314         && !config.getTypeHandlerRegistry().hasTypeHandler(resultMapBuilder.type())) {
315       javaClass = MetaClass.forClass(resultMapBuilder.type()).getSetterType(propertyName);
316     }
317     if (javaClass == null && statementName != null) {
318       javaClass = List.class;
319     }
320 
321     JdbcType jdbcTypeEnum = null;
322     if (jdbcType != null) {
323       jdbcTypeEnum = JdbcType.valueOf(jdbcType);
324     }
325 
326     TypeHandler typeHandler = null;
327     if (javaClass != null) {
328       typeHandler = config.getTypeHandlerRegistry().getTypeHandler(javaClass, jdbcTypeEnum);
329     }
330     try {
331       if (callback != null && callback.length() > 0) {
332         Object o = config.getTypeAliasRegistry().resolveAlias(callback).newInstance();
333         if (o instanceof TypeHandlerCallback) {
334           typeHandler = new TypeHandlerCallbackAdapter((TypeHandlerCallback) o);
335         }
336       }
337     } catch (Exception e) {
338       throw new RuntimeException("Error occurred during custom type handler configuration.  Cause: " + e, e);
339     }
340     if (typeHandler == null && config.getTypeHandlerRegistry().hasTypeHandler(resultMapBuilder.type())) {
341       typeHandler = config.getTypeHandlerRegistry().getTypeHandler(resultMapBuilder.type());
342     }
343 
344     if (typeHandler == null) {
345       Class resultClass = resultMapBuilder.type();
346       if (resultClass != null && !Map.class.isAssignableFrom(resultClass)) {
347         MetaClass metaResultClass = MetaClass.forClass(resultClass);
348         Class resultType = null;
349         if (metaResultClass.hasGetter(propertyName)) {
350           resultType = metaResultClass.getGetterType(propertyName);
351         } else if (metaResultClass.hasSetter(propertyName)) {
352           resultType = metaResultClass.getSetterType(propertyName);
353         }
354         if (resultType != null) {
355           typeHandler = config.getTypeHandlerRegistry().getTypeHandler(resultType);
356         }
357       } else {
358         typeHandler = config.getTypeHandlerRegistry().getUnknownTypeHandler();
359       }
360     }
361 
362     List<ResultMapping> composites = parseCompositeColumnName(columnName);
363     if (composites.size() > 0) {
364       ResultMapping first = composites.get(0);
365       columnName = first.getColumn();
366     }
367 
368     ResultMapping.Builder resultMappingBuilder = new ResultMapping.Builder(config, propertyName, columnName, typeHandler);
369     resultMappingBuilder.javaType(javaClass);
370     resultMappingBuilder.nestedQueryId(statementName);
371     resultMappingBuilder.nestedResultMapId(resultMapName);
372     resultMappingBuilder.jdbcType(jdbcTypeEnum);
373     resultMappingBuilder.composites(composites);
374     if (groupByProperties != null && groupByProperties.contains(propertyName)) {
375       List<ResultFlag> flags = new ArrayList<ResultFlag>();
376       resultMappingBuilder.flags(flags);
377     }
378     resultMappingList.add(resultMappingBuilder.build());
379   }

381   private List<ResultMapping> parseCompositeColumnName(String columnName) {
382     List<ResultMapping> composites = new ArrayList<ResultMapping>();
383     if (columnName != null) {
384       if (columnName.indexOf('=') > -1
385           || columnName.indexOf(',') > -1) {
386         StringTokenizer parser = new StringTokenizer(columnName, "{}=, ", false);
387         while (parser.hasMoreTokens()) {
388           String property = parser.nextToken();
389           String column = parser.nextToken();
390           ResultMapping.Builder complexBuilder = new ResultMapping.Builder(config, property, column, config.getTypeHandlerRegistry().getUnknownTypeHandler());
391           composites.add(complexBuilder.build());
392         }
393       }
394     }
395     return composites;
396   }

413   public void sqlMapparameterMapparameter(XNode context) throws Exception {
414     String nullValue = context.getStringAttribute("nullValue");
415     if (nullValue != null) {
416       throw new UnsupportedOperationException("Null value subsitution is not supported by iBATIS 3.");
417     }
418 
419     String propertyName = context.getStringAttribute("property");
420     String jdbcType = context.getStringAttribute("jdbcType");
421     String javaType = context.getStringAttribute("javaType");
422     String resultMap = context.getStringAttribute("resultMap");
423     String mode = context.getStringAttribute("mode");
424     String callback = context.getStringAttribute("typeHandler");
425     String numericScaleProp = context.getStringAttribute("numericScale");
426 
427     Class javaClass = null;
428     try {
429       if (javaType != null && javaType.length() > 0) {
430         javaClass = config.getTypeAliasRegistry().resolveAlias(javaType);
431       }
432     } catch (Exception e) {
433       throw new RuntimeException("Error setting javaType on parameter mapping.  Cause: " + e);
434     }
435 
436     JdbcType jdbcTypeEnum = null;
437     if (jdbcType != null) {
438       jdbcTypeEnum = JdbcType.valueOf(jdbcType);
439     }
440 
441     TypeHandler typeHandler = null;
442     if (javaClass != null) {
443       typeHandler = config.getTypeHandlerRegistry().getTypeHandler(javaClass, jdbcTypeEnum);
444     }
445     if (callback != null) {
446       Object o = config.getTypeAliasRegistry().resolveAlias(callback).newInstance();
447       if (o instanceof TypeHandlerCallback) {
448         typeHandler = new TypeHandlerCallbackAdapter((TypeHandlerCallback) o);
449       }
450     }
451     if (typeHandler == null && config.getTypeHandlerRegistry().hasTypeHandler(parameterMapBuilder.type())) {
452       typeHandler = config.getTypeHandlerRegistry().getTypeHandler(parameterMapBuilder.type());
453     }
454     if (typeHandler == null) {
455       Class parameterClass = parameterMapBuilder.type();
456       if (parameterClass != null && !Map.class.isAssignableFrom(parameterClass)) {
457         MetaClass metaParamClass = MetaClass.forClass(parameterClass);
458         Class paramType = null;
459         if (metaParamClass.hasGetter(propertyName)) {
460           paramType = metaParamClass.getGetterType(propertyName);
461         } else if (metaParamClass.hasSetter(propertyName)) {
462           paramType = metaParamClass.getSetterType(propertyName);
463         }
464         if (paramType != null) {
465           typeHandler = config.getTypeHandlerRegistry().getTypeHandler(paramType);
466         }
467       } else {
468         typeHandler = config.getTypeHandlerRegistry().getUnknownTypeHandler();
469       }
470     }
471 
472     ParameterMode paramModeEnum = ParameterMode.IN;
473     if (mode != null) {
474       paramModeEnum = ParameterMode.valueOf(mode);
475     }
476 
477     Integer numericScale = null;
478     if (numericScaleProp != null) {
479       numericScale = new Integer(numericScaleProp);
480     }
481 
482     ParameterMapping.Builder parameterMappingBuilder = new ParameterMapping.Builder(config, propertyName, typeHandler);
483     parameterMappingBuilder.javaType(javaClass);
484     parameterMappingBuilder.jdbcType(jdbcTypeEnum);
485     parameterMappingBuilder.mode(paramModeEnum);
486     parameterMappingBuilder.numericScale(numericScale);
487     parameterMappingBuilder.resultMapId(resultMap);
488 
489     parameterMappingList.add(parameterMappingBuilder.build());
490   }

35  public class Ibatis2Configuration extends Configuration {
36  
37    private TransactionManager transactionManager;
38  
39    private DataSource dataSource;
40  
41  
42    private FlushCacheInterceptor flushCachePlugin;
43  
44    private Map<String, Boolean> postSelectKeyMap;
45  
46    public Ibatis2Configuration() {
47      setAutoMappingBehavior(AutoMappingBehavior.FULL);
48      setUseGeneratedKeys(false);
49      this.flushCachePlugin = new FlushCacheInterceptor();
50      this.addInterceptor(flushCachePlugin);
51      this.postSelectKeyMap = new HashMap<String, Boolean>();
52      registerDefaultTypeAliases();
53    }
54  
55    private void registerDefaultTypeAliases() {
56      // TRANSACTION ALIASES
57      // changed to JDBC2 because it collides with the default typealias defined in mb3
58      this.getTypeAliasRegistry().registerAlias("JDBC2", JdbcTransactionConfig.class.getName());
59      this.getTypeAliasRegistry().registerAlias("JTA", JtaTransactionConfig.class.getName());
60      this.getTypeAliasRegistry().registerAlias("EXTERNAL", ExternalTransactionConfig.class.getName());
61  
62      // DATA SOURCE ALIASES
63      this.getTypeAliasRegistry().registerAlias("SIMPLE", SimpleDataSourceFactory.class.getName());
64      this.getTypeAliasRegistry().registerAlias("DBCP", DbcpDataSourceFactory.class.getName());
65  //    this.getTypeAliasRegistry().registerAlias("JNDI", JndiDataSourceFactory.class.getName());
66  
67      // CACHE ALIASES
68  //    this.getTypeAliasRegistry().registerAlias("FIFO", FifoCache.class.getName());
69  //    this.getTypeAliasRegistry().registerAlias("LRU", LruCache.class.getName());
70      this.getTypeAliasRegistry().registerAlias("MEMORY", SoftCache.class.getName());
71    }
72  
73    public TransactionManager getTransactionManager() {
74      return transactionManager;
75    }
76  
77    public void setTransactionManager(TransactionManager transactionManager) {
78      this.transactionManager = transactionManager;
79    }
80  
81    public DataSource getDataSource() {
82      return dataSource;
83    }
84  
85    public void setDataSource(DataSource dataSource) {
86      this.dataSource = dataSource;
87    }
88  
89    public FlushCacheInterceptor getFlushCachePlugin() {
90      return flushCachePlugin;
91    }
92  
93    public void setPostSelectKey(String statement, boolean postSelectKey) {
94      this.postSelectKeyMap.put(statement, postSelectKey);
95    }
96  
97    public boolean isPostSelectKey(String statement) {
98      Boolean postSelectKey = postSelectKeyMap.get(statement);
99      return postSelectKey == null || postSelectKey;
100   }

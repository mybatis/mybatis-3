37    public int doStartFragment(SqlTagContext ctx, SqlTag tag, Object parameterObject) {
38  
39      ctx.pushRemoveFirstPrependMarker(tag);
40  
41      if (isCondition(ctx, tag, parameterObject)) {
42        return INCLUDE_BODY;
43      } else {
44        return SKIP_BODY;
45      }

48    public int doEndFragment(SqlTagContext ctx, SqlTag tag, Object parameterObject, StringBuffer bodyContent) {
49  
50      IterateContext iterate = ctx.peekIterateContext();
51  
52      if (null != iterate && iterate.isAllowNext()) {
53        iterate.next();
54        iterate.setAllowNext(false);
55        if (!iterate.hasNext()) {
56          iterate.setFinal(true);
57        }
58      }
59  
60      //iteratePropertyReplace(bodyContent,ctx.peekIterateContext());
61  
62      return super.doEndFragment(ctx, tag, parameterObject, bodyContent);
63    }

65    protected long compare(SqlTagContext ctx, SqlTag tag, Object parameterObject) {
66      String comparePropertyName = tag.getComparePropertyAttr();
67      String compareValue = tag.getCompareValueAttr();
68  
69      String prop = getResolvedProperty(ctx, tag);
70      Object value1;
71      Class type;
72  
73      MetaObject metaObject = SystemMetaObject.forObject(parameterObject);
74      if (prop != null) {
75        value1 = metaObject.getValue(prop);
76        type = metaObject.getGetterType(prop);
77      } else {
78        value1 = parameterObject;
79        if (value1 != null) {
80          type = parameterObject.getClass();
81        } else {
82          type = Object.class;
83        }
84      }
85      if (comparePropertyName != null) {
86        Object value2 = metaObject.getValue(comparePropertyName);
87        return compareValues(type, value1, value2);
88      } else if (compareValue != null) {
89        return compareValues(type, value1, compareValue);
90      } else {
91        throw new RuntimeException("Error comparing in conditional fragment.  Uknown 'compare to' values.");
92      }
93    }
94  
95    protected long compareValues(Class type, Object value1, Object value2) {
96      long result;
97  
98      if (value1 == null || value2 == null) {
99        result = value1 == value2 ? 0 : NOT_COMPARABLE;
100     } else {
101       if (value2.getClass() != type) {
102         value2 = convertValue(type, value2.toString());
103       }
104       if (value2 instanceof String && type != String.class) {
105         value1 = value1.toString();
106       }
107       if (!(value1 instanceof Comparable && value2 instanceof Comparable)) {
108         value1 = value1.toString();
109         value2 = value2.toString();
110       }
111       result = ((Comparable) value1).compareTo(value2);
112     }
113 
114     return result;
115   }

117   protected Object convertValue(Class type, String value) {
118     if (type == String.class) {
119       return value;
120     } else if (type == Byte.class || type == byte.class) {
121       return Byte.valueOf(value);
122     } else if (type == Short.class || type == short.class) {
123       return Short.valueOf(value);
124     } else if (type == Character.class || type == char.class) {
125       return value.charAt(0);
126     } else if (type == Integer.class || type == int.class) {
127       return Integer.valueOf(value);
128     } else if (type == Long.class || type == long.class) {
129       return Long.valueOf(value);
130     } else if (type == Float.class || type == float.class) {
131       return Float.valueOf(value);
132     } else if (type == Double.class || type == double.class) {
133       return Double.valueOf(value);
134     } else if (type == Boolean.class || type == boolean.class) {
135       return Boolean.valueOf(value);
136     } else if (type == Date.class) {
137       return format(DATE_MASK, value);
138     } else if (type == BigInteger.class) {
139       return new BigInteger(value);
140     } else if (type == BigDecimal.class) {
141       return new BigDecimal(value);
142     } else {
143       return value;
144     }
145 
146   }

148   private static Date format(String format, String datetime) {
149     try {
150       return new SimpleDateFormat(format).parse(datetime);
151     } catch (ParseException e) {
152       throw new SqlMapException("Error parsing default null value date.  Format must be '" + format + "'. Cause: " + e);
153     }
154   }

156   /*
157    * This method will add the proper index values to an indexed property
158    * string if we are inside an iterate tag
159    *
160    * @param ctx
161    * @param tag
162    */
163   protected String getResolvedProperty(SqlTagContext ctx, SqlTag tag) {
164     String prop = tag.getPropertyAttr();
165     IterateContext itCtx = ctx.peekIterateContext();
166 
167     if (prop != null) {
168 
169       if (null != itCtx && itCtx.isAllowNext()) {
170         itCtx.next();
171         itCtx.setAllowNext(false);
172         if (!itCtx.hasNext()) {
173           itCtx.setFinal(true);
174         }
175       }
176 
177       if (prop.indexOf(START_INDEX) > -1) {
178         if (itCtx != null) {
179           prop = itCtx.addIndexToTagProperty(prop);
180         }
181       }
182     }
183 
184     return prop;
185   }

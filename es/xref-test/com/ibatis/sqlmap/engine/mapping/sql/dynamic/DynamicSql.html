34  public class DynamicSql implements Sql, DynamicParent {

36    private List children = new ArrayList();

38    private ThreadLocal<SqlTagContext> sqlTagContext = new ThreadLocal<SqlTagContext>();

40    private Configuration configuration;
41    private TypeHandlerRegistry typeHandlerRegistry;
42    private List<ParameterMapping> parameterMappings;

44    public DynamicSql(Configuration configuration) {
45      this.parameterMappings = new ArrayList<ParameterMapping>();
46      this.configuration = configuration;
47      this.typeHandlerRegistry = configuration.getTypeHandlerRegistry();
48    }

50    public String getSql(Object parameterObject) {
51      if (sqlTagContext.get() == null) {
52        sqlTagContext.set(process(parameterObject));
53      }
54      return sqlTagContext.get().getDynamicSql();
55    }

57    public List<ParameterMapping> getParameterMappings(Object parameterObject) {
58      if (sqlTagContext.get() == null) {
59        sqlTagContext.set(process(parameterObject));
60      }
61      return sqlTagContext.get().getParameterMappings();
62    }

64    private SqlTagContext process(Object parameterObject) {
65      SqlTagContext ctx = new SqlTagContext();
66      List localChildren = children;
67      processBodyChildren(ctx, parameterObject, localChildren.iterator());
68  
69      String dynSql = ctx.getBodyText();
70  
71      // Processes $substitutions$ after DynamicSql
72      if (SimpleDynamicSql.isSimpleDynamicSql(dynSql)) {
73        dynSql = new SimpleDynamicSql(dynSql, ctx.getParameterMappings(), typeHandlerRegistry).getSql(parameterObject);
74      }
75  
76      ctx.setDynamicSql(dynSql);
77      for (ParameterMapping mapping : parameterMappings) {
78        ctx.addParameterMapping(mapping);
79      }
80      return ctx;
81    }

83    private void processBodyChildren(SqlTagContext ctx, Object parameterObject, Iterator localChildren) {
84      PrintWriter out = ctx.getWriter();
85      processBodyChildren(ctx, parameterObject, localChildren, out);
86    }

88    private void processBodyChildren(SqlTagContext ctx, Object parameterObject, Iterator localChildren, PrintWriter out) {
89      while (localChildren.hasNext()) {
90        SqlChild child = (SqlChild) localChildren.next();
91        if (child instanceof SqlText) {
92          SqlText sqlText = (SqlText) child;
93          String sqlStatement = sqlText.getText();
94          if (sqlText.isWhiteSpace()) {
95            out.print(sqlStatement);
96          } else if (!sqlText.isPostParseRequired()) {
97  
98            // BODY OUT
99            out.print(sqlStatement);
100 
101           List<ParameterMapping> mappings = sqlText.getParameterMappings();
102           if (mappings != null) {
103             for (int i = 0, n = mappings.size(); i < n; i++) {
104               ctx.addParameterMapping(mappings.get(i));
105             }
106           }
107         } else {
108 
109           IterateContext itCtx = ctx.peekIterateContext();
110 
111           if (null != itCtx && itCtx.isAllowNext()) {
112             itCtx.next();
113             itCtx.setAllowNext(false);
114             if (!itCtx.hasNext()) {
115               itCtx.setFinal(true);
116             }
117           }
118 
119           if (itCtx != null) {
120             StringBuffer sqlStatementBuffer = new StringBuffer(sqlStatement);
121             iteratePropertyReplace(sqlStatementBuffer, itCtx);
122             sqlStatement = sqlStatementBuffer.toString();
123           }
124 
125           sqlText = new InlineParameterMapParser(configuration).parseInlineParameterMap(sqlStatement);
126 
127           List<ParameterMapping> mappings = sqlText.getParameterMappings();
128           out.print(sqlText.getText());
129           if (mappings != null) {
130             for (int i = 0, n = mappings.size(); i < n; i++) {
131               ctx.addParameterMapping(mappings.get(i));
132             }
133           }
134         }
135       } else if (child instanceof SqlTag) {
136         SqlTag tag = (SqlTag) child;
137         SqlTagHandler handler = tag.getHandler();
138         int response;
139         do {
140           StringWriter sw = new StringWriter();
141           PrintWriter pw = new PrintWriter(sw);
142 
143           response = handler.doStartFragment(ctx, tag, parameterObject);
144           if (response != SqlTagHandler.SKIP_BODY) {
145 
146             processBodyChildren(ctx, parameterObject, tag.getChildren(), pw);
147             pw.flush();
148             pw.close();
149             StringBuffer body = sw.getBuffer();
150             response = handler.doEndFragment(ctx, tag, parameterObject, body);
151             handler.doPrepend(ctx, tag, parameterObject, body);
152 
153             if (response != SqlTagHandler.SKIP_BODY) {
154               if (body.length() > 0) {
155                 out.print(body.toString());
156               }
157             }
158 
159           }
160         } while (response == SqlTagHandler.REPEAT_BODY);
161 
162         ctx.popRemoveFirstPrependMarker(tag);
163 
164         if (ctx.peekIterateContext() != null && ctx.peekIterateContext().getTag() == tag) {
165           ctx.setAttribute(ctx.peekIterateContext().getTag(), null);
166           ctx.popIterateContext();
167         }
168 
169       }
170     }
171   }

173   protected void iteratePropertyReplace(StringBuffer bodyContent, IterateContext iterate) {
174     if (iterate != null) {
175       String[] mappings = new String[]{"#", "$"};
176       for (String mapping : mappings) {
177         int startIndex = 0;
178         int endIndex = -1;
179         while (startIndex > -1 && startIndex < bodyContent.length()) {
180           startIndex = bodyContent.indexOf(mapping, endIndex + 1);
181           endIndex = bodyContent.indexOf(mapping, startIndex + 1);
182           if (startIndex > -1 && endIndex > -1) {
183             String replacement = iterate.addIndexToTagProperty(bodyContent.substring(startIndex + 1, endIndex));
184             bodyContent.replace(startIndex + 1, endIndex, replacement);
185           }
186         }
187       }
188     }
189   }

191   protected static void replace(StringBuffer buffer, String find, String replace) {
192     int pos = buffer.toString().indexOf(find);
193     int len = find.length();
194     while (pos > -1) {
195       buffer.replace(pos, pos + len, replace);
196       pos = buffer.toString().indexOf(find);
197     }
198   }

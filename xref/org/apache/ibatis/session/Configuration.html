449   public Executor newExecutor(Transaction transaction, ExecutorType executorType, boolean autoCommit) {
450     executorType = executorType == null ? defaultExecutorType : executorType;
451     executorType = executorType == null ? ExecutorType.SIMPLE : executorType;
452     Executor executor;
453     if (ExecutorType.BATCH == executorType) {
454       executor = new BatchExecutor(this, transaction);
455     } else if (ExecutorType.REUSE == executorType) {
456       executor = new ReuseExecutor(this, transaction);
457     } else {
458       executor = new SimpleExecutor(this, transaction);
459     }
460     if (cacheEnabled) {
461       executor = new CachingExecutor(executor, autoCommit);
462     }
463     executor = (Executor) interceptorChain.pluginAll(executor);
464     return executor;
465   }

654   /*
655    * Parses all the unprocessed statement nodes in the cache. It is recommended
656    * to call this method once all the mappers are added as it provides fail-fast
657    * statement validation.
658    */
659   protected void buildAllStatements() {
660     if (!incompleteResultMaps.isEmpty()) {
661       synchronized (incompleteResultMaps) {
662         // This always throws a BuilderException.
663         incompleteResultMaps.iterator().next().resolve();
664       }
665     }
666     if (!incompleteCacheRefs.isEmpty()) {
667       synchronized (incompleteCacheRefs) {
668         // This always throws a BuilderException.
669         incompleteCacheRefs.iterator().next().resolveCacheRef();
670       }
671     }
672     if (!incompleteStatements.isEmpty()) {
673       synchronized (incompleteStatements) {
674         // This always throws a BuilderException.
675         incompleteStatements.iterator().next().parseStatementNode();
676       }
677     }
678   }

680   /*
681    * Extracts namespace from fully qualified statement id.
682    *
683    * @param statementId
684    * @return namespace or null when id does not contain period.
685    */
686   protected String extractNamespace(String statementId) {
687     int lastPeriod = statementId.lastIndexOf('.');
688     return lastPeriod > 0 ? statementId.substring(0, lastPeriod) : null;
689   }

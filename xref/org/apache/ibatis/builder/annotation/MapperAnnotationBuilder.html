110   public void parse() {
111     String resource = type.toString();
112     if (!configuration.isResourceLoaded(resource)) {
113       loadXmlResource();
114       configuration.addLoadedResource(resource);
115       assistant.setCurrentNamespace(type.getName());
116       parseCache();
117       parseCacheRef();
118       Method[] methods = type.getMethods();
119       for (Method method : methods) {
120         try {
121           parseStatement(method);
122         } catch (IncompleteElementException e) {
123           configuration.addIncompleteMethod(new MethodResolver(this, method));
124         }
125       }
126     }
127     parsePendingMethods();
128   }

130   private void parsePendingMethods() {
131     Collection<MethodResolver> incompleteMethods = configuration.getIncompleteMethods();
132     synchronized (incompleteMethods) {
133       Iterator<MethodResolver> iter = incompleteMethods.iterator();
134       while (iter.hasNext()) {
135         try {
136           iter.next().resolve();
137           iter.remove();
138         } catch (IncompleteElementException e) {
139           // This method is still missing a resource
140         }
141       }
142     }
143   }

145   private void loadXmlResource() {
146     // Spring may not know the real resource name so we check a flag
147     // to prevent loading again a resource twice
148     // this flag is set at XMLMapperBuilder#bindMapperForNamespace
149     if (!configuration.isResourceLoaded("namespace:" + type.getName())) {
150       String xmlResource = type.getName().replace('.', '/') + ".xml";
151       InputStream inputStream = null;
152       try {
153         inputStream = Resources.getResourceAsStream(type.getClassLoader(), xmlResource);
154       } catch (IOException e) {
155         // ignore, resource is not required
156       }
157       if (inputStream != null) {
158         XMLMapperBuilder xmlParser = new XMLMapperBuilder(inputStream, assistant.getConfiguration(), xmlResource, configuration.getSqlFragments(), type.getName());
159         xmlParser.parse();
160       }
161     }
162   }

178   private String parseResultMap(Method method) {
179     Class<?> returnType = getReturnType(method);
180     ConstructorArgs args = method.getAnnotation(ConstructorArgs.class);
181     Results results = method.getAnnotation(Results.class);
182     TypeDiscriminator typeDiscriminator = method.getAnnotation(TypeDiscriminator.class);
183     String resultMapId = generateResultMapName(method);
184     applyResultMap(resultMapId, returnType, argsIf(args), resultsIf(results), typeDiscriminator);
185     return resultMapId;
186   }

188   private String generateResultMapName(Method method) {
189     StringBuilder suffix = new StringBuilder();
190     for (Class<?> c : method.getParameterTypes()) {
191       suffix.append("-");
192       suffix.append(c.getSimpleName());
193     }
194     if (suffix.length() < 1) {
195       suffix.append("-void");
196     }
197     return type.getName() + "." + method.getName() + suffix;
198   }

200   private void applyResultMap(String resultMapId, Class<?> returnType, Arg[] args, Result[] results, TypeDiscriminator discriminator) {
201     List<ResultMapping> resultMappings = new ArrayList<ResultMapping>();
202     applyConstructorArgs(args, returnType, resultMappings);
203     applyResults(results, returnType, resultMappings);
204     Discriminator disc = applyDiscriminator(resultMapId, returnType, discriminator);
205     assistant.addResultMap(resultMapId, returnType, null, disc, resultMappings, null); // TODO add AutoMappingBehaviour
206     createDiscriminatorResultMaps(resultMapId, returnType, discriminator);
207   }

209   private void createDiscriminatorResultMaps(String resultMapId, Class<?> resultType, TypeDiscriminator discriminator) {
210     if (discriminator != null) {
211       for (Case c : discriminator.cases()) {
212         String caseResultMapId = resultMapId + "-" + c.value();
213         List<ResultMapping> resultMappings = new ArrayList<ResultMapping>();
214         applyConstructorArgs(c.constructArgs(), resultType, resultMappings); // issue #136
215         applyResults(c.results(), resultType, resultMappings);
216         assistant.addResultMap(caseResultMapId, c.type(), resultMapId, null, resultMappings, null); // TODO add AutoMappingBehaviour
217       }
218     }

221   private Discriminator applyDiscriminator(String resultMapId, Class<?> resultType, TypeDiscriminator discriminator) {
222     if (discriminator != null) {
223       String column = discriminator.column();
224       Class<?> javaType = discriminator.javaType() == void.class ? String.class : discriminator.javaType();
225       JdbcType jdbcType = discriminator.jdbcType() == JdbcType.UNDEFINED ? null : discriminator.jdbcType();
226       Class<? extends TypeHandler<?>> typeHandler = discriminator.typeHandler() == UnknownTypeHandler.class ? null : discriminator.typeHandler();
227       Case[] cases = discriminator.cases();
228       Map<String, String> discriminatorMap = new HashMap<String, String>();
229       for (Case c : cases) {
230         String value = c.value();
231         String caseResultMapId = resultMapId + "-" + value;
232         discriminatorMap.put(value, caseResultMapId);
233       }
234       return assistant.buildDiscriminator(resultType, column, javaType, jdbcType, typeHandler, discriminatorMap);
235     }
236     return null;
237   }

239   void parseStatement(Method method) {
240     Class<?> parameterTypeClass = getParameterType(method);
241     LanguageDriver languageDriver = getLanguageDriver(method);
242     SqlSource sqlSource = getSqlSourceFromAnnotations(method, parameterTypeClass, languageDriver);
243     if (sqlSource != null) {
244       Options options = method.getAnnotation(Options.class);
245       final String mappedStatementId = type.getName() + "." + method.getName();
246       Integer fetchSize = null;
247       Integer timeout = null;
248       StatementType statementType = StatementType.PREPARED;
249       ResultSetType resultSetType = ResultSetType.FORWARD_ONLY;
250       SqlCommandType sqlCommandType = getSqlCommandType(method);
251       boolean isSelect = sqlCommandType == SqlCommandType.SELECT;
252       boolean flushCache = !isSelect;
253       boolean useCache = isSelect;
254 
255       KeyGenerator keyGenerator;
256       String keyProperty = "id";
257       String keyColumn = null;
258       if (SqlCommandType.INSERT.equals(sqlCommandType)) {
259         // first check for SelectKey annotation - that overrides everything else
260         SelectKey selectKey = method.getAnnotation(SelectKey.class);
261         if (selectKey != null) {
262           keyGenerator = handleSelectKeyAnnotation(selectKey, mappedStatementId, getParameterType(method), languageDriver);
263           keyProperty = selectKey.keyProperty();
264         } else {
265           if (options == null) {
266             keyGenerator = configuration.isUseGeneratedKeys() ? new Jdbc3KeyGenerator() : new NoKeyGenerator();
267           } else {
268             keyGenerator = options.useGeneratedKeys() ? new Jdbc3KeyGenerator() : new NoKeyGenerator();
269             keyProperty = options.keyProperty();
270             keyColumn = options.keyColumn();
271           }
272         }
273       } else {
274         keyGenerator = new NoKeyGenerator();
275       }
276 
277       if (options != null) {
278         flushCache = options.flushCache();
279         useCache = options.useCache();
280         fetchSize = options.fetchSize() > -1 || options.fetchSize() == Integer.MIN_VALUE ? options.fetchSize() : null; //issue #348
281         timeout = options.timeout() > -1 ? options.timeout() : null;
282         statementType = options.statementType();
283         resultSetType = options.resultSetType();
284       }
285 
286       String resultMapId = null;
287       ResultMap resultMapAnnotation = method.getAnnotation(ResultMap.class);
288       if (resultMapAnnotation != null) {
289         String[] resultMaps = resultMapAnnotation.value();
290         StringBuilder sb = new StringBuilder();
291         for (String resultMap : resultMaps) {
292           if (sb.length() > 0) sb.append(",");
293           sb.append(resultMap);
294         }
295         resultMapId = sb.toString();
296       } else if (isSelect) {
297         resultMapId = parseResultMap(method);
298       }
299 
300       assistant.addMappedStatement(
301           mappedStatementId,
302           sqlSource,
303           statementType,
304           sqlCommandType,
305           fetchSize,
306           timeout,
307           null,                             // ParameterMapID
308           parameterTypeClass,
309           resultMapId,    // ResultMapID
310           getReturnType(method),
311           resultSetType,
312           flushCache,
313           useCache,
314           false, // TODO issue #577
315           keyGenerator,
316           keyProperty,
317           keyColumn,
318           null,
319           languageDriver,
320           null);
321     }
322   }

333   private Class<?> getParameterType(Method method) {
334     Class<?> parameterType = null;
335     Class<?>[] parameterTypes = method.getParameterTypes();
336     for (int i = 0; i < parameterTypes.length; i++) {
337       if (!RowBounds.class.isAssignableFrom(parameterTypes[i]) && !ResultHandler.class.isAssignableFrom(parameterTypes[i])) {
338         if (parameterType == null) {
339           parameterType = parameterTypes[i];
340         } else {
341           parameterType = Map.class;
342         }
343       }
344     }
345     return parameterType;
346   }

348   private Class<?> getReturnType(Method method) {
349     Class<?> returnType = method.getReturnType();
350     if (void.class.equals(returnType)) { // issue #508
351       ResultType rt = method.getAnnotation(ResultType.class);
352       if (rt != null) {
353         returnType = rt.value();
354       } 
355     } else if (Collection.class.isAssignableFrom(returnType)) {
356       Type returnTypeParameter = method.getGenericReturnType();
357       if (returnTypeParameter instanceof ParameterizedType) {
358         Type[] actualTypeArguments = ((ParameterizedType) returnTypeParameter).getActualTypeArguments();
359         if (actualTypeArguments != null && actualTypeArguments.length == 1) {
360           returnTypeParameter = actualTypeArguments[0];
361           if (returnTypeParameter instanceof Class) {
362             returnType = (Class<?>) returnTypeParameter;
363           } else if (returnTypeParameter instanceof ParameterizedType) { // (issue #443) actual type can be a also a parametrized type
364             returnType = (Class<?>) ((ParameterizedType) returnTypeParameter).getRawType();
365           } else if (returnTypeParameter instanceof GenericArrayType) {
366             Class<?> componentType = (Class<?>) ((GenericArrayType) returnTypeParameter).getGenericComponentType();
367             returnType = Array.newInstance(componentType, 0).getClass(); // (issue #525) support List<byte[]>
368           }
369         }
370       }
371     } else if (method.isAnnotationPresent(MapKey.class) && Map.class.isAssignableFrom(returnType)) {
372       // (issue 504) Do not look into Maps if there is not MapKey annotation
373       Type returnTypeParameter = method.getGenericReturnType();
374       if (returnTypeParameter instanceof ParameterizedType) {
375         Type[] actualTypeArguments = ((ParameterizedType) returnTypeParameter).getActualTypeArguments();
376         if (actualTypeArguments != null && actualTypeArguments.length == 2) {
377           returnTypeParameter = actualTypeArguments[1];
378           if (returnTypeParameter instanceof Class) {
379             returnType = (Class<?>) returnTypeParameter;
380           } else if (returnTypeParameter instanceof ParameterizedType) { // (issue 443) actual type can be a also a parametrized type
381             returnType = (Class<?>) ((ParameterizedType) returnTypeParameter).getRawType();
382           }
383         }
384       }
385     }
386 
387     return returnType;
388   }

390   private SqlSource getSqlSourceFromAnnotations(Method method, Class<?> parameterType, LanguageDriver languageDriver) {
391     try {
392       Class<? extends Annotation> sqlAnnotationType = getSqlAnnotationType(method);
393       Class<? extends Annotation> sqlProviderAnnotationType = getSqlProviderAnnotationType(method);
394       if (sqlAnnotationType != null) {
395         if (sqlProviderAnnotationType != null) {
396           throw new BindingException("You cannot supply both a static SQL and SqlProvider to method named " + method.getName());
397         }
398         Annotation sqlAnnotation = method.getAnnotation(sqlAnnotationType);
399         final String[] strings = (String[]) sqlAnnotation.getClass().getMethod("value").invoke(sqlAnnotation);
400         return buildSqlSourceFromStrings(strings, parameterType, languageDriver);
401       } else if (sqlProviderAnnotationType != null) {
402         Annotation sqlProviderAnnotation = method.getAnnotation(sqlProviderAnnotationType);
403         return new ProviderSqlSource(assistant.getConfiguration(), sqlProviderAnnotation);
404       }
405       return null;
406     } catch (Exception e) {
407       throw new BuilderException("Could not find value method on SQL annotation.  Cause: " + e, e);
408     }
409   }

411   private SqlSource buildSqlSourceFromStrings(String[] strings, Class<?> parameterTypeClass, LanguageDriver languageDriver) {
412     final StringBuilder sql = new StringBuilder();
413     for (String fragment : strings) {
414       sql.append(fragment);
415       sql.append(" ");
416     }
417     return languageDriver.createSqlSource(configuration, sql.toString(), parameterTypeClass);
418   }

420   private SqlCommandType getSqlCommandType(Method method) {
421     Class<? extends Annotation> type = getSqlAnnotationType(method);
422 
423     if (type == null) {
424       type = getSqlProviderAnnotationType(method);
425 
426       if (type == null) {
427         return SqlCommandType.UNKNOWN;
428       }
429 
430       if (type == SelectProvider.class) {
431         type = Select.class;
432       } else if (type == InsertProvider.class) {
433         type = Insert.class;
434       } else if (type == UpdateProvider.class) {
435         type = Update.class;
436       } else if (type == DeleteProvider.class) {
437         type = Delete.class;
438       }
439     }
440 
441     return SqlCommandType.valueOf(type.getSimpleName().toUpperCase(Locale.ENGLISH));
442   }

462   private void applyResults(Result[] results, Class<?> resultType, List<ResultMapping> resultMappings) {
463     for (Result result : results) {
464       ArrayList<ResultFlag> flags = new ArrayList<ResultFlag>();
465       if (result.id()) flags.add(ResultFlag.ID);
466       ResultMapping resultMapping = assistant.buildResultMapping(
467           resultType,
468           nullOrEmpty(result.property()),
469           nullOrEmpty(result.column()),
470           result.javaType() == void.class ? null : result.javaType(),
471           result.jdbcType() == JdbcType.UNDEFINED ? null : result.jdbcType(),
472           hasNestedSelect(result) ? nestedSelectId(result) : null,
473           null,
474           null,
475           null,
476           result.typeHandler() == UnknownTypeHandler.class ? null : result.typeHandler(),
477           flags,
478           null,
479           null);
480       resultMappings.add(resultMapping);
481     }
482   }

534   private KeyGenerator handleSelectKeyAnnotation(SelectKey selectKeyAnnotation, String baseStatementId, Class<?> parameterTypeClass, LanguageDriver languageDriver) {
535     String id = baseStatementId + SelectKeyGenerator.SELECT_KEY_SUFFIX;
536     Class<?> resultTypeClass = selectKeyAnnotation.resultType();
537     StatementType statementType = selectKeyAnnotation.statementType();
538     String keyProperty = selectKeyAnnotation.keyProperty();
539     boolean executeBefore = selectKeyAnnotation.before();
540 
541     // defaults
542     boolean useCache = false;
543     KeyGenerator keyGenerator = new NoKeyGenerator();
544     Integer fetchSize = null;
545     Integer timeout = null;
546     boolean flushCache = false;
547     String parameterMap = null;
548     String resultMap = null;
549     ResultSetType resultSetTypeEnum = null;
550 
551     SqlSource sqlSource = buildSqlSourceFromStrings(selectKeyAnnotation.statement(), parameterTypeClass, languageDriver);
552     SqlCommandType sqlCommandType = SqlCommandType.SELECT;
553 
554     assistant.addMappedStatement(id, sqlSource, statementType, sqlCommandType, fetchSize, timeout, parameterMap, parameterTypeClass, resultMap, resultTypeClass, resultSetTypeEnum,
555         flushCache, useCache, false,
556         keyGenerator, keyProperty, null, null, languageDriver, null);
557 
558     id = assistant.applyCurrentNamespace(id, false);
559 
560     MappedStatement keyStatement = configuration.getMappedStatement(id, false);
561     SelectKeyGenerator answer = new SelectKeyGenerator(keyStatement, executeBefore);
562     configuration.addKeyGenerator(id, answer);
563     return answer;
564   }

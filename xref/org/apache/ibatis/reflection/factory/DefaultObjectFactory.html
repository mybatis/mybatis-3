33  public class DefaultObjectFactory implements ObjectFactory, Serializable {
34  
35    private static final long serialVersionUID = -8855120656740914948L;

37    public <T> T create(Class<T> type) {
38      return create(type, null, null);
39    }

41    public <T> T create(Class<T> type, List<Class<?>> constructorArgTypes, List<Object> constructorArgs) {
42      Class<?> classToCreate = resolveInterface(type);
43      @SuppressWarnings("unchecked")
44      // we know types are assignable
45      T created = (T) instantiateClass(classToCreate, constructorArgTypes, constructorArgs);
46      return created;
47    }
48  
49    public void setProperties(Properties properties) {
50      // no props for default
51    }

53    private <T> T instantiateClass(Class<T> type, List<Class<?>> constructorArgTypes, List<Object> constructorArgs) {
54      try {
55        Constructor<T> constructor;
56        if (constructorArgTypes == null || constructorArgs == null) {
57          constructor = type.getDeclaredConstructor();
58          if (!constructor.isAccessible()) {
59            constructor.setAccessible(true);
60          }
61          return constructor.newInstance();
62        }
63        constructor = type.getDeclaredConstructor(constructorArgTypes.toArray(new Class[constructorArgTypes.size()]));
64        if (!constructor.isAccessible()) {
65          constructor.setAccessible(true);
66        }
67        return constructor.newInstance(constructorArgs.toArray(new Object[constructorArgs.size()]));
68      } catch (Exception e) {
69        StringBuilder argTypes = new StringBuilder();
70        if (constructorArgTypes != null) {
71          for (Class<?> argType : constructorArgTypes) {
72            argTypes.append(argType.getSimpleName());
73            argTypes.append(",");
74          }
75        }
76        StringBuilder argValues = new StringBuilder();
77        if (constructorArgs != null) {
78          for (Object argValue : constructorArgs) {
79            argValues.append(String.valueOf(argValue));
80            argValues.append(",");
81          }
82        }
83        throw new ReflectionException("Error instantiating " + type + " with invalid types (" + argTypes + ") or values (" + argValues + "). Cause: " + e, e);
84      }
85    }
86  
87    protected Class<?> resolveInterface(Class<?> type) {
88      Class<?> classToCreate;
89      if (type == List.class || type == Collection.class || type == Iterable.class) {
90        classToCreate = ArrayList.class;
91      } else if (type == Map.class) {
92        classToCreate = HashMap.class;
93      } else if (type == SortedSet.class) { // issue #510 Collections Support
94        classToCreate = TreeSet.class;
95      } else if (type == Set.class) {
96        classToCreate = HashSet.class;
97      } else {
98        classToCreate = type;
99      }
100     return classToCreate;
101   }

103   public <T> boolean isCollection(Class<T> type) {
104     return Collection.class.isAssignableFrom(type);
105   }

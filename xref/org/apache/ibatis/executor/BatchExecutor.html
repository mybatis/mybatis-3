16  package org.apache.ibatis.executor;
17  
18  import java.sql.BatchUpdateException;
19  import java.sql.Connection;
20  import java.sql.SQLException;
21  import java.sql.Statement;
22  import java.util.ArrayList;
23  import java.util.Collections;
24  import java.util.List;
25  
26  import org.apache.ibatis.executor.keygen.Jdbc3KeyGenerator;
27  import org.apache.ibatis.executor.keygen.KeyGenerator;
28  import org.apache.ibatis.executor.statement.StatementHandler;
29  import org.apache.ibatis.mapping.BoundSql;
30  import org.apache.ibatis.mapping.MappedStatement;
31  import org.apache.ibatis.session.Configuration;
32  import org.apache.ibatis.session.ResultHandler;
33  import org.apache.ibatis.session.RowBounds;
34  import org.apache.ibatis.transaction.Transaction;
35  
36  public class BatchExecutor extends BaseExecutor {
37  
38    public static final int BATCH_UPDATE_RETURN_VALUE = Integer.MIN_VALUE + 1002;
39  
40    private final List<Statement> statementList = new ArrayList<Statement>();
41    private final List<BatchResult> batchResultList = new ArrayList<BatchResult>();
42    private String currentSql;
43    private MappedStatement currentStatement;
44  
45    public BatchExecutor(Configuration configuration, Transaction transaction) {
46      super(configuration, transaction);
47    }
48  
49    public int doUpdate(MappedStatement ms, Object parameterObject) throws SQLException {
50      final Configuration configuration = ms.getConfiguration();
51      final StatementHandler handler = configuration.newStatementHandler(this, ms, parameterObject, RowBounds.DEFAULT, null, null);
52      final BoundSql boundSql = handler.getBoundSql();
53      final String sql = boundSql.getSql();
54      final Statement stmt;
55      if (sql.equals(currentSql) && ms.equals(currentStatement)) {
56        int last = statementList.size() - 1;
57        stmt = statementList.get(last);
58        BatchResult batchResult = batchResultList.get(last);
59        batchResult.addParameterObject(parameterObject);
60      } else {
61        Connection connection = getConnection(ms.getStatementLog());
62        stmt = handler.prepare(connection);
63        currentSql = sql;
64        currentStatement = ms;
65        statementList.add(stmt);
66        batchResultList.add(new BatchResult(ms, sql, parameterObject));
67      }
68      handler.parameterize(stmt);
69      handler.batch(stmt);
70      return BATCH_UPDATE_RETURN_VALUE;
71    }
72  
73    public <E> List<E> doQuery(MappedStatement ms, Object parameterObject, RowBounds rowBounds, ResultHandler resultHandler, BoundSql boundSql)
74        throws SQLException {
75      Statement stmt = null;
76      try {
77        flushStatements();
78        Configuration configuration = ms.getConfiguration();
79        StatementHandler handler = configuration.newStatementHandler(this, ms, parameterObject, rowBounds, resultHandler, boundSql);
80        Connection connection = getConnection(ms.getStatementLog());
81        stmt = handler.prepare(connection);
82        handler.parameterize(stmt);
83        return handler.<E>query(stmt, resultHandler);
84      } finally {
85        closeStatement(stmt);
86      }
87    }
88  
89    public List<BatchResult> doFlushStatements(boolean isRollback) throws SQLException {
90      try {
91        List<BatchResult> results = new ArrayList<BatchResult>();
92        if (isRollback) {
93          return Collections.emptyList();
94        } else {
95          for (int i = 0, n = statementList.size(); i < n; i++) {
96            Statement stmt = statementList.get(i);
97            BatchResult batchResult = batchResultList.get(i);
98            try {
99              batchResult.setUpdateCounts(stmt.executeBatch());
100             MappedStatement ms = batchResult.getMappedStatement();
101             List<Object> parameterObjects = batchResult.getParameterObjects();
102             KeyGenerator keyGenerator = ms.getKeyGenerator();
103             if (keyGenerator instanceof Jdbc3KeyGenerator) {
104               Jdbc3KeyGenerator jdbc3KeyGenerator = (Jdbc3KeyGenerator) keyGenerator;
105               jdbc3KeyGenerator.processBatch(ms, stmt, parameterObjects);
106             } else {
107               for (Object parameter : parameterObjects) {
108                 keyGenerator.processAfter(this, ms, stmt, parameter);
109               }
110             }
111           } catch (BatchUpdateException e) {
112             StringBuffer message = new StringBuffer();
113             message.append(batchResult.getMappedStatement().getId())
114                 .append(" (batch index #")
115                 .append(i + 1)
116                 .append(")")
117                 .append(" failed.");
118             if (i > 0) {
119               message.append(" ")
120                   .append(i)
121                   .append(" prior sub executor(s) completed successfully, but will be rolled back.");
122             }
123             throw new BatchExecutorException(message.toString(), e, results, batchResult);
124           }
125           results.add(batchResult);
126         }
127         return results;
128       }
129     } finally {
130       for (Statement stmt : statementList) {
131         closeStatement(stmt);
132       }
133       currentSql = null;
134       statementList.clear();
135       batchResultList.clear();
136     }
137   }

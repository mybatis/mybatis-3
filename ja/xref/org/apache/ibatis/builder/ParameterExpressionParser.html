16  package org.apache.ibatis.builder;
17  
18  import java.util.HashMap;
19  import java.util.Map;
20  
21  /**
22   * Inline parameter expression parser.
23   * Supported grammar (simplified):
24   * <pre>
25   * inline-parameter = (propertyName | expression) oldJdbcType attributes
26   * propertyName = /expression language's property navigation path/
27   * expression = '(' /expression language's expression/ ')'
28   * oldJdbcType = ':' /any valid jdbc type/
29   * attributes = (',' attribute)*
30   * attribute = name '=' value
31   * </pre>
32   */
33  public class ParameterExpressionParser {
34  
35    public static Map<String, String> parse(String expression) {
36      Map<String, String> map = new HashMap<String, String>();
37      parse(expression.toCharArray(), 0, map);
38      return map;
39    }
40  
41    private static void parse(char[] expression, int p, Map<String, String> map) {
42      p = skipWS(expression, p);
43      if (expression[p] == '(') {
44        expression(expression, p + 1, map);
45      } else {
46        property(expression, p, map);
47      }
48    }
49  
50    private static void expression(char[] expression, int left, Map<String, String> map) {
51      int match = 1;
52      int right = left + 1;
53      while (match > 0) {
54        if (expression[right] == ')') {
55          match--;
56        } else if (expression[right] == '(') {
57          match++;
58        }
59        right++;
60      }
61      map.put("expression", new String(expression, left, right - left - 1));
62      jdbcTypeOpt(expression, right, map);
63    }
64  
65    private static void property(char[] expression, int p, Map<String, String> map) {
66      int left = skipWS(expression, p);
67      if (left < expression.length) {
68        int right = skipUntil(expression, left, ',', ':');
69        map.put("property", new String(expression, left, right - left));
70        jdbcTypeOpt(expression, right, map);
71      }
72    }
73  
74    private static int skipWS(char[] expression, int p) {
75      for (int i = p; i < expression.length; i++) {
76        char c = expression[i];
77        if (!isWhitespaceBreak(c) && c != ' ') {
78          return i;
79        }
80      }
81      return expression.length;
82    }
83  
84    private static int skipUntil(char[] expression, int p, final char to) {
85      for (int i = p; i < expression.length; i++) {
86        char c = expression[i];
87        if (isWhitespaceBreak(c) || c == to) {
88          return i;
89        }
90      }
91      return expression.length;
92    }
93  
94    private static int skipUntil(char[] expression, int p, char to1, char to2) {
95      for (int i = p; i < expression.length; i++) {
96        char c = expression[i];
97        if (isWhitespaceBreak(c) || c == to1 || c == to2) {
98          return i;
99        }
100     }
101     return expression.length;
102   }
103 
104   private static boolean isWhitespaceBreak(char c) {
105     return c == '\t' || c == '\n' || c == '\r';
106   }
107 
108   private static void jdbcTypeOpt(char[] expression, int p, Map<String, String> map) {
109     p = skipWS(expression, p);
110     if (p < expression.length) {
111       if (expression[p] == ':') {
112         jdbcType(expression, p + 1, map);
113       } else if (expression[p] == ',') {
114         option(expression, p + 1, map);
115       } else {
116         throw new BuilderException("Parsing error in {" + new String(expression) + "} in position " + p);
117       }
118     }
119   }
120 
121   private static void jdbcType(char[] expression, int p, Map<String, String> map) {
122     int left = skipWS(expression, p);
123     int right = skipUntil(expression, left, ',');
124     if (right > left) {
125       map.put("jdbcType", new String(expression, left, right - left));
126     } else {
127       throw new BuilderException("Parsing error in {" + new String(expression) + "} in position " + p);
128     }
129     option(expression, right+1, map);
130   }
131 
132   private static void option(char[] expression, int p, Map<String, String> map) {
133     int left = skipWS(expression, p);
134     if (left < expression.length) {
135       int right = skipUntil(expression, left, '=');
136       String name = new String(expression, left, right - left);
137       left = skipWS(expression, right) + 1;
138       right = skipUntil(expression, left, ',');
139       String value = new String(expression, left, right - left);
140       map.put(name, value);
141       option(expression, right + 1, map);
142     }
143   }

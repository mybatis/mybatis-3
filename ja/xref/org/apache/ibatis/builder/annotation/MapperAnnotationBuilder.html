104   public MapperAnnotationBuilder(Configuration configuration, Class<?> type) {
105     String resource = type.getName().replace('.', '/') + ".java (best guess)";
106     this.assistant = new MapperBuilderAssistant(configuration, resource);
107     this.configuration = configuration;
108     this.type = type;
109 
110     sqlAnnotationTypes.add(Select.class);
111     sqlAnnotationTypes.add(Insert.class);
112     sqlAnnotationTypes.add(Update.class);
113     sqlAnnotationTypes.add(Delete.class);
114 
115     sqlProviderAnnotationTypes.add(SelectProvider.class);
116     sqlProviderAnnotationTypes.add(InsertProvider.class);
117     sqlProviderAnnotationTypes.add(UpdateProvider.class);
118     sqlProviderAnnotationTypes.add(DeleteProvider.class);
119   }

121   public void parse() {
122     String resource = type.toString();
123     if (!configuration.isResourceLoaded(resource)) {
124       loadXmlResource();
125       configuration.addLoadedResource(resource);
126       assistant.setCurrentNamespace(type.getName());
127       parseCache();
128       parseCacheRef();
129       Method[] methods = type.getMethods();
130       for (Method method : methods) {
131         try {
132           // issue #237
133           if (!method.isBridge()) {
134             parseStatement(method);
135           }
136         } catch (IncompleteElementException e) {
137           configuration.addIncompleteMethod(new MethodResolver(this, method));
138         }
139       }
140     }
141     parsePendingMethods();
142   }
143 
144   private void parsePendingMethods() {
145     Collection<MethodResolver> incompleteMethods = configuration.getIncompleteMethods();
146     synchronized (incompleteMethods) {
147       Iterator<MethodResolver> iter = incompleteMethods.iterator();
148       while (iter.hasNext()) {
149         try {
150           iter.next().resolve();
151           iter.remove();
152         } catch (IncompleteElementException e) {
153           // This method is still missing a resource
154         }
155       }
156     }
157   }

159   private void loadXmlResource() {
160     // Spring may not know the real resource name so we check a flag
161     // to prevent loading again a resource twice
162     // this flag is set at XMLMapperBuilder#bindMapperForNamespace
163     if (!configuration.isResourceLoaded("namespace:" + type.getName())) {
164       String xmlResource = type.getName().replace('.', '/') + ".xml";
165       InputStream inputStream = null;
166       try {
167         inputStream = Resources.getResourceAsStream(type.getClassLoader(), xmlResource);
168       } catch (IOException e) {
169         // ignore, resource is not required
170       }
171       if (inputStream != null) {
172         XMLMapperBuilder xmlParser = new XMLMapperBuilder(inputStream, assistant.getConfiguration(), xmlResource, configuration.getSqlFragments(), type.getName());
173         xmlParser.parse();
174       }
175     }
176   }

178   private void parseCache() {
179     CacheNamespace cacheDomain = type.getAnnotation(CacheNamespace.class);
180     if (cacheDomain != null) {
181       Integer size = cacheDomain.size() == 0 ? null : cacheDomain.size();
182       Long flushInterval = cacheDomain.flushInterval() == 0 ? null : cacheDomain.flushInterval();
183       Properties props = convertToProperties(cacheDomain.properties());
184       assistant.useNewCache(cacheDomain.implementation(), cacheDomain.eviction(), flushInterval, size, cacheDomain.readWrite(), cacheDomain.blocking(), props);
185     }
186   }

188   private Properties convertToProperties(Property[] properties) {
189     if (properties.length == 0) {
190       return null;
191     }
192     Properties props = new Properties();
193     for (Property property : properties) {
194       props.setProperty(property.name(),
195           PropertyParser.parse(property.value(), configuration.getVariables()));
196     }
197     return props;
198   }

200   private void parseCacheRef() {
201     CacheNamespaceRef cacheDomainRef = type.getAnnotation(CacheNamespaceRef.class);
202     if (cacheDomainRef != null) {
203       Class<?> refType = cacheDomainRef.value();
204       String refName = cacheDomainRef.name();
205       if (refType == void.class && refName.isEmpty()) {
206         throw new BuilderException("Should be specified either value() or name() attribute in the @CacheNamespaceRef");
207       }
208       if (refType != void.class && !refName.isEmpty()) {
209         throw new BuilderException("Cannot use both value() and name() attribute in the @CacheNamespaceRef");
210       }
211       String namespace = (refType != void.class) ? refType.getName() : refName;
212       assistant.useCacheRef(namespace);
213     }
214   }

216   private String parseResultMap(Method method) {
217     Class<?> returnType = getReturnType(method);
218     ConstructorArgs args = method.getAnnotation(ConstructorArgs.class);
219     Results results = method.getAnnotation(Results.class);
220     TypeDiscriminator typeDiscriminator = method.getAnnotation(TypeDiscriminator.class);
221     String resultMapId = generateResultMapName(method);
222     applyResultMap(resultMapId, returnType, argsIf(args), resultsIf(results), typeDiscriminator);
223     return resultMapId;
224   }
225 
226   private String generateResultMapName(Method method) {
227     Results results = method.getAnnotation(Results.class);
228     if (results != null && !results.id().isEmpty()) {
229       return type.getName() + "." + results.id();
230     }
231     StringBuilder suffix = new StringBuilder();
232     for (Class<?> c : method.getParameterTypes()) {
233       suffix.append("-");
234       suffix.append(c.getSimpleName());
235     }
236     if (suffix.length() < 1) {
237       suffix.append("-void");
238     }
239     return type.getName() + "." + method.getName() + suffix;
240   }

242   private void applyResultMap(String resultMapId, Class<?> returnType, Arg[] args, Result[] results, TypeDiscriminator discriminator) {
243     List<ResultMapping> resultMappings = new ArrayList<ResultMapping>();
244     applyConstructorArgs(args, returnType, resultMappings);
245     applyResults(results, returnType, resultMappings);
246     Discriminator disc = applyDiscriminator(resultMapId, returnType, discriminator);
247     // TODO add AutoMappingBehaviour
248     assistant.addResultMap(resultMapId, returnType, null, disc, resultMappings, null);
249     createDiscriminatorResultMaps(resultMapId, returnType, discriminator);
250   }

252   private void createDiscriminatorResultMaps(String resultMapId, Class<?> resultType, TypeDiscriminator discriminator) {
253     if (discriminator != null) {
254       for (Case c : discriminator.cases()) {
255         String caseResultMapId = resultMapId + "-" + c.value();
256         List<ResultMapping> resultMappings = new ArrayList<ResultMapping>();
257         // issue #136
258         applyConstructorArgs(c.constructArgs(), resultType, resultMappings);
259         applyResults(c.results(), resultType, resultMappings);
260         // TODO add AutoMappingBehaviour
261         assistant.addResultMap(caseResultMapId, c.type(), resultMapId, null, resultMappings, null);
262       }
263     }
264   }

266   private Discriminator applyDiscriminator(String resultMapId, Class<?> resultType, TypeDiscriminator discriminator) {
267     if (discriminator != null) {
268       String column = discriminator.column();
269       Class<?> javaType = discriminator.javaType() == void.class ? String.class : discriminator.javaType();
270       JdbcType jdbcType = discriminator.jdbcType() == JdbcType.UNDEFINED ? null : discriminator.jdbcType();
271       @SuppressWarnings("unchecked")
272       Class<? extends TypeHandler<?>> typeHandler = (Class<? extends TypeHandler<?>>)
273               (discriminator.typeHandler() == UnknownTypeHandler.class ? null : discriminator.typeHandler());
274       Case[] cases = discriminator.cases();
275       Map<String, String> discriminatorMap = new HashMap<String, String>();
276       for (Case c : cases) {
277         String value = c.value();
278         String caseResultMapId = resultMapId + "-" + value;
279         discriminatorMap.put(value, caseResultMapId);
280       }
281       return assistant.buildDiscriminator(resultType, column, javaType, jdbcType, typeHandler, discriminatorMap);
282     }
283     return null;
284   }

286   void parseStatement(Method method) {
287     Class<?> parameterTypeClass = getParameterType(method);
288     LanguageDriver languageDriver = getLanguageDriver(method);
289     SqlSource sqlSource = getSqlSourceFromAnnotations(method, parameterTypeClass, languageDriver);
290     if (sqlSource != null) {
291       Options options = method.getAnnotation(Options.class);
292       final String mappedStatementId = type.getName() + "." + method.getName();
293       Integer fetchSize = null;
294       Integer timeout = null;
295       StatementType statementType = StatementType.PREPARED;
296       ResultSetType resultSetType = ResultSetType.FORWARD_ONLY;
297       SqlCommandType sqlCommandType = getSqlCommandType(method);
298       boolean isSelect = sqlCommandType == SqlCommandType.SELECT;
299       boolean flushCache = !isSelect;
300       boolean useCache = isSelect;
301 
302       KeyGenerator keyGenerator;
303       String keyProperty = "id";
304       String keyColumn = null;
305       if (SqlCommandType.INSERT.equals(sqlCommandType) || SqlCommandType.UPDATE.equals(sqlCommandType)) {
306         // first check for SelectKey annotation - that overrides everything else
307         SelectKey selectKey = method.getAnnotation(SelectKey.class);
308         if (selectKey != null) {
309           keyGenerator = handleSelectKeyAnnotation(selectKey, mappedStatementId, getParameterType(method), languageDriver);
310           keyProperty = selectKey.keyProperty();
311         } else if (options == null) {
312           keyGenerator = configuration.isUseGeneratedKeys() ? new Jdbc3KeyGenerator() : new NoKeyGenerator();
313         } else {
314           keyGenerator = options.useGeneratedKeys() ? new Jdbc3KeyGenerator() : new NoKeyGenerator();
315           keyProperty = options.keyProperty();
316           keyColumn = options.keyColumn();
317         }
318       } else {
319         keyGenerator = new NoKeyGenerator();
320       }
321 
322       if (options != null) {
323         if (FlushCachePolicy.TRUE.equals(options.flushCache())) {
324           flushCache = true;
325         } else if (FlushCachePolicy.FALSE.equals(options.flushCache())) {
326           flushCache = false;
327         }
328         useCache = options.useCache();
329         fetchSize = options.fetchSize() > -1 || options.fetchSize() == Integer.MIN_VALUE ? options.fetchSize() : null; //issue #348
330         timeout = options.timeout() > -1 ? options.timeout() : null;
331         statementType = options.statementType();
332         resultSetType = options.resultSetType();
333       }
334 
335       String resultMapId = null;
336       ResultMap resultMapAnnotation = method.getAnnotation(ResultMap.class);
337       if (resultMapAnnotation != null) {
338         String[] resultMaps = resultMapAnnotation.value();
339         StringBuilder sb = new StringBuilder();
340         for (String resultMap : resultMaps) {
341           if (sb.length() > 0) {
342             sb.append(",");
343           }
344           sb.append(resultMap);
345         }
346         resultMapId = sb.toString();
347       } else if (isSelect) {
348         resultMapId = parseResultMap(method);
349       }
350 
351       assistant.addMappedStatement(
352           mappedStatementId,
353           sqlSource,
354           statementType,
355           sqlCommandType,
356           fetchSize,
357           timeout,
358           // ParameterMapID
359           null,
360           parameterTypeClass,
361           resultMapId,
362           getReturnType(method),
363           resultSetType,
364           flushCache,
365           useCache,
366           // TODO gcode issue #577
367           false,
368           keyGenerator,
369           keyProperty,
370           keyColumn,
371           // DatabaseID
372           null,
373           languageDriver,
374           // ResultSets
375           options != null ? nullOrEmpty(options.resultSets()) : null);
376     }
377   }

388   private Class<?> getParameterType(Method method) {
389     Class<?> parameterType = null;
390     Class<?>[] parameterTypes = method.getParameterTypes();
391     for (Class<?> currentParameterType : parameterTypes) {
392       if (!RowBounds.class.isAssignableFrom(currentParameterType) && !ResultHandler.class.isAssignableFrom(currentParameterType)) {
393         if (parameterType == null) {
394           parameterType = currentParameterType;
395         } else {
396           // issue #135
397           parameterType = ParamMap.class;
398         }
399       }
400     }
401     return parameterType;
402   }

404   private Class<?> getReturnType(Method method) {
405     Class<?> returnType = method.getReturnType();
406     Type resolvedReturnType = TypeParameterResolver.resolveReturnType(method, type);
407     if (resolvedReturnType instanceof Class) {
408       returnType = (Class<?>) resolvedReturnType;
409       if (returnType.isArray()) {
410         returnType = returnType.getComponentType();
411       }
412       // gcode issue #508
413       if (void.class.equals(returnType)) {
414         ResultType rt = method.getAnnotation(ResultType.class);
415         if (rt != null) {
416           returnType = rt.value();
417         }
418       }
419     } else if (resolvedReturnType instanceof ParameterizedType) {
420       ParameterizedType parameterizedType = (ParameterizedType) resolvedReturnType;
421       Class<?> rawType = (Class<?>) parameterizedType.getRawType();
422       if (Collection.class.isAssignableFrom(rawType) || Cursor.class.isAssignableFrom(rawType)) {
423         Type[] actualTypeArguments = parameterizedType.getActualTypeArguments();
424         if (actualTypeArguments != null && actualTypeArguments.length == 1) {
425           Type returnTypeParameter = actualTypeArguments[0];
426           if (returnTypeParameter instanceof Class<?>) {
427             returnType = (Class<?>) returnTypeParameter;
428           } else if (returnTypeParameter instanceof ParameterizedType) {
429             // (gcode issue #443) actual type can be a also a parameterized type
430             returnType = (Class<?>) ((ParameterizedType) returnTypeParameter).getRawType();
431           } else if (returnTypeParameter instanceof GenericArrayType) {
432             Class<?> componentType = (Class<?>) ((GenericArrayType) returnTypeParameter).getGenericComponentType();
433             // (gcode issue #525) support List<byte[]>
434             returnType = Array.newInstance(componentType, 0).getClass();
435           }
436         }
437       } else if (method.isAnnotationPresent(MapKey.class) && Map.class.isAssignableFrom(rawType)) {
438         // (gcode issue 504) Do not look into Maps if there is not MapKey annotation
439         Type[] actualTypeArguments = parameterizedType.getActualTypeArguments();
440           if (actualTypeArguments != null && actualTypeArguments.length == 2) {
441             Type returnTypeParameter = actualTypeArguments[1];
442             if (returnTypeParameter instanceof Class<?>) {
443               returnType = (Class<?>) returnTypeParameter;
444             } else if (returnTypeParameter instanceof ParameterizedType) {
445               // (gcode issue 443) actual type can be a also a parameterized type
446               returnType = (Class<?>) ((ParameterizedType) returnTypeParameter).getRawType();
447             }
448           }
449       }
450     }
451 
452     return returnType;
453   }

455   private SqlSource getSqlSourceFromAnnotations(Method method, Class<?> parameterType, LanguageDriver languageDriver) {
456     try {
457       Class<? extends Annotation> sqlAnnotationType = getSqlAnnotationType(method);
458       Class<? extends Annotation> sqlProviderAnnotationType = getSqlProviderAnnotationType(method);
459       if (sqlAnnotationType != null) {
460         if (sqlProviderAnnotationType != null) {
461           throw new BindingException("You cannot supply both a static SQL and SqlProvider to method named " + method.getName());
462         }
463         Annotation sqlAnnotation = method.getAnnotation(sqlAnnotationType);
464         final String[] strings = (String[]) sqlAnnotation.getClass().getMethod("value").invoke(sqlAnnotation);
465         return buildSqlSourceFromStrings(strings, parameterType, languageDriver);
466       } else if (sqlProviderAnnotationType != null) {
467         Annotation sqlProviderAnnotation = method.getAnnotation(sqlProviderAnnotationType);
468         return new ProviderSqlSource(assistant.getConfiguration(), sqlProviderAnnotation);
469       }
470       return null;
471     } catch (Exception e) {
472       throw new BuilderException("Could not find value method on SQL annotation.  Cause: " + e, e);
473     }
474   }

476   private SqlSource buildSqlSourceFromStrings(String[] strings, Class<?> parameterTypeClass, LanguageDriver languageDriver) {
477     final StringBuilder sql = new StringBuilder();
478     for (String fragment : strings) {
479       sql.append(fragment);
480       sql.append(" ");
481     }
482     return languageDriver.createSqlSource(configuration, sql.toString().trim(), parameterTypeClass);
483   }

485   private SqlCommandType getSqlCommandType(Method method) {
486     Class<? extends Annotation> type = getSqlAnnotationType(method);
487 
488     if (type == null) {
489       type = getSqlProviderAnnotationType(method);
490 
491       if (type == null) {
492         return SqlCommandType.UNKNOWN;
493       }
494 
495       if (type == SelectProvider.class) {
496         type = Select.class;
497       } else if (type == InsertProvider.class) {
498         type = Insert.class;
499       } else if (type == UpdateProvider.class) {
500         type = Update.class;
501       } else if (type == DeleteProvider.class) {
502         type = Delete.class;
503       }
504     }
505 
506     return SqlCommandType.valueOf(type.getSimpleName().toUpperCase(Locale.ENGLISH));
507   }

517   private Class<? extends Annotation> chooseAnnotationType(Method method, Set<Class<? extends Annotation>> types) {
518     for (Class<? extends Annotation> type : types) {
519       Annotation annotation = method.getAnnotation(type);
520       if (annotation != null) {
521         return type;
522       }
523     }
524     return null;
525   }

527   private void applyResults(Result[] results, Class<?> resultType, List<ResultMapping> resultMappings) {
528     for (Result result : results) {
529       List<ResultFlag> flags = new ArrayList<ResultFlag>();
530       if (result.id()) {
531         flags.add(ResultFlag.ID);
532       }
533       @SuppressWarnings("unchecked")
534       Class<? extends TypeHandler<?>> typeHandler = (Class<? extends TypeHandler<?>>)
535               ((result.typeHandler() == UnknownTypeHandler.class) ? null : result.typeHandler());
536       ResultMapping resultMapping = assistant.buildResultMapping(
537           resultType,
538           nullOrEmpty(result.property()),
539           nullOrEmpty(result.column()),
540           result.javaType() == void.class ? null : result.javaType(),
541           result.jdbcType() == JdbcType.UNDEFINED ? null : result.jdbcType(),
542           hasNestedSelect(result) ? nestedSelectId(result) : null,
543           null,
544           null,
545           null,
546           typeHandler,
547           flags,
548           null,
549           null,
550           isLazy(result));
551       resultMappings.add(resultMapping);
552     }
553   }

555   private String nestedSelectId(Result result) {
556     String nestedSelect = result.one().select();
557     if (nestedSelect.length() < 1) {
558       nestedSelect = result.many().select();
559     }
560     if (!nestedSelect.contains(".")) {
561       nestedSelect = type.getName() + "." + nestedSelect;
562     }
563     return nestedSelect;
564   }

566   private boolean isLazy(Result result) {
567     boolean isLazy = configuration.isLazyLoadingEnabled();
568     if (result.one().select().length() > 0 && FetchType.DEFAULT != result.one().fetchType()) {
569       isLazy = (result.one().fetchType() == FetchType.LAZY);
570     } else if (result.many().select().length() > 0 && FetchType.DEFAULT != result.many().fetchType()) {
571       isLazy = (result.many().fetchType() == FetchType.LAZY);
572     }
573     return isLazy;
574   }
575   
576   private boolean hasNestedSelect(Result result) {
577     if (result.one().select().length() > 0 && result.many().select().length() > 0) {
578       throw new BuilderException("Cannot use both @One and @Many annotations in the same @Result");
579     }
580     return result.one().select().length() > 0 || result.many().select().length() > 0;  
581   }

583   private void applyConstructorArgs(Arg[] args, Class<?> resultType, List<ResultMapping> resultMappings) {
584     for (Arg arg : args) {
585       List<ResultFlag> flags = new ArrayList<ResultFlag>();
586       flags.add(ResultFlag.CONSTRUCTOR);
587       if (arg.id()) {
588         flags.add(ResultFlag.ID);
589       }
590       @SuppressWarnings("unchecked")
591       Class<? extends TypeHandler<?>> typeHandler = (Class<? extends TypeHandler<?>>)
592               (arg.typeHandler() == UnknownTypeHandler.class ? null : arg.typeHandler());
593       ResultMapping resultMapping = assistant.buildResultMapping(
594           resultType,
595           null,
596           nullOrEmpty(arg.column()),
597           arg.javaType() == void.class ? null : arg.javaType(),
598           arg.jdbcType() == JdbcType.UNDEFINED ? null : arg.jdbcType(),
599           nullOrEmpty(arg.select()),
600           nullOrEmpty(arg.resultMap()),
601           null,
602           null,
603           typeHandler,
604           flags,
605           null,
606           null,
607           false);
608       resultMappings.add(resultMapping);
609     }
610   }

624   private KeyGenerator handleSelectKeyAnnotation(SelectKey selectKeyAnnotation, String baseStatementId, Class<?> parameterTypeClass, LanguageDriver languageDriver) {
625     String id = baseStatementId + SelectKeyGenerator.SELECT_KEY_SUFFIX;
626     Class<?> resultTypeClass = selectKeyAnnotation.resultType();
627     StatementType statementType = selectKeyAnnotation.statementType();
628     String keyProperty = selectKeyAnnotation.keyProperty();
629     String keyColumn = selectKeyAnnotation.keyColumn();
630     boolean executeBefore = selectKeyAnnotation.before();
631 
632     // defaults
633     boolean useCache = false;
634     KeyGenerator keyGenerator = new NoKeyGenerator();
635     Integer fetchSize = null;
636     Integer timeout = null;
637     boolean flushCache = false;
638     String parameterMap = null;
639     String resultMap = null;
640     ResultSetType resultSetTypeEnum = null;
641 
642     SqlSource sqlSource = buildSqlSourceFromStrings(selectKeyAnnotation.statement(), parameterTypeClass, languageDriver);
643     SqlCommandType sqlCommandType = SqlCommandType.SELECT;
644 
645     assistant.addMappedStatement(id, sqlSource, statementType, sqlCommandType, fetchSize, timeout, parameterMap, parameterTypeClass, resultMap, resultTypeClass, resultSetTypeEnum,
646         flushCache, useCache, false,
647         keyGenerator, keyProperty, keyColumn, null, languageDriver, null);
648 
649     id = assistant.applyCurrentNamespace(id, false);
650 
651     MappedStatement keyStatement = configuration.getMappedStatement(id, false);
652     SelectKeyGenerator answer = new SelectKeyGenerator(keyStatement, executeBefore);
653     configuration.addKeyGenerator(id, answer);
654     return answer;
655   }

102     private void resolveTypeHandler() {
103       if (parameterMapping.typeHandler == null) {
104         if (parameterMapping.javaType != null) {
105           Configuration configuration = parameterMapping.configuration;
106           TypeHandlerRegistry typeHandlerRegistry = configuration.getTypeHandlerRegistry();
107           parameterMapping.typeHandler = typeHandlerRegistry.getTypeHandler(parameterMapping.javaType, parameterMapping.jdbcType);
108         }
109       }
110     }

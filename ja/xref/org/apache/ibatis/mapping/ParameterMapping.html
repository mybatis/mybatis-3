109     public ParameterMapping build() {
110       resolveTypeHandler();
111       validate();
112       return parameterMapping;
113     }
114 
115     private void validate() {
116       if (ResultSet.class.equals(parameterMapping.javaType)) {
117         if (parameterMapping.resultMapId == null) { 
118           throw new IllegalStateException("Missing resultmap in property '"  
119               + parameterMapping.property + "'.  " 
120               + "Parameters of type java.sql.ResultSet require a resultmap.");
121         }            
122       } else {
123         if (parameterMapping.typeHandler == null) { 
124           throw new IllegalStateException("Type handler was null on parameter mapping for property '"  
125               + parameterMapping.property + "'.  " 
126               + "It was either not specified and/or could not be found for the javaType / jdbcType combination specified.");
127         }
128       }
129     }
130 
131     private void resolveTypeHandler() {
132       if (parameterMapping.typeHandler == null && parameterMapping.javaType != null) {
133         Configuration configuration = parameterMapping.configuration;
134         TypeHandlerRegistry typeHandlerRegistry = configuration.getTypeHandlerRegistry();
135         parameterMapping.typeHandler = typeHandlerRegistry.getTypeHandler(parameterMapping.javaType, parameterMapping.jdbcType);
136       }
137     }

37    private XNode context;

39    private Class<?> parameterType;

42      this(configuration, context, null);
43    }

45    public XMLScriptBuilder(Configuration configuration, XNode context, Class<?> parameterType) {
46      super(configuration);
47      this.context = context;
48      this.parameterType = parameterType;
49    }
50  
51    public SqlSource parseScriptNode() {
52      List<SqlNode> contents = parseDynamicTags(context);
53      MixedSqlNode rootSqlNode = new MixedSqlNode(contents);
54      SqlSource sqlSource = null;
55      if (isDynamic) {
56        sqlSource = new DynamicSqlSource(configuration, rootSqlNode);
57      } else {
58        sqlSource = new RawSqlSource(configuration, rootSqlNode, parameterType);
59      }
60      return sqlSource;
61    }
62  
63    List<SqlNode> parseDynamicTags(XNode node) {
64      List<SqlNode> contents = new ArrayList<SqlNode>();
65      NodeList children = node.getNode().getChildNodes();
66      for (int i = 0; i < children.getLength(); i++) {
67        XNode child = node.newXNode(children.item(i));
68        if (child.getNode().getNodeType() == Node.CDATA_SECTION_NODE || child.getNode().getNodeType() == Node.TEXT_NODE) {
69          String data = child.getStringBody("");
70          TextSqlNode textSqlNode = new TextSqlNode(data);
71          if (textSqlNode.isDynamic()) {
72            contents.add(textSqlNode);
73            isDynamic = true;
74          } else {
75            contents.add(new StaticTextSqlNode(data));
76          }
77        } else if (child.getNode().getNodeType() == Node.ELEMENT_NODE) { // issue #628
78          String nodeName = child.getNode().getNodeName();
79          NodeHandler handler = nodeHandlers(nodeName);
80          if (handler == null) {
81            throw new BuilderException("Unknown element <" + nodeName + "> in SQL statement.");
82          }
83          handler.handleNode(child, contents);
84          isDynamic = true;
85        }
86      }
87      return contents;
88    }
89  
90    NodeHandler nodeHandlers(String nodeName) {
91      Map<String, NodeHandler> map = new HashMap<String, NodeHandler>();
92      map.put("trim", new TrimHandler());
93      map.put("where", new WhereHandler());
94      map.put("set", new SetHandler());
95      map.put("foreach", new ForEachHandler());
96      map.put("if", new IfHandler());
97      map.put("choose", new ChooseHandler());
98      map.put("when", new IfHandler());
99      map.put("otherwise", new OtherwiseHandler());
100     map.put("bind", new BindHandler());
101     return map.get(nodeName);
102   }

104   private interface NodeHandler {
105     void handleNode(XNode nodeToHandle, List<SqlNode> targetContents);
106   }
107 
108   private class BindHandler implements NodeHandler {
109     public BindHandler() {
110       // Prevent Synthetic Access
111     }
112 
113     @Override
114     public void handleNode(XNode nodeToHandle, List<SqlNode> targetContents) {
115       final String name = nodeToHandle.getStringAttribute("name");
116       final String expression = nodeToHandle.getStringAttribute("value");
117       final VarDeclSqlNode node = new VarDeclSqlNode(name, expression);
118       targetContents.add(node);
119     }
120   }
121 
122   private class TrimHandler implements NodeHandler {
123     public TrimHandler() {
124       // Prevent Synthetic Access
125     }
126 
127     @Override
128     public void handleNode(XNode nodeToHandle, List<SqlNode> targetContents) {
129       List<SqlNode> contents = parseDynamicTags(nodeToHandle);
130       MixedSqlNode mixedSqlNode = new MixedSqlNode(contents);
131       String prefix = nodeToHandle.getStringAttribute("prefix");
132       String prefixOverrides = nodeToHandle.getStringAttribute("prefixOverrides");
133       String suffix = nodeToHandle.getStringAttribute("suffix");
134       String suffixOverrides = nodeToHandle.getStringAttribute("suffixOverrides");
135       TrimSqlNode trim = new TrimSqlNode(configuration, mixedSqlNode, prefix, prefixOverrides, suffix, suffixOverrides);
136       targetContents.add(trim);
137     }
138   }

216   private class ChooseHandler implements NodeHandler {
217     public ChooseHandler() {
218       // Prevent Synthetic Access
219     }
220 
221     @Override
222     public void handleNode(XNode nodeToHandle, List<SqlNode> targetContents) {
223       List<SqlNode> whenSqlNodes = new ArrayList<SqlNode>();
224       List<SqlNode> otherwiseSqlNodes = new ArrayList<SqlNode>();
225       handleWhenOtherwiseNodes(nodeToHandle, whenSqlNodes, otherwiseSqlNodes);
226       SqlNode defaultSqlNode = getDefaultSqlNode(otherwiseSqlNodes);
227       ChooseSqlNode chooseSqlNode = new ChooseSqlNode(whenSqlNodes, defaultSqlNode);
228       targetContents.add(chooseSqlNode);
229     }
230 
231     private void handleWhenOtherwiseNodes(XNode chooseSqlNode, List<SqlNode> ifSqlNodes, List<SqlNode> defaultSqlNodes) {
232       List<XNode> children = chooseSqlNode.getChildren();
233       for (XNode child : children) {
234         String nodeName = child.getNode().getNodeName();
235         NodeHandler handler = nodeHandlers(nodeName);
236         if (handler instanceof IfHandler) {
237           handler.handleNode(child, ifSqlNodes);
238         } else if (handler instanceof OtherwiseHandler) {
239           handler.handleNode(child, defaultSqlNodes);
240         }
241       }
242     }
243 
244     private SqlNode getDefaultSqlNode(List<SqlNode> defaultSqlNodes) {
245       SqlNode defaultSqlNode = null;
246       if (defaultSqlNodes.size() == 1) {
247         defaultSqlNode = defaultSqlNodes.get(0);
248       } else if (defaultSqlNodes.size() > 1) {
249         throw new BuilderException("Too many default (otherwise) elements in choose statement.");
250       }
251       return defaultSqlNode;
252     }

28    private static final String CLOSE = "close";
29    private static final Class<?>[] IFACES = new Class<?>[] { Connection.class };

31    private int hashCode = 0;

41    /*
42     * Constructor for SimplePooledConnection that uses the Connection and PooledDataSource passed in
43     *
44     * @param connection - the connection that is to be presented as a pooled connection
45     * @param dataSource - the dataSource that the connection is from
46     */
47    public PooledConnection(Connection connection, PooledDataSource dataSource) {
48      this.hashCode = connection.hashCode();
49      this.realConnection = connection;
50      this.dataSource = dataSource;
51      this.createdTimestamp = System.currentTimeMillis();
52      this.lastUsedTimestamp = System.currentTimeMillis();
53      this.valid = true;
54      this.proxyConnection = (Connection) Proxy.newProxyInstance(Connection.class.getClassLoader(), IFACES, this);
55    }

57    /*
58     * Invalidates the connection
59     */
60    public void invalidate() {

64    /*
65     * Method to see if the connection is usable
66     *
67     * @return True if the connection is usable
68     */
69    public boolean isValid() {
70      return valid && realConnection != null && dataSource.pingConnection(this);
71    }

213   public boolean equals(Object obj) {
214     if (obj instanceof PooledConnection) {
215       return realConnection.hashCode() == (((PooledConnection) obj).realConnection.hashCode());
216     } else if (obj instanceof Connection) {
217       return hashCode == obj.hashCode();
218     } else {
219       return false;
220     }

223   /*
224    * Required for InvocationHandler implementation.
225    *
226    * @param proxy  - not used
227    * @param method - the method to be executed
228    * @param args   - the parameters to be passed to the method
229    * @see java.lang.reflect.InvocationHandler#invoke(Object, java.lang.reflect.Method, Object[])
230    */
231   public Object invoke(Object proxy, Method method, Object[] args)
232       throws Throwable {
233     String methodName = method.getName();
234     if (CLOSE.hashCode() == methodName.hashCode() && CLOSE.equals(methodName)) {
235       dataSource.pushConnection(this);
236       return null;
237     } else {
238       try {
239         if (method.getDeclaringClass() != Object.class) {
240           // issue #578. 
241           // toString() should never fail
242           // throw an SQLException instead of a Runtime
243           checkConnection();
244         }
245         return method.invoke(realConnection, args);
246       } catch (Throwable t) {
247         throw ExceptionUtil.unwrapThrowable(t);
248       }
249     }
250   }
251 
252   private void checkConnection() throws SQLException {
253     if (!valid) {
254       throw new SQLException("Error accessing PooledConnection. Connection is invalid.");
255     }
256   }

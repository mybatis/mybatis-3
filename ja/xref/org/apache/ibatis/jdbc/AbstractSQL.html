281   private static class SafeAppendable {
282     private final Appendable a;
283     private boolean empty = true;
284 
285     public SafeAppendable(Appendable a) {
286       super();
287       this.a = a;
288     }
289 
290     public SafeAppendable append(CharSequence s) {
291       try {
292         if (empty && s.length() > 0) {
293           empty = false;
294         }
295         a.append(s);
296       } catch (IOException e) {
297         throw new RuntimeException(e);
298       }
299       return this;
300     }
301 
302     public boolean isEmpty() {
303       return empty;
304     }
305 
306   }

308   private static class SQLStatement {
309 
310     public enum StatementType {
311       DELETE, INSERT, SELECT, UPDATE
312     }
313 
314     StatementType statementType;
315     List<String> sets = new ArrayList<String>();
316     List<String> select = new ArrayList<String>();
317     List<String> tables = new ArrayList<String>();
318     List<String> join = new ArrayList<String>();
319     List<String> innerJoin = new ArrayList<String>();
320     List<String> outerJoin = new ArrayList<String>();
321     List<String> leftOuterJoin = new ArrayList<String>();
322     List<String> rightOuterJoin = new ArrayList<String>();
323     List<String> where = new ArrayList<String>();
324     List<String> having = new ArrayList<String>();
325     List<String> groupBy = new ArrayList<String>();
326     List<String> orderBy = new ArrayList<String>();
327     List<String> lastList = new ArrayList<String>();
328     List<String> columns = new ArrayList<String>();
329     List<String> values = new ArrayList<String>();
330     boolean distinct;
331 
332     public SQLStatement() {
333         // Prevent Synthetic Access
334     }
335 
336     private void sqlClause(SafeAppendable builder, String keyword, List<String> parts, String open, String close,
337                            String conjunction) {
338       if (!parts.isEmpty()) {
339         if (!builder.isEmpty()) {
340           builder.append("\n");
341         }
342         builder.append(keyword);
343         builder.append(" ");
344         builder.append(open);
345         String last = "________";
346         for (int i = 0, n = parts.size(); i < n; i++) {
347           String part = parts.get(i);
348           if (i > 0 && !part.equals(AND) && !part.equals(OR) && !last.equals(AND) && !last.equals(OR)) {
349             builder.append(conjunction);
350           }
351           builder.append(part);
352           last = part;
353         }
354         builder.append(close);
355       }
356     }
357 
358     private String selectSQL(SafeAppendable builder) {
359       if (distinct) {
360         sqlClause(builder, "SELECT DISTINCT", select, "", "", ", ");
361       } else {
362         sqlClause(builder, "SELECT", select, "", "", ", ");
363       }
364 
365       sqlClause(builder, "FROM", tables, "", "", ", ");
366       sqlClause(builder, "JOIN", join, "", "", "\nJOIN ");
367       sqlClause(builder, "INNER JOIN", innerJoin, "", "", "\nINNER JOIN ");
368       sqlClause(builder, "OUTER JOIN", outerJoin, "", "", "\nOUTER JOIN ");
369       sqlClause(builder, "LEFT OUTER JOIN", leftOuterJoin, "", "", "\nLEFT OUTER JOIN ");
370       sqlClause(builder, "RIGHT OUTER JOIN", rightOuterJoin, "", "", "\nRIGHT OUTER JOIN ");
371       sqlClause(builder, "WHERE", where, "(", ")", " AND ");
372       sqlClause(builder, "GROUP BY", groupBy, "", "", ", ");
373       sqlClause(builder, "HAVING", having, "(", ")", " AND ");
374       sqlClause(builder, "ORDER BY", orderBy, "", "", ", ");
375       return builder.toString();
376     }
377 
378     private String insertSQL(SafeAppendable builder) {
379       sqlClause(builder, "INSERT INTO", tables, "", "", "");
380       sqlClause(builder, "", columns, "(", ")", ", ");
381       sqlClause(builder, "VALUES", values, "(", ")", ", ");
382       return builder.toString();
383     }
384 
385     private String deleteSQL(SafeAppendable builder) {
386       sqlClause(builder, "DELETE FROM", tables, "", "", "");
387       sqlClause(builder, "WHERE", where, "(", ")", " AND ");
388       return builder.toString();
389     }
390 
391     private String updateSQL(SafeAppendable builder) {
392 
393       sqlClause(builder, "UPDATE", tables, "", "", "");
394       sqlClause(builder, "SET", sets, "", "", ", ");
395       sqlClause(builder, "WHERE", where, "(", ")", " AND ");
396       return builder.toString();
397     }
398 
399     public String sql(Appendable a) {
400       SafeAppendable builder = new SafeAppendable(a);
401       if (statementType == null) {
402         return null;
403       }
404 
405       String answer;
406 
407       switch (statementType) {
408         case DELETE:
409           answer = deleteSQL(builder);
410           break;
411 
412         case INSERT:
413           answer = insertSQL(builder);
414           break;
415 
416         case SELECT:
417           answer = selectSQL(builder);
418           break;
419 
420         case UPDATE:
421           answer = updateSQL(builder);
422           break;
423 
424         default:
425           answer = null;
426       }
427 
428       return answer;
429     }

91    public void runScript(Reader reader) {
92      setAutoCommit();
93  
94      try {
95        if (sendFullScript) {
96          executeFullScript(reader);
97        } else {
98          executeLineByLine(reader);
99        }
100     } finally {
101       rollbackConnection();
102     }
103   }

105   private void executeFullScript(Reader reader) {
106     StringBuilder script = new StringBuilder();
107     try {
108       BufferedReader lineReader = new BufferedReader(reader);
109       String line;
110       while ((line = lineReader.readLine()) != null) {
111         script.append(line);
112         script.append(LINE_SEPARATOR);
113       }
114       executeStatement(script.toString());
115       commitConnection();
116     } catch (Exception e) {
117       String message = "Error executing: " + script + ".  Cause: " + e;
118       printlnError(message);
119       throw new RuntimeSqlException(message, e);
120     }
121   }
122 
123   private void executeLineByLine(Reader reader) {
124     StringBuilder command = new StringBuilder();
125     try {
126       BufferedReader lineReader = new BufferedReader(reader);
127       String line;
128       while ((line = lineReader.readLine()) != null) {
129         command = handleLine(command, line);
130       }
131       commitConnection();
132       checkForMissingLineTerminator(command);
133     } catch (Exception e) {
134       String message = "Error executing: " + command + ".  Cause: " + e;
135       printlnError(message);
136       throw new RuntimeSqlException(message, e);
137     }
138   }

148   private void setAutoCommit() {
149     try {
150       if (autoCommit != connection.getAutoCommit()) {
151         connection.setAutoCommit(autoCommit);
152       }
153     } catch (Throwable t) {
154       throw new RuntimeSqlException("Could not set AutoCommit to " + autoCommit + ". Cause: " + t, t);

158   private void commitConnection() {
159     try {
160       if (!connection.getAutoCommit()) {
161         connection.commit();
162       }
163     } catch (Throwable t) {
164       throw new RuntimeSqlException("Could not commit transaction. Cause: " + t, t);
165     }
166   }

168   private void rollbackConnection() {
169     try {
170       if (!connection.getAutoCommit()) {
171         connection.rollback();
172       }
173     } catch (Throwable t) {
174       // ignore
175     }
176   }

178   private void checkForMissingLineTerminator(StringBuilder command) {
179     if (command != null && command.toString().trim().length() > 0) {
180       throw new RuntimeSqlException("Line missing end-of-line terminator (" + delimiter + ") => " + command);
181     }
182   }

184   private StringBuilder handleLine(StringBuilder command, String line) throws SQLException, UnsupportedEncodingException {
185     String trimmedLine = line.trim();
186     if (lineIsComment(trimmedLine)) {
187       println(trimmedLine);
188     } else if (commandReadyToExecute(trimmedLine)) {
189       command.append(line.substring(0, line.lastIndexOf(delimiter)));
190       command.append(LINE_SEPARATOR);
191       println(command);
192       executeStatement(command.toString());
193       command.setLength(0);
194     } else if (trimmedLine.length() > 0) {
195       command.append(line);
196       command.append(LINE_SEPARATOR);
197     }
198     return command;
199   }

201   private boolean lineIsComment(String trimmedLine) {
202     return trimmedLine.startsWith("//") || trimmedLine.startsWith("--");
203   }
204 
205   private boolean commandReadyToExecute(String trimmedLine) {
206     // issue #561 remove anything after the delimiter
207     return !fullLineDelimiter && trimmedLine.contains(delimiter) || fullLineDelimiter && trimmedLine.equals(delimiter);
208   }

210   private void executeStatement(String command) throws SQLException, UnsupportedEncodingException {
211     boolean hasResults = false;
212     Statement statement = connection.createStatement();
213     statement.setEscapeProcessing(escapeProcessing);
214     String sql = command;
215     if (removeCRs)
216       sql = sql.replaceAll("\r\n", "\n");
217     if (stopOnError) {
218       hasResults = statement.execute(sql);
219     } else {
220       try {
221         hasResults = statement.execute(sql);
222       } catch (SQLException e) {
223         String message = "Error executing: " + command + ".  Cause: " + e;
224         printlnError(message);
225       }
226     }
227     printResults(statement, hasResults);
228     try {
229       statement.close();
230     } catch (Exception e) {
231       // Ignore to workaround a bug in some connection pools
232     }
233   }
234 
235   private void printResults(Statement statement, boolean hasResults) {
236     try {
237       if (hasResults) {
238         ResultSet rs = statement.getResultSet();
239         if (rs != null) {
240           ResultSetMetaData md = rs.getMetaData();
241           int cols = md.getColumnCount();
242           for (int i = 0; i < cols; i++) {
243             String name = md.getColumnLabel(i + 1);
244             print(name + "\t");
245           }
246           println("");
247           while (rs.next()) {
248             for (int i = 0; i < cols; i++) {
249               String value = rs.getString(i + 1);
250               print(value + "\t");
251             }
252             println("");
253           }
254         }
255       }
256     } catch (SQLException e) {
257       printlnError("Error printing results: " + e.getMessage());
258     }
259   }

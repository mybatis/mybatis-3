25  /**
26   * The primary Java interface for working with MyBatis.
27   * Through this interface you can execute commands, get mappers and manage transactions.
28   *
29   */
30  public interface SqlSession extends Closeable {
31  
32    /**
33     * Retrieve a single row mapped from the statement key
34     * @param <T> the returned object type
35     * @param statement
36     * @return Mapped object
37     */
38    <T> T selectOne(String statement);
39  
40    /**
41     * Retrieve a single row mapped from the statement key and parameter.
42     * @param <T> the returned object type
43     * @param statement Unique identifier matching the statement to use.
44     * @param parameter A parameter object to pass to the statement.
45     * @return Mapped object
46     */
47    <T> T selectOne(String statement, Object parameter);
48  
49    /**
50     * Retrieve a list of mapped objects from the statement key and parameter.
51     * @param <E> the returned list element type
52     * @param statement Unique identifier matching the statement to use.
53     * @return List of mapped object
54     */
55    <E> List<E> selectList(String statement);
56  
57    /**
58     * Retrieve a list of mapped objects from the statement key and parameter.
59     * @param <E> the returned list element type
60     * @param statement Unique identifier matching the statement to use.
61     * @param parameter A parameter object to pass to the statement.
62     * @return List of mapped object
63     */
64    <E> List<E> selectList(String statement, Object parameter);
65  
66    /**
67     * Retrieve a list of mapped objects from the statement key and parameter,
68     * within the specified row bounds.
69     * @param <E> the returned list element type
70     * @param statement Unique identifier matching the statement to use.
71     * @param parameter A parameter object to pass to the statement.
72     * @param rowBounds  Bounds to limit object retrieval
73     * @return List of mapped object
74     */
75    <E> List<E> selectList(String statement, Object parameter, RowBounds rowBounds);
76  
77    /**
78     * The selectMap is a special case in that it is designed to convert a list
79     * of results into a Map based on one of the properties in the resulting
80     * objects.
81     * Eg. Return a of Map[Integer,Author] for selectMap("selectAuthors","id")
82     * @param <K> the returned Map keys type
83     * @param <V> the returned Map values type
84     * @param statement Unique identifier matching the statement to use.
85     * @param mapKey The property to use as key for each value in the list.
86     * @return Map containing key pair data.
87     */
88    <K, V> Map<K, V> selectMap(String statement, String mapKey);
89  
90    /**
91     * The selectMap is a special case in that it is designed to convert a list
92     * of results into a Map based on one of the properties in the resulting
93     * objects.
94     * @param <K> the returned Map keys type
95     * @param <V> the returned Map values type
96     * @param statement Unique identifier matching the statement to use.
97     * @param parameter A parameter object to pass to the statement.
98     * @param mapKey The property to use as key for each value in the list.
99     * @return Map containing key pair data.
100    */
101   <K, V> Map<K, V> selectMap(String statement, Object parameter, String mapKey);
102 
103   /**
104    * The selectMap is a special case in that it is designed to convert a list
105    * of results into a Map based on one of the properties in the resulting
106    * objects.
107    * @param <K> the returned Map keys type
108    * @param <V> the returned Map values type
109    * @param statement Unique identifier matching the statement to use.
110    * @param parameter A parameter object to pass to the statement.
111    * @param mapKey The property to use as key for each value in the list.
112    * @param rowBounds  Bounds to limit object retrieval
113    * @return Map containing key pair data.
114    */
115   <K, V> Map<K, V> selectMap(String statement, Object parameter, String mapKey, RowBounds rowBounds);
116 
117   /**
118    * Retrieve a single row mapped from the statement key and parameter
119    * using a {@code ResultHandler}.
120    * @param statement Unique identifier matching the statement to use.

125   void select(String statement, Object parameter, ResultHandler handler);

127   /**

134   void select(String statement, ResultHandler handler);

144   void select(String statement, Object parameter, RowBounds rowBounds, ResultHandler handler);

146   /**

149    * @return int The number of rows affected by the insert.
150    */
151   int insert(String statement);
152 
153   /**
154    * Execute an insert statement with the given parameter object. Any generated
155    * autoincrement values or selectKey entries will modify the given parameter
156    * object properties. Only the number of rows affected will be returned.
157    * @param statement Unique identifier matching the statement to execute.
158    * @param parameter A parameter object to pass to the statement.
159    * @return int The number of rows affected by the insert.
160    */
161   int insert(String statement, Object parameter);
162 
163   /**
164    * Execute an update statement. The number of rows affected will be returned.
165    * @param statement Unique identifier matching the statement to execute.
166    * @return int The number of rows affected by the update.
167    */
168   int update(String statement);
169 
170   /**
171    * Execute an update statement. The number of rows affected will be returned.
172    * @param statement Unique identifier matching the statement to execute.
173    * @param parameter A parameter object to pass to the statement.
174    * @return int The number of rows affected by the update.
175    */
176   int update(String statement, Object parameter);
177 
178   /**
179    * Execute a delete statement. The number of rows affected will be returned.
180    * @param statement Unique identifier matching the statement to execute.
181    * @return int The number of rows affected by the delete.
182    */
183   int delete(String statement);
184 
185   /**
186    * Execute a delete statement. The number of rows affected will be returned.
187    * @param statement Unique identifier matching the statement to execute.
188    * @param parameter A parameter object to pass to the statement.
189    * @return int The number of rows affected by the delete.
190    */
191   int delete(String statement, Object parameter);
192 
193   /**
194    * Flushes batch statements and commits database connection.
195    * Note that database connection will not be committed if no updates/deletes/inserts were called.
196    * To force the commit call {@link SqlSession#commit(boolean)}
197    */
198   void commit();
199 
200   /**
201    * Flushes batch statements and commits database connection.
202    * @param force forces connection commit
203    */
204   void commit(boolean force);
205 
206   /**
207    * Discards pending batch statements and rolls database connection back.
208    * Note that database connection will not be rolled back if no updates/deletes/inserts were called.
209    * To force the rollback call {@link SqlSession#rollback(boolean)}
210    */
211   void rollback();
212 
213   /**
214    * Discards pending batch statements and rolls database connection back.
215    * Note that database connection will not be rolled back if no updates/deletes/inserts were called.
216    * @param force forces connection rollback
217    */
218   void rollback(boolean force);

241   Configuration getConfiguration();

249   <T> T getMapper(Class<T> type);

251   /**
252    * Retrieves inner database connection
253    * @return Connection
254    */
255   Connection getConnection();

16  package org.apache.ibatis.executor;
17  
18  /**
19   * @author Clinton Begin
20   */
21  public class ErrorContext {
22  
23    private static final String LINE_SEPARATOR = System.getProperty("line.separator","\n");
24    private static final ThreadLocal<ErrorContext> LOCAL = new ThreadLocal<ErrorContext>();
25  
26    private ErrorContext stored;
27    private String resource;
28    private String activity;
29    private String object;
30    private String message;
31    private String sql;
32    private Throwable cause;
33  
34    private ErrorContext() {
35    }
36  
37    public static ErrorContext instance() {
38      ErrorContext context = LOCAL.get();
39      if (context == null) {
40        context = new ErrorContext();
41        LOCAL.set(context);
42      }
43      return context;
44    }
45  
46    public ErrorContext store() {
47      stored = this;
48      LOCAL.set(new ErrorContext());
49      return LOCAL.get();
50    }
51  
52    public ErrorContext recall() {
53      if (stored != null) {
54        LOCAL.set(stored);
55        stored = null;
56      }
57      return LOCAL.get();
58    }
59  
60    public ErrorContext resource(String resource) {
61      this.resource = resource;
62      return this;
63    }
64  
65    public ErrorContext activity(String activity) {
66      this.activity = activity;
67      return this;
68    }
69  
70    public ErrorContext object(String object) {
71      this.object = object;
72      return this;
73    }
74  
75    public ErrorContext message(String message) {
76      this.message = message;
77      return this;
78    }
79  
80    public ErrorContext sql(String sql) {
81      this.sql = sql;
82      return this;
83    }
84  
85    public ErrorContext cause(Throwable cause) {
86      this.cause = cause;
87      return this;
88    }
89  
90    public ErrorContext reset() {
91      resource = null;
92      activity = null;
93      object = null;
94      message = null;
95      sql = null;
96      cause = null;
97      LOCAL.remove();
98      return this;
99    }
100 
101   public String toString() {
102     StringBuffer description = new StringBuffer();
103 
104     // message
105     if (this.message != null) {
106       description.append(LINE_SEPARATOR);
107       description.append("### ");
108       description.append(this.message);
109     }
110 
111     // resource
112     if (resource != null) {
113       description.append(LINE_SEPARATOR);
114       description.append("### The error may exist in ");
115       description.append(resource);
116     }
117 
118     // object
119     if (object != null) {
120       description.append(LINE_SEPARATOR);
121       description.append("### The error may involve ");
122       description.append(object);
123     }
124 
125     // activity
126     if (activity != null) {
127       description.append(LINE_SEPARATOR);
128       description.append("### The error occurred while ");
129       description.append(activity);
130     }
131 
132     // activity
133     if (sql != null) {
134       description.append(LINE_SEPARATOR);
135       description.append("### SQL: ");
136       description.append(sql.replace('\n', ' ').replace('\r', ' ').replace('\t', ' ').trim());
137     }
138 
139     // cause
140     if (cause != null) {
141       description.append(LINE_SEPARATOR);
142       description.append("### Cause: ");
143       description.append(cause.toString());
144     }
145 
146     return description.toString();
147   }

16  package org.apache.ibatis.executor.loader;
17  
18  import java.sql.SQLException;
19  import java.util.HashMap;
20  import java.util.Locale;
21  import java.util.Map;
22  import java.util.Set;
23  
24  import org.apache.ibatis.executor.ExecutorException;
25  import org.apache.ibatis.reflection.MetaObject;
26  
27  public class ResultLoaderMap {
28  
29    private final Map<String, LoadPair> loaderMap = new HashMap<String, LoadPair>();
30  
31    public void addLoader(String property, MetaObject metaResultObject, ResultLoader resultLoader) {
32      String upperFirst = getUppercaseFirstProperty(property);
33      if (!upperFirst.equalsIgnoreCase(property) && loaderMap.containsKey(upperFirst)) {
34        throw new ExecutorException("Nested lazy loaded result property '" + property +
35                "' for query id '" + resultLoader.mappedStatement.getId() +
36                " already exists in the result map. The leftmost property of all lazy loaded properties must be unique within a result map.");
37      }
38      loaderMap.put(upperFirst, new LoadPair(property, metaResultObject, resultLoader));
39    }
40    
41    public Set<String> getPropertyNames() {
42      return loaderMap.keySet();
43    }
44  
45    public int size() {
46      return loaderMap.size();
47    }
48  
49    public boolean hasLoader(String property) {
50      return loaderMap.containsKey(property.toUpperCase(Locale.ENGLISH));
51    }
52    
53    public boolean load(String property) throws SQLException {
54      LoadPair pair = loaderMap.remove(property.toUpperCase(Locale.ENGLISH));
55      if (pair != null) {
56        pair.load();
57        return true;
58      }
59      return false;
60    }
61  
62    public void loadAll() throws SQLException {
63      final Set<String> methodNameSet = loaderMap.keySet();
64      String[] methodNames = methodNameSet.toArray(new String[methodNameSet.size()]);
65      for (String methodName : methodNames) {
66        load(methodName);
67      }
68    }
69  
70    private static String getUppercaseFirstProperty(String property) {
71      String[] parts = property.split("\\.");
72      return parts[0].toUpperCase(Locale.ENGLISH);
73    }
74  
75    private static class LoadPair {
76      private String property;
77      private MetaObject metaResultObject;
78      private ResultLoader resultLoader;
79  
80      private LoadPair(String property, MetaObject metaResultObject, ResultLoader resultLoader) {
81        this.property = property;
82        this.metaResultObject = metaResultObject;
83        this.resultLoader = resultLoader;
84      }
85  
86      public void load() throws SQLException {
87        metaResultObject.setValue(property, resultLoader.loadResult());
88      }
89    }
90  }

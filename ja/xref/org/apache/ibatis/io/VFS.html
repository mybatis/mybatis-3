21  import java.net.URL;
22  import java.util.ArrayList;
23  import java.util.Arrays;
24  import java.util.Collections;
25  import java.util.List;

27  import org.apache.ibatis.logging.Log;
28  import org.apache.ibatis.logging.LogFactory;

30  /**
31   * Provides a very simple API for accessing resources within an application server.
32   * 
33   * @author Ben Gunter
34   */
35  public abstract class VFS {
36    private static final Log log = LogFactory.getLog(ResolverUtil.class);
37  
38    /** The built-in implementations. */
39    public static final Class<?>[] IMPLEMENTATIONS = { JBoss6VFS.class, DefaultVFS.class };
40  
41    /** The list to which implementations are added by {@link #addImplClass(Class)}. */
42    public static final List<Class<? extends VFS>> USER_IMPLEMENTATIONS = new ArrayList<Class<? extends VFS>>();
43  
44    /** Singleton instance. */
45    private static VFS instance;
46  
47    /**
48     * Get the singleton {@link VFS} instance. If no {@link VFS} implementation can be found for the
49     * current environment, then this method returns null.
50     */
51    @SuppressWarnings("unchecked")
52    public static VFS getInstance() {
53      if (instance != null)
54        return instance;
55  
56      // Try the user implementations first, then the built-ins
57      List<Class<? extends VFS>> impls = new ArrayList<Class<? extends VFS>>();
58      impls.addAll(USER_IMPLEMENTATIONS);
59      impls.addAll(Arrays.asList((Class<? extends VFS>[]) IMPLEMENTATIONS));
60  
61      // Try each implementation class until a valid one is found
62      VFS vfs = null;
63      for (int i = 0; vfs == null || !vfs.isValid(); i++) {
64        Class<? extends VFS> impl = impls.get(i);
65        try {
66          vfs = impl.newInstance();
67          if (vfs == null || !vfs.isValid()) {
68            log.debug("VFS implementation " + impl.getName() +
69                " is not valid in this environment.");
70          }
71        } catch (InstantiationException e) {
72          log.error("Failed to instantiate " + impl, e);
73          return null;
74        } catch (IllegalAccessException e) {
75          log.error("Failed to instantiate " + impl, e);
76          return null;
77        }
78      }
79  
80      log.debug("Using VFS adapter " + vfs.getClass().getName());
81      return VFS.instance = vfs;
82    }
83  
84    /**
85     * Adds the specified class to the list of {@link VFS} implementations. Classes added in this
86     * manner are tried in the order they are added and before any of the built-in implementations.
87     * 
88     * @param clazz The {@link VFS} implementation class to add.
89     */
90    public static void addImplClass(Class<? extends VFS> clazz) {
91      if (clazz != null)
92        USER_IMPLEMENTATIONS.add(clazz);
93    }
94  
95    /** Get a class by name. If the class is not found then return null. */
96    protected static Class<?> getClass(String className) {
97      try {
98        return Thread.currentThread().getContextClassLoader().loadClass(className);
99  //      return ReflectUtil.findClass(className);
100     } catch (ClassNotFoundException e) {
101       log.debug("Class not found: " + className);
102       return null;
103     }
104   }
105 
106   /**
107    * Get a method by name and parameter types. If the method is not found then return null.
108    * 
109    * @param clazz The class to which the method belongs.
110    * @param methodName The name of the method.
111    * @param parameterTypes The types of the parameters accepted by the method.
112    */
113   protected static Method getMethod(Class<?> clazz, String methodName, Class<?>... parameterTypes) {
114     try {
115       if (clazz == null)
116         return null;
117       else
118         return clazz.getMethod(methodName, parameterTypes);
119     } catch (SecurityException e) {
120       log.error("Security exception looking for method " + clazz.getName() + "." + methodName + ".  Cause: " + e);
121       return null;
122     } catch (NoSuchMethodException e) {
123       log.error("Method not found " + clazz.getName() + "." + methodName + "." + methodName + ".  Cause: " + e);
124       return null;
125     }
126   }
127 
128   /**
129    * Invoke a method on an object and return whatever it returns.
130    * 
131    * @param method The method to invoke.
132    * @param object The instance or class (for static methods) on which to invoke the method.
133    * @param parameters The parameters to pass to the method.
134    * @return Whatever the method returns.
135    * @throws IOException If I/O errors occur
136    * @throws StripesRuntimeException If anything else goes wrong
137    */
138   @SuppressWarnings("unchecked")
139   protected static <T> T invoke(Method method, Object object, Object... parameters)
140       throws IOException, RuntimeException {
141     try {
142       return (T) method.invoke(object, parameters);
143     } catch (IllegalArgumentException e) {
144       throw new RuntimeException(e);
145     } catch (IllegalAccessException e) {
146       throw new RuntimeException(e);
147     } catch (InvocationTargetException e) {
148       if (e.getTargetException() instanceof IOException)
149         throw (IOException) e.getTargetException();
150       else
151         throw new RuntimeException(e);
152     }
153   }
154 
155   /**
156    * Get a list of {@link URL}s from the context classloader for all the resources found at the
157    * specified path.
158    * 
159    * @param path The resource path.
160    * @return A list of {@link URL}s, as returned by {@link ClassLoader#getResources(String)}.
161    * @throws IOException If I/O errors occur
162    */
163   protected static List<URL> getResources(String path) throws IOException {
164     return Collections.list(Thread.currentThread().getContextClassLoader().getResources(path));
165   }
166 
167   /** Return true if the {@link VFS} implementation is valid for the current environment. */
168   public abstract boolean isValid();
169 
170   /**
171    * Recursively list the full resource path of all the resources that are children of the
172    * resource identified by a URL.
173    * 
174    * @param url The URL that identifies the resource to list.
175    * @param forPath The path to the resource that is identified by the URL. Generally, this is the
176    *            value passed to {@link #getResources(String)} to get the resource URL.
177    * @return A list containing the names of the child resources.
178    * @throws IOException If I/O errors occur
179    */
180   protected abstract List<String> list(URL url, String forPath) throws IOException;
181 
182   /**
183    * Recursively list the full resource path of all the resources that are children of all the
184    * resources found at the specified path.
185    * 
186    * @param path The path of the resource(s) to list.
187    * @return A list containing the names of the child resources.
188    * @throws IOException If I/O errors occur
189    */
190   public List<String> list(String path) throws IOException {
191     List<String> names = new ArrayList<String>();
192     for (URL url : getResources(path)) {
193       names.addAll(list(url, path));
194     }
195     return names;
196   }

75    /*
76     * Returns the URL of the resource on the classpath
77     *
78     * @param loader   The classloader used to fetch the resource
79     * @param resource The resource to find
80     * @return The resource
81     * @throws java.io.IOException If the resource cannot be found or read
82     */
83    public static URL getResourceURL(ClassLoader loader, String resource) throws IOException {
84      URL url = classLoaderWrapper.getResourceAsURL(resource, loader);
85      if (url == null) throw new IOException("Could not find resource " + resource);
86      return url;

114   /*
115    * Returns a resource on the classpath as a Properties object
116    *
117    * @param resource The resource to find

121   public static Properties getResourceAsProperties(String resource) throws IOException {
122     Properties props = new Properties();

145   /*

152   public static Reader getResourceAsReader(String resource) throws IOException {
153     Reader reader;
154     if (charset == null) {

203   /*
204    * Gets a URL as an input stream
205    *
206    * @param urlString - the URL to get
207    * @return An input stream with the data from the URL
208    * @throws java.io.IOException If the resource cannot be found or read
209    */
210   public static InputStream getUrlAsStream(String urlString) throws IOException {
211     URL url = new URL(urlString);
212     URLConnection conn = url.openConnection();
213     return conn.getInputStream();
214   }

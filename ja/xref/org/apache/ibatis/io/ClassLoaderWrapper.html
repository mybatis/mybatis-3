104   /*
105    * Try to get a resource from a group of classloaders
106    *
107    * @param resource    - the resource to get
108    * @param classLoader - the classloaders to examine
109    * @return the resource or null
110    */
111   InputStream getResourceAsStream(String resource, ClassLoader[] classLoader) {
112     for (ClassLoader cl : classLoader) {
113       if (null != cl) {
114 
115         // try to find the resource as passed
116         InputStream returnValue = cl.getResourceAsStream(resource);
117 
118         // now, some class loaders want this leading "/", so we'll add it and try again if we didn't find the resource
119         if (null == returnValue) returnValue = cl.getResourceAsStream("/" + resource);
120 
121         if (null != returnValue) return returnValue;
122       }
123     }
124     return null;
125   }

127   /*
128    * Get a resource as a URL using the current class path
129    *
130    * @param resource    - the resource to locate
131    * @param classLoader - the class loaders to examine
132    * @return the resource or null
133    */
134   URL getResourceAsURL(String resource, ClassLoader[] classLoader) {
135 
136     URL url;
137 
138     for (ClassLoader cl : classLoader) {
139 
140       if (null != cl) {
141 
142         // look for the resource as passed in...
143         url = cl.getResource(resource);
144 
145         // ...but some class loaders want this leading "/", so we'll add it
146         // and try again if we didn't find the resource
147         if (null == url) url = cl.getResource("/" + resource);
148 
149         // "It's always in the last place I look for it!"
150         // ... because only an idiot would keep looking for it after finding it, so stop looking already.
151         if (null != url) return url;
152 
153       }
154 
155     }
156 
157     // didn't find it anywhere.
158     return null;
159 
160   }

162   /*
163    * Attempt to load a class from a group of classloaders
164    *
165    * @param name        - the class to load
166    * @param classLoader - the group of classloaders to examine
167    * @return the class
168    * @throws ClassNotFoundException - Remember the wisdom of Judge Smails: Well, the world needs ditch diggers, too.
169    */
170   Class<?> classForName(String name, ClassLoader[] classLoader) throws ClassNotFoundException {
171 
172     for (ClassLoader cl : classLoader) {
173 
174       if (null != cl) {
175 
176         try {
177 
178           Class<?> c = Class.forName(name, true, cl);
179 
180           if (null != c) return c;
181 
182         } catch (ClassNotFoundException e) {
183           // we'll ignore this until all classloaders fail to locate the class
184         }
185 
186       }
187 
188     }
189 
190     throw new ClassNotFoundException("Cannot find class: " + name);
191 
192   }

194   ClassLoader[] getClassLoaders(ClassLoader classLoader) {
195     return new ClassLoader[]{
196         classLoader, 
197         defaultClassLoader, 
198         Thread.currentThread().getContextClassLoader(), 
199         getClass().getClassLoader(), 
200         systemClassLoader}; 
201   }

24  import java.io.Reader;

27  public class NodeletParserTest {
28  
29    @Test
30    public void shouldParseAttribute() throws Exception {
31      NodeEventParser parser = new NodeEventParser();
32      NodeHandler handler = new NodeHandler();
33      parser.addNodeletHandler(handler);
34      parser.setVariables(new Properties() {
35        {
36          setProperty("id_var", "1234567890");
37        }
38      });
39      Reader resource = Resources.getResourceAsReader("com/resources/nodelet_test.xml");
40      parser.parse(resource);
41      Employee emp = handler.getEmployee();
42      assertEquals(1234567890, emp.getId());
43      assertEquals("Jim", emp.getFirstName());
44      assertEquals("Smith", emp.getLastName());
45      assertEquals(new Date(1970 - 1900, 6 - 1, 15), emp.getBirthDate());
46      assertEquals(5.8, emp.getHeight(), 0.0001);
47      assertEquals("ft", emp.getHeightUnits());
48      assertEquals(200, emp.getWeight(), 0.0001);
49      assertEquals("lbs", emp.getWeightUnits());
50    }

53    public static class NodeHandler {
54  
55      private Employee employee = new Employee();
56      private int year;
57      private int month;
58      private int day;

60      public Employee getEmployee() {
61        return employee;
62      }
63  
64      @NodeEvent("/employee")
65      public void id(XNode node) {
66        employee.setId(node.getIntAttribute("id", 0));
67      }
68  
69      @NodeEvent("/employee/first_name")
70      public void firstName(XNode node) {
71        employee.setFirstName(node.getStringBody(""));
72      }
73  
74      @NodeEvent("/employee/last_name")
75      public void lastName(XNode node) {
76        employee.setLastName(node.getStringBody(""));
77      }
78  
79      @NodeEvent("/employee/birth_date/year")
80      public void year(XNode node) {
81        year = node.getIntBody(0);
82      }
83  
84      @NodeEvent("/employee/birth_date/month")
85      public void month(XNode node) {
86        month = node.getIntBody(0);
87      }
88  
89      @NodeEvent("/employee/birth_date/day")
90      public void day(XNode node) {
91        day = node.getIntBody(0);
92      }
93  
94      @NodeEvent("/employee/birth_date/end()")
95      public void birth_date(XNode node) {
96        employee.setBirthDate(new Date(year - 1900, month - 1, day));
97      }
98  
99      @NodeEvent("/employee/height")
100     public void height(XNode node) {
101       employee.setHeight(node.getDoubleBody(0.0));
102       employee.setHeightUnits(node.getStringAttribute("units", ""));
103     }
104 
105     @NodeEvent("/employee/weight")
106     public void weight(XNode node) {

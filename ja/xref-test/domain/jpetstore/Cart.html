16  package domain.jpetstore;
17  
18  import java.io.Serializable;
19  import java.math.BigDecimal;
20  import java.util.*;
21  
22  public class Cart implements Serializable {
23  
24    private final Map itemMap = Collections.synchronizedMap(new HashMap());
25    private final List itemList = new ArrayList();
26  
27    public Iterator getCartItems() {
28      return itemList.iterator();
29    }
30  
31    public List getCartItemList() {
32      return itemList;
33    }
34  
35    public int getNumberOfItems() {
36      return itemList.size();
37    }
38  
39    public boolean containsItemId(String itemId) {
40      return itemMap.containsKey(itemId);
41    }
42  
43    public void addItem(Item item, boolean isInStock) {
44      CartItem cartItem = (CartItem) itemMap.get(item.getItemId());
45      if (cartItem == null) {
46        cartItem = new CartItem();
47        cartItem.setItem(item);
48        cartItem.setQuantity(0);
49        cartItem.setInStock(isInStock);
50        itemMap.put(item.getItemId(), cartItem);
51        itemList.add(cartItem);
52      }
53      cartItem.incrementQuantity();
54    }
55  
56  
57    public Item removeItemById(String itemId) {
58      CartItem cartItem = (CartItem) itemMap.remove(itemId);
59      if (cartItem == null) {
60        return null;
61      } else {
62        itemList.remove(cartItem);
63        return cartItem.getItem();
64      }
65    }
66  
67    public void incrementQuantityByItemId(String itemId) {
68      CartItem cartItem = (CartItem) itemMap.get(itemId);
69      cartItem.incrementQuantity();
70    }
71  
72    public void setQuantityByItemId(String itemId, int quantity) {
73      CartItem cartItem = (CartItem) itemMap.get(itemId);
74      cartItem.setQuantity(quantity);
75    }
76  
77    public BigDecimal getSubTotal() {
78      BigDecimal subTotal = new BigDecimal("0");
79      Iterator items = getCartItems();
80      while (items.hasNext()) {
81        CartItem cartItem = (CartItem) items.next();
82        Item item = cartItem.getItem();
83        BigDecimal listPrice = item.getListPrice();
84        BigDecimal quantity = new BigDecimal(String.valueOf(cartItem.getQuantity()));
85        subTotal = subTotal.add(listPrice.multiply(quantity));
86      }
87      return subTotal;
88    }
89  
90  }

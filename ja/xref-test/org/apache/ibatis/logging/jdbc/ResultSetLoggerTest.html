16  package org.apache.ibatis.logging.jdbc;

18  import static org.mockito.Mockito.verify;
19  import static org.mockito.Mockito.when;
20  
21  import java.sql.ResultSet;
22  import java.sql.ResultSetMetaData;
23  import java.sql.SQLException;
24  import java.sql.Types;
25  
26  import org.apache.ibatis.logging.Log;
27  import org.apache.ibatis.logging.jdbc.ResultSetLogger;
28  import org.junit.Test;
29  import org.junit.runner.RunWith;
30  import org.mockito.Mock;
31  import org.mockito.runners.MockitoJUnitRunner;
32  
33  @RunWith(MockitoJUnitRunner.class)
34  public class ResultSetLoggerTest {
35  
36    @Mock
37    private ResultSet rs;
38  
39    @Mock
40    private Log log;
41  
42    @Mock
43    private ResultSetMetaData metaData;
44  
45    public void setup(int type) throws SQLException {
46      when(rs.next()).thenReturn(true);
47      when(rs.getMetaData()).thenReturn(metaData);
48      when(metaData.getColumnCount()).thenReturn(1);
49      when(metaData.getColumnType(1)).thenReturn(type);
50      when(metaData.getColumnLabel(1)).thenReturn("ColumnName");
51      when(rs.getString(1)).thenReturn("value");
52      when(log.isTraceEnabled()).thenReturn(true);
53      ResultSet resultSet = ResultSetLogger.newInstance(rs, log);
54      resultSet.next();
55    }
56  
57    @Test
58    public void shouldNotPrintBlobs() throws SQLException {
59      setup(Types.LONGNVARCHAR);
60      verify(log).trace("<==    Columns: ColumnName");
61      verify(log).trace("<==        Row: <<BLOB>>");
62    }
63  
64    @Test
65    public void shouldPrintVarchars() throws SQLException {
66      setup(Types.VARCHAR);
67      verify(log).trace("<==    Columns: ColumnName");
68      verify(log).trace("<==        Row: value");
69    }

71  }

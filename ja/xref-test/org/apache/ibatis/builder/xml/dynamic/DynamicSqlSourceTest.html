275   @Test
276   public void shouldIterateOnceForEachItemInCollection() throws Exception {
277     final HashMap<String, String[]> parameterObject = new HashMap<String, String[]>() {{
278       put("array", new String[]{"one", "two", "three"});
279     }};
280     final String expected = "SELECT * FROM BLOG WHERE ID in (  one = ? AND two = ? AND three = ? )";
281     DynamicSqlSource source = createDynamicSqlSource(
282         new TextSqlNode("SELECT * FROM BLOG WHERE ID in"),
283         new ForEachSqlNode(new Configuration(),mixedContents(new TextSqlNode("${item} = #{item}")), "array", "index", "item", "(", ")", "AND"));
284     BoundSql boundSql = source.getBoundSql(parameterObject);
285     assertEquals(expected, boundSql.getSql());
286     assertEquals(3, boundSql.getParameterMappings().size());
287     assertEquals("__frch_item_0", boundSql.getParameterMappings().get(0).getProperty());
288     assertEquals("__frch_item_1", boundSql.getParameterMappings().get(1).getProperty());
289     assertEquals("__frch_item_2", boundSql.getParameterMappings().get(2).getProperty());
290   }

292   private DynamicSqlSource createDynamicSqlSource(SqlNode... contents) throws IOException, SQLException {
293     createBlogDataSource();
294     final String resource = "org/apache/ibatis/builder/MapperConfig.xml";
295     final Reader reader = Resources.getResourceAsReader(resource);
296     SqlSessionFactory sqlMapper = new SqlSessionFactoryBuilder().build(reader);
297     Configuration configuration = sqlMapper.getConfiguration();
298     MixedSqlNode sqlNode = mixedContents(contents);
299     return new DynamicSqlSource(configuration, sqlNode);
300   }
301 
302   private MixedSqlNode mixedContents(SqlNode... contents) {
303     return new MixedSqlNode(Arrays.asList(contents));
304   }

89    public static class EnumOrderTypeHandler<E extends Enum<E>> extends BaseTypeHandler<E> {
90  
91      private E[] constants;
92  
93      public EnumOrderTypeHandler(Class<E> javaType) {
94        constants = javaType.getEnumConstants();
95      }
96  
97      @Override
98      public void setNonNullParameter(PreparedStatement ps, int i, E parameter, JdbcType jdbcType) throws SQLException {
99        ps.setInt(i, parameter.ordinal() + 1); // 0 means NULL so add +1
100     }
101 
102     @Override
103     public E getNullableResult(ResultSet rs, String columnName) throws SQLException {
104       int index = rs.getInt(columnName) - 1;
105       return index < 0 ? null : constants[index];
106     }
107 
108     @Override
109     public E getNullableResult(ResultSet rs, int columnIndex) throws SQLException {
110       int index = rs.getInt(rs.getInt(columnIndex)) - 1;
111       return index < 0 ? null : constants[index];
112     }
113 
114     @Override
115     public E getNullableResult(CallableStatement cs, int columnIndex) throws SQLException {
116       int index = cs.getInt(columnIndex) - 1;
117       return index < 0 ? null : constants[index];
118     }
119   }

121   @Test
122   public void registerJavaTypeInitializingTypeHandler() {
123     final String MAPPER_CONFIG = "<?xml version=\"1.0\" encoding=\"UTF-8\" ?>\n"
124         + "<!DOCTYPE configuration PUBLIC \"-//mybatis.org//DTD Config 3.0//EN\" \"http://mybatis.org/dtd/mybatis-3-config.dtd\">\n" 
125         + "<configuration>\n" 
126         + "  <typeHandlers>\n"
127         + "    <typeHandler javaType=\"org.apache.ibatis.builder.XmlConfigBuilderTest$MyEnum\"\n"
128         + "      handler=\"org.apache.ibatis.builder.XmlConfigBuilderTest$EnumOrderTypeHandler\"/>\n" 
129         + "  </typeHandlers>\n" 
130         + "</configuration>\n";
131 
132     XMLConfigBuilder builder = new XMLConfigBuilder(new StringReader(MAPPER_CONFIG));
133     builder.parse();
134 
135     TypeHandlerRegistry typeHandlerRegistry = builder.getConfiguration().getTypeHandlerRegistry();
136     TypeHandler<MyEnum> typeHandler = typeHandlerRegistry.getTypeHandler(MyEnum.class);
137 
138     assertTrue(typeHandler instanceof EnumOrderTypeHandler);
139     assertArrayEquals(MyEnum.values(), ((EnumOrderTypeHandler) typeHandler).constants);
140   }

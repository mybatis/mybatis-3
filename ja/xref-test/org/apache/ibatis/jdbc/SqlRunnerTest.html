32    @Test
33    public void shouldSelectOne() throws Exception {
34      DataSource ds = createUnpooledDataSource(JPETSTORE_PROPERTIES);
35      runScript(ds, JPETSTORE_DDL);
36      runScript(ds, JPETSTORE_DATA);
37      Connection connection = ds.getConnection();
38      SqlRunner exec = new SqlRunner(connection);
39      Map<String, Object> row = exec.selectOne("SELECT * FROM PRODUCT WHERE PRODUCTID = ?", "FI-SW-01");
40      connection.close();
41      assertEquals("FI-SW-01", row.get("PRODUCTID"));
42    }

56    @Test
57    public void shouldInsert() throws Exception {
58      DataSource ds = createUnpooledDataSource(BLOG_PROPERTIES);
59      runScript(ds, BLOG_DDL);
60      Connection connection = ds.getConnection();
61      SqlRunner exec = new SqlRunner(connection);
62      exec.setUseGeneratedKeySupport(true);
63      int id = exec.insert("INSERT INTO author (username, password, email, bio) VALUES (?,?,?,?)", "someone", "******", "someone@apache.org", Null.LONGVARCHAR);
64      Map<String,Object> row = exec.selectOne("SELECT * FROM author WHERE username = ?", "someone");
65      connection.rollback();
66      connection.close();
67      assertTrue(SqlRunner.NO_GENERATED_KEY != id);
68      assertEquals("someone", row.get("USERNAME"));
69    }
70  
71    @Test
72    public void shouldUpdateCategory() throws Exception {
73      DataSource ds = createUnpooledDataSource(JPETSTORE_PROPERTIES);
74      runScript(ds, JPETSTORE_DDL);
75      runScript(ds, JPETSTORE_DATA);
76      Connection connection = ds.getConnection();
77      SqlRunner exec = new SqlRunner(connection);
78      int count = exec.update("update product set category = ? where productid = ?", "DOGS", "FI-SW-01");
79      Map<String,Object> row = exec.selectOne("SELECT * FROM PRODUCT WHERE PRODUCTID = ?", "FI-SW-01");
80      connection.close();
81      assertEquals("DOGS", row.get("CATEGORY"));
82      assertEquals(1, count);
83    }

85    @Test
86    public void shouldDeleteOne() throws Exception {
87      DataSource ds = createUnpooledDataSource(JPETSTORE_PROPERTIES);
88      runScript(ds, JPETSTORE_DDL);
89      runScript(ds, JPETSTORE_DATA);
90      Connection connection = ds.getConnection();
91      SqlRunner exec = new SqlRunner(connection);
92      int count = exec.delete("delete from item");
93      List<Map<String,Object>> rows = exec.selectAll("SELECT * FROM ITEM");
94      connection.close();
95      assertEquals(28, count);
96      assertEquals(0, rows.size());
97    }
98  
99    @Test
100   public void shouldDemonstrateDDLThroughRunMethod() throws Exception {
101     DataSource ds = createUnpooledDataSource(JPETSTORE_PROPERTIES);
102     Connection connection = ds.getConnection();
103     SqlRunner exec = new SqlRunner(connection);
104     exec.run("CREATE TABLE BLAH(ID INTEGER)");
105     exec.run("insert into BLAH values (1)");
106     List<Map<String,Object>> rows = exec.selectAll("SELECT * FROM BLAH");
107     exec.run("DROP TABLE BLAH");
108     connection.close();
109     assertEquals(1, rows.size());
110   }

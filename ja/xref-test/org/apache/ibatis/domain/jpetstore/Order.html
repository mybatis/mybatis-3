274   public void initOrder(Account account, Cart cart) {
275 
276     username = account.getUsername();
277     orderDate = new Date();
278 
279     shipToFirstName = account.getFirstName();
280     shipToLastName = account.getLastName();
281     shipAddress1 = account.getAddress1();
282     shipAddress2 = account.getAddress2();
283     shipCity = account.getCity();
284     shipState = account.getState();
285     shipZip = account.getZip();
286     shipCountry = account.getCountry();
287 
288     billToFirstName = account.getFirstName();
289     billToLastName = account.getLastName();
290     billAddress1 = account.getAddress1();
291     billAddress2 = account.getAddress2();
292     billCity = account.getCity();
293     billState = account.getState();
294     billZip = account.getZip();
295     billCountry = account.getCountry();
296 
297     totalPrice = cart.getSubTotal();
298 
299     creditCard = "999 9999 9999 9999";
300     expiryDate = "12/03";
301     cardType = "Visa";
302     courier = "UPS";
303     locale = "CA";
304     status = "P";
305 
306 
307     Iterator<CartItem> i = cart.getCartItems();
308     while (i.hasNext()) {
309       CartItem cartItem = (CartItem) i.next();
310       addLineItem(cartItem);
311     }
312 
313   }
314 
315   public void addLineItem(CartItem cartItem) {
316     LineItem lineItem = new LineItem(lineItems.size() + 1, cartItem);
317     addLineItem(lineItem);
318   }

320   public void addLineItem(LineItem lineItem) {
321     lineItems.add(lineItem);
322   }

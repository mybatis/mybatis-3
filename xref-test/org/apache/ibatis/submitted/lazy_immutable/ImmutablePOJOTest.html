16  package org.apache.ibatis.submitted.lazy_immutable;
17  
18  import org.apache.ibatis.io.Resources;
19  import org.apache.ibatis.jdbc.ScriptRunner;
20  
21  import java.io.PrintWriter;
22  import java.io.Reader;
23  import java.sql.Connection;
24  import java.sql.DriverManager;
25  
26  import org.apache.ibatis.session.SqlSessionFactoryBuilder;
27  import org.apache.ibatis.session.SqlSession;
28  import org.apache.ibatis.session.SqlSessionFactory;
29  import org.junit.BeforeClass;
30  import org.junit.Test;
31  
32  import static org.junit.Assert.*;
33  
34  public final class ImmutablePOJOTest {
35  
36      private static final Integer POJO_ID = 1;
37      private static SqlSessionFactory factory;
38  
39      @BeforeClass
40      public static void setupClass() throws Exception {
41          Connection conn = null;
42  
43          try {
44              Class.forName("org.hsqldb.jdbcDriver");
45              conn = DriverManager.getConnection("jdbc:hsqldb:mem:lazy_immutable", "sa", "");
46  
47              Reader reader = Resources.getResourceAsReader("org/apache/ibatis/submitted/lazy_immutable/CreateDB.sql");
48  
49              ScriptRunner runner = new ScriptRunner(conn);
50              runner.setLogWriter(null);
51              runner.setErrorLogWriter(new PrintWriter(System.err));
52              runner.runScript(reader);
53              conn.commit();
54              reader.close();
55  
56              reader = Resources.getResourceAsReader("org/apache/ibatis/submitted/lazy_immutable/ibatisConfig.xml");
57              factory = new SqlSessionFactoryBuilder().build(reader);
58              reader.close();
59          } finally {
60              if (conn != null) {
61                  conn.close();
62              }
63          }
64      }
65  
66      @Test
67      public void testLoadLazyImmutablePOJO() {
68          final SqlSession session = factory.openSession();
69          try {
70              final ImmutablePOJOMapper mapper = session.getMapper(ImmutablePOJOMapper.class);
71              final ImmutablePOJO pojo = mapper.getImmutablePOJO(POJO_ID);
72  
73              assertEquals(POJO_ID, pojo.getId());
74              assertNotNull("Description should not be null.", pojo.getDescription());
75              assertFalse("Description should not be empty.", pojo.getDescription().length() == 0);
76          } finally {
77              session.close();
78          }
79      }
80  
81  }

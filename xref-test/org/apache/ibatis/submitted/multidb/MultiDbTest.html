32  public class MultiDbTest {
33  
34    protected static SqlSessionFactory sqlSessionFactory;
35    protected static SqlSessionFactory sqlSessionFactory2;
36  
37    @BeforeClass
38    public static void setUp() throws Exception {
39      Connection conn = null;
40  
41      try {
42        Class.forName("org.hsqldb.jdbcDriver");
43        conn = DriverManager.getConnection("jdbc:hsqldb:mem:multidb", "sa", "");
44  
45        Reader reader = Resources.getResourceAsReader("org/apache/ibatis/submitted/multidb/CreateDB.sql");
46  
47        ScriptRunner runner = new ScriptRunner(conn);
48        runner.setLogWriter(null);
49        runner.setErrorLogWriter(null);
50        runner.runScript(reader);
51        conn.commit();
52        reader.close();
53  
54        reader = Resources.getResourceAsReader("org/apache/ibatis/submitted/multidb/MultiDbConfig.xml");
55        sqlSessionFactory = new SqlSessionFactoryBuilder().build(reader);
56        reader.close();
57      } finally {
58        if (conn != null) {
59          conn.close();
60        }
61      }
62    }
63  
64    @Test
65    public void shouldExecuteHsqlQuery() {
66      SqlSession sqlSession = sqlSessionFactory.openSession();
67      try {
68        MultiDbMapper mapper = sqlSession.getMapper(MultiDbMapper.class);
69        String answer = mapper.select1(1);
70        assertEquals("hsql", answer);
71      } finally {
72        sqlSession.close();
73      }
74    }
75

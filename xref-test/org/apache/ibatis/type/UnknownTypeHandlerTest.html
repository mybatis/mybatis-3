16  package org.apache.ibatis.type;
17  
18  import java.sql.SQLException;
19  
20  import static org.junit.Assert.assertEquals;
21  import static org.mockito.Mockito.doThrow;
22  import static org.mockito.Mockito.verify;
23  import static org.mockito.Mockito.when;
24  import static org.mockito.Mockito.spy;
25  
26  import org.apache.ibatis.executor.result.ResultMapException;
27  import org.junit.Assert;
28  import org.junit.Test;
29  
30  public class UnknownTypeHandlerTest extends BaseTypeHandlerTest {
31  
32    private static final TypeHandler<Object> TYPE_HANDLER = spy(new UnknownTypeHandler(new TypeHandlerRegistry()));
33  
34    @Override
35    @Test
36    public void shouldSetParameter() throws Exception {
37      TYPE_HANDLER.setParameter(ps, 1, "Hello", null);
38      verify(ps).setString(1, "Hello");
39    }
40  
41    @Override
42    @Test
43    public void shouldGetResultFromResultSetByName() throws Exception {
44      when(rs.getMetaData()).thenReturn(rsmd);
45      when(rsmd.getColumnCount()).thenReturn(1);
46      when(rsmd.getColumnName(1)).thenReturn("column");
47      when(rsmd.getColumnClassName(1)).thenReturn(String.class.getName());
48      when(rsmd.getColumnType(1)).thenReturn(JdbcType.VARCHAR.TYPE_CODE);
49      when(rs.getString("column")).thenReturn("Hello");
50      when(rs.wasNull()).thenReturn(false);
51      assertEquals("Hello", TYPE_HANDLER.getResult(rs, "column"));
52    }
53  
54    @Override
55    public void shouldGetResultNullFromResultSetByName() throws Exception {
56      // Unnecessary
57    }
58  
59    @Override
60    @Test
61    public void shouldGetResultFromResultSetByPosition() throws Exception {
62      when(rs.getMetaData()).thenReturn(rsmd);
63      when(rsmd.getColumnClassName(1)).thenReturn(String.class.getName());
64      when(rsmd.getColumnType(1)).thenReturn(JdbcType.VARCHAR.TYPE_CODE);
65      when(rs.getString(1)).thenReturn("Hello");
66      when(rs.wasNull()).thenReturn(false);
67      assertEquals("Hello", TYPE_HANDLER.getResult(rs, 1));
68    }
69  
70    @Override
71    public void shouldGetResultNullFromResultSetByPosition() throws Exception {
72      // Unnecessary
73    }
74  
75    @Override
76    @Test
77    public void shouldGetResultFromCallableStatement() throws Exception {
78      when(cs.getObject(1)).thenReturn("Hello");
79      when(cs.wasNull()).thenReturn(false);
80      assertEquals("Hello", TYPE_HANDLER.getResult(cs, 1));
81    }
82  
83    @Override
84    public void shouldGetResultNullFromCallableStatement() throws Exception {
85      // Unnecessary
86    }
87  
88    @Test
89    public void setParameterWithNullParameter() throws Exception {
90      TYPE_HANDLER.setParameter(ps, 0, null, JdbcType.INTEGER);
91      verify(ps).setNull(0, JdbcType.INTEGER.TYPE_CODE);
92    }
93  
94    @Test
95    public void setParameterWithNullParameterThrowsException() throws SQLException {
96      doThrow(new SQLException("invalid column")).when(ps).setNull(1, JdbcType.INTEGER.TYPE_CODE);
97      try {
98        TYPE_HANDLER.setParameter(ps, 1, null, JdbcType.INTEGER);
99        Assert.fail("Should have thrown a TypeException");
100     } catch (Exception e) {
101       Assert.assertTrue("Expected TypedException", e instanceof TypeException);
102       Assert.assertTrue("Parameter index is in exception", e.getMessage().contains("parameter #1"));
103     }
104   }
105 
106   @Test
107   public void setParameterWithNonNullParameterThrowsException() throws SQLException {
108     doThrow(new SQLException("invalid column")).when((UnknownTypeHandler)TYPE_HANDLER).setNonNullParameter(ps, 1, 99, JdbcType.INTEGER);
109     try {
110       TYPE_HANDLER.setParameter(ps, 1, 99, JdbcType.INTEGER);
111       Assert.fail("Should have thrown a TypeException");
112     } catch (Exception e) {
113       Assert.assertTrue("Expected TypedException", e instanceof TypeException);
114       Assert.assertTrue("Parameter index is in exception", e.getMessage().contains("parameter #1"));
115     }
116   }
117 
118   @Test
119   public void getResultWithResultSetAndColumnNameThrowsException() throws SQLException {
120     doThrow(new SQLException("invalid column")).when((UnknownTypeHandler)TYPE_HANDLER).getNullableResult(rs, "foo");
121     try {
122       TYPE_HANDLER.getResult(rs, "foo");
123       Assert.fail("Should have thrown a ResultMapException");
124     } catch (Exception e) {
125       Assert.assertTrue("Expected ResultMapException", e instanceof ResultMapException);
126       Assert.assertTrue("column name is not in exception", e.getMessage().contains("column 'foo'"));
127     }
128   }
129 
130   @Test

134       TYPE_HANDLER.getResult(rs, 1);

136     } catch (Exception e) {

139     }
140   }

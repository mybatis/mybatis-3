60    /*
61     * This is a test case for iBATIS JIRA-250 "SELECT statement
62     * returns unexpected result when 'groupBy' and 'nullValue'
63     * are specified in resultMaps."
64     * <p/>
65     * The problem was that when a child object in a resultmap only
66     * contained null values it would still be created when one of the
67     * properties of the child object contained a nullValue attribute
68     * in the ResultMap. The nullValue would be applied before checking
69     * whether all properties of the child were 'null', so at least 1
70     * property would always be non-null (the reason for the
71     * nullValue attribute).
72     * <p/>
73     * The fix is to first check whether all properties of the child are
74     * 'null', and if the child object contains at least 1 non-null property
75     * to then only create the child object and apply the nullValue attribute.
76     *
77     * @throws Exception none should be thrown.
78     */
79    public void testGroupByJIRA250() throws Exception {
80      List list = sqlMap.queryForList("getAllProductCategoriesJIRA250", null);
81      Category cat = (Category) list.get(0);
82      assertEquals(0, cat.getProductList().size());
83    }

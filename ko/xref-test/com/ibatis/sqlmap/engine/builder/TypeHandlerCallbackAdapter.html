430   public static class CallableStatementResultSet implements InvocationHandler {
431 
432     private CallableStatement cs;
433 
434     private CallableStatementResultSet(CallableStatement cs) {
435       this.cs = cs;
436     }
437 
438     public static ResultSet newProxy(CallableStatement cs) {
439       return (ResultSet) Proxy.newProxyInstance(cs.getClass().getClassLoader(),new Class[]{ResultSet.class}, new CallableStatementResultSet(cs));
440     }
441 
442     public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
443       Method csMethod = cs.getClass().getMethod(method.getName(),method.getParameterTypes());
444       return csMethod.invoke(cs,args);
445     }

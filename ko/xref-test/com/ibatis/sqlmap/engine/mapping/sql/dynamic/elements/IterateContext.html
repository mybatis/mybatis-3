62    public IterateContext(Object collection, SqlTag tag, IterateContext parent) {
63      this.parent = parent;
64      this.tag = tag;
65      if (collection instanceof Collection) {
66        this.iterator = ((Collection) collection).iterator();
67      } else if (collection instanceof Iterator) {
68        this.iterator = ((Iterator) collection);
69      } else if (collection.getClass().isArray()) {
70        List list = arrayToList(collection);
71        this.iterator = list.iterator();
72      } else {
73        throw new SqlMapException("ParameterObject or property was not a Collection, Array or Iterator.");
74      }
75    }

108   private List arrayToList(Object array) {
109     List list = null;
110     if (array instanceof Object[]) {
111       list = Arrays.asList((Object[]) array);
112     } else {
113       list = new ArrayList();
114       for (int i = 0, n = Array.getLength(array); i < n; i++) {
115         list.add(Array.get(array, i));
116       }
117     }
118     return list;
119   }

188   /*
189    * Returns the last property of any bean specified in this IterateContext.
190    *
191    * @return The last property of any bean specified in this IterateContext.
192    */
193   public String getEndProperty() {
194     if (parent != null) {
195       int parentPropertyIndex = property.indexOf(parent.getProperty());
196       if (parentPropertyIndex > -1) {
197         int endPropertyIndex1 = property.indexOf(']', parentPropertyIndex);
198         int endPropertyIndex2 = property.indexOf('.', parentPropertyIndex);
199         return property.substring(parentPropertyIndex + Math.max(endPropertyIndex1, endPropertyIndex2) + 1, property.length());
200       } else {
201         return property;
202       }
203     } else {
204       return property;
205     }
206   }

208   /*
209    * Replaces value of a tag property to match it's value with current iteration and all other iterations.
210    *
211    * @param tagProperty the property of a TagHandler.
212    * @return A Map containing the modified tag property in PROCESS_STRING key and the index where the modification occured in PROCESS_INDEX key.
213    */
214   protected Map processTagProperty(String tagProperty) {
215     if (parent != null) {
216       Map parentResult = parent.processTagProperty(tagProperty);
217       return this.addIndex((String) parentResult.get(PROCESS_STRING), (Integer) parentResult.get(PROCESS_INDEX));
218     } else {
219       return this.addIndex(tagProperty, 0);
220     }
221   }
222 
223   /*
224    * Replaces value of a tag property to match it's value with current iteration and all other iterations.
225    *
226    * @param tagProperty the property of a TagHandler.
227    * @return The tag property with all "[]" replaced with the correct iteration value.
228    */
229   public String addIndexToTagProperty(String tagProperty) {
230     Map map = this.processTagProperty(tagProperty);
231     return (String) map.get(PROCESS_STRING);
232   }

234   /*
235    * Adds index value to the first found property matching this Iteration starting at index startIndex.
236    *
237    * @param input      The input String.
238    * @param startIndex The index where search for property begins.
239    * @return A Map containing the modified tag property in PROCESS_STRING key and the index where the modification occured in PROCESS_INDEX key.
240    */
241   protected Map addIndex(String input, int startIndex) {
242     if (input != null && input.startsWith("[")) {
243       input = "_collection" + input;
244     }
245     String endProperty = getEndProperty() + "[";
246     int propertyIndex = input.indexOf(endProperty, startIndex);
247     int modificationIndex = 0;
248     // Is the iterate property in the tag property at all?
249     if (propertyIndex > -1) {
250       // Make sure the tag property does not already have a number.
251       if (input.charAt(propertyIndex + endProperty.length()) == ']') {
252         // Add iteration number to property.
253         input = input.substring(0, propertyIndex + endProperty.length()) + this.getIndex() + input.substring(propertyIndex + endProperty.length());
254         modificationIndex = propertyIndex + endProperty.length();
255       }
256     }
257     Map ret = new HashMap();
258     ret.put(PROCESS_INDEX, new Integer(modificationIndex));
259     ret.put(PROCESS_STRING, input);
260     return ret;
261   }

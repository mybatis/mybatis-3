23

25    private int quantity;

27    private BigDecimal total;
28  
29    public boolean isInStock() {

31    }
32  
33    public void setInStock(boolean inStock) {
34      this.inStock = inStock;
35    }
36

38      return total;
39    }
40  
41    public Item getItem() {
42      return item;
43    }
44  
45    public void setItem(Item item) {
46      this.item = item;

48    }
49  
50    public int getQuantity() {
51      return quantity;
52    }
53  
54    public void setQuantity(int quantity) {
55      this.quantity = quantity;
56      calculateTotal();
57    }
58  
59    public void incrementQuantity() {
60      quantity++;
61      calculateTotal();
62    }
63  
64    private void calculateTotal() {
65      if (item != null && item.getListPrice() != null) {
66        total = item.getListPrice().multiply(new BigDecimal(quantity));
67      } else {
68        total = null;
69      }
70    }

75    private ClassIntrospector(Class clazz) {
76      className = clazz.getName();
77      addMethods(clazz);
78      readablePropertyNames = (String[]) getMethods.keySet().toArray(new String[getMethods.keySet().size()]);
79      writeablePropertyNames = (String[]) setMethods.keySet().toArray(new String[setMethods.keySet().size()]);
80    }

82    private void addMethods(Class cls) {
83      Method[] methods = getAllMethodsForClass(cls);
84      for (int i = 0; i < methods.length; i++) {
85        String name = methods[i].getName();
86        if (name.startsWith("set") && name.length() > 3) {
87          if (methods[i].getParameterTypes().length == 1) {
88            name = dropCase(name);
89            if (setMethods.containsKey(name)) {
90              throw new RuntimeException("Illegal overloaded setter method for property " + name + " in class " + cls.getName() +
91                  ".  This breaks the JavaBeans specification and can cause unpredicatble results.");
92            }
93            setMethods.put(name, methods[i]);
94            setTypes.put(name, methods[i].getParameterTypes()[0]);
95          }
96        } else if (name.startsWith("get") && name.length() > 3) {
97          if (methods[i].getParameterTypes().length == 0) {
98            name = dropCase(name);
99            getMethods.put(name, methods[i]);
100           getTypes.put(name, methods[i].getReturnType());
101         }
102       } else if (name.startsWith("is") && name.length() > 2) {
103         if (methods[i].getParameterTypes().length == 0) {
104           name = dropCase(name);
105           getMethods.put(name, methods[i]);
106           getTypes.put(name, methods[i].getReturnType());
107         }
108       }
109       name = null;
110     }
111   }

113   private Method[] getAllMethodsForClass(Class cls) {
114     if (cls.isInterface()) {
115       // interfaces only have public methods - so the
116       // simple call is all we need (this will also get superinterface methods)
117       return cls.getMethods();
118     } else {
119       // need to get all the declared methods in this class
120       // and any super-class - then need to set access appropriatly
121       // for private methods
122       return getClassMethods(cls);
123     }
124   }

126   /*
127    * This method returns an array containing all methods
128    * declared in this class and any superclass.
129    * We use this method, instead of the simpler Class.getMethods(),
130    * because we want to look for private methods as well.
131    *
132    * @param cls
133    * @return
134    */
135   private Method[] getClassMethods(Class cls) {
136     HashMap uniqueMethods = new HashMap();
137     Class currentClass = cls;
138     while (currentClass != null) {
139       addUniqueMethods(uniqueMethods, currentClass.getDeclaredMethods());
140 
141       // we also need to look for interface methods - 
142       // because the class may be abstract
143       Class[] interfaces = currentClass.getInterfaces();
144       for (int i = 0; i < interfaces.length; i++) {
145         addUniqueMethods(uniqueMethods, interfaces[i].getMethods());
146       }
147 
148       currentClass = currentClass.getSuperclass();
149     }
150 
151     Collection methods = uniqueMethods.values();
152 
153     return (Method[]) methods.toArray(new Method[methods.size()]);
154   }
155 
156   private void addUniqueMethods(HashMap uniqueMethods, Method[] methods) {
157     for (int i = 0; i < methods.length; i++) {
158       Method currentMethod = methods[i];
159       String signature = getSignature(currentMethod);
160       // check to see if the method is already known
161       // if it is known, then an extended class must have
162       // overridden a method
163       if (!uniqueMethods.containsKey(signature)) {
164         if (canAccessPrivateMethods()) {
165           try {
166             currentMethod.setAccessible(true);
167           } catch (Exception e) {
168             // Ignored. This is only a final precaution, nothing we can do.
169           }
170         }
171 
172         uniqueMethods.put(signature, currentMethod);
173       }
174     }
175   }

177   private String getSignature(Method method) {
178     StringBuffer sb = new StringBuffer();
179     sb.append(method.getName());
180     Class[] parameters = method.getParameterTypes();
181 
182     for (int i = 0; i < parameters.length; i++) {
183       if (i == 0) {
184         sb.append(':');
185       } else {
186         sb.append(',');
187       }
188       sb.append(parameters[i].getName());
189     }
190 
191     return sb.toString();
192   }

194   private boolean canAccessPrivateMethods() {
195     try {
196       System.getSecurityManager().checkPermission(new ReflectPermission("suppressAccessChecks"));
197       return true;
198     } catch (SecurityException e) {
199       return false;
200     } catch (NullPointerException e) {
201       return true;
202     }
203   }

205   private static String dropCase(String name) {
206     if (name.startsWith("is")) {
207       name = name.substring(2);
208     } else if (name.startsWith("get") || name.startsWith("set")) {
209       name = name.substring(3);
210     } else {
211       throw new RuntimeException("Error parsing property name '" + name + "'.  Didn't start with 'is', 'get' or 'set'.");
212     }
213 
214     if (name.length() == 1 || (name.length() > 1 && !Character.isUpperCase(name.charAt(1)))) {
215       name = name.substring(0, 1).toLowerCase(Locale.US) + name.substring(1);
216     }
217 
218     return name;
219   }

325    * Tells us if the class passed in is a knwon common type

327    * @param clazz The class to check
328    * @return True if the class is known
329    */
330   public static boolean isKnownType(Class clazz) {
331     if (SIMPLE_TYPE_SET.contains(clazz)) {
332       return true;
333     } else if (Collection.class.isAssignableFrom(clazz)) {
334       return true;
335     } else if (Map.class.isAssignableFrom(clazz)) {
336       return true;
337     } else if (List.class.isAssignableFrom(clazz)) {
338       return true;
339     } else if (Set.class.isAssignableFrom(clazz)) {
340       return true;
341     } else if (Iterator.class.isAssignableFrom(clazz)) {
342       return true;
343     } else {
344       return false;
345     }
346   }

348   /*
349    * Gets an instance of ClassInfo for the specified class.
350    *
351    * @param clazz The class for which to lookup the method cache.
352    * @return The method cache for the class
353    */
354   public static ClassIntrospector getInstance(Class clazz) {
355     if (cacheEnabled) {
356       synchronized (clazz) {
357         ClassIntrospector cache = (ClassIntrospector) CLASS_INFO_MAP.get(clazz);
358         if (cache == null) {
359           cache = new ClassIntrospector(clazz);
360           CLASS_INFO_MAP.put(clazz, cache);
361         }
362         return cache;
363       }
364     } else {
365       return new ClassIntrospector(clazz);
366     }
367   }

373   /*
374    * Examines a Throwable object and gets it's root cause
375    *
376    * @param t - the exception to examine
377    * @return The root cause
378    */
379   public static Throwable unwrapThrowable(Throwable t) {
380     Throwable t2 = t;
381     while (true) {
382       if (t2 instanceof InvocationTargetException) {
383         t2 = ((InvocationTargetException) t).getTargetException();
384       } else if (t instanceof UndeclaredThrowableException) {
385         t2 = ((UndeclaredThrowableException) t).getUndeclaredThrowable();
386       } else {
387         return t2;
388       }
389     }
390   }

16  package org.apache.ibatis.transaction.managed;
17  
18  import static org.junit.Assert.assertEquals;
19  import static org.mockito.Mockito.verify;
20  import static org.mockito.Mockito.verifyNoMoreInteractions;
21  
22  import java.sql.Connection;
23  import java.util.Properties;
24  
25  import org.apache.ibatis.BaseDataTest;
26  import org.apache.ibatis.transaction.Transaction;
27  import org.apache.ibatis.transaction.TransactionFactory;
28  import org.junit.Test;
29  import org.junit.runner.RunWith;
30  import org.mockito.Mock;
31  import org.mockito.junit.MockitoJUnitRunner;
32  
33  @RunWith(MockitoJUnitRunner.class)
34  public class ManagedTransactionFactoryTest extends BaseDataTest {
35  
36    @Mock
37    private Connection conn;
38  
39    @Test
40    public void shouldEnsureThatCallsToManagedTransactionAPIDoNotForwardToManagedConnections() throws Exception {
41      TransactionFactory tf = new ManagedTransactionFactory();
42      tf.setProperties(new Properties());
43      Transaction tx = tf.newTransaction(conn);
44      assertEquals(conn, tx.getConnection());
45      tx.commit();
46      tx.rollback();
47      tx.close();
48      verify(conn).close();
49    }
50  
51    @Test
52    public void shouldEnsureThatCallsToManagedTransactionAPIDoNotForwardToManagedConnectionsAndDoesNotCloseConnection() throws Exception {
53      TransactionFactory tf = new ManagedTransactionFactory();
54      Properties props = new Properties();
55      props.setProperty("closeConnection", "false");
56      tf.setProperties(props);
57      Transaction tx = tf.newTransaction(conn);
58      assertEquals(conn, tx.getConnection());
59      tx.commit();
60      tx.rollback();
61      tx.close();
62      verifyNoMoreInteractions(conn);
63    }
64  
65  }

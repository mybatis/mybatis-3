36  public class ForEachTest {
37  
38    private static SqlSessionFactory sqlSessionFactory;
39  
40    @Rule
41    public ExpectedException ex = ExpectedException.none();
42  
43    @BeforeClass
44    public static void setUp() throws Exception {
45      // create a SqlSessionFactory
46      Reader reader = Resources.getResourceAsReader("org/apache/ibatis/submitted/foreach/mybatis-config.xml");
47      sqlSessionFactory = new SqlSessionFactoryBuilder().build(reader);
48      reader.close();
49  
50      // populate in-memory database
51      SqlSession session = sqlSessionFactory.openSession();
52      Connection conn = session.getConnection();
53      reader = Resources.getResourceAsReader("org/apache/ibatis/submitted/foreach/CreateDB.sql");
54      ScriptRunner runner = new ScriptRunner(conn);
55      runner.setLogWriter(null);
56      runner.runScript(reader);
57      reader.close();
58      session.close();
59    }
60  
61    @Test
62    public void shouldGetAUser() {
63      SqlSession sqlSession = sqlSessionFactory.openSession();
64      try {
65        Mapper mapper = sqlSession.getMapper(Mapper.class);
66        User testProfile = new User();
67        testProfile.setId(2);
68        User friendProfile = new User();
69        friendProfile.setId(6);
70        List<User> friendList = new ArrayList<User>();
71        friendList.add(friendProfile);
72        testProfile.setFriendList(friendList);
73        User user = mapper.getUser(testProfile);
74        Assert.assertEquals("User6", user.getName());
75      } finally {
76        sqlSession.close();
77      }
78    }
79  
80    @Test
81    public void shouldHandleComplexNullItem() {
82      SqlSession sqlSession = sqlSessionFactory.openSession();
83      try {
84        Mapper mapper = sqlSession.getMapper(Mapper.class);
85        User user1 = new User();
86        user1.setId(2);
87        user1.setName("User2");
88        List<User> users = new ArrayList<User>();
89        users.add(user1);
90        users.add(null);
91        int count = mapper.countByUserList(users);
92        Assert.assertEquals(1, count);
93      } finally {
94        sqlSession.close();
95      }
96    }
97  
98    @Test
99    public void shouldHandleMoreComplexNullItem() {
100     SqlSession sqlSession = sqlSessionFactory.openSession();
101     try {
102       Mapper mapper = sqlSession.getMapper(Mapper.class);
103       User user1 = new User();
104       User bestFriend = new User();
105       bestFriend.setId(5);
106       user1.setBestFriend(bestFriend);
107       List<User> users = new ArrayList<User>();
108       users.add(user1);
109       users.add(null);
110       int count = mapper.countByBestFriend(users);
111       Assert.assertEquals(1, count);
112     } finally {
113       sqlSession.close();
114     }
115   }
116 
117   @Test
118   public void nullItemInContext() {
119     SqlSession sqlSession = sqlSessionFactory.openSession();
120     try {
121       Mapper mapper = sqlSession.getMapper(Mapper.class);
122       User user1 = new User();
123       user1.setId(3);
124       List<User> users = new ArrayList<User>();
125       users.add(user1);
126       users.add(null);
127       String name = mapper.selectWithNullItemCheck(users);
128       Assert.assertEquals("User3", name);
129     } finally {
130       sqlSession.close();
131     }
132   }

134   @Test
135   public void shouldReportMissingPropertyName() {
136     ex.expect(PersistenceException.class);
137     ex.expectMessage("There is no getter for property named 'idd' in 'class org.apache.ibatis.submitted.foreach.User'");
138 
139     SqlSession sqlSession = sqlSessionFactory.openSession();
140     try {
141       Mapper mapper = sqlSession.getMapper(Mapper.class);
142       mapper.typoInItemProperty(Arrays.asList(new User()));
143     } finally {
144       sqlSession.close();
145     }
146   }

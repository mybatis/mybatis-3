42    @BeforeClass
43    public static void initDatabase() throws Exception {
44      Connection conn = null;
45  
46      try {
47        Class.forName("org.hsqldb.jdbcDriver");
48        conn = DriverManager.getConnection("jdbc:hsqldb:mem:sptest", "sa", "");
49  
50        Reader reader = Resources.getResourceAsReader("org/apache/ibatis/submitted/sptests/CreateDB.sql");
51  
52        ScriptRunner runner = new ScriptRunner(conn);
53        runner.setDelimiter("go");
54        runner.setLogWriter(null);
55        runner.setErrorLogWriter(null);
56        runner.runScript(reader);
57        conn.commit();
58        reader.close();
59  
60        reader = Resources.getResourceAsReader("org/apache/ibatis/submitted/sptests/MapperConfig.xml");
61        sqlSessionFactory = new SqlSessionFactoryBuilder().build(reader);
62        reader.close();
63      } finally {
64        if (conn != null) {
65          conn.close();
66        }
67      }
68    }

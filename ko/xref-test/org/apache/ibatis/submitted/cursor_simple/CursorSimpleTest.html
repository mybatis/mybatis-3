57      @Test
58      public void shouldGetAllUser() {
59          SqlSession sqlSession = sqlSessionFactory.openSession();
60          Mapper mapper = sqlSession.getMapper(Mapper.class);
61          Cursor<User> usersCursor = mapper.getAllUsers();
62          try {
63              Assert.assertFalse(usersCursor.isOpen());
64  
65              // Cursor is just created, current index is -1
66              Assert.assertEquals(-1, usersCursor.getCurrentIndex());
67  
68              Iterator<User> iterator = usersCursor.iterator();
69  
70              // Check if hasNext, fetching is started
71              Assert.assertTrue(iterator.hasNext());
72              Assert.assertTrue(usersCursor.isOpen());
73              Assert.assertFalse(usersCursor.isConsumed());
74  
75              // next() has not been called, index is still -1
76              Assert.assertEquals(-1, usersCursor.getCurrentIndex());
77  
78              User user = iterator.next();
79              Assert.assertEquals("User1", user.getName());
80              Assert.assertEquals(0, usersCursor.getCurrentIndex());
81  
82              user = iterator.next();
83              Assert.assertEquals("User2", user.getName());
84              Assert.assertEquals(1, usersCursor.getCurrentIndex());
85  
86              user = iterator.next();
87              Assert.assertEquals("User3", user.getName());
88              Assert.assertEquals(2, usersCursor.getCurrentIndex());
89  
90              user = iterator.next();
91              Assert.assertEquals("User4", user.getName());
92              Assert.assertEquals(3, usersCursor.getCurrentIndex());
93  
94              // Check no more elements
95              Assert.assertFalse(iterator.hasNext());
96              Assert.assertFalse(usersCursor.isOpen());
97              Assert.assertTrue(usersCursor.isConsumed());
98          } finally {
99              sqlSession.close();
100         }
101     }

103     @Test
104     public void testCursorClosedOnSessionClose() {
105         SqlSession sqlSession = sqlSessionFactory.openSession();
106         Mapper mapper = sqlSession.getMapper(Mapper.class);
107         Cursor<User> usersCursor = mapper.getAllUsers();
108         try {
109             Assert.assertFalse(usersCursor.isOpen());
110 
111             Iterator<User> iterator = usersCursor.iterator();
112 
113             // Check if hasNext, fetching is started
114             Assert.assertTrue(iterator.hasNext());
115             Assert.assertTrue(usersCursor.isOpen());
116             Assert.assertFalse(usersCursor.isConsumed());
117 
118             // Consume only the first result
119             User user = iterator.next();
120             Assert.assertEquals("User1", user.getName());
121 
122             // Check there is still remaining elements
123             Assert.assertTrue(iterator.hasNext());
124             Assert.assertTrue(usersCursor.isOpen());
125             Assert.assertFalse(usersCursor.isConsumed());
126         } finally {
127             sqlSession.close();
128         }
129 
130         // The cursor was not fully consumed, but it should be close since we closed the session
131         Assert.assertFalse(usersCursor.isOpen());
132         Assert.assertFalse(usersCursor.isConsumed());
133     }

242 
243             // Consume only the first result
244             User user = iterator.next();
245             Assert.assertEquals("User1", user.getName());
246 
247             usersCursor.close();
248             // Check multiple close are no-op
249             usersCursor.close();
250

254             Assert.assertFalse(usersCursor.isConsumed());
255         } finally {
256             sqlSession.close();

260     @Test
261     public void testCursorUsageAfterClose() throws IOException {
262         SqlSession sqlSession = sqlSessionFactory.openSession();
263         Mapper mapper = sqlSession.getMapper(Mapper.class);
264         Cursor<User> usersCursor = mapper.getAllUsers();
265 
266         try {
267             Iterator<User> iterator = usersCursor.iterator();
268             User user = iterator.next();
269             Assert.assertEquals("User1", user.getName());
270             Assert.assertEquals(0, usersCursor.getCurrentIndex());
271 
272             user = iterator.next();
273             Assert.assertEquals("User2", user.getName());
274             Assert.assertEquals(1, usersCursor.getCurrentIndex());
275 
276             usersCursor.close();
277 
278             // hasNext now return false, since the cursor is closed
279             Assert.assertFalse(iterator.hasNext());
280             Assert.assertFalse(usersCursor.isOpen());
281             Assert.assertFalse(usersCursor.isConsumed());
282 
283             // trying next() will fail
284             iterator.next();
285 
286             Assert.fail("We should have failed since Cursor is closed");
287         } catch (CursorException e) {
288             // We had an exception and current index has not changed
289             Assert.assertEquals(1, usersCursor.getCurrentIndex());
290             return;
291         } finally {
292             sqlSession.close();
293         }
294 
295         Assert.fail("Should have returned earlier");
296     }

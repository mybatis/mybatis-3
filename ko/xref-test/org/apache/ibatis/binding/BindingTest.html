615   @Test
616   public void shouldCacheMapperMethod() throws Exception {
617     final SqlSession session = sqlSessionFactory.openSession();
618     try {
619 
620       // Create another mapper instance with a method cache we can test against:
621       final MapperProxyFactory<BoundBlogMapper> mapperProxyFactory = new MapperProxyFactory<BoundBlogMapper>(BoundBlogMapper.class);
622       assertEquals(BoundBlogMapper.class, mapperProxyFactory.getMapperInterface());
623       final BoundBlogMapper mapper = mapperProxyFactory.newInstance(session);
624       assertNotSame(mapper, mapperProxyFactory.newInstance(session));
625       assertTrue(mapperProxyFactory.getMethodCache().isEmpty());
626 
627       // Mapper methods we will call later:
628       final Method selectBlog = BoundBlogMapper.class.getMethod("selectBlog", Integer.TYPE);
629       final Method selectBlogByIdUsingConstructor = BoundBlogMapper.class.getMethod("selectBlogByIdUsingConstructor", Integer.TYPE);
630 
631       // Call mapper method and verify it is cached:
632       mapper.selectBlog(1);
633       assertEquals(1, mapperProxyFactory.getMethodCache().size());
634       assertTrue(mapperProxyFactory.getMethodCache().containsKey(selectBlog));
635       final MapperMethod cachedSelectBlog = mapperProxyFactory.getMethodCache().get(selectBlog);
636 
637       // Call mapper method again and verify the cache is unchanged:
638       session.clearCache();
639       mapper.selectBlog(1);
640       assertEquals(1, mapperProxyFactory.getMethodCache().size());
641       assertSame(cachedSelectBlog, mapperProxyFactory.getMethodCache().get(selectBlog));
642 
643       // Call another mapper method and verify that it shows up in the cache as well:
644       session.clearCache();
645       mapper.selectBlogByIdUsingConstructor(1);
646       assertEquals(2, mapperProxyFactory.getMethodCache().size());
647       assertSame(cachedSelectBlog, mapperProxyFactory.getMethodCache().get(selectBlog));
648       assertTrue(mapperProxyFactory.getMethodCache().containsKey(selectBlogByIdUsingConstructor));
649 
650     } finally {
651       session.close();
652     }
653   }

135   @Override
136   public void handleOutputParameters(CallableStatement cs) throws SQLException {
137     final Object parameterObject = parameterHandler.getParameterObject();
138     final MetaObject metaParam = configuration.newMetaObject(parameterObject);
139     final List<ParameterMapping> parameterMappings = boundSql.getParameterMappings();
140     for (int i = 0; i < parameterMappings.size(); i++) {
141       final ParameterMapping parameterMapping = parameterMappings.get(i);
142       if (parameterMapping.getMode() == ParameterMode.OUT || parameterMapping.getMode() == ParameterMode.INOUT) {
143         if (ResultSet.class.equals(parameterMapping.getJavaType())) {
144           handleRefCursorOutputParameter((ResultSet) cs.getObject(i + 1), parameterMapping, metaParam);
145         } else {
146           final TypeHandler<?> typeHandler = parameterMapping.getTypeHandler();
147           metaParam.setValue(parameterMapping.getProperty(), typeHandler.getResult(cs, i + 1));
148         }
149       }
150     }
151   }
152 
153   private void handleRefCursorOutputParameter(ResultSet rs, ParameterMapping parameterMapping, MetaObject metaParam) throws SQLException {
154     if (rs == null) {
155       return;
156     }
157     try {
158       final String resultMapId = parameterMapping.getResultMapId();
159       final ResultMap resultMap = configuration.getResultMap(resultMapId);
160       final DefaultResultHandler resultHandler = new DefaultResultHandler(objectFactory);
161       final ResultSetWrapper rsw = new ResultSetWrapper(rs, configuration);
162       handleRowValues(rsw, resultMap, resultHandler, new RowBounds(), null);
163       metaParam.setValue(parameterMapping.getProperty(), resultHandler.getResultList());
164     } finally {
165       // issue #228 (close resultsets)
166       closeResultSet(rs);
167     }
168   }

173   @Override
174   public List<Object> handleResultSets(Statement stmt) throws SQLException {
175     ErrorContext.instance().activity("handling results").object(mappedStatement.getId());
176 
177     final List<Object> multipleResults = new ArrayList<Object>();
178 
179     int resultSetCount = 0;
180     ResultSetWrapper rsw = getFirstResultSet(stmt);
181 
182     List<ResultMap> resultMaps = mappedStatement.getResultMaps();
183     int resultMapCount = resultMaps.size();
184     validateResultMapsCount(rsw, resultMapCount);
185     while (rsw != null && resultMapCount > resultSetCount) {
186       ResultMap resultMap = resultMaps.get(resultSetCount);
187       handleResultSet(rsw, resultMap, multipleResults, null);
188       rsw = getNextResultSet(stmt);
189       cleanUpAfterHandlingResultSet();
190       resultSetCount++;
191     }
192 
193     String[] resultSets = mappedStatement.getResultSets();
194     if (resultSets != null) {
195       while (rsw != null && resultSetCount < resultSets.length) {
196         ResultMapping parentMapping = nextResultMaps.get(resultSets[resultSetCount]);
197         if (parentMapping != null) {
198           String nestedResultMapId = parentMapping.getNestedResultMapId();
199           ResultMap resultMap = configuration.getResultMap(nestedResultMapId);
200           handleResultSet(rsw, resultMap, null, parentMapping);
201         }
202         rsw = getNextResultSet(stmt);
203         cleanUpAfterHandlingResultSet();
204         resultSetCount++;
205       }
206     }
207 
208     return collapseSingleResultList(multipleResults);
209   }

211   @Override
212   public <E> Cursor<E> handleCursorResultSets(Statement stmt) throws SQLException {
213     ErrorContext.instance().activity("handling cursor results").object(mappedStatement.getId());
214 
215     ResultSetWrapper rsw = getFirstResultSet(stmt);
216 
217     List<ResultMap> resultMaps = mappedStatement.getResultMaps();
218 
219     int resultMapCount = resultMaps.size();
220     validateResultMapsCount(rsw, resultMapCount);
221     if (resultMapCount != 1) {
222       throw new ExecutorException("Cursor results cannot be mapped to multiple resultMaps");
223     }
224 
225     ResultMap resultMap = resultMaps.get(0);
226     return new DefaultCursor<E>(this, resultMap, rsw, rowBounds);
227   }

229   private ResultSetWrapper getFirstResultSet(Statement stmt) throws SQLException {
230     ResultSet rs = stmt.getResultSet();
231     while (rs == null) {
232       // move forward to get the first resultset in case the driver
233       // doesn't return the resultset as the first result (HSQLDB 2.1)
234       if (stmt.getMoreResults()) {
235         rs = stmt.getResultSet();
236       } else {
237         if (stmt.getUpdateCount() == -1) {
238           // no more results. Must be no resultset
239           break;
240         }
241       }
242     }
243     return rs != null ? new ResultSetWrapper(rs, configuration) : null;
244   }
245 
246   private ResultSetWrapper getNextResultSet(Statement stmt) throws SQLException {
247     // Making this method tolerant of bad JDBC drivers
248     try {
249       if (stmt.getConnection().getMetaData().supportsMultipleResultSets()) {
250         // Crazy Standard JDBC way of determining if there are more results
251         if (!((!stmt.getMoreResults()) && (stmt.getUpdateCount() == -1))) {
252           ResultSet rs = stmt.getResultSet();
253           return rs != null ? new ResultSetWrapper(rs, configuration) : null;
254         }
255       }
256     } catch (Exception e) {
257       // Intentionally ignored.
258     }
259     return null;
260   }

264       if (rs != null) {
265         rs.close();
266       }

268       // ignore

276   private void validateResultMapsCount(ResultSetWrapper rsw, int resultMapCount) {
277     if (rsw != null && resultMapCount < 1) {
278       throw new ExecutorException("A query was run and no Result Maps were found for the Mapped Statement '" + mappedStatement.getId()
279           + "'.  It's likely that neither a Result Type nor a Result Map was specified.");
280     }
281   }

283   private void handleResultSet(ResultSetWrapper rsw, ResultMap resultMap, List<Object> multipleResults, ResultMapping parentMapping) throws SQLException {
284     try {
285       if (parentMapping != null) {
286         handleRowValues(rsw, resultMap, null, RowBounds.DEFAULT, parentMapping);
287       } else {
288         if (resultHandler == null) {
289           DefaultResultHandler defaultResultHandler = new DefaultResultHandler(objectFactory);
290           handleRowValues(rsw, resultMap, defaultResultHandler, rowBounds, null);
291           multipleResults.add(defaultResultHandler.getResultList());
292         } else {
293           handleRowValues(rsw, resultMap, resultHandler, rowBounds, null);
294         }
295       }
296     } finally {
297       // issue #228 (close resultsets)
298       closeResultSet(rsw.getResultSet());
299     }
300   }

304     return multipleResults.size() == 1 ? (List<Object>) multipleResults.get(0) : multipleResults;

311   public void handleRowValues(ResultSetWrapper rsw, ResultMap resultMap, ResultHandler<?> resultHandler, RowBounds rowBounds, ResultMapping parentMapping) throws SQLException {
312     if (resultMap.hasNestedResultMaps()) {
313       ensureNoRowBounds();
314       checkResultHandler();
315       handleRowValuesForNestedResultMap(rsw, resultMap, resultHandler, rowBounds, parentMapping);
316     } else {
317       handleRowValuesForSimpleResultMap(rsw, resultMap, resultHandler, rowBounds, parentMapping);
318     }
319   }

321   private void ensureNoRowBounds() {
322     if (configuration.isSafeRowBoundsEnabled() && rowBounds != null && (rowBounds.getLimit() < RowBounds.NO_ROW_LIMIT || rowBounds.getOffset() > RowBounds.NO_ROW_OFFSET)) {
323       throw new ExecutorException("Mapped Statements with nested result mappings cannot be safely constrained by RowBounds. "
324           + "Use safeRowBoundsEnabled=false setting to bypass this check.");
325     }
326   }
327 
328   protected void checkResultHandler() {
329     if (resultHandler != null && configuration.isSafeResultHandlerEnabled() && !mappedStatement.isResultOrdered()) {
330       throw new ExecutorException("Mapped Statements with nested result mappings cannot be safely used with a custom ResultHandler. "
331           + "Use safeResultHandlerEnabled=false setting to bypass this check "
332           + "or ensure your statement returns ordered data and set resultOrdered=true on it.");
333     }
334   }

336   private void handleRowValuesForSimpleResultMap(ResultSetWrapper rsw, ResultMap resultMap, ResultHandler<?> resultHandler, RowBounds rowBounds, ResultMapping parentMapping)
337       throws SQLException {
338     DefaultResultContext<Object> resultContext = new DefaultResultContext<Object>();
339     skipRows(rsw.getResultSet(), rowBounds);
340     while (shouldProcessMoreRows(resultContext, rowBounds) && rsw.getResultSet().next()) {
341       ResultMap discriminatedResultMap = resolveDiscriminatedResultMap(rsw.getResultSet(), resultMap, null);
342       Object rowValue = getRowValue(rsw, discriminatedResultMap);
343       storeObject(resultHandler, resultContext, rowValue, parentMapping, rsw.getResultSet());
344     }
345   }
346 
347   private void storeObject(ResultHandler<?> resultHandler, DefaultResultContext<Object> resultContext, Object rowValue, ResultMapping parentMapping, ResultSet rs) throws SQLException {
348     if (parentMapping != null) {
349       linkToParents(rs, parentMapping, rowValue);
350     } else {
351       callResultHandler(resultHandler, resultContext, rowValue);
352     }
353   }
354 
355   @SuppressWarnings("unchecked" /* because ResultHandler<?> is always ResultHandler<Object>*/)
356   private void callResultHandler(ResultHandler<?> resultHandler, DefaultResultContext<Object> resultContext, Object rowValue) {
357     resultContext.nextResultObject(rowValue);
358     ((ResultHandler<Object>)resultHandler).handleResult(resultContext);
359   }
360 
361   private boolean shouldProcessMoreRows(ResultContext<?> context, RowBounds rowBounds) throws SQLException {
362     return !context.isStopped() && context.getResultCount() < rowBounds.getLimit();
363   }
364 
365   private void skipRows(ResultSet rs, RowBounds rowBounds) throws SQLException {
366     if (rs.getType() != ResultSet.TYPE_FORWARD_ONLY) {
367       if (rowBounds.getOffset() != RowBounds.NO_ROW_OFFSET) {
368         rs.absolute(rowBounds.getOffset());
369       }
370     } else {
371       for (int i = 0; i < rowBounds.getOffset(); i++) {
372         rs.next();
373       }
374     }
375   }

381   private Object getRowValue(ResultSetWrapper rsw, ResultMap resultMap) throws SQLException {
382     final ResultLoaderMap lazyLoader = new ResultLoaderMap();
383     Object rowValue = createResultObject(rsw, resultMap, lazyLoader, null);
384     if (rowValue != null && !hasTypeHandlerForResultObject(rsw, resultMap.getType())) {
385       final MetaObject metaObject = configuration.newMetaObject(rowValue);
386       boolean foundValues = this.useConstructorMappings;
387       if (shouldApplyAutomaticMappings(resultMap, false)) {
388         foundValues = applyAutomaticMappings(rsw, resultMap, metaObject, null) || foundValues;
389       }
390       foundValues = applyPropertyMappings(rsw, resultMap, metaObject, lazyLoader, null) || foundValues;
391       foundValues = lazyLoader.size() > 0 || foundValues;
392       rowValue = (foundValues || configuration.isReturnInstanceForEmptyRow()) ? rowValue : null;
393     }
394     return rowValue;
395   }

397   private boolean shouldApplyAutomaticMappings(ResultMap resultMap, boolean isNested) {
398     if (resultMap.getAutoMapping() != null) {
399       return resultMap.getAutoMapping();
400     } else {
401       if (isNested) {
402         return AutoMappingBehavior.FULL == configuration.getAutoMappingBehavior();
403       } else {
404         return AutoMappingBehavior.NONE != configuration.getAutoMappingBehavior();
405       }
406     }
407   }

413   private boolean applyPropertyMappings(ResultSetWrapper rsw, ResultMap resultMap, MetaObject metaObject, ResultLoaderMap lazyLoader, String columnPrefix)
414       throws SQLException {
415     final List<String> mappedColumnNames = rsw.getMappedColumnNames(resultMap, columnPrefix);
416     boolean foundValues = false;
417     final List<ResultMapping> propertyMappings = resultMap.getPropertyResultMappings();
418     for (ResultMapping propertyMapping : propertyMappings) {
419       String column = prependPrefix(propertyMapping.getColumn(), columnPrefix);
420       if (propertyMapping.getNestedResultMapId() != null) {
421         // the user added a column attribute to a nested result map, ignore it
422         column = null;
423       }
424       if (propertyMapping.isCompositeResult()
425           || (column != null && mappedColumnNames.contains(column.toUpperCase(Locale.ENGLISH)))
426           || propertyMapping.getResultSet() != null) {
427         Object value = getPropertyMappingValue(rsw.getResultSet(), metaObject, propertyMapping, lazyLoader, columnPrefix);
428         // issue #541 make property optional
429         final String property = propertyMapping.getProperty();
430         if (property == null) {
431           continue;
432         } else if (value == DEFERED) {
433           foundValues = true;
434           continue;
435         }
436         if (value != null) {
437           foundValues = true;
438         }
439         if (value != null || (configuration.isCallSettersOnNulls() && !metaObject.getSetterType(property).isPrimitive())) {
440           // gcode issue #377, call setter on nulls (value is not 'found')
441           metaObject.setValue(property, value);
442         }
443       }
444     }
445     return foundValues;
446   }
447 
448   private Object getPropertyMappingValue(ResultSet rs, MetaObject metaResultObject, ResultMapping propertyMapping, ResultLoaderMap lazyLoader, String columnPrefix)
449       throws SQLException {
450     if (propertyMapping.getNestedQueryId() != null) {
451       return getNestedQueryMappingValue(rs, metaResultObject, propertyMapping, lazyLoader, columnPrefix);
452     } else if (propertyMapping.getResultSet() != null) {
453       addPendingChildRelation(rs, metaResultObject, propertyMapping);   // TODO is that OK?
454       return DEFERED;
455     } else {
456       final TypeHandler<?> typeHandler = propertyMapping.getTypeHandler();
457       final String column = prependPrefix(propertyMapping.getColumn(), columnPrefix);
458       return typeHandler.getResult(rs, column);
459     }
460   }

462   private List<UnMappedColumnAutoMapping> createAutomaticMappings(ResultSetWrapper rsw, ResultMap resultMap, MetaObject metaObject, String columnPrefix) throws SQLException {
463     final String mapKey = resultMap.getId() + ":" + columnPrefix;
464     List<UnMappedColumnAutoMapping> autoMapping = autoMappingsCache.get(mapKey);
465     if (autoMapping == null) {
466       autoMapping = new ArrayList<UnMappedColumnAutoMapping>();
467       final List<String> unmappedColumnNames = rsw.getUnmappedColumnNames(resultMap, columnPrefix);
468       for (String columnName : unmappedColumnNames) {
469         String propertyName = columnName;
470         if (columnPrefix != null && !columnPrefix.isEmpty()) {
471           // When columnPrefix is specified,
472           // ignore columns without the prefix.
473           if (columnName.toUpperCase(Locale.ENGLISH).startsWith(columnPrefix)) {
474             propertyName = columnName.substring(columnPrefix.length());
475           } else {
476             continue;
477           }
478         }
479         final String property = metaObject.findProperty(propertyName, configuration.isMapUnderscoreToCamelCase());
480         if (property != null && metaObject.hasSetter(property)) {
481           final Class<?> propertyType = metaObject.getSetterType(property);
482           if (typeHandlerRegistry.hasTypeHandler(propertyType, rsw.getJdbcType(columnName))) {
483             final TypeHandler<?> typeHandler = rsw.getTypeHandler(propertyType, columnName);
484             autoMapping.add(new UnMappedColumnAutoMapping(columnName, property, typeHandler, propertyType.isPrimitive()));
485           } else {
486             configuration.getAutoMappingUnknownColumnBehavior()
487                     .doAction(mappedStatement, columnName, property, propertyType);
488           }
489         } else{
490           configuration.getAutoMappingUnknownColumnBehavior()
491                   .doAction(mappedStatement, columnName, (property != null) ? property : propertyName, null);
492         }
493       }
494       autoMappingsCache.put(mapKey, autoMapping);
495     }
496     return autoMapping;
497   }
498   
499   private boolean applyAutomaticMappings(ResultSetWrapper rsw, ResultMap resultMap, MetaObject metaObject, String columnPrefix) throws SQLException {
500     List<UnMappedColumnAutoMapping> autoMapping = createAutomaticMappings(rsw, resultMap, metaObject, columnPrefix);
501     boolean foundValues = false;
502     if (autoMapping.size() > 0) {
503       for (UnMappedColumnAutoMapping mapping : autoMapping) {
504         final Object value = mapping.typeHandler.getResult(rsw.getResultSet(), mapping.column);
505         if (value != null) {
506           foundValues = true;
507         }
508         if (value != null || (configuration.isCallSettersOnNulls() && !mapping.primitive)) {
509           // gcode issue #377, call setter on nulls (value is not 'found')
510           metaObject.setValue(mapping.property, value);
511         }
512       }
513     }
514     return foundValues;
515   }

519   private void linkToParents(ResultSet rs, ResultMapping parentMapping, Object rowValue) throws SQLException {
520     CacheKey parentKey = createKeyForMultipleResults(rs, parentMapping, parentMapping.getColumn(), parentMapping.getForeignColumn());
521     List<PendingRelation> parents = pendingRelations.get(parentKey);
522     if (parents != null) {
523       for (PendingRelation parent : parents) {
524         if (parent != null && rowValue != null) {
525             linkObjects(parent.metaObject, parent.propertyMapping, rowValue);
526         }
527       }
528     }
529   }

531   private void addPendingChildRelation(ResultSet rs, MetaObject metaResultObject, ResultMapping parentMapping) throws SQLException {
532     CacheKey cacheKey = createKeyForMultipleResults(rs, parentMapping, parentMapping.getColumn(), parentMapping.getColumn());
533     PendingRelation deferLoad = new PendingRelation();
534     deferLoad.metaObject = metaResultObject;
535     deferLoad.propertyMapping = parentMapping;
536     List<PendingRelation> relations = pendingRelations.get(cacheKey);
537     // issue #255
538     if (relations == null) {
539       relations = new ArrayList<DefaultResultSetHandler.PendingRelation>();
540       pendingRelations.put(cacheKey, relations);
541     }
542     relations.add(deferLoad);
543     ResultMapping previous = nextResultMaps.get(parentMapping.getResultSet());
544     if (previous == null) {
545       nextResultMaps.put(parentMapping.getResultSet(), parentMapping);
546     } else {
547       if (!previous.equals(parentMapping)) {
548         throw new ExecutorException("Two different properties are mapped to the same resultSet");
549       }
550     }
551   }

553   private CacheKey createKeyForMultipleResults(ResultSet rs, ResultMapping resultMapping, String names, String columns) throws SQLException {
554     CacheKey cacheKey = new CacheKey();
555     cacheKey.update(resultMapping);
556     if (columns != null && names != null) {
557       String[] columnsArray = columns.split(",");
558       String[] namesArray = names.split(",");
559       for (int i = 0 ; i < columnsArray.length ; i++) {
560         Object value = rs.getString(columnsArray[i]);
561         if (value != null) {
562           cacheKey.update(namesArray[i]);
563           cacheKey.update(value);
564         }
565       }
566     }
567     return cacheKey;
568   }

574   private Object createResultObject(ResultSetWrapper rsw, ResultMap resultMap, ResultLoaderMap lazyLoader, String columnPrefix) throws SQLException {
575     this.useConstructorMappings = false; // reset previous mapping result
576     final List<Class<?>> constructorArgTypes = new ArrayList<Class<?>>();
577     final List<Object> constructorArgs = new ArrayList<Object>();
578     Object resultObject = createResultObject(rsw, resultMap, constructorArgTypes, constructorArgs, columnPrefix);
579     if (resultObject != null && !hasTypeHandlerForResultObject(rsw, resultMap.getType())) {
580       final List<ResultMapping> propertyMappings = resultMap.getPropertyResultMappings();
581       for (ResultMapping propertyMapping : propertyMappings) {
582         // issue gcode #109 && issue #149
583         if (propertyMapping.getNestedQueryId() != null && propertyMapping.isLazy()) {
584           resultObject = configuration.getProxyFactory().createProxy(resultObject, lazyLoader, configuration, objectFactory, constructorArgTypes, constructorArgs);
585           break;
586         }
587       }
588     }
589     this.useConstructorMappings = (resultObject != null && !constructorArgTypes.isEmpty()); // set current mapping result
590     return resultObject;
591   }

593   private Object createResultObject(ResultSetWrapper rsw, ResultMap resultMap, List<Class<?>> constructorArgTypes, List<Object> constructorArgs, String columnPrefix)
594       throws SQLException {
595     final Class<?> resultType = resultMap.getType();
596     final MetaClass metaType = MetaClass.forClass(resultType, reflectorFactory);
597     final List<ResultMapping> constructorMappings = resultMap.getConstructorResultMappings();
598     if (hasTypeHandlerForResultObject(rsw, resultType)) {
599       return createPrimitiveResultObject(rsw, resultMap, columnPrefix);
600     } else if (!constructorMappings.isEmpty()) {
601       return createParameterizedResultObject(rsw, resultType, constructorMappings, constructorArgTypes, constructorArgs, columnPrefix);
602     } else if (resultType.isInterface() || metaType.hasDefaultConstructor()) {
603       return objectFactory.create(resultType);
604     } else if (shouldApplyAutomaticMappings(resultMap, false)) {
605       return createByConstructorSignature(rsw, resultType, constructorArgTypes, constructorArgs, columnPrefix);
606     }
607     throw new ExecutorException("Do not know how to create an instance of " + resultType);
608   }

610   Object createParameterizedResultObject(ResultSetWrapper rsw, Class<?> resultType, List<ResultMapping> constructorMappings,
611       List<Class<?>> constructorArgTypes, List<Object> constructorArgs, String columnPrefix) {
612     boolean foundValues = false;
613     for (ResultMapping constructorMapping : constructorMappings) {
614       final Class<?> parameterType = constructorMapping.getJavaType();
615       final String column = constructorMapping.getColumn();
616       final Object value;
617       try {
618         if (constructorMapping.getNestedQueryId() != null) {
619           value = getNestedQueryConstructorValue(rsw.getResultSet(), constructorMapping, columnPrefix);
620         } else if (constructorMapping.getNestedResultMapId() != null) {
621           final ResultMap resultMap = configuration.getResultMap(constructorMapping.getNestedResultMapId());
622           value = getRowValue(rsw, resultMap);
623         } else {
624           final TypeHandler<?> typeHandler = constructorMapping.getTypeHandler();
625           value = typeHandler.getResult(rsw.getResultSet(), prependPrefix(column, columnPrefix));
626         }
627       } catch (ResultMapException e) {
628         throw new ExecutorException("Could not process result for mapping: " + constructorMapping, e);
629       } catch (SQLException e) {
630         throw new ExecutorException("Could not process result for mapping: " + constructorMapping, e);
631       }
632       constructorArgTypes.add(parameterType);
633       constructorArgs.add(value);
634       foundValues = value != null || foundValues;
635     }
636     return foundValues ? objectFactory.create(resultType, constructorArgTypes, constructorArgs) : null;
637   }

639   private Object createByConstructorSignature(ResultSetWrapper rsw, Class<?> resultType, List<Class<?>> constructorArgTypes, List<Object> constructorArgs,
640       String columnPrefix) throws SQLException {
641     for (Constructor<?> constructor : resultType.getDeclaredConstructors()) {
642       if (typeNames(constructor.getParameterTypes()).equals(rsw.getClassNames())) {
643         boolean foundValues = false;
644         for (int i = 0; i < constructor.getParameterTypes().length; i++) {
645           Class<?> parameterType = constructor.getParameterTypes()[i];
646           String columnName = rsw.getColumnNames().get(i);
647           TypeHandler<?> typeHandler = rsw.getTypeHandler(parameterType, columnName);
648           Object value = typeHandler.getResult(rsw.getResultSet(), prependPrefix(columnName, columnPrefix));
649           constructorArgTypes.add(parameterType);
650           constructorArgs.add(value);
651           foundValues = value != null || foundValues;
652         }
653         return foundValues ? objectFactory.create(resultType, constructorArgTypes, constructorArgs) : null;
654       }
655     }
656     throw new ExecutorException("No constructor found in " + resultType.getName() + " matching " + rsw.getClassNames());
657   }

660     List<String> names = new ArrayList<String>();

664     return names;

668     final Class<?> resultType = resultMap.getType();
669     final String columnName;

671       final List<ResultMapping> resultMappingList = resultMap.getResultMappings();

673       columnName = prependPrefix(mapping.getColumn(), columnPrefix);
674     } else {

677     final TypeHandler<?> typeHandler = rsw.getTypeHandler(resultType, columnName);

685   private Object getNestedQueryConstructorValue(ResultSet rs, ResultMapping constructorMapping, String columnPrefix) throws SQLException {
686     final String nestedQueryId = constructorMapping.getNestedQueryId();
687     final MappedStatement nestedQuery = configuration.getMappedStatement(nestedQueryId);
688     final Class<?> nestedQueryParameterType = nestedQuery.getParameterMap().getType();
689     final Object nestedQueryParameterObject = prepareParameterForNestedQuery(rs, constructorMapping, nestedQueryParameterType, columnPrefix);
690     Object value = null;
691     if (nestedQueryParameterObject != null) {
692       final BoundSql nestedBoundSql = nestedQuery.getBoundSql(nestedQueryParameterObject);
693       final CacheKey key = executor.createCacheKey(nestedQuery, nestedQueryParameterObject, RowBounds.DEFAULT, nestedBoundSql);
694       final Class<?> targetType = constructorMapping.getJavaType();
695       final ResultLoader resultLoader = new ResultLoader(configuration, executor, nestedQuery, nestedQueryParameterObject, targetType, key, nestedBoundSql);
696       value = resultLoader.loadResult();
697     }
698     return value;
699   }

701   private Object getNestedQueryMappingValue(ResultSet rs, MetaObject metaResultObject, ResultMapping propertyMapping, ResultLoaderMap lazyLoader, String columnPrefix)
702       throws SQLException {
703     final String nestedQueryId = propertyMapping.getNestedQueryId();
704     final String property = propertyMapping.getProperty();
705     final MappedStatement nestedQuery = configuration.getMappedStatement(nestedQueryId);
706     final Class<?> nestedQueryParameterType = nestedQuery.getParameterMap().getType();
707     final Object nestedQueryParameterObject = prepareParameterForNestedQuery(rs, propertyMapping, nestedQueryParameterType, columnPrefix);
708     Object value = null;
709     if (nestedQueryParameterObject != null) {
710       final BoundSql nestedBoundSql = nestedQuery.getBoundSql(nestedQueryParameterObject);
711       final CacheKey key = executor.createCacheKey(nestedQuery, nestedQueryParameterObject, RowBounds.DEFAULT, nestedBoundSql);
712       final Class<?> targetType = propertyMapping.getJavaType();
713       if (executor.isCached(nestedQuery, key)) {
714         executor.deferLoad(nestedQuery, metaResultObject, property, key, targetType);
715         value = DEFERED;
716       } else {
717         final ResultLoader resultLoader = new ResultLoader(configuration, executor, nestedQuery, nestedQueryParameterObject, targetType, key, nestedBoundSql);
718         if (propertyMapping.isLazy()) {
719           lazyLoader.addLoader(property, metaResultObject, resultLoader);
720           value = DEFERED;
721         } else {
722           value = resultLoader.loadResult();
723         }
724       }
725     }
726     return value;
727   }

739     if (typeHandlerRegistry.hasTypeHandler(parameterType)) {
740       typeHandler = typeHandlerRegistry.getTypeHandler(parameterType);
741     } else {
742       typeHandler = typeHandlerRegistry.getUnknownTypeHandler();
743     }
744     return typeHandler.getResult(rs, prependPrefix(resultMapping.getColumn(), columnPrefix));

747   private Object prepareCompositeKeyParameter(ResultSet rs, ResultMapping resultMapping, Class<?> parameterType, String columnPrefix) throws SQLException {
748     final Object parameterObject = instantiateParameterObject(parameterType);
749     final MetaObject metaObject = configuration.newMetaObject(parameterObject);
750     boolean foundValues = false;
751     for (ResultMapping innerResultMapping : resultMapping.getComposites()) {
752       final Class<?> propType = metaObject.getSetterType(innerResultMapping.getProperty());
753       final TypeHandler<?> typeHandler = typeHandlerRegistry.getTypeHandler(propType);
754       final Object propValue = typeHandler.getResult(rs, prependPrefix(innerResultMapping.getColumn(), columnPrefix));
755       // issue #353 & #560 do not execute nested query if key is null
756       if (propValue != null) {
757         metaObject.setValue(innerResultMapping.getProperty(), propValue);
758         foundValues = true;
759       }
760     }
761     return foundValues ? parameterObject : null;
762   }

778   public ResultMap resolveDiscriminatedResultMap(ResultSet rs, ResultMap resultMap, String columnPrefix) throws SQLException {
779     Set<String> pastDiscriminators = new HashSet<String>();
780     Discriminator discriminator = resultMap.getDiscriminator();
781     while (discriminator != null) {
782       final Object value = getDiscriminatorValue(rs, discriminator, columnPrefix);
783       final String discriminatedMapId = discriminator.getMapIdFor(String.valueOf(value));
784       if (configuration.hasResultMap(discriminatedMapId)) {
785         resultMap = configuration.getResultMap(discriminatedMapId);
786         Discriminator lastDiscriminator = discriminator;
787         discriminator = resultMap.getDiscriminator();
788         if (discriminator == lastDiscriminator || !pastDiscriminators.add(discriminatedMapId)) {
789           break;
790         }
791       } else {
792         break;
793       }
794     }
795     return resultMap;
796   }

798   private Object getDiscriminatorValue(ResultSet rs, Discriminator discriminator, String columnPrefix) throws SQLException {
799     final ResultMapping resultMapping = discriminator.getResultMapping();
800     final TypeHandler<?> typeHandler = resultMapping.getTypeHandler();
801     return typeHandler.getResult(rs, prependPrefix(resultMapping.getColumn(), columnPrefix));
802   }

804   private String prependPrefix(String columnName, String prefix) {
805     if (columnName == null || columnName.length() == 0 || prefix == null || prefix.length() == 0) {
806       return columnName;
807     }
808     return prefix + columnName;
809   }

815   private void handleRowValuesForNestedResultMap(ResultSetWrapper rsw, ResultMap resultMap, ResultHandler<?> resultHandler, RowBounds rowBounds, ResultMapping parentMapping) throws SQLException {
816     final DefaultResultContext<Object> resultContext = new DefaultResultContext<Object>();
817     skipRows(rsw.getResultSet(), rowBounds);
818     Object rowValue = previousRowValue;
819     while (shouldProcessMoreRows(resultContext, rowBounds) && rsw.getResultSet().next()) {
820       final ResultMap discriminatedResultMap = resolveDiscriminatedResultMap(rsw.getResultSet(), resultMap, null);
821       final CacheKey rowKey = createRowKey(discriminatedResultMap, rsw, null);
822       Object partialObject = nestedResultObjects.get(rowKey);
823       // issue #577 && #542
824       if (mappedStatement.isResultOrdered()) {
825         if (partialObject == null && rowValue != null) {
826           nestedResultObjects.clear();
827           storeObject(resultHandler, resultContext, rowValue, parentMapping, rsw.getResultSet());
828         }
829         rowValue = getRowValue(rsw, discriminatedResultMap, rowKey, null, partialObject);
830       } else {
831         rowValue = getRowValue(rsw, discriminatedResultMap, rowKey, null, partialObject);
832         if (partialObject == null) {
833           storeObject(resultHandler, resultContext, rowValue, parentMapping, rsw.getResultSet());
834         }
835       }
836     }
837     if (rowValue != null && mappedStatement.isResultOrdered() && shouldProcessMoreRows(resultContext, rowBounds)) {
838       storeObject(resultHandler, resultContext, rowValue, parentMapping, rsw.getResultSet());
839       previousRowValue = null;
840     } else if (rowValue != null) {
841       previousRowValue = rowValue;
842     }
843   }

849   private Object getRowValue(ResultSetWrapper rsw, ResultMap resultMap, CacheKey combinedKey, String columnPrefix, Object partialObject) throws SQLException {
850     final String resultMapId = resultMap.getId();
851     Object rowValue = partialObject;
852     if (rowValue != null) {
853       final MetaObject metaObject = configuration.newMetaObject(rowValue);
854       putAncestor(rowValue, resultMapId, columnPrefix);
855       applyNestedResultMappings(rsw, resultMap, metaObject, columnPrefix, combinedKey, false);
856       ancestorObjects.remove(resultMapId);
857     } else {
858       final ResultLoaderMap lazyLoader = new ResultLoaderMap();
859       rowValue = createResultObject(rsw, resultMap, lazyLoader, columnPrefix);
860       if (rowValue != null && !hasTypeHandlerForResultObject(rsw, resultMap.getType())) {
861         final MetaObject metaObject = configuration.newMetaObject(rowValue);
862         boolean foundValues = this.useConstructorMappings;
863         if (shouldApplyAutomaticMappings(resultMap, true)) {
864           foundValues = applyAutomaticMappings(rsw, resultMap, metaObject, columnPrefix) || foundValues;
865         }
866         foundValues = applyPropertyMappings(rsw, resultMap, metaObject, lazyLoader, columnPrefix) || foundValues;
867         putAncestor(rowValue, resultMapId, columnPrefix);
868         foundValues = applyNestedResultMappings(rsw, resultMap, metaObject, columnPrefix, combinedKey, true) || foundValues;
869         ancestorObjects.remove(resultMapId);
870         foundValues = lazyLoader.size() > 0 || foundValues;
871         rowValue = (foundValues || configuration.isReturnInstanceForEmptyRow()) ? rowValue : null;
872       }
873       if (combinedKey != CacheKey.NULL_CACHE_KEY) {
874         nestedResultObjects.put(combinedKey, rowValue);
875       }
876     }
877     return rowValue;
878   }

880   private void putAncestor(Object resultObject, String resultMapId, String columnPrefix) {
881     ancestorObjects.put(resultMapId, resultObject);
882   }

888   private boolean applyNestedResultMappings(ResultSetWrapper rsw, ResultMap resultMap, MetaObject metaObject, String parentPrefix, CacheKey parentRowKey, boolean newObject) {
889     boolean foundValues = false;
890     for (ResultMapping resultMapping : resultMap.getPropertyResultMappings()) {
891       final String nestedResultMapId = resultMapping.getNestedResultMapId();
892       if (nestedResultMapId != null && resultMapping.getResultSet() == null) {
893         try {
894           final String columnPrefix = getColumnPrefix(parentPrefix, resultMapping);
895           final ResultMap nestedResultMap = getNestedResultMap(rsw.getResultSet(), nestedResultMapId, columnPrefix);
896           if (resultMapping.getColumnPrefix() == null) {
897             // try to fill circular reference only when columnPrefix
898             // is not specified for the nested result map (issue #215)
899             Object ancestorObject = ancestorObjects.get(nestedResultMapId);
900             if (ancestorObject != null) {
901               if (newObject) {
902                 linkObjects(metaObject, resultMapping, ancestorObject); // issue #385
903               }
904               continue;
905             }
906           } 
907           final CacheKey rowKey = createRowKey(nestedResultMap, rsw, columnPrefix);
908           final CacheKey combinedKey = combineKeys(rowKey, parentRowKey);
909           Object rowValue = nestedResultObjects.get(combinedKey);
910           boolean knownValue = (rowValue != null);
911           instantiateCollectionPropertyIfAppropriate(resultMapping, metaObject); // mandatory
912           if (anyNotNullColumnHasValue(resultMapping, columnPrefix, rsw)) {
913             rowValue = getRowValue(rsw, nestedResultMap, combinedKey, columnPrefix, rowValue);
914             if (rowValue != null && !knownValue) {
915               linkObjects(metaObject, resultMapping, rowValue);
916               foundValues = true;
917             }
918           }
919         } catch (SQLException e) {
920           throw new ExecutorException("Error getting nested result map values for '" + resultMapping.getProperty() + "'.  Cause: " + e, e);
921         }
922       }
923     }
924     return foundValues;
925   }

927   private String getColumnPrefix(String parentPrefix, ResultMapping resultMapping) {
928     final StringBuilder columnPrefixBuilder = new StringBuilder();
929     if (parentPrefix != null) {
930       columnPrefixBuilder.append(parentPrefix);
931     }
932     if (resultMapping.getColumnPrefix() != null) {
933       columnPrefixBuilder.append(resultMapping.getColumnPrefix());
934     }
935     return columnPrefixBuilder.length() == 0 ? null : columnPrefixBuilder.toString().toUpperCase(Locale.ENGLISH);
936   }

938   private boolean anyNotNullColumnHasValue(ResultMapping resultMapping, String columnPrefix, ResultSetWrapper rsw) throws SQLException {
939     Set<String> notNullColumns = resultMapping.getNotNullColumns();
940     if (notNullColumns != null && !notNullColumns.isEmpty()) {
941       ResultSet rs = rsw.getResultSet();
942       for (String column: notNullColumns) {
943         rs.getObject(prependPrefix(column, columnPrefix));
944         if (!rs.wasNull()) {
945           return true;
946         }
947       }
948       return false;
949     } else if (columnPrefix != null) {
950       for (String columnName : rsw.getColumnNames()) {
951         if (columnName.toUpperCase().startsWith(columnPrefix.toUpperCase())) {
952           return true;
953         }
954       }
955       return false;
956     }
957     return true;
958   }

969   private CacheKey createRowKey(ResultMap resultMap, ResultSetWrapper rsw, String columnPrefix) throws SQLException {
970     final CacheKey cacheKey = new CacheKey();
971     cacheKey.update(resultMap.getId());
972     List<ResultMapping> resultMappings = getResultMappingsForRowKey(resultMap);
973     if (resultMappings.size() == 0) {
974       if (Map.class.isAssignableFrom(resultMap.getType())) {
975         createRowKeyForMap(rsw, cacheKey);
976       } else {
977         createRowKeyForUnmappedProperties(resultMap, rsw, cacheKey, columnPrefix);
978       }
979     } else {
980       createRowKeyForMappedProperties(resultMap, rsw, cacheKey, resultMappings, columnPrefix);
981     }
982     if (cacheKey.getUpdateCount() < 2) {
983       return CacheKey.NULL_CACHE_KEY;
984     }    
985     return cacheKey;
986   }

988   private CacheKey combineKeys(CacheKey rowKey, CacheKey parentRowKey) {
989     if (rowKey.getUpdateCount() > 1 && parentRowKey.getUpdateCount() > 1) {
990       CacheKey combinedKey;
991       try {
992         combinedKey = rowKey.clone();
993       } catch (CloneNotSupportedException e) {
994         throw new ExecutorException("Error cloning cache key.  Cause: " + e, e);
995       }
996       combinedKey.update(parentRowKey);
997       return combinedKey;
998     }
999     return CacheKey.NULL_CACHE_KEY;
1000   }

1010   private void createRowKeyForMappedProperties(ResultMap resultMap, ResultSetWrapper rsw, CacheKey cacheKey, List<ResultMapping> resultMappings, String columnPrefix) throws SQLException {
1011     for (ResultMapping resultMapping : resultMappings) {
1012       if (resultMapping.getNestedResultMapId() != null && resultMapping.getResultSet() == null) {
1013         // Issue #392
1014         final ResultMap nestedResultMap = configuration.getResultMap(resultMapping.getNestedResultMapId());
1015         createRowKeyForMappedProperties(nestedResultMap, rsw, cacheKey, nestedResultMap.getConstructorResultMappings(),
1016             prependPrefix(resultMapping.getColumnPrefix(), columnPrefix));
1017       } else if (resultMapping.getNestedQueryId() == null) {
1018         final String column = prependPrefix(resultMapping.getColumn(), columnPrefix);
1019         final TypeHandler<?> th = resultMapping.getTypeHandler();
1020         List<String> mappedColumnNames = rsw.getMappedColumnNames(resultMap, columnPrefix);
1021         // Issue #114
1022         if (column != null && mappedColumnNames.contains(column.toUpperCase(Locale.ENGLISH))) {
1023           final Object value = th.getResult(rsw.getResultSet(), column);
1024           if (value != null || configuration.isReturnInstanceForEmptyRow()) {
1025             cacheKey.update(column);
1026             cacheKey.update(value);
1027           }
1028         }
1029       }
1030     }
1031   }

1033   private void createRowKeyForUnmappedProperties(ResultMap resultMap, ResultSetWrapper rsw, CacheKey cacheKey, String columnPrefix) throws SQLException {
1034     final MetaClass metaType = MetaClass.forClass(resultMap.getType(), reflectorFactory);
1035     List<String> unmappedColumnNames = rsw.getUnmappedColumnNames(resultMap, columnPrefix);
1036     for (String column : unmappedColumnNames) {
1037       String property = column;
1038       if (columnPrefix != null && !columnPrefix.isEmpty()) {
1039         // When columnPrefix is specified, ignore columns without the prefix.
1040         if (column.toUpperCase(Locale.ENGLISH).startsWith(columnPrefix)) {
1041           property = column.substring(columnPrefix.length());
1042         } else {
1043           continue;
1044         }
1045       }
1046       if (metaType.findProperty(property, configuration.isMapUnderscoreToCamelCase()) != null) {
1047         String value = rsw.getResultSet().getString(column);
1048         if (value != null) {
1049           cacheKey.update(column);
1050           cacheKey.update(value);
1051         }
1052       }
1053     }
1054   }

1077   private Object instantiateCollectionPropertyIfAppropriate(ResultMapping resultMapping, MetaObject metaObject) {
1078     final String propertyName = resultMapping.getProperty();
1079     Object propertyValue = metaObject.getValue(propertyName);
1080     if (propertyValue == null) {
1081       Class<?> type = resultMapping.getJavaType();
1082       if (type == null) {
1083         type = metaObject.getSetterType(propertyName);
1084       }
1085       try {
1086         if (objectFactory.isCollection(type)) {
1087           propertyValue = objectFactory.create(type);
1088           metaObject.setValue(propertyName, propertyValue);
1089           return propertyValue;
1090         }
1091       } catch (Exception e) {
1092         throw new ExecutorException("Error instantiating collection property for result '" + resultMapping.getProperty() + "'.  Cause: " + e, e);
1093       }
1094     } else if (objectFactory.isCollection(propertyValue.getClass())) {
1095       return propertyValue;
1096     }
1097     return null;
1098   }

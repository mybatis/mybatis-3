16  package org.apache.ibatis.scripting.xmltags;
17  
18  import java.lang.reflect.AccessibleObject;
19  import java.lang.reflect.Member;
20  import java.util.Map;
21  
22  import ognl.MemberAccess;
23  
24  import org.apache.ibatis.reflection.Reflector;
25  
26  /**
27   * The {@link MemberAccess} class that based on <a href=
28   * 'https://github.com/jkuhnert/ognl/blob/OGNL_3_2_1/src/java/ognl/DefaultMemberAccess.java'>DefaultMemberAccess</a>.
29   *
30   * @author Kazuki Shimizu
31   * @since 3.5.0
32   *
33   * @see <a href=
34   *      'https://github.com/jkuhnert/ognl/blob/OGNL_3_2_1/src/java/ognl/DefaultMemberAccess.java'>DefaultMemberAccess</a>
35   * @see <a href='https://github.com/jkuhnert/ognl/issues/47'>#47 of ognl</a>
36   */
37  class OgnlMemberAccess implements MemberAccess {
38  
39    private final boolean canControlMemberAccessible;
40  
41    OgnlMemberAccess() {
42      this.canControlMemberAccessible = Reflector.canControlMemberAccessible();
43    }
44  
45    @Override
46    public Object setup(Map context, Object target, Member member, String propertyName) {
47      Object result = null;
48      if (isAccessible(context, target, member, propertyName)) {
49        AccessibleObject accessible = (AccessibleObject) member;
50        if (!accessible.isAccessible()) {
51          result = Boolean.FALSE;
52          accessible.setAccessible(true);
53        }
54      }
55      return result;
56    }
57  
58    @Override
59    public void restore(Map context, Object target, Member member, String propertyName,
60        Object state) {
61      // Flipping accessible flag is not thread safe. See #1648
62    }
63  
64    @Override
65    public boolean isAccessible(Map context, Object target, Member member, String propertyName) {
66      return canControlMemberAccessible;
67    }
68  
69  }

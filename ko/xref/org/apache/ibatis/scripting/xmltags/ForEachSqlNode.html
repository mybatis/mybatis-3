18  import java.util.Map;

20  import org.apache.ibatis.parsing.GenericTokenParser;
21  import org.apache.ibatis.parsing.TokenHandler;
22  import org.apache.ibatis.session.Configuration;

24  public class ForEachSqlNode implements SqlNode {
25    public static final String ITEM_PREFIX = "__frch_";
26  
27    private ExpressionEvaluator evaluator;
28    private String collectionExpression;
29    private SqlNode contents;
30    private String open;
31    private String close;
32    private String separator;
33    private String item;
34    private String index;
35    private Configuration configuration;
36  
37    public ForEachSqlNode(Configuration configuration, SqlNode contents, String collectionExpression, String index, String item, String open, String close, String separator) {
38      this.evaluator = new ExpressionEvaluator();
39      this.collectionExpression = collectionExpression;
40      this.contents = contents;
41      this.open = open;
42      this.close = close;
43      this.separator = separator;
44      this.index = index;
45      this.item = item;
46      this.configuration = configuration;
47    }
48  
49    public boolean apply(DynamicContext context) {
50      Map<String, Object> bindings = context.getBindings();
51      final Iterable<?> iterable = evaluator.evaluateIterable(collectionExpression, bindings);
52      boolean first = true;
53      applyOpen(context);
54      int i = 0;
55      for (Object o : iterable) {
56        DynamicContext oldContext = context;
57        if (first) {
58          context = new PrefixedContext(context, "");
59        } else {
60          if (separator != null) {
61            context = new PrefixedContext(context, separator);
62          } else {
63            context = new PrefixedContext(context, "");
64          }
65        }
66        int uniqueNumber = context.getUniqueNumber();
67        if (o instanceof Map.Entry) { // Issue #709 
68          @SuppressWarnings("unchecked") 
69          Map.Entry<Object, Object> mapEntry = (Map.Entry<Object, Object>) o;
70          applyIndex(context, mapEntry.getKey(), uniqueNumber);
71          applyItem(context, mapEntry.getValue(), uniqueNumber);
72        } else {
73          applyIndex(context, i, uniqueNumber);
74          applyItem(context, o, uniqueNumber);
75        }
76        contents.apply(new FilteredDynamicContext(configuration, context, index, item, uniqueNumber));
77        if (first) first = !((PrefixedContext) context).isPrefixApplied();
78        context = oldContext;
79        i++;
80      }
81      applyClose(context);
82      return true;
83    }
84  
85    private void applyIndex(DynamicContext context, Object o, int i) {
86      if (index != null) {
87        context.bind(index, o);
88        context.bind(itemizeItem(index, i), o);
89      }
90    }
91  
92    private void applyItem(DynamicContext context, Object o, int i) {
93      if (item != null) {
94        context.bind(item, o);
95        context.bind(itemizeItem(item, i), o);
96      }
97    }
98  
99    private void applyOpen(DynamicContext context) {
100     if (open != null) {
101       context.appendSql(open);
102     }
103   }
104 
105   private void applyClose(DynamicContext context) {
106     if (close != null) {
107       context.appendSql(close);
108     }
109   }
110 
111   private static String itemizeItem(String item, int i) {
112     return new StringBuilder(ITEM_PREFIX).append(item).append("_").append(i).toString();
113   }

115   private static class FilteredDynamicContext extends DynamicContext {
116     private DynamicContext delegate;
117     private int index;
118     private String itemIndex;
119     private String item;
120 
121     public FilteredDynamicContext(Configuration configuration,DynamicContext delegate, String itemIndex, String item, int i) {
122       super(configuration, null);
123       this.delegate = delegate;
124       this.index = i;
125       this.itemIndex = itemIndex;
126       this.item = item;
127     }
128 
129     @Override
130     public Map<String, Object> getBindings() {
131       return delegate.getBindings();
132     }
133 
134     @Override
135     public void bind(String name, Object value) {
136       delegate.bind(name, value);
137     }
138 
139     @Override
140     public String getSql() {
141       return delegate.getSql();
142     }
143 
144     @Override
145     public void appendSql(String sql) {
146       GenericTokenParser parser = new GenericTokenParser("#{", "}", new TokenHandler() {
147         public String handleToken(String content) {
148           String newContent = content.replaceFirst(item, itemizeItem(item, index));
149           if (itemIndex != null && newContent.equals(content)) {
150             newContent = content.replaceFirst(itemIndex, itemizeItem(itemIndex, index));
151           }
152           return new StringBuilder("#{").append(newContent).append("}").toString();
153         }
154       });
155 
156       delegate.appendSql(parser.parse(sql));
157     }
158 
159     @Override
160     public int getUniqueNumber() {
161       return delegate.getUniqueNumber();
162     }
163 
164   }

167   private class PrefixedContext extends DynamicContext {
168     private DynamicContext delegate;
169     private String prefix;
170     private boolean prefixApplied;
171 
172     public PrefixedContext(DynamicContext delegate, String prefix) {
173       super(configuration, null);
174       this.delegate = delegate;
175       this.prefix = prefix;
176       this.prefixApplied = false;
177     }
178 
179     public boolean isPrefixApplied() {
180       return prefixApplied;
181     }
182 
183     @Override
184     public Map<String, Object> getBindings() {
185       return delegate.getBindings();
186     }
187 
188     @Override
189     public void bind(String name, Object value) {
190       delegate.bind(name, value);
191     }
192 
193     @Override
194     public void appendSql(String sql) {
195       if (!prefixApplied && sql != null && sql.trim().length() > 0) {
196         delegate.appendSql(prefix);
197         prefixApplied = true;
198       }
199       delegate.appendSql(sql);
200     }
201 
202     @Override
203     public String getSql() {
204       return delegate.getSql();
205     }
206 
207     @Override
208     public int getUniqueNumber() {
209       return delegate.getUniqueNumber();
210     }
211   }

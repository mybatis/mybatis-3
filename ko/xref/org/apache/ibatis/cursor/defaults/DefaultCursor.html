16  package org.apache.ibatis.cursor.defaults;
17  
18  import org.apache.ibatis.cursor.Cursor;
19  import org.apache.ibatis.executor.resultset.DefaultResultSetHandler;
20  import org.apache.ibatis.executor.resultset.ResultSetWrapper;
21  import org.apache.ibatis.mapping.ResultMap;
22  import org.apache.ibatis.session.ResultContext;
23  import org.apache.ibatis.session.ResultHandler;
24  import org.apache.ibatis.session.RowBounds;
25  
26  import java.sql.ResultSet;
27  import java.sql.SQLException;
28  import java.sql.Statement;
29  import java.util.Iterator;
30  import java.util.NoSuchElementException;
31  
32  /**
33   * This is the default implementation of a MyBatis Cursor.
34   * This implementation is not thread safe.
35   *
36   * @author Guillaume Darmont / guillaume@dropinocean.com
37   */
38  public class DefaultCursor<T> implements Cursor<T> {
39  
40      // ResultSetHandler stuff
41      private final DefaultResultSetHandler resultSetHandler;
42      private final ResultMap resultMap;
43      private final ResultSetWrapper rsw;
44      private final RowBounds rowBounds;
45      private final ObjectWrapperResultHandler<T> objectWrapperResultHandler = new ObjectWrapperResultHandler<T>();
46  
47      private final CursorIterator cursorIterator = new CursorIterator();
48      private boolean iteratorRetrieved;
49  
50      private CursorStatus status = CursorStatus.CREATED;
51      private int indexWithRowBound = -1;
52  
53      private enum CursorStatus {
54  
55          /**
56           * A freshly created cursor, database ResultSet consuming has not started
57           */
58          CREATED,
59          /**
60           * A cursor currently in use, database ResultSet consuming has started
61           */
62          OPEN,
63          /**
64           * A closed cursor, not fully consumed
65           */
66          CLOSED,
67          /**
68           * A fully consumed cursor, a consumed cursor is always closed
69           */
70          CONSUMED
71      }
72  
73      public DefaultCursor(DefaultResultSetHandler resultSetHandler, ResultMap resultMap, ResultSetWrapper rsw, RowBounds rowBounds) {
74          this.resultSetHandler = resultSetHandler;
75          this.resultMap = resultMap;
76          this.rsw = rsw;
77          this.rowBounds = rowBounds;
78      }
79  
80      @Override
81      public boolean isOpen() {
82          return status == CursorStatus.OPEN;
83      }
84  
85      @Override
86      public boolean isConsumed() {
87          return status == CursorStatus.CONSUMED;
88      }
89  
90      @Override
91      public int getCurrentIndex() {
92          return rowBounds.getOffset() + cursorIterator.iteratorIndex;
93      }
94  
95      @Override
96      public Iterator<T> iterator() {
97          if (iteratorRetrieved) {
98              throw new IllegalStateException("Cannot open more than one iterator on a Cursor");
99          }
100         iteratorRetrieved = true;
101         return cursorIterator;
102     }
103 
104     @Override
105     public void close() {
106         if (isClosed()) {
107             return;
108         }
109 
110         ResultSet rs = rsw.getResultSet();
111         try {
112             if (rs != null) {
113                 Statement statement = rs.getStatement();
114 
115                 rs.close();
116                 if (statement != null) {
117                     statement.close();
118                 }
119             }
120             status = CursorStatus.CLOSED;
121         } catch (SQLException e) {
122             // ignore
123         }
124     }
125 
126     protected T fetchNextUsingRowBound() {
127         T result = fetchNextObjectFromDatabase();
128         while (result != null && indexWithRowBound < rowBounds.getOffset()) {
129             result = fetchNextObjectFromDatabase();
130         }
131         return result;
132     }
133 
134     protected T fetchNextObjectFromDatabase() {
135         if (isClosed()) {
136             return null;
137         }
138 
139         try {
140             status = CursorStatus.OPEN;
141             resultSetHandler.handleRowValues(rsw, resultMap, objectWrapperResultHandler, RowBounds.DEFAULT, null);
142         } catch (SQLException e) {
143             throw new RuntimeException(e);
144         }
145 
146         T next = objectWrapperResultHandler.result;
147         if (next != null) {
148             indexWithRowBound++;
149         }
150         // No more object or limit reached
151         if (next == null || (getReadItemsCount() == rowBounds.getOffset() + rowBounds.getLimit())) {
152             close();
153             status = CursorStatus.CONSUMED;
154         }
155         objectWrapperResultHandler.result = null;
156 
157         return next;
158     }
159 
160     private boolean isClosed() {
161         return status == CursorStatus.CLOSED || status == CursorStatus.CONSUMED;
162     }
163 
164     private int getReadItemsCount() {
165         return indexWithRowBound + 1;
166     }
167 
168     private static class ObjectWrapperResultHandler<T> implements ResultHandler<T> {
169 
170         private T result;
171 
172         @Override
173         public void handleResult(ResultContext<? extends T> context) {
174             this.result = context.getResultObject();
175             context.stop();
176         }
177     }
178 
179     private class CursorIterator implements Iterator<T> {
180 
181         /**
182          * Holder for the next object to be returned
183          */
184         T object;
185 
186         /**
187          * Index of objects returned using next(), and as such, visible to users.
188          */
189         int iteratorIndex = -1;
190 
191         @Override
192         public boolean hasNext() {
193             if (object == null) {
194                 object = fetchNextUsingRowBound();
195             }
196             return object != null;
197         }
198 
199         @Override
200         public T next() {
201             // Fill next with object fetched from hasNext()
202             T next = object;
203 
204             if (next == null) {
205                 next = fetchNextUsingRowBound();
206             }
207 
208             if (next != null) {
209                 object = null;
210                 iteratorIndex++;
211                 return next;
212             }
213             throw new NoSuchElementException();
214         }
215 
216         @Override
217         public void remove() {
218             throw new UnsupportedOperationException("Cannot remove element from Cursor");
219         }
220     }
221 }

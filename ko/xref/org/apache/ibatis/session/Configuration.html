554   public Executor newExecutor(Transaction transaction, ExecutorType executorType) {
555     executorType = executorType == null ? defaultExecutorType : executorType;
556     executorType = executorType == null ? ExecutorType.SIMPLE : executorType;
557     Executor executor;
558     if (ExecutorType.BATCH == executorType) {
559       executor = new BatchExecutor(this, transaction);
560     } else if (ExecutorType.REUSE == executorType) {
561       executor = new ReuseExecutor(this, transaction);
562     } else {
563       executor = new SimpleExecutor(this, transaction);
564     }
565     if (cacheEnabled) {
566       executor = new CachingExecutor(executor);
567     }
568     executor = (Executor) interceptorChain.pluginAll(executor);
569     return executor;
570   }

754   /*
755    * Parses all the unprocessed statement nodes in the cache. It is recommended
756    * to call this method once all the mappers are added as it provides fail-fast
757    * statement validation.
758    */
759   protected void buildAllStatements() {
760     if (!incompleteResultMaps.isEmpty()) {
761       synchronized (incompleteResultMaps) {
762         // This always throws a BuilderException.
763         incompleteResultMaps.iterator().next().resolve();
764       }
765     }
766     if (!incompleteCacheRefs.isEmpty()) {
767       synchronized (incompleteCacheRefs) {
768         // This always throws a BuilderException.
769         incompleteCacheRefs.iterator().next().resolveCacheRef();
770       }
771     }
772     if (!incompleteStatements.isEmpty()) {
773       synchronized (incompleteStatements) {
774         // This always throws a BuilderException.
775         incompleteStatements.iterator().next().parseStatementNode();
776       }
777     }
778     if (!incompleteMethods.isEmpty()) {
779       synchronized (incompleteMethods) {
780         // This always throws a BuilderException.
781         incompleteMethods.iterator().next().resolve();
782       }
783     }
784   }

792   protected String extractNamespace(String statementId) {

797   // Slow but a one time cost. A better solution is welcome.
798   protected void checkGloballyForDiscriminatedNestedResultMaps(ResultMap rm) {
799     if (rm.hasNestedResultMaps()) {
800       for (Map.Entry<String, ResultMap> entry : resultMaps.entrySet()) {
801         Object value = entry.getValue();
802         if (value instanceof ResultMap) {
803           ResultMap entryResultMap = (ResultMap) value;
804           if (!entryResultMap.hasNestedResultMaps() && entryResultMap.getDiscriminator() != null) {
805             Collection<String> discriminatedResultMapNames = entryResultMap.getDiscriminator().getDiscriminatorMap().values();
806             if (discriminatedResultMapNames.contains(rm.getId())) {
807               entryResultMap.forceNestedResultMaps();
808             }
809           }
810         }
811       }
812     }
813   }
814 
815   // Slow but a one time cost. A better solution is welcome.
816   protected void checkLocallyForDiscriminatedNestedResultMaps(ResultMap rm) {
817     if (!rm.hasNestedResultMaps() && rm.getDiscriminator() != null) {
818       for (Map.Entry<String, String> entry : rm.getDiscriminator().getDiscriminatorMap().entrySet()) {
819         String discriminatedResultMapName = entry.getValue();
820         if (hasResultMap(discriminatedResultMapName)) {
821           ResultMap discriminatedResultMap = resultMaps.get(discriminatedResultMapName);
822           if (discriminatedResultMap.hasNestedResultMaps()) {
823             rm.forceNestedResultMaps();
824             break;
825           }
826         }
827       }
828     }

831   protected static class StrictMap<V> extends HashMap<String, V> {
832 
833     private static final long serialVersionUID = -4950446264854982944L;
834     private final String name;
835 
836     public StrictMap(String name, int initialCapacity, float loadFactor) {
837       super(initialCapacity, loadFactor);
838       this.name = name;
839     }
840 
841     public StrictMap(String name, int initialCapacity) {
842       super(initialCapacity);
843       this.name = name;
844     }
845 
846     public StrictMap(String name) {
847       super();
848       this.name = name;
849     }
850 
851     public StrictMap(String name, Map<String, ? extends V> m) {
852       super(m);
853       this.name = name;
854     }
855 
856     @SuppressWarnings("unchecked")
857     public V put(String key, V value) {
858       if (containsKey(key)) {
859         throw new IllegalArgumentException(name + " already contains value for " + key);
860       }
861       if (key.contains(".")) {
862         final String shortKey = getShortName(key);
863         if (super.get(shortKey) == null) {
864           super.put(shortKey, value);
865         } else {
866           super.put(shortKey, (V) new Ambiguity(shortKey));
867         }
868       }
869       return super.put(key, value);
870     }
871 
872     public V get(Object key) {
873       V value = super.get(key);
874       if (value == null) {
875         throw new IllegalArgumentException(name + " does not contain value for " + key);
876       }
877       if (value instanceof Ambiguity) {
878         throw new IllegalArgumentException(((Ambiguity) value).getSubject() + " is ambiguous in " + name
879             + " (try using the full name including the namespace, or rename one of the entries)");
880       }
881       return value;
882     }
883 
884     private String getShortName(String key) {
885       final String[] keyParts = key.split("\\.");
886       return keyParts[keyParts.length - 1];
887     }

890       final private String subject;
891 
892       public Ambiguity(String subject) {
893         this.subject = subject;
894       }
895 
896       public String getSubject() {
897         return subject;
898       }
899     }

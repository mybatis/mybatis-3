37  /**
38   * A default implementation of {@link VFS} that works for most application servers.
39   * 
40   * @author Ben Gunter
41   */
42  public class DefaultVFS extends VFS {
43    private static final Log log = LogFactory.getLog(ResolverUtil.class);

45    /** The magic header that indicates a JAR (ZIP) file. */
46    private static final byte[] JAR_MAGIC = { 'P', 'K', 3, 4 };

48    @Override
49    public boolean isValid() {
50      return true;

53    @Override
54    public List<String> list(URL url, String path) throws IOException {
55      InputStream is = null;
56      try {
57        List<String> resources = new ArrayList<String>();
58  
59        // First, try to find the URL of a JAR file containing the requested resource. If a JAR
60        // file is found, then we'll list child resources by reading the JAR.
61        URL jarUrl = findJarForResource(url);
62        if (jarUrl != null) {
63          is = jarUrl.openStream();
64          log.debug("Listing " + url);
65          resources = listResources(new JarInputStream(is), path);
66        }
67        else {
68          List<String> children = new ArrayList<String>();
69          try {
70            if (isJar(url)) {
71              // Some versions of JBoss VFS might give a JAR stream even if the resource
72              // referenced by the URL isn't actually a JAR
73              is = url.openStream();
74              JarInputStream jarInput = new JarInputStream(is);
75              log.debug("Listing " + url);
76              for (JarEntry entry; (entry = jarInput.getNextJarEntry()) != null;) {
77                log.debug("Jar entry: " + entry.getName());
78                children.add(entry.getName());
79              }
80            }
81            else {
82              /*
83               * Some servlet containers allow reading from directory resources like a
84               * text file, listing the child resources one per line. However, there is no
85               * way to differentiate between directory and file resources just by reading
86               * them. To work around that, as each line is read, try to look it up via
87               * the class loader as a child of the current resource. If any line fails
88               * then we assume the current resource is not a directory.
89               */
90              is = url.openStream();
91              BufferedReader reader = new BufferedReader(new InputStreamReader(is));
92              List<String> lines = new ArrayList<String>();
93              for (String line; (line = reader.readLine()) != null;) {
94                log.debug("Reader entry: " + line);
95                lines.add(line);
96                if (getResources(path + "/" + line).isEmpty()) {
97                  lines.clear();
98                  break;
99                }
100             }
101 
102             if (!lines.isEmpty()) {
103               log.debug("Listing " + url);
104               children.addAll(lines);
105             }
106           }
107         } catch (FileNotFoundException e) {
108           /*
109            * For file URLs the openStream() call might fail, depending on the servlet
110            * container, because directories can't be opened for reading. If that happens,
111            * then list the directory directly instead.
112            */
113           if ("file".equals(url.getProtocol())) {
114             File file = new File(url.getFile());
115             log.debug("Listing directory " + file.getAbsolutePath());
116             if (file.isDirectory()) {
117               log.debug("Listing " + url);
118               children = Arrays.asList(file.list());
119             }
120           }
121           else {
122             // No idea where the exception came from so rethrow it
123             throw e;
124           }
125         }
126 
127         // The URL prefix to use when recursively listing child resources
128         String prefix = url.toExternalForm();
129         if (!prefix.endsWith("/"))
130           prefix = prefix + "/";
131 
132         // Iterate over immediate children, adding files and recursing into directories
133         for (String child : children) {
134           String resourcePath = path + "/" + child;
135           resources.add(resourcePath);
136           URL childUrl = new URL(prefix + child);
137           resources.addAll(list(childUrl, resourcePath));
138         }
139       }
140 
141       return resources;
142     } finally {
143       try {
144         if (is != null)
145           is.close();
146       } catch (Exception e) {
147       }
148     }
149   }

151   /**
152    * List the names of the entries in the given {@link JarInputStream} that begin with the
153    * specified {@code path}. Entries will match with or without a leading slash.
154    * 
155    * @param jar The JAR input stream
156    * @param path The leading path to match
157    * @return The names of all the matching entries
158    * @throws IOException If I/O errors occur
159    */
160   protected List<String> listResources(JarInputStream jar, String path) throws IOException {
161     // Include the leading and trailing slash when matching names
162     if (!path.startsWith("/"))
163       path = "/" + path;
164     if (!path.endsWith("/"))
165       path = path + "/";
166 
167     // Iterate over the entries and collect those that begin with the requested path
168     List<String> resources = new ArrayList<String>();
169     for (JarEntry entry; (entry = jar.getNextJarEntry()) != null;) {
170       if (!entry.isDirectory()) {
171         // Add leading slash if it's missing
172         String name = entry.getName();
173         if (!name.startsWith("/"))
174           name = "/" + name;
175 
176         // Check file name
177         if (name.startsWith(path)) {
178           log.debug("Found resource: " + name);
179           resources.add(name.substring(1)); // Trim leading slash
180         }
181       }
182     }
183     return resources;
184   }

186   /**
187    * Attempts to deconstruct the given URL to find a JAR file containing the resource referenced
188    * by the URL. That is, assuming the URL references a JAR entry, this method will return a URL
189    * that references the JAR file containing the entry. If the JAR cannot be located, then this
190    * method returns null.
191    * 
192    * @param url The URL of the JAR entry.
193    * @return The URL of the JAR file, if one is found. Null if not.
194    * @throws MalformedURLException
195    */
196   protected URL findJarForResource(URL url) throws MalformedURLException {
197     log.debug("Find JAR URL: " + url);
198 
199     // If the file part of the URL is itself a URL, then that URL probably points to the JAR
200     try {
201       for (;;) {
202         url = new URL(url.getFile());
203         log.debug("Inner URL: " + url);
204       }
205     } catch (MalformedURLException e) {
206       // This will happen at some point and serves as a break in the loop
207     }
208 
209     // Look for the .jar extension and chop off everything after that
210     StringBuilder jarUrl = new StringBuilder(url.toExternalForm());
211     int index = jarUrl.lastIndexOf(".jar");
212     if (index >= 0) {
213       jarUrl.setLength(index + 4);
214       log.debug("Extracted JAR URL: " + jarUrl);
215     }
216     else {
217       log.debug("Not a JAR: " + jarUrl);
218       return null;
219     }
220 
221     // Try to open and test it
222     try {
223       URL testUrl = new URL(jarUrl.toString());
224       if (isJar(testUrl)) {
225         return testUrl;
226       }
227       else {
228         // WebLogic fix: check if the URL's file exists in the filesystem.
229         log.debug("Not a JAR: " + jarUrl);
230         jarUrl.replace(0, jarUrl.length(), testUrl.getFile());
231         File file = new File(jarUrl.toString());
232 
233         // File name might be URL-encoded
234         if (!file.exists()) {
235           try {
236             file = new File(URLEncoder.encode(jarUrl.toString(), "UTF-8"));
237           } catch (UnsupportedEncodingException e) {
238             throw new RuntimeException("Unsupported encoding?  UTF-8?  That's unpossible.");
239           }
240         }
241 
242         if (file.exists()) {
243           log.debug("Trying real file: " + file.getAbsolutePath());
244           testUrl = file.toURI().toURL();
245           if (isJar(testUrl)) {
246             return testUrl;
247           }
248         }
249       }
250     } catch (MalformedURLException e) {
251       log.warn("Invalid JAR URL: " + jarUrl);
252     }
253 
254     log.debug("Not a JAR: " + jarUrl);
255     return null;
256   }

258   /**
259    * Converts a Java package name to a path that can be looked up with a call to
260    * {@link ClassLoader#getResources(String)}.
261    * 
262    * @param packageName The Java package name to convert to a path
263    */
264   protected String getPackagePath(String packageName) {
265     return packageName == null ? null : packageName.replace('.', '/');
266   }

268   /**
269    * Returns true if the resource located at the given URL is a JAR file.
270    * 
271    * @param url The URL of the resource to test.
272    */
273   protected boolean isJar(URL url) {
274     return isJar(url, new byte[JAR_MAGIC.length]);
275   }
276 
277   /**
278    * Returns true if the resource located at the given URL is a JAR file.
279    * 
280    * @param url The URL of the resource to test.
281    * @param buffer A buffer into which the first few bytes of the resource are read. The buffer
282    *            must be at least the size of {@link #JAR_MAGIC}. (The same buffer may be reused
283    *            for multiple calls as an optimization.)
284    */
285   protected boolean isJar(URL url, byte[] buffer) {
286     InputStream is = null;
287     try {
288       is = url.openStream();
289       is.read(buffer, 0, JAR_MAGIC.length);
290       if (Arrays.equals(buffer, JAR_MAGIC)) {
291         log.debug("Found JAR: " + url);
292         return true;
293       }
294     } catch (Exception e) {
295       // Failure to read the stream means this is not a JAR
296     } finally {
297       try {
298         is.close();
299       } catch (Exception e) {
300       }
301     }
302 
303     return false;
304   }

27  /**
28   * <p>ResolverUtil is used to locate classes that are available in the/a class path and meet
29   * arbitrary conditions. The two most common conditions are that a class implements/extends
30   * another class, or that is it annotated with a specific annotation. However, through the use
31   * of the {@link Test} class it is possible to search using arbitrary conditions.</p>
32   * <p/>
33   * <p>A ClassLoader is used to locate all locations (directories and jar files) in the class
34   * path that contain classes within certain packages, and then to load those classes and
35   * check them. By default the ClassLoader returned by
36   * {@code Thread.currentThread().getContextClassLoader()} is used, but this can be overridden
37   * by calling {@link #setClassLoader(ClassLoader)} prior to invoking any of the {@code find()}
38   * methods.</p>
39   * <p/>
40   * <p>General searches are initiated by calling the
41   * {@link #find(org.apache.ibatis.io.ResolverUtil.Test, String)} ()} method and supplying
42   * a package name and a Test instance. This will cause the named package <b>and all sub-packages</b>
43   * to be scanned for classes that meet the test. There are also utility methods for the common
44   * use cases of scanning multiple packages for extensions of particular classes, or classes
45   * annotated with a specific annotation.</p>
46   * <p/>
47   * <p>The standard usage pattern for the ResolverUtil class is as follows:</p>
48   * <p/>
49   * <pre>
50   * ResolverUtil&lt;ActionBean&gt; resolver = new ResolverUtil&lt;ActionBean&gt;();
51   * resolver.findImplementation(ActionBean.class, pkg1, pkg2);
52   * resolver.find(new CustomTest(), pkg1);
53   * resolver.find(new CustomTest(), pkg2);
54   * Collection&lt;ActionBean&gt; beans = resolver.getClasses();
55   * </pre>
56   *
57   * @author Tim Fennell
58   */
59  public class ResolverUtil<T> {
60    /*
61     * An instance of Log to use for logging in this class.
62     */
63    private static final Log log = LogFactory.getLog(ResolverUtil.class);
64  
65    /**
66     * A simple interface that specifies how to test classes to determine if they
67     * are to be included in the results produced by the ResolverUtil.
68     */
69    public static interface Test {
70      /**
71       * Will be called repeatedly with candidate classes. Must return True if a class
72       * is to be included in the results, false otherwise.
73       */
74      boolean matches(Class<?> type);
75    }
76  
77    /**
78     * A Test that checks to see if each class is assignable to the provided class. Note
79     * that this test will match the parent type itself if it is presented for matching.
80     */
81    public static class IsA implements Test {
82      private Class<?> parent;
83  
84      /** Constructs an IsA test using the supplied Class as the parent class/interface. */
85      public IsA(Class<?> parentType) {
86        this.parent = parentType;
87      }
88  
89      /** Returns true if type is assignable to the parent type supplied in the constructor. */
90      public boolean matches(Class<?> type) {
91        return type != null && parent.isAssignableFrom(type);
92      }
93  
94      @Override
95      public String toString() {
96        return "is assignable to " + parent.getSimpleName();
97      }
98    }

100   /**
101    * A Test that checks to see if each class is annotated with a specific annotation. If it
102    * is, then the test returns true, otherwise false.
103    */
104   public static class AnnotatedWith implements Test {
105     private Class<? extends Annotation> annotation;
106 
107     /** Constructs an AnnotatedWith test for the specified annotation type. */
108     public AnnotatedWith(Class<? extends Annotation> annotation) {
109       this.annotation = annotation;
110     }
111 
112     /** Returns true if the type is annotated with the class provided to the constructor. */
113     public boolean matches(Class<?> type) {
114       return type != null && type.isAnnotationPresent(annotation);
115     }
116 
117     @Override
118     public String toString() {
119       return "annotated with @" + annotation.getSimpleName();
120     }
121   }

123   /** The set of matches being accumulated. */
124   private Set<Class<? extends T>> matches = new HashSet<Class<? extends T>>();
125 
126   /**
127    * The ClassLoader to use when looking for classes. If null then the ClassLoader returned
128    * by Thread.currentThread().getContextClassLoader() will be used.
129    */
130   private ClassLoader classloader;
131 
132   /**
133    * Provides access to the classes discovered so far. If no calls have been made to
134    * any of the {@code find()} methods, this set will be empty.
135    *
136    * @return the set of classes that have been discovered.
137    */
138   public Set<Class<? extends T>> getClasses() {
139     return matches;
140   }
141 
142   /**
143    * Returns the classloader that will be used for scanning for classes. If no explicit
144    * ClassLoader has been set by the calling, the context class loader will be used.
145    *
146    * @return the ClassLoader that will be used to scan for classes
147    */
148   public ClassLoader getClassLoader() {
149     return classloader == null ? Thread.currentThread().getContextClassLoader() : classloader;
150   }

152   /**
153    * Sets an explicit ClassLoader that should be used when scanning for classes. If none
154    * is set then the context classloader will be used.
155    *
156    * @param classloader a ClassLoader to use when scanning for classes
157    */
158   public void setClassLoader(ClassLoader classloader) {
159     this.classloader = classloader;
160   }

162   /**
163    * Attempts to discover classes that are assignable to the type provided. In the case
164    * that an interface is provided this method will collect implementations. In the case
165    * of a non-interface class, subclasses will be collected.  Accumulated classes can be
166    * accessed by calling {@link #getClasses()}.
167    *
168    * @param parent the class of interface to find subclasses or implementations of
169    * @param packageNames one or more package names to scan (including subpackages) for classes
170    */
171   public ResolverUtil<T> findImplementations(Class<?> parent, String... packageNames) {
172     if (packageNames == null)
173       return this;
174 
175     Test test = new IsA(parent);
176     for (String pkg : packageNames) {
177       find(test, pkg);
178     }
179 
180     return this;
181   }
182 
183   /**
184    * Attempts to discover classes that are annotated with the annotation. Accumulated
185    * classes can be accessed by calling {@link #getClasses()}.
186    *
187    * @param annotation the annotation that should be present on matching classes
188    * @param packageNames one or more package names to scan (including subpackages) for classes
189    */
190   public ResolverUtil<T> findAnnotated(Class<? extends Annotation> annotation, String... packageNames) {
191     if (packageNames == null)
192       return this;
193 
194     Test test = new AnnotatedWith(annotation);
195     for (String pkg : packageNames) {
196       find(test, pkg);
197     }
198 
199     return this;
200   }
201 
202   /**
203    * Scans for classes starting at the package provided and descending into subpackages.
204    * Each class is offered up to the Test as it is discovered, and if the Test returns
205    * true the class is retained.  Accumulated classes can be fetched by calling
206    * {@link #getClasses()}.
207    *
208    * @param test an instance of {@link Test} that will be used to filter classes
209    * @param packageName the name of the package from which to start scanning for
210    *        classes, e.g. {@code net.sourceforge.stripes}
211    */
212   public ResolverUtil<T> find(Test test, String packageName) {
213     String path = getPackagePath(packageName);
214 
215     try {
216       List<String> children = VFS.getInstance().list(path);
217       for (String child : children) {
218         if (child.endsWith(".class"))
219           addIfMatching(test, child);
220       }
221     } catch (IOException ioe) {
222       log.error("Could not read package: " + packageName, ioe);
223     }
224 
225     return this;
226   }
227 
228   /**
229    * Converts a Java package name to a path that can be looked up with a call to
230    * {@link ClassLoader#getResources(String)}.
231    * 
232    * @param packageName The Java package name to convert to a path
233    */
234   protected String getPackagePath(String packageName) {
235     return packageName == null ? null : packageName.replace('.', '/');
236   }
237 
238   /**
239    * Add the class designated by the fully qualified class name provided to the set of
240    * resolved classes if and only if it is approved by the Test supplied.
241    *
242    * @param test the test used to determine if the class matches
243    * @param fqn the fully qualified name of a class
244    */
245   @SuppressWarnings("unchecked")
246   protected void addIfMatching(Test test, String fqn) {
247     try {
248       String externalName = fqn.substring(0, fqn.indexOf('.')).replace('/', '.');
249       ClassLoader loader = getClassLoader();
250       log.debug("Checking to see if class " + externalName + " matches criteria [" + test + "]");
251 
252       Class<?> type = loader.loadClass(externalName);
253       if (test.matches(type)) {
254         matches.add((Class<T>) type);
255       }
256     } catch (Throwable t) {
257       log.warn("Could not examine class '" + fqn + "'" + " due to a " +
258           t.getClass().getName() + " with message: " + t.getMessage());
259     }
260   }
